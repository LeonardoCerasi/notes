\section{Electron-positron annihilation}

Amplitude $ e^- e^+ \ra \mu^- \mu^+ $ using Feynman rules (which are valid at all perturbation orders):
\begin{equation*}
  \begin{split}
    \mat
    & = \bar{v}_e(\ve{p}_2 , r_2) (-\img e \gamma^\alpha) u_e(\ve{p}_1 , r_1) \frac{-\img \eta_{\alpha \beta}}{q^2} \bar{u}_\mu(\ve{p}_3 , r_3) (-\img e \gamma^\beta) v_\mu(\ve{p}_4 , r_4) \\
    & = \img e^2 [\bar{v}_e(\ve{p}_2 , r_2) \gamma^\alpha ue(\ve{p}_1 , r_1)] [\bar{u}_\mu(\ve{p}_3 , r_3) \gamma^\beta v(\ve{p}_4 , r_4)] \frac{\eta_{\alpha \beta}}{q^2}
  \end{split}
\end{equation*}
with $ p_1 + p_2 = p_3 + p_4 $. We can use the fact that spinors are helicity eigenstates (see slides for eqs, derived in second/third lecture).

Assume we have an unpolarized beam of $ e^- e^+ $ and that we do not measure final polarizations for $ \mu^+ \mu^- $: in this case we need to evaluate the averaged amplitude:
\begin{equation}
  \braket{\abs{\mat}^2} = \frac{1}{4} \sum_{\text{polarizations}} \abs{\mat}^2
\end{equation}
where the $ \frac{1}{4} $ factor comes from the fact that there are four possible initial polarizations and they are equally probable ($ ++ $, $ +- $, $ -+ $, $ -- $ helicities).

Assume to be in the CM frame, i.e. $ \ve{p}_1 = - \ve{p}_2 $ and $ \ve{p}_3 = - \ve{p}_4 $, and to be in the ultra-relativistic limit, i.e. $ \sqrt{s} \equiv E_\text{cm} = (p_1 + p_2)^2 \gg m_e^2 , m_\mu^2 $. WLOG, assume that $ e^+ e^- $ are collimated along the $ z $-axis, so that:
\begin{equation*}
  p_1 = (E, 0, 0, E)
  \qquad \qquad
  p_2 = (E, 0, 0, -E)
\end{equation*}
and assume that $ \mu^+ \mu^- $ move in the $ xz $-plane, so that:
\begin{equation*}
  p_3 = (E, E \sin \theta, 0, E \cos \theta)
  \qquad \quad
  p_4 = (E, -E \sin \theta, 0, -E \cos \theta)
\end{equation*}
To compute the Dirac spinors for $ e^- $, take the $ E \gg m_e $ and $ \theta , \varphi \ra 0 $ of the eqs:
\begin{equation*}
  u_{e \uparrow} = \sqrt{E}
  \begin{pmatrix}
    1 \\ 0 \\ 1 \\ 0
  \end{pmatrix}
  \qquad \qquad
  u_{e \downarrow} = \sqrt{E}
  \begin{pmatrix}
    0 \\ 1 \\ 0 \\ -1
  \end{pmatrix}
\end{equation*}
For the $ e^+ $, take $ E \gg m_e $ and $ \theta , \varphi \ra \pi $ instead:
\begin{equation*}
  v_{e \uparrow} = \sqrt{E}
  \begin{pmatrix}
    1 \\ 0 \\ -1 \\ 0
  \end{pmatrix}
  \qquad \qquad
  v_{e \downarrow} = \sqrt{E}
  \begin{pmatrix}
    0 \\ -1 \\ 0 \\ -1
  \end{pmatrix}
\end{equation*}
To compute the Dirac spinors for $ \mu^- $, take the $ E \gg m_\mu $ and $ \varphi \ra 0 $ of the eqs:
\begin{equation*}
  u_{\mu \uparrow} = \sqrt{E}
  \begin{pmatrix}
    \cos \frac{\theta}{2} \\ \sin \frac{\theta}{2} \\ \cos \frac{\theta}{2} \\ \sin \frac{\theta}{2}
  \end{pmatrix}
  \qquad \qquad
  u_{e \downarrow} = \sqrt{E}
  \begin{pmatrix}
    - \sin \frac{\theta}{2} \\ \cos \frac{\theta}{2} \\ \sin \frac{\theta}{2} \\ - \cos \frac{\theta}{2}
  \end{pmatrix}
\end{equation*}
For the $ \mu^+ $, take $ E \gg m_\mu $ and $ \theta \ra \pi - \theta , \varphi \ra \pi $ instead:
\begin{equation*}
  v_{\mu \uparrow} = \sqrt{E}
  \begin{pmatrix}
    \cos \frac{\theta}{2} \\ \sin \frac{\theta}{2} \\ - \cos \frac{\theta}{2} \\ - \sin \frac{\theta}{2}
  \end{pmatrix}
  \qquad \qquad
  v_{e \downarrow} = \sqrt{E}
  \begin{pmatrix}
    \sin \frac{\theta}{2} \\ - \cos \frac{\theta}{2} \\ \sin \frac{\theta}{2} \\ - \cos \frac{\theta}{2}
  \end{pmatrix}
\end{equation*}

Now, we can compute the electron current for all possible polarizations:
\begin{equation}
  j_e^\alpha = \bar{v}_e \gamma^\alpha u_e
\end{equation}
\begin{equation*}
  j_e^0 = \bar{v}_e \gamma^0 u_e = v_e\dg \gamma^0 \gamma^0 u_e = v_e\dg u_e = v_{e,1}^* u_{e,1} + v_{e,2}^* u_{e,2} + v_{e,3}^* u_{e,3} + v_{e,4}^* u_{e,4}
\end{equation*}
\begin{equation*}
  j_e^1 = \bar{v}_e \gamma^1 u_e = v_e\dg \gamma^0 \gamma^1 u_e = v_{e,1}^* u_{e,4} + v_{e,2}^* u_{e,3} + v_{e,3}^* u_{e,2} + v_{e,4}^* u_{e,1}
\end{equation*}
\begin{equation*}
  j_e^2 = \bar{v}_e \gamma^1 u_e = v_e\dg \gamma^0 \gamma^2 u_e = -\img [ v_{e,1}^* u_{e,4} + v_{e,2}^* u_{e,3} + v_{e,3}^* u_{e,2} + v_{e,4}^* u_{e,1} ]
\end{equation*}
\begin{equation*}
  j_e^1 = \bar{v}_e \gamma^1 u_e = v_e\dg \gamma^0 \gamma^1 u_e = v_{e,1}^* u_{e,3} - v_{e,2}^* u_{e,4} + v_{e,3}^* u_{e,1} - v_{e,4}^* u_{e,2}
\end{equation*}
Using the above expressions:
\begin{equation*}
  j_{e \uparrow \uparrow}^\alpha = E (0,0,0,0)
\end{equation*}
\begin{equation*}
  j_{e \downarrow \downarrow}^\alpha = E (0,0,0,0)
\end{equation*}
\begin{equation*}
  j_{e \downarrow \uparrow}^\alpha = 2E (0,-1,\img,0)
\end{equation*}
\begin{equation*}
  j_{e \uparrow \downarrow}^\alpha = 2E (0,-1,-\img,0)
\end{equation*}

Doing the same calculation for the muon current (in the ultra-relativistic limit, electrons and muons are equal, since their only difference is their mass):
\begin{equation*}
  j_{\mu \uparrow \uparrow}^\alpha = E (0,0,0,0)
\end{equation*}
\begin{equation*}
  j_{\mu \downarrow \downarrow}^\alpha = E (0,0,0,0)
\end{equation*}
\begin{equation*}
  j_{\mu \downarrow \uparrow}^\alpha = 2E (0, -\cos \theta, \img, \sin \theta)
\end{equation*}
\begin{equation*}
  j_{\mu \uparrow \downarrow}^\alpha = 2E (0, - \cos \theta, -\img, \sin \theta)
\end{equation*}

It is now possible to evaluate the scattering amplitude in all non-vanishing cases (with $ \eta = (+,-,-,-) $ and $ q^2 = (2E)^2 $):
\begin{equation*}
  \mat_{\uparrow \downarrow \ra \uparrow \downarrow} = \img e^2 j_{e \uparrow \downarrow}^\alpha j_{\mu \uparrow \downarrow}^\beta \frac{\eta_{\alpha \beta}}{q^2} = \frac{\img e^2}{q^2} (2E)^2 \left[ -\cos \theta - 1 \right] = - \img e^2 \left[ 1 + \cos \theta \right]
\end{equation*}
\begin{equation*}
  \abs{\mat_{\uparrow \downarrow \ra \uparrow \downarrow}}^2 = \alpha^2 (4\pi)^2 (1 + \cos \theta)^2
\end{equation*}
\begin{equation*}
  \abs{\mat_{\downarrow \uparrow \ra \downarrow \uparrow}}^2 = \alpha^2 (4\pi)^2 (1 + \cos \theta)^2
\end{equation*}

\begin{equation*}
  \mat_{\uparrow \downarrow \ra \downarrow \uparrow} = \img e^2 j_{e \uparrow \downarrow}^\alpha j_{\mu \downarrow \uparrow}^\beta \frac{\eta_{\alpha \beta}}{q^2} = \img e^2 \left[ 1 - \cos \theta \right]
\end{equation*}
\begin{equation*}
  \abs{\mat_{\uparrow \downarrow \ra \downarrow \uparrow}}^2 = \alpha^2 (4\pi)^2 (1 - \cos \theta)^2
\end{equation*}
\begin{equation*}
  \abs{\mat_{\downarrow \uparrow \ra \downarrow \uparrow}}^2 = \alpha^2 (4\pi)^2 (1 - \cos \theta)^2
\end{equation*}

Finally, the averaged amplitude is:
\begin{equation*}
  \begin{split}
    \braket{\abs{\mat}^2}
    & = \frac{1}{4} \left[ \abs{\mat_{\uparrow \downarrow \ra \uparrow \downarrow}}^2 + \abs{\mat_{\downarrow \uparrow \ra \downarrow \uparrow}}^2 + \abs{\mat_{\uparrow \downarrow \ra \downarrow \uparrow}}^2 + \abs{\mat_{\downarrow \uparrow \ra \downarrow \uparrow}}^2 \right] \\
    & = \alpha^2 (4\pi)^2 \frac{1}{2} \left[ 1 + \cos^2 \theta + 2 \cos \theta + 1 + \cos^2 \theta - 2 \cos \theta \right] \\
    & = \alpha^2 (4\pi)^2 [1 + \cos^2 \theta]
  \end{split}
\end{equation*}

The cross-section can be computed using the formula derived a few lectures ago:
\begin{equation}
  \frac{\dd \theta}{\dd \Omega^*} = \frac{1}{64\pi^2 s} \frac{\abs{\ve{p}_4^*}}{\abs{\ve{p}_1^*}} \braket{\abs{\mat}^2}
\end{equation}
where $ \Omega^* $ is the solid angle in the CM frame. Therefore, for the $ e^+ e^- \ra \mu^+ \mu^- $ scattering:
\begin{equation}
  \frac{\dd \sigma}{\dd \Omega^*} = \frac{\alpha^2}{4s} (1 + \cos^2 \theta)
\end{equation}
This cross-section is a parabola in $ \cos \theta $: clearly, there is a forward-backward symmetry ($ \cos \theta \lra - \cos \theta $), linked to the conservation of angular momentum (in particular of spin, in this case).

Consider first the $ + - \ra - + $ case with $ \theta = \pi $: this case is suppressed, since in the initial state the total spin is along the positive $ z $-axis, while in the final state it is along the negative $ z $-axis, thus violating angular momentum conservation. % --> + -->  ===>  <-- + <--  (spins)
This contribution is proportional to $ \alpha (1 + \cos \theta) $, while the $ + - \ra + - $ contribution is proportional to $ \alpha (1 - \cos \theta) $, which result in the forward-backward symmetry.

\subsection{Chirality}

Chirality eigenstates are eigenstates of $ \gamma^5 \defeq \img \gamma^0 \gamma^1 \gamma^2 \gamma^3 $ (recall that $ (\gamma^5)^2 = \mt{I}_4 $, $ {\gamma^5}\dg = \gamma^5 $ and $ \{\gamma^5 , \gamma^\mu\} = 0 \,\,\forall \mu = 0,1,2,3 $). These eigenstates are found as:
\begin{equation*}
  \gamma^5 u = \lambda u
  \quad \implies \quad
  \gamma^5
  \begin{pmatrix}
    u_A \\ u_B
  \end{pmatrix}
  = \lambda
  \begin{pmatrix}
    u_A \\ u_B
  \end{pmatrix}
  \quad \implies \quad
  \begin{cases}
    u_A = \lambda u_B  \\
    u_B = \lambda u_A  \\
  \end{cases}
  \quad \implies \quad
  \begin{cases}
    u_A = \lambda^2 u_A  \\
    u_B = \lambda^2 u_B
  \end{cases}
\end{equation*}
Clearly, $ \lambda = \pm 1 $: eigenstates with $ \lambda = +1 $ are called right-handed and $ \lambda = -1 $ left-handed.

\section{Trace techniques}

To compute $ \abs{\mat}^2 $, we need to evaluate $ \mat^* $. Consider again $ e^+ e^- \ra \mu^+ \mu^- $, which has:
\begin{equation*}
  \mat = \frac{\img e^2}{q^2} \bar{v}_e(p_2) \gamma^\alpha u_e(p_1) \eta_{\alpha \beta} \bar{u}_\mu(p_3) \gamma^\beta v_\mu(p_4)
\end{equation*}
The conjugates of the vector currents are found as:
\begin{equation*}
  (\bar{v} \gamma^\alpha u)^* = (v\dg \gamma^0 \gamma^\alpha u)^* = (v\dg \gamma^0 \gamma^\alpha u)\dg = u\dg {\gamma^\alpha}\dg \gamma^0 v = u\dg \gamma^0 \gamma^\alpha v = \bar{u} \gamma^\alpha v
\end{equation*}
Where we used $ {\gamma^\alpha}\dg \gamma^0 = \gamma^0 \gamma^\alpha $. Therefore:
\begin{equation*}
  \begin{split}
    \abs{\mat}^2
    & = \frac{e^4}{q^4} \left[ \bar{v}_e(\ve{p}_2 , r_2) \gamma^\alpha u_e(\ve{p}_1 , r_1) \bar{u}_\mu(\ve{p}_3 , r_3) \gamma_\alpha v_\mu(\ve{p}_4 , r_4) \right] \times \\
    & \qquad \qquad \qquad \qquad \left[ \bar{u}_e(\ve{p}_1 , r_1) \gamma^\beta v_e(\ve{p}_2 , r_2) \bar{v}_\mu(\ve{p}_4 , r_4) \gamma_\beta u_\mu(\ve{p}_3 , r_3) \right] \\
    & = \frac{e^4}{q^4} \left[ \bar{v}_e(\ve{p}_2 , r_2) \gamma^\alpha u_e(\ve{p}_1 , r_1) \bar{u}_e(\ve{p}_1 , r_1) \gamma^\beta v_e(\ve{p}_2 , r_2) \right] \times \\
    & \qquad \qquad \qquad \qquad \left[ \bar{u}_\mu(\ve{p}_3 , r_3) \gamma_\alpha v_\mu(\ve{p}_4 , r_4) \bar{v}_\mu(\ve{p}_4 , r_4) \gamma_\beta u_\mu(\ve{p}_3 , r_3) \right]
  \end{split}
\end{equation*}
It is now possible to evaluate the averaged matrix element (with explicit spinor indices):
\begin{equation*}
  \begin{split}
    \braket{\abs{\mat}^2}
    & = \frac{e^4}{4q^4} \sum_{r_1 , r_2 , r_3 , r_4} \left[ \bar{v}_{e,a}(\ve{p}_2 , r_2) \gamma^\alpha_{ab} u_{e,b}(\ve{p}_1 , r_1) \bar{u}_{e,c}(\ve{p}_1 , r_1) \gamma^\beta_{cd} v_{e,d}(\ve{p}_2 , r_2) \right] \times \\
    & \qquad \qquad \qquad \qquad \left[ \bar{u}_{\mu,f}(\ve{p}_3 , r_3) \gamma_{\alpha,fg} v_{\mu,g}(\ve{p}_4 , r_4) \bar{v}_{\mu,h}(\ve{p}_4 , r_4) \gamma_{\beta,hi} u_{\mu,i}(\ve{p}_3 , r_3) \right]
  \end{split}
\end{equation*}
This expression can be simplified using completeness relations for spinors:
\begin{equation}
  \sum_s u(\ve{p},s) \bar{u}(\ve{p},s) = \slashed{p} + m
  \qquad \qquad
  \sum_s v(\ve{p},s) \bar{v}(\ve{p},s) = \slashed{p} - m
\end{equation}
With appropriate spinor indices:
\begin{equation*}
  \begin{split}
    \braket{\abs{\mat}^2}
    & = \frac{e^4}{4q^4} \left[ (\slashed{p}_2 - m_e)_{da} \gamma^\alpha_{ab} (\slashed{p}_1 + m_e)_{bc} \gamma^\beta_{cd} \right] \left[ (\slashed{p}_3 - m_\mu)_{if} \gamma_{\alpha,fg} (\slashed{p}_4 - m_\mu)_{gh} \gamma_{\beta,hi} \right] \\
    & = \frac{e^4}{4q^4} \tr \left[ (\slashed{p}_2 - m_e) \gamma^\alpha (\slashed{p}_1 + m_e) \gamma^\beta \right] \tr \left[ (\slashed{p}_3 - m_\mu) \gamma_{\alpha} (\slashed{p}_4 - m_\mu) \gamma_{\beta} \right]
  \end{split}
\end{equation*}
In the ultra-relativistic limit masses are negligible, thus:
\begin{equation*}
  \begin{split}
    \braket{\abs{\mat}^2}
    & = \frac{e^4}{4q^4} \tr \left[ \slashed{p}_2 \gamma^\alpha \slashed{p}_1 \gamma^\beta \right] \tr \left[ \slashed{p}_3 \gamma_{\alpha} \slashed{p}_4 \gamma_{\beta} \right] \\
    & = \frac{e^4}{4q^4} {p_1}_\mu {p_2}_\nu p_3^\rho p_4^\sigma \tr \{ \gamma^\nu \gamma^\alpha \gamma^\mu \gamma^\beta \} \tr \{ \gamma_\rho \gamma_\alpha \gamma_\sigma \gamma_\beta \} \\
  \end{split}
\end{equation*}
Using the trace identity:
\begin{equation}
  \tr \{ \gamma_\mu \gamma_\nu \gamma_\alpha \gamma_\beta \} = 4 \left( \eta_{\mu \nu} \eta_{\alpha \beta} - \eta_{\mu \alpha} \eta_{\nu \beta} + \eta_{\mu \beta} \eta_{\nu \alpha} \right)
\end{equation}
the unpolarized matrix element becomes:
\begin{equation*}
  \begin{split}
    \braket{\abs{\mat}^2}
    & = \frac{4e^4}{q^4} \left[ p_2^\alpha p_1^\beta + p_1^\alpha p_2^\beta - (p_1 \cdot p_2) \eta^{\alpha \beta} \right] \left[ {p_3}_\alpha {p_4}_\beta + {p_4}_\alpha {p_3}_\beta - (p_3 \cdot p_4) \eta_{\alpha \beta} \right] \\
    & = \frac{4e^4}{q^4} \left[ 2 (p_1 \cdot p_3) (p_2 \cdot p_4) + 2 (p_1 \cdot p_4) (p_2 \cdot p_3) - 4 (p_1 \cdot p_2) (p_3 \cdot p_4) + 4 (p_1 \cdot p_2) (p_3 \cdot p_4) \right] \\
    & = \frac{8e^4}{q^4} \left[ (p_1 \cdot p_3) (p_2 \cdot p_4) + (p_1 \cdot p_4) (p_2 \cdot p_3) \right]
  \end{split}
\end{equation*}
This expression can be explicited in the CM frame, where:
\begin{align*}
  & p_1 = (E, 0, 0, E)
  \qquad \qquad
  & p_2 = (E, 0, 0, -E)
  \\
  & p_3 = (E, E \sin \theta, 0, E \cos \theta)
  \qquad \quad
  & p_4 = (E, -E \sin \theta, 0, -E \cos \theta)
\end{align*}
\begin{equation*}
  \braket{\abs{\mat}^2} = \frac{8e^4}{q^4} \left[ E^4 \left( 1 + \cos \theta \right)^2 + E^4 \left( 1 - \cos \theta \right)^2 \right]
\end{equation*}
Using $ q^2 = (p_1 + p_2)^2 = 4E^2 $, the final expression is found to be:
\begin{equation}
  \braket{\abs{\mat}^2} = e^4 \left[ 1 + \cos^2 \theta \right]
\end{equation}
which is the same result obtained with Dirac spinors.

\section{Crossing symmetry}

Now, we want to compute the cross-section for $ e^- \mu^- \ra e^- \mu^- $. In this case:
\begin{equation*}
  \mat_{e\mu \ra e\mu} = \frac{\img e^2}{q^2} \bar{u}_e(\ve{p}_3 , r_3) \gamma^\alpha u_e(\ve{p}_1 , r_1) \bar{u}_\mu(\ve{p}_4 ,r_4) \gamma_\alpha u_\mu(\ve{p}_2 , r_2)
\end{equation*}
\begin{equation*}
  \mat_{ee \ra \mu\mu} = \frac{\img e^2}{q^2} \bar{v}_e(\ve{p}_2 , r_2) \gamma^\alpha u_e(\ve{p}_1 , r_1) \bar{u}_\mu(\ve{p}_3 , r_3) \gamma_\alpha v_\mu(\ve{p}_4 , r_4)
\end{equation*}
The similarity between these matrix elements is not an accident: indeed, $ \mat_{e\mu \ra e\mu} $ can be obtained from $ \mat_{ee \ra \mu\mu} $ by substitutions:
\begin{align*}
  \bar{v}_e(\ve{p}_2 , r_2) &\longrightarrow \bar{u}_e(\ve{p}_3 , r_3)
  \\
  \bar{u}_\mu(\ve{p}_3 , r_3) &\longrightarrow \bar{u}_\mu(\ve{p}_4 , r_4)
  \\
  v_\mu(\ve{p}_4 , r_4) &\longrightarrow u_\mu(\ve{p}_2 , r_2)
\end{align*}
In order to take into account that the completeness relations for $ v $ and $ u $ differ by a negative sign in the mass term, the previous changes are equivalent to:
\begin{equation}
  p_2 \mapsto -p_3
  \qquad
  p_3 \mapsto p_4
  \qquad
  p_4 \mapsto -p_2
\end{equation}
In terms of Mandelstam variables:
\begin{equation}
  \braket{\abs{\mat_{ee \ra \mu\mu}^2}} = 2e^4 \frac{t^2 + u^2}{s^2}
\end{equation}
and the changes are:
\begin{equation}
  s \mapsto t
  \qquad
  t \mapsto u
  \qquad
  u \mapsto s
\end{equation}
hence:
\begin{equation}
  \braket{\abs{\mat_{e\mu \ra e\mu}}^2} = 2e^4 \frac{u^2 + s^2}{t^2}
\end{equation}
In general, \bctxt{crossing symmetry} states that scattering amplitudes do not change when exchanging an incoming particle with an outgoing antiparticle with momentum reversed, and vice versa.

Consider the annihilation $ e^- e^+ \ra f \bar{f} $, where $ f $ is a generic fermion which might not be relativistic. Then it can be shown that:
\begin{equation}
  \sigma_{e\bar{e} \ra f\bar{f}} = \frac{4\pi\alpha^2 Q_f^2}{3s} \beta_f \frac{3 - \beta_f^2}{2}
\end{equation}
where $ Q_f $ is the electric charge of $ f $ in units of $ e $ and the $ \beta $-factor of $ f $ is:
\begin{equation}
  \beta_f^2 \equiv \frac{p_f}{E_f} = 1 - \frac{4m_f^2}{s}
\end{equation}

\chapter{Lecture 17/03}

The electron (probe) starts interacting with the constituents of the proton: the cross-section becomes an incoherent sum over all possible electron-parton scatterings, reproducing experimental results. However, to compute these cross-section the parton distribution functions are needed.

The structure function of the proton is (Callan--Gross relation):
\begin{equation*}
  F_2^{ep}(x,Q^2) = 2x F_1^{eq}(x,Q^2) = x \sum_i Q_i^2 q_i^p(x)
\end{equation*}
For up and down quarks $ Q_u = \frac{2}{3} $ and $ Q_d = - \frac{1}{3} $, hence:
\begin{equation*}
  F_2^{ep}(x,Q^2) = x \left[ \frac{4}{9} u^p(x) + \frac{1}{9} d^p(x) + \frac{4}{9} \bar{u}^p(x) + \frac{1}{3} \bar{d}^p(x) + \dots \right]
\end{equation*}
where the contributions from heavier particles (which are virtual in the proton, due to their large mass) are neglected. Doing the same thing for the neutron:
\begin{equation*}
  F_2^{en}(x,Q^2) = x \left[ \frac{4}{9} u^n(x) + \frac{1}{9} d^n(x) + \frac{4}{9} \bar{u}^n(x) + \frac{1}{3} \bar{d}^n(x) + \dots \right]
\end{equation*}
At first order, the problem is equivalent to the neutron by making the transformation $ u \lra d $, hence we expect $ d^n(x) = u^p(x) $ and $ d^p(x) = u^n(x) $ (and respective charge conjugates). In order to simplify the notation, define:
\begin{equation*}
  u(x) \equiv u^p(x) = d^n(x)
  \qquad \qquad
  d(x) = d^p(x) = u^n(x)
\end{equation*}
so that:
\begin{equation}
  F_2^{ep}(x,Q^2) = x \left[ \frac{4}{9} u(x) + \frac{1}{9} d(x) + \frac{4}{9} \bar{u}(x) + \frac{1}{3} \bar{d}(x) + \dots \right]
\end{equation}
\begin{equation}
  F_2^{en}(x,Q^2) = x \left[ \frac{4}{9} d(x) + \frac{1}{9} u(x) + \frac{4}{9} \bar{d}(x) + \frac{1}{3} \bar{u}(x) + \dots \right]
\end{equation}
Then:
\begin{equation*}
  \int_0^1 \dd x \, F_2^{ep}(x) = \frac{4}{9} f_u + \frac{1}{9} f_d
  \qquad \qquad
  \int_0^1 \dd x \, F_2^{en}(x) = \frac{4}{9} f_d + \frac{1}{9} f_u
\end{equation*}
where the momentum fractions are defined as:
\begin{equation*}
  f_u \equiv \int_0^1 \dd x \, x \left[ u(x) + \bar{u}(x) \right]
  \qquad \quad
  f_d \equiv \int_0^1 \dd x \, x \left[ d(x) + \bar{d}(x) \right]
\end{equation*}
From SLAC data (see notes) $ \int_0^1 \dd x \, F_2^{ep}(x) = 0.18 $, and similarly $ \int_0^1 \dd x \, F_2^{en}(x) = 0.12 $, thus:
\begin{equation*}
  \begin{cases}
    \frac{4}{9} f_u + \frac{1}{9} f_d = 0.18 \\
    \frac{1}{9} f_u + \frac{4}{9} f_d = 0.12 \\
  \end{cases}
  \quad \implies \quad
  f_u \simeq 0.36
  \qquad
  f_d \simeq 0.18
\end{equation*}
The fraction $ f_u $, which is the momentum fraction of up quarks in the proton and down quarks in the neutron, is twice the fraction $ f_d $, suggesting that $ p = uud $ and $ n = udd $. But doing the sum $ f_u + f_d \simeq 0.51 $, thus half the momentum of nucleons is missing: this is evidence that gluons\footnotemark carry much of the total momentum.

\footnotetext{Gluons cannot be studied with electromagnetic probes, since they only interact through the strong force.}

Studying the structure function of deuterium $ \ch{D} = pn $:
\begin{equation*}
  F_2^{e\ch{D}}(x,Q^2) = \frac{1}{2} \left[ F_2^{ep}(x,Q^2) + F_2^{en}(x,Q^2) \right] = \frac{5}{18} x \left[ u(x) + \bar{u}(x) + d(x) + \bar{d}(x) \right]
\end{equation*}
results into:
\begin{equation*}
  \int_0^1 \dd x \, F_2^{e\ch{D}}(x) = \frac{5}{18} ( f_u + f_d )
  \quad \implies \quad
  \frac{\int_0^1 \dd x \, F_2^{e\ch{D}}(x)}{\int_0^1 \dd x \, F_2^{ep}(x)} = \frac{5}{2} \frac{f_u + f_d}{4 f_u + f_d} = 0.84
  \quad \implies \quad
  \frac{f_d}{f_u} \simeq 0.52
\end{equation*}
which confirms the preceding result.

Now, we rewrite $ F_2 $ in terms of parton distribution functions for valence and sea quarks. Sea quarks are produced by gluons in pairs $ q\bar{q} $, while valence quarks are the main constituents (e.g. $ p = uud $). Writing $ u(x) = u_\text{v}(x) + u_\text{s}(x) $, $ d(x) = d_\text{v}(x) + d_\text{s}(x) $, $ \bar{u}(x) = \bar{u}_\text{s}(x) $ and $ \bar{d}(x) = \bar{d}_\text{s}(x) $ (since, being the proton a particle, antiparticles can only be produced in $ q\bar{q} $ pairs), we expect:
\begin{equation*}
  \int_0^1 \dd x \, u_\text{v}(x) = 2
  \qquad \qquad
  \int_0^1 \dd x \, d_\text{v}(x) = 1
\end{equation*}
Since sea quarks are produced in pairs, we expect that $ u_\text{s}(x) = \bar{u}_\text{s}(x) = d_\text{s}(x) = \bar{d}_\text{s}(x) \equiv \text{s}(x) $, so that:
\begin{equation*}
  F_2^{ep}(x) = x \left[ \frac{4}{9} u_\text{v}(x) + \frac{1}{9} d_\text{v}(x) + \frac{10}{9} \text{s}(x) \right]
\end{equation*}
\begin{equation*}
  F_2^{en}(x) = x \left[ \frac{4}{9} d_\text{v}(x) + \frac{1}{9} u_\text{v}(x) + \frac{10}{9} \text{s}(x) \right]
\end{equation*}
Then:
\begin{equation*}
  \frac{F_2^{en}(x)}{F_2^{ep}(x)} = \frac{4 d_\text{v}(x) + u_\text{v}(x) + 10 \text{s}(x)}{d u_\text{v}(x) + d_\text{v}(x) + 10 \text{s}(x)}
\end{equation*}
We expect that:
\begin{itemize}
  \item for $ x \ra 0 $, i.e. very low carried momentum fraction, the sea quarks dominate, hence the ratio is expected to go to $ 1 $;
  \item for $ x \ra 1 $, i.e. very high carried momentum fraction, the sea quarks are negligible, hence using the approximation $ u_\text{v}(x) \sim 2 d_\text{v}(x) $ the ratio is expected to go to $ \frac{2}{3} $.
\end{itemize}
From SLAC data (see notes), the $ x \ra 0 $ limit matches experimental data, confirming that sea quarks dominate, while the $ x \ra 1 $ is wrong: from experimental data, in the $ x \ra 1 $ limit the ratio goes to $ \frac{1}{4} $. We still don't know precisely why this is the case, but the most simple explanation comes from the Pauli exclusion principle: if one quarks carries $ x \ra 1 $, the other two both carry $ x \ra 0 $, but if the latter ones are the up quarks, this violates the Pauli exclusion principle. For this reason, the $ x \ra 1 $ limit can only be determined by an up quark carrying almost all momentum, hence setting $ d_\text{v} \sim 0 $ the result $ \frac{1}{4} $ is recovered.

\paragraph{Violation of Bjorken scaling}

Bjorken scaling implies that $ F_2(x,Q^2) $ is independent of $ Q^2 $, however experimental data shows that this is true only for $ 0.01 \ll x \ll 0.5 $: this means that $ F_2(x,Q^2) $ is actually weakly dependent on $ Q^2 $. For $ x \ll 0.01 $, $ F_2 $ increases with $ Q^2 $, while for $ x \gg 0.5 $ it decreases with $ Q^2 $.

This violation of Bjorken scaling is expected: higher $ Q^2 $ means lower probe wavelength, hence higher resolution power, allowing for a better measurement of sea particles (thus shifting the structure function towards low $ x $, since sea quarks carry lower momentum fractions).

\paragraph{Asimmetry in sea quarks}

The expectation of all sea quarks' distribution function being equal (to $ \text{s}(x) $) is wrong too, probably due to the Pauli exclusion principle again.

\begin{example}{Exercise on deep inelastic scattering $ ep $}{}
  Assume that $ e(p_1) $, $ p(p_2) $ and the scattered electron $ e(p_3) $ have momenta:
  \begin{equation*}
    p_1 = (E_1 , 0 , 0 , E_1)
    \qquad \qquad
    p_" = (E_2 , 0 , 0 , - E_2)
    \qquad \qquad
    p_3 = (E_3 , E_3 \sin \theta , 0 , E_3 \cos \theta)
  \end{equation*}
  so that $ q^2 = (p_1 - p_3)^2 \simeq - 2 p_1 \cdot p_3 = - 2 E_1 E_3 (1 - \cos \theta) $ and $ p_2 \cdot q = 2 E_1 E_2 - E_2 E_3 (1 + \cos \theta) $, hence:
  \begin{equation*}
    x \defeq \frac{Q^2}{2 p_2 \cdot q} \simeq - \frac{q^2}{2 p_2 \cdot q} = \frac{E_3}{E_2} \frac{1 - \cos \theta}{2 - \frac{E_3}{E_1} (1 + \cos \theta)}
  \end{equation*}
  Assuming $ \theta \simeq 150\deg $: $ Q^2 \simeq 3 \cdot 10^4 \gev^2 $ and $ x \simeq 0.7 $.
\end{example}

\section{Decays in QED}

At tree-level, the only QED decays are decays of bound states. The simplest one is positronium decay: $ e^+ e^- \ra \gamma \gamma $. In this case, the mediator of the decay is a virtual electron. The matrix element is found by crossing-symmetry from the Compton scattering, but in this case the amplitude is not computed averaging over spins, since now $ e^+ e^- $ form a ground state. Assuming the ground state of positronium, which is a singlet state (like the helium atom). At tree level, there are the $ t $-channel and the $ u $-channel, which have to be coherently summed as they are undistinguishable:
\begin{equation*}
  \begin{split}
    \mat_t
    & = \bar{v}(p_2) (-\img e \gamma^\mu) \frac{\img(\slashed{q} + m_e)}{q^2 - m_e^2} (-\img \gamma^\nu) u(p_1) \varepsilon_\mu^*(p_4) \varepsilon_\nu^*(p_3) \\
    & = - \img e^2 \bar{v}(p_2) \slashed{\varepsilon}^*(p_4) \frac{\slashed{p}_1 - \slashed{p}_3 + m_e}{(p_1 - p_3)^2 - m_2^2} \slashed{\varepsilon}^*(p_3) u(p_1)
  \end{split}
\end{equation*}
\begin{equation*}
  \begin{split}
    \mat_u
    & = \bar{v}(p_2) (-\img e \gamma^\mu) \frac{\img(\slashed{q} + m_e)}{q^2 - m_e^2} (-\img \gamma^\nu) u(p_1) \varepsilon_\mu^*(p_3) \varepsilon_\nu^*(p_4) \\
    & = - \img e^2 \bar{v}(p_2) \slashed{\varepsilon}^*(p_3) \frac{\slashed{p}_1 - \slashed{p}_4 + m_e}{(p_1 - p_4)^2 - m_2^2} \slashed{\varepsilon}^*(p_4) u(p_1)
  \end{split}
\end{equation*}
The total amplitude then is:
\begin{equation*}
  \man = - \img e^2 \bar{v}(p_2) \left[ \slashed{\varepsilon}^*(p_4)  \frac{\slashed{p}_1 - \slashed{p}_3 + m_e}{(p_1 - p_3)^2 - m_2^2} \slashed{\varepsilon}^*(p_3) + \slashed{\varepsilon}^*(p_3) \frac{\slashed{p}_1 - \slashed{p}_4 + m_e}{(p_1 - p_4)^2 - m_2^2} \slashed{\varepsilon}^*(p_4) \right] u(p_1)
\end{equation*}
Assume to be in the rest frame of the positronium, so that:
\begin{equation*}
  p_1 = (m_e,0,0,0)
  \qquad p_2 = (m_e,0,0,0)
  \qquad p_3 = m_e (1,0,0,1)
  \qquad p_4 = m_E (1,0,0,-1)
\end{equation*}
Then $ (p_1 - p_3)^2 = - 2 m_2^2 $ and $ \slashed{p}_1 \slashed{\varepsilon}^*(p_3) = - \slashed{\varepsilon}^*(p_3) \slashed{p}_1 + 2 p_1 \cdot \varepsilon^*(p_3) $, but $ p_1 $ is purely temporal while $ \varepsilon(p_3) $ is purely spatial, hence $ p_1 \cdot \varepsilon^*(p_3) = 0 $ and $ \slashed{p}_1 \slashed{\varepsilon}^*(p_3) = - \slashed{\varepsilon}^*(p_3) \slashed{p}_1 $. Analogously $ \slashed{p}_3 \slashed{\varepsilon}^*(p_3) = - \slashed{\varepsilon}^*(p_3) \slashed{p}_3 $, since $ p_0 \cdot \varepsilon(p_3) = 0 $ by the Lorentz condition.
