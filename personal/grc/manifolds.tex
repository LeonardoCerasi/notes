\section{Smooth manifolds}
\label{sec:manifolds}

\subsection{Topological manifolds}

\begin{definition}{Topological space}{}
  The \bcdef{topology} $ \tp $ of a set $ X $ is a family of subsets of $ X $, i.e. $ \tp \subseteq \pwst{X} $, defined as \bcdef{open sets}, with the following properties:
  \begin{enumerate}
    \item $ \emptyset,X \in \tp $;
    \item $ O_{\alpha},O_{\beta} \in \tp \implies O_{\alpha}\cap O_{\beta} \in \tp $;
    \item $ \{O_{\alpha}\}_{\alpha \in \mathcal{I}} \subset \tp \implies \bigcup_{\alpha \in \mathcal{I}} O_{\alpha} \in \tp $.
  \end{enumerate}
  A \bcdef{topological space} $ M $ is a set of points, endowed with a topology $ \tp $.
\end{definition}

Given a topological space $ (M,\tp) $, $ O \in \tp $ is a \emph{neighbourhood} of a point $ p \in M $ if $ p \in O $: then, $ (M,\tp) $ is \bctxt{Hausdorff} if $ \forall p,q \in M \,\, \exists O_1, O_2 \in \tp : p \in O_1 , q \in O_2 \land O_1 \cap O_2 = \emptyset $.

Topological spaces allow to introduce the concept of continuity: given two topological spaces $ (M_1, \tp_1) $ and $ (M_2, \tp_2) $, a map $ f : M_1 \ra M_2 $ is \bctxt{continuous} if $ O \in \tp_2 \implies f^{-1}(O)\in \tp_1 $.

\begin{definition}{Homeomorphism}{}
  Given two topological spaces $ (M_1, \tp_1) $ and $ (M_2, \tp_2) $, a map $ f : M_1 \rightarrow M_2 $ is a \bcdef{homeomorphism} if it is bijective and bicontinuous, i.e. both $ f $ and $ f^{-1} $ are continuous.
\end{definition}

The composition of two homeomorphisms $ f : M_1 \ra M_2 $ and $ g : M_2 \ra M_3 $ is still a homeomorphism $ g \circ f : M_1 \ra M_3 $, since both bijectivity and continuity are preserved by composition.

\begin{definition}{Topological manifold}{topological-manifold}
  An $ n $-dimensional \bcdef{topological manifold} $ \mathcal{M} $ is a Hausdorff topological space such that:
  \begin{enumerate}
    \item $ \man $ is \emph{second-countable}, i.e. there exists a countable basis for $ \tp(\man) $;
    \item $ \man $ is locally homeomorphic to $ \R^n $, i.e. $ \forall p\in\man \, \exists U \in \tp(\man) : p \in U \land \exists \varphi : U \rightarrow O \in \tp(\R^n) $ homeomorphism;
  \end{enumerate}
\end{definition}

Note that, by definition, every topological manifold has a well-defined dimension. Indeed, using the theory of de Rham cohomology (see \secref{ssec:derham}), it is possible to prove the following theorem.

\begin{theorem}{Topological invariance of dimension}{topological-invariance-dimension}
  A non-empty $ n $-dimensional topological manifold cannot be homeomorphic to an $ m $-dimensional topoplogical manifold, unless $ m = n $.
\end{theorem}

The empty set is excluded since it is an $ n $-dimensional topological manifold for all $ n \in \N $.

\subsubsection{Coordinate charts}

\begin{definition}{Coordinate chart}{}
  Let $ M $ be a topological $ n $-manifold. A \bcdef{coordinate chart} (or just a \emph{chart}) on $ M $ is a pair $ (U , \varphi) $, where $ U \in \tp(M) $ and $ \varphi : U \ra \varphi(U) \in \tp(\R^n) $ is a homeomorphism.
\end{definition}

Given a chart $ (U , \varphi) $, the set $ U $ is called a \emph{coordinate domain}, while $ \varphi $ is a \emph{(local) coordinate map}, whose component functions $ \varphi = (x^1 , \dots , x^n) $ are called \emph{local coordinates} on $ U $.

By \dref{def:topological-manifold}, each point $ p \in M $ is contained in the coordinate domain of a chart: if $ \varphi(p) = 0 $ in a chart $ (U , \varphi) $, then this chart is said to be ``centered at $ p $".

\begin{example}{Spheres}{topological-spheres}
  Consider the unit $ n $-sphere $ \Sn{n} $, with $ n \in \N_0 $. It is a second-countable Hausdorff topological space, since it is a topological subspace of $ \R^{n+1} $. To show local Euclideanity, for each index $ i = 1, \dots, n+1 $ let $ U_i^+ $ denote the subset of $ \R^{n+1} $ where the $ i^\text{th} $ coordinate if positive:
  \begin{equation*}
    U_i^+ \equiv \{(x^1 , \dots , x^{n+1}) \in \R^{n+1} : x^i > 0\}
  \end{equation*}
  and similarly $ U_i^- $ the set with $ x^i < 0 $. Then, let $ f : \Dn{n} \ra \R $ be the continuous function:
  \begin{equation*}
    f(u) = \sqrt{1 - \norm{u}^2}
  \end{equation*}
  For each $ i = 1 , \dots , n+1 $, it is easy to check that $ U_i^\pm \cap \Sn{n} $ is the graph of the function:
  \begin{equation*}
    x^i = \pm f(x^1 , \dots , x^{i-1} , x^{i+1} , \dots , x^{n+1})
  \end{equation*}
  Thus, each subset $ U_i^\pm \cap \Sn{n} \subset \Sn{n} $ is locally Euclidean of dimension $ n $, and the local coordinate maps are given by (see \figref{fig:sn-coordinates}):
  \begin{equation*}
    \varphi_i^\pm : U_i^\pm \cap \Sn{n} \ra \Dn{n}
    \qquad
    \varphi_i^\pm(x^1 , \dots , x^{n+1}) = (x^1 , \dots , x^{i-1} , x^{i+1} , \dots , x^{n+1})
  \end{equation*}
  Since each point of $ \Sn{n} $ is in the domain of at least one of these $ 2n+2 $ charts, $ \Sn{n} $ is a topological $ n $-manifold.
\end{example}

\begin{example}{Tori and product manifolds}{product-manifolds}
  For $ n \in \N $, the $ n $-torus is the product space $ \Tn{n} \defeq \Sn{1} \times \dots \times \Sn{1} $ of $ n $ copies of $ \Sn{1} $. By \exref{ex:topological-spheres}, this is a topological $ n $-space.

  Indeed, suppose $ M_1 , \dots , M_k $ are topological manifolds of dimensions $ n_1 , \dots , n_k $ respectively. The product space $ M_1 \times \dots \times M_k $ is itself a second-countable Hausdorff space, since it is a finite product of second-countable Hausdorff spaces, hence only local Euclideanity needs to be checked. Given any point $ (p_1 , \dots , p_k) \in M_1 \times \dots \times \dots M_k $, choose a chart $ (U_i , \varphi_i) $ on each $ M_i $ such that $ p_i \in U_i $. Then, the product map:
  \begin{equation*}
    \varphi_1 \times \dots \times \varphi_k : U_1 \times \dots \times U_k \ra \varphi_1(U_1) \times \dots \times \varphi_k(U_k) \in \tp(\R^{n_1 + \dots + n_k})
  \end{equation*}
  is a homeomorphism, thus $ M_1 \times \dots \times M_k $ is a topological manifold of dimension $ n_1 + \dots + n_k $.
\end{example}

A set of charts $ \atl = \{U_\alpha , \varphi_\alpha\} $ which cover $ \man $, i.e. such that $ \bigcup_\alpha U_\alpha = \man $ is called an \bctxt{atlas}. In general, an atlas may be contained in a larger atlas: a \emph{maximal atlas} is an atlas which is not properly contained in any larger atlas.

\subsection{Smooth manifolds}

Given $ U \in \tp(\R^n) $ and $ V \in \tp(\R^m) $, a function $ f : U \ra V $ is \bctxt{smooth} if each of its components is $ \cfk{\infty}(U) $. If $ f $ is also bijective and has a smooth inverse function, then it is a \emph{diffeomorphism}.

\begin{definition}{Smooth compatibility}{smooth}
  Let $ \man $ be a topological $ n $-manifold. Two charts $ (U , \varphi) $ and $ (V , \psi) $ of $ \man $ are \bcdef{smoothly compatible} if either $ U \cap V = \emptyset $ or $ \psi \circ \varphi^{-1} : \varphi(U \cap V) \ra \psi(U \cap V) $ is a diffeomorphism.
\end{definition}

If any two charts in an atlas are smoothly compatible, then the atlas is a \bctxt{smooth atlas}.

The same definition applies to weaker conditions on the differentiability of a function, i.e. $ \cfk{k} $ functions which are continuously differentiable up to the $ k^\text{th} $ derivative, and to real-analytic functions, i.e. $ \cra $ functions which can be expressed as a convergent power series in a neighbourhood of each point.

\begin{definition}{Smooth manifold}{smooth-manifold}
  An $ n $-dimensional \bcdef{smooth manifold} is an $ n $-dimensional topological manifold equipped with a maximal smooth atlas.
\end{definition}

This definition can be extended to \emph{$ \cfk{k} $ manifolds} and \emph{real-analytic manifolds} by considering $ \cfk{k} $ of $ \cra $ atlases, respectively.

\subsubsection{Smooth charts}

Each smooth coordinate map $ \varphi_{\alpha} $ provides a coordinate system on the smooth coordinate domain $ U_{\alpha} $: $ \varphi_{\alpha}(p) = \left( x^1(p), \dots, x^n(p) \right) $. The \emph{transition functions} $ \varphi_{\beta} \circ \varphi_{\alpha}^{-1} $ are therefore coordinate transformations on overlapping regions of the manifold, represented as diffeomorphism between open subsets of $ \R^n $.

\begin{example}{Finite-dimensional vector spaces}{}
  Let $ V $ be an $ n $-dimensional $ \R $-vector space. Any norm on $ V $ determines a topology, which can be shown to be independent on the choice of norm. With this topology, $ V $ is a topological $ n $-manifold, and a natural smooth structure on it can be defined as follows.

  Each ordered basis $ (E_1 , \dots , E_n) $ of $ V $ defines a basis isomorphism $ \beta : \R^n \ra V $ as:
  \begin{equation*}
    \beta(x) = \sum_{i = 1}^n x^i E_i
  \end{equation*}
  which is clearly continuous, i.e. a homeomorphism, so $ (V , \beta^{-1}) $ is a chart on $ V $. If $ (\tilde{E}_1 , \dots , \tilde{E}_n) $ is any other ordered basis of $ V $, denoting with $ \tilde{\beta} $ the corresponding homeomorphism, then $ \exists \mt{A} \in \GL{n,\R} $ such that (denoting the elements of $ \mt{A} $ as $ A_{ij} \equiv \tensor{A}{_i^j} $):
  \begin{equation*}
    E_i = \sum_{j = 1}^n \tensor{A}{_i^j} \tilde{E}_j
  \end{equation*}
  The transition map between the two charts $ (V , \beta^{-1}) $ and $ (V , \tilde{\beta}^{-1}) $ is then given by $ \tilde{\beta}^{-1} \circ \beta(x) = \tilde{x} $, where:
  \begin{equation*}
    \sum_{j = 1}^n \tilde{x}^j \tilde{E}_j = \sum_{i = 1}^n x^i E_i = \sum_{i,j = 1}^n x^i \tensor{A}{_i^j} \tilde{E}_j
    \quad \implies \quad
    \tilde{x}^j = \sum_{i = 1}^n x^i \tensor{A}{_i^j}
  \end{equation*}
  Since $ \mt{A} $ is invertible, this is clearly a diffeomorphism, so any two charts on $ V $ are smoothly compatible, and the collection of all such charts thus defines a maximal smooth atlas on $ V $, called the \emph{standard smooth structure} on $ V $.
\end{example}

The construction of a smooth manifold is structured in two stages: starting from a topological space, first check that it is a topological manifold, and then specify a smooth atlas on it. However, it is convenient to combine these steps into a single one, especially when starting with a set which is not already equipped with a topology.

\begin{lemma}{Smooth chart lemma}{smooth-chart-lemma}
  Let $ M $ be a set, and consider a collection $ \{U_\alpha\} $ of subsets of $ M $ together with maps $ \varphi_\alpha : U_\alpha \ra \R^n $ such that:
  \begin{enumerate}
    \item \label{item:scl-1} $ \varphi_\alpha $ is a bijection between $ U_\alpha $ and $ \varphi_\alpha(U_\alpha) \in \tp(\R^n) $ for all $ \alpha $;
    \item \label{item:scl-2} $ \varphi_\alpha(U_\alpha \cap U_\beta) , \varphi_\beta(U_\alpha \cap U_\beta) \in \tp(\R^n) $ for all $ \alpha , \beta $;
    \item \label{item:scl-3} if $ U_\alpha \cap U_\beta \neq \emptyset $, then $ \varphi_\beta \circ \varphi_\alpha^{-1} : \varphi_\alpha(U_\alpha \cap U_\beta) \ra \varphi_\beta(U_\alpha \cap U_\beta) $ is smooth;
    \item \label{item:scl-4} $ M = \bigcup_{\alpha \in \mathcal{I}} U_\alpha $, with $ \mathcal{I} $ countable;
    \item \label{item:scl-5} given $ p \neq q \in M $, either $ \exists \alpha : p,q \in U_\alpha $, or $ \exists \alpha , \beta : p \in U_\alpha , q \in U_\beta \land U_\alpha \cap U_\beta = \emptyset $.
  \end{enumerate}
  Then, $ M $ has a unique smooth manifold structure with $ \tp = \{\varphi_\alpha^{-1}(V) : V \in \tp(\R^n)\} $ and $ \mathcal{A} = \{(U_\alpha , \varphi_\alpha)\} $.
\end{lemma}

\begin{proofbox}
  \begin{proof}
    To define a topology on $ M $, take all sets of the form $ \varphi_\alpha^{-1}(V) $, with $ V \in \tp(\R^n) $, as a basis. To prove that this is indeed the basis for a topology, consider two basis sets $ \varphi_\alpha^{-1}(V) $ and $ \varphi_\beta^{-1}(W) $ with non-empty intersection: WTS $ \forall p \in \varphi_\alpha^{-1}(V) \cap \varphi_\beta^{-1}(W) \, \exists B $ basis set such that $ p \in B \land B \ssq \varphi_\alpha^{-1}(V) \cap \varphi_\beta^{-1}(W) $.

    It suffices to show that $ \varphi_\alpha^{-1}(V) \cap \varphi_\beta^{-1}(W) $ is itself a basis set (see \figref{fig:scl}). To see this, note that \itemref{item:scl-3} implies that $ (\varphi_\beta \circ \varphi_\alpha^{-1})^{-1}(W) $ is an open subset of $ \varphi_\alpha(U_\alpha \cap U_\beta) $ (as $ W \in \tp(\R^n) $), while \itemref{item:scl-2} implies that $ \varphi_\alpha(U_\alpha \cap U_\beta) \in \tp(\R^n) $. It follows that (see \figref{fig:scl}):
    \begin{equation*}
      \varphi_\alpha^{-1}(V) \cap \varphi_\beta^{-1}(W) = \varphi_\alpha^{-1}(V \cap (\varphi_\beta \circ \varphi_\alpha^{-1})^{-1}(W))
    \end{equation*}
    is also a basis set, as $ V \cap (\varphi_\beta \circ \varphi_\alpha^{-1})^{-1}(W) \in \tp(\R^n) $. Therefore, $ M $ is a topological space with topology $ \tp = \{\varphi_\alpha^{-1}(V) : V \in \tp(\R^n)\} $.

    By this definition, each map $ \varphi_\alpha : U_\alpha \ra \varphi_\alpha(U_\alpha) \in \tp(\R^n) $ is a homeomorphism, so $ (M,\tp) $ is a topological $ n $-manifold, since the Hausdorff property follows directly from \itemref{item:scl-5}, while second-countability from \itemref{item:scl-4}. Finally, \itemref{item:scl-3} guarantees that the collection $ \atl = \{(U_\alpha , \varphi_\alpha)\} $ is a smooth atlas over $ M $, making $ (M,\tp,\atl) $ a smooth $ n $-manifold.
  \end{proof}
\end{proofbox}

  \centering
  \includegraphics[width = 0.83 \textwidth]{smooth-chart-lemma.png}
  \caption{Smooth chart lemma.}
  \label{fig:scl}
\end{figure}

\subsubsection{Smooth maps}

With the introduction of a smooth structure on a manifold, it is possible to import the concepts of Analysis on $ \R^n $ to functions between manifolds. In the reminder of these notes, the term ``function" typically denotes maps whose codomain is $ \R^k $, with $ k \in \N $.

\begin{definition}{Smooth function}{}
  Let $ (\man , \atl) $ be a smooth $ n $-manifold. A function $ f : \man \ra \R^k $, with $ k \in \N $, is a \bcdef{smooth function} if $ \forall p \in \man \, \exists (U , \varphi) \in \atl : p \in U \land f \circ \varphi^{-1} \in \cfk{\infty}(\varphi(U)) $.
\end{definition}

The function $ \hat{f}_\varphi \equiv f \circ \varphi^{-1} : \varphi(U) \ra \R^k $ is the \emph{coordinate representation} of $ f $: by definition, a function is smooth if and only if its coordinate representation is smooth in some smooth chart around each point. By smooth compatibility, a smooth function $ f : \man \ra \R^k $ is smooth on every smooth chart of $ \man $, hence smooth functions have smooth coordinate representations in every smooth chart.

The most important case is that of smooth real-valued functions $ f : \man \ra \R $: the set of all such functions is denoted by $ \cm $, and it is an $ \R $-vector space under pointwise addition and constant multiplication.

\begin{definition}{Smooth map}{}
  Let $ (\man_1 , \atl_1) , (\man_2 , \atl_2) $ be smooth manifolds. A map $ f : \man_1 \ra \man_2 $ is a \bcdef{smooth map} if $ \forall p \in \man_1 \, \exists (U,\varphi) \in \atl_1 , (V,\psi) \in \atl_2 : p \in U , f(p) \in V \land f(U) \ssq V \land \psi \circ f \circ \varphi^{-1} \in \cfk{\infty}(\varphi(U)) $.
\end{definition}

From this definition of continuity it follows that smoothness implies continuity.

\begin{proposition}{Smoothness and continuity}{}
  Every smooth map is continuous.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Suppose $ f : \man_1 \ra \man_2 $ is smooth. Given $ p \in \man_1 $, by the smoothness of $ f $ there are smooth charts $ (U,\varphi) : p \in U $ and $ (V,\psi) : f(p) \in V $ such that $ f(U) \ssq V $ and $ \psi \circ f \circ \varphi^{-1} : \varphi(U) \ra \psi(V) $ is smooth, hence continuous. Since $ \varphi : U \ra \varphi(U) $ and $ \psi : V \ra \psi(V) $ are homeomorphisms:
    \begin{equation*}
      f \vert_U = \psi^{-1} \circ (\psi \circ f \circ \varphi^{-1}) \circ \varphi : U \ra V
    \end{equation*}
    which is a composition of continuous maps. Since $ f $ is continuous in a neighbourhood of each point, it is continuous on all $ \mat_1 $.
  \end{proof}
\end{proofbox}

The condition that, for a map $ f : \mat_1 \ra \mat_2 $ to be smooth, for all $ p \in \mat_1 $ there need to be coordinate domains $ U \ni p $ and $ V \ni f(p) $ such that $ f(U) \ssq V $ is included in the definition of smoothness precisely so that smoothness automatically implies continuity.

There are simple classes of maps which are automatically smooth, since they reduce to composition of smooth maps between subsets of Euclidean spaces. In particular:
\begin{itemize}
  \item every constant map is smooth;
  \item $ \id_\man $ is smooth;
  \item given an open submanifold $ \mathcal{U} \ssq \man $, the inclusion map $ U \hookrightarrow \man $ is smooth;
  \item the composition of smooth maps is a smooth map.
\end{itemize}

\begin{proposition}{Smooth maps on product manifolds}{product-smooth-maps}
  Let $ \man_1 , \dots , \man_k $ and $ \man $ be smooth manifolds, and for each $ i = 1 , \dots , k $ let $ \pi_i : \man_1 \times \dots \times \man_k \ra \man_i $ be the projection onto $ \man_i $. A map $ f : \man \ra \man_1 \times \dots \times \man_k $ is smooth if and only if each of the component maps $ f_i = \pi_i \circ f : \man \ra \man_i $ is smooth.
\end{proposition}

In general, to check that a map is smooth, it suffices to show that each of its component map is either smooth or a composition of smooth maps.

\begin{example}{Circles and tori}{}
  If the circle $ \Sn{1} $ is given its standard smooth structure, the map $ \varepsilon : \R \ra \Sn{1} $ defined by $ \varepsilon(t) = e^{2\pi \img t} $ is smooth, because with respect to any angle coordinate $ \theta $ on $ \Sn{1} $ it has a coordinate representation $ \hat{\varepsilon}_\theta = 2\pi t + c $, for some constant $ c \in \R $.

  By \pref{prop:product-smooth-maps}, the map $ \varepsilon_n : \R^n \ra \Tn{n} $ defined as $ \varepsilon_n(t_1 , \dots , t_n) = (e^{2\pi \img t_1} , \dots , e^{2\pi \img t_n}) $ is smooth.
\end{example}

\begin{example}{Inclusions on spheres}{}
  Consider the $ n $-sphere $ \Sn{n} $ with its standard smooth structure. The inclusion map $ \inc : \Sn{n} \hookrightarrow \R^{n+1} $ is clearly continuous, since it is the inclusion map of a topological space. Its coordinate representation with respect to any of the charts introduced in \exref{ex:topological-spheres} is:
  \begin{equation*}
    \hat{\inc}_\pm(u^1 , \dots , u^n) = \inc \circ (\varphi_i^\pm)^{-1}(u^1 , \dots , u^n) = (u^1 , \dots , u^{i-1} , \pm \sqrt{1 - \norm{u}^2} , u^{i+1} , \dots , u^n)
  \end{equation*}
  which is smooth on its domain $ \Dn{n} $, since $ \norm{u} < 1 $.
\end{example}

The most interesting class of smooth maps are smooth bijections with a smooth inverse, known as \bctxt{diffeomorphisms}. Since smoothness implies continuity, a diffeomorphism is also a homeomorphism; moreover, every composition of diffeomorphisms is a diffeomorphism, since composition preserves both continuity and smoothness, and by \pref{prop:product-smooth-maps} every finite product of diffeomorphisms is a diffeomorphism.

Two smooth manifolds are said to be \emph{diffeomorphic} if there exists a diffeomorphisms between them: this is clearly an equivalence relation on the class of all smooth manifolds. A weaker version of \tref{th:topological-invariance-dimension} holds in the case of diffeomorphic manifolds.

\begin{theorem}{Diffeomorphism invariance of dimension}{}
  A non-empty $ n $-dimensional smooth manifold cannot be diffeomorphic to an $ m $-dimensional smooth manifold, unless $ m = n $.
\end{theorem}

Like two topological spaces are considered to be ``the same" if they are homeomorphic, two smooth manifolds are essentially indistinguishable if they are diffeomorphic. A natural question to ask, then, is whether a given topological manifold admits smooth structures which are not diffeomorphic to each other.

\begin{proposition}{Uniqueness of smooth structures in low dimension}{}
  Every $ n $-dimensional topological manifold, with $ n \in \N_{\leq 3} $, has a smooth structure that is unique up to diffeomorphism.
\end{proposition}

On the contrary, in higher dimensions it turns out that there are cases of topological manifolds which admit several non-diffeomorphic smooth structures. For example, the Euclidean space $ \R^n $ admits a unique smooth structure (up to diffeomorphism), except for $ n = 4 $: in this case, $ \R^4 $ admits uncountably-many non-diffeomorphic distinct smooth structures (seldom called \emph{fake $ \R^4 $'s}). Another example is $ \Sn{7} $, which admits exactly $ 15 $ diffeomorphism classes of smooth structures. Finally, in any dimension $ n \in \N_{\geq 4} $ there exist compact topological manifolds that have no smooth structures at all.

\begin{example}{$ 0 $-dimensional manifolds}{}
  A topological $ 0 $-manifold $ \man $ is just a countable discrete space. For each point $ p \in \man $, the only neighbourhood of $ p $ that is homeomorphic to an open subset of $ \R^0 $ is $ \{p\} $ itself, and there is exactly one coordinate map $ \varphi : \{p\} \ra \R^0 $. Thus, the set of all charts on $ \man $ trivially satisfies the smooth compatibility condition, since they have no overlapping coordinate domains, hence each $ 0 $-dimensional manifold has a unique smooth structure.
\end{example}

\subsection{Manifolds with boundary}

In the analysis of integration on smooth manifolds (\secref{sec:integration}), one often encounters spaces which would be smooth except for a ``boundary" of some sort, e.g. closed balls in $ \R^n $ or closed hemispheres in $ \Sn{n} $. To describe these objects, the definition of manifold needs to be extended. To do so, define the \emph{closed $ n $-dimensional upper half-plane} $ \U^n \ssq \R^n $ as:
\begin{equation}
  \U^n \defeq \{(x^1 , \dots , x^n) \in \R^n : x^n \geq 0\}
\end{equation}
Clearly, for $ n > 0 $ this means that $ \Int \U^n = \{x^n > 0\} $ and $ \pa \U^n = \{x^n = 0\} $, while for $ n = 0 $ trivially $ \U^0 = \R^0 = \{0\} $, hence $ \Int \U^0 = \R^0 $ and $ \pa \U^0 = \emptyset $.

\begin{definition}{Topologival manifold with boundary}{}
  An $ n $-dimensional \bcdef{topological manifold with boundary} $ \man $ is a second-countable Hausdorff space in which every point has a neighbourhood homeomorphic to either an open subset of $ \R^n $ or a (relatively) open subset of $ \U^n $.
\end{definition}

\begin{figure}
  \centering
  \includegraphics[width = 0.75 \textwidth]{manifold-boundary.png}
  \caption{A manifold with boundary.}
  \label{fig:man-bound}
\end{figure}

The intuition behind this definition is depicted in \figref{fig:man-bound}. A chart $ (U,\varphi) $ of $ \man $ is then classified as a \emph{interior chart} if $ \varphi(U) \in \tp(\R^n) : \varphi(U) \cap \pa \U^n = \emptyset $, or as a \emph{boundary chart} if $ \varphi(U) \in \tp(\U^n) : \varphi(U) \cap \pa \U^n \neq \emptyset $. Boundary chart whose image is of the form $ \ball_r(x) \cap \U^n $ for some $ x \in \pa \U^n $ and $ r \in \R^+ $ is called a coordinate half-ball.

A point $ p \in \man $ is an interior point of $ \man $ if it is in the domain of some interior chart, while it is a boundary point of $ \man $ if it is in the domain of some boundary chart $ (U,\varphi) : \varphi(p) \in \pa \U^n $: the set of all boundary points of $ \man $ is the \bctxt{boundary} $ \pa \man $ of $ \man $.

\begin{theorem}{Topological invariance of boundary}{}
  Let $ \man $ be a topological manifold with boundary. Then, each point of $ \man $ is either a boundary point of an interior point, but not both.
\end{theorem}

Therefore, $ \pa \man \cap \Int \man = \emptyset $ and $ \man = \Int \man \sqcup \pa \man $.

\begin{observation}{Topological boundary and manifold boundary}{}
  The notions of topological boundary and manifold boundary are well-distinct: a manifold with boundary may have a non-empty manifold boundary, irrespective of whether it has a topological boundary as a subset of some other topological space.

  A clear example is the closed unit ball $ \Dn{n} $. Indeed, it is a topological manifold with boundary whose manifold boundary is $ \pa \Dn{n} = \Sn{n-1} $. When viewed as a subset of $ \R^n $, the topological boundary coincides with the manifold boundary; however, if $ \Dn{n} $ is thought of as a subset of $ \R^{n+1} $, its topological boundary is $ \Dn{n} $ itself.
\end{observation}

Note that a topological manifold with a boundary does not in general satisfy the definition of topological manifold. On the contrary, all topological manifolds are topological manifolds with a boundary, whose boundary is empty. In the following, the term manifold is used to denote a manifold that may or may not have a boundary, with further specification given only if required.

\begin{proposition}{Characterization of manifolds with boundary}{}
  Let $ \man $ be a topological $ n $-manifold with boundary. Then:
  \begin{enumerate}
    \item $ \Int \man \ssq \man $ is open and a topological $ n $-manifold without boundary;
    \item $ \pa \man \ssq \man $ is closed and a topological $ (n-1) $-manifold without boundary;
    \item $ \man $ is a topological manifold if and only if $ \pa \man = \emptyset $;
    \item if $ n = 0 $, then $ \pa \man = \emptyset $ and $ \man $ is a topological $ 0 $-manifold.
  \end{enumerate}
\end{proposition}

\subsubsection{Smooth manifolds with boundary}

Recall that a map from an arbitrary $ A \ssq \R^n $ to $ \R^k $ is said to be \emph{smooth} if in a neighbourhood of each point of $ A $ it admits an extension to a smooth map defined on an open subset of $ \R^n $. Thus, given $ U \in \tp(\U^n) $, a map $ f : U \ra \R^k $ is smooth if for each $ x \in U $ there exist $ \tilde{U} \in \tp(\U^n) : x \in \tilde{U} $ and a smooth map $ \tilde{f} : \tilde{U} \ra \R^k $ that agrees with $ f $ on $ \tilde{U} \cap \U^n $ (see \figref{fig:bound-smooth}). Clearly, the restriction of $ f $ to $ U \cap \Int \U^n $ is smooth in the usual sense. Moreover, all partial derivatives of $ f $ at points of $ U \cap \pa \U^n $ are determined by continuity by their values in $ \Int \U^n $, hence they are independent on the choice of extension.

\begin{figure}
  \centering
  \includegraphics[width = 0.70 \textwidth]{bound-smooth.png}
  \caption{Smoothness of maps on open subsets $ \U^n $.}
  \label{fig:bound-smooth}
\end{figure}

\begin{example}{Functions on the plane}{}
  Let $ \Int \Dn{2} \ssq \R^2 $ be the open unit disk, let $ U = \U^2 \cap \Int \Dn{2} $ and $ f : U \ra \R $ defined as $ f(x,y) = \sqrt{1 - x^2 - y^2} $. This expression naturally extends to all $ \Int \Dn{2} $ in a smooth way, hence $ f $ is smooth on $ U $. On the other hand, $ g(x,y) = \sqrt{y} $ has no smooth extension to any neighbourhood of $ (0,0) \in \R^2 $, since $ \pa_y g \ra \infty $ as $ y \ra 0 $. Therefore, $ g $ is not smooth on $ U $.
\end{example}

With this notion of smoothness, \dref{def:smooth-manifold} directly generalizes to smooth manifolds with boundary. As noted before, every smooth manifold without boundary is automatically a smooth manifold with boundary, whose boundary is empty. Moreover, if $ \man $ is a smooth manifold with boundary, then $ \Int \man $ is an open smooth submanifold of $ \man $ without boundary.

The construction of smooth structures on finite products illustrated in \exref{ex:product-manifolds} does not extend directly to smooth manifolds with boundary. This is due to the fact that $ \U^n \times \U^m $ is not itself a half-space, hence a finite product of smooth manifolds with boundary cannot generally be considered as a smooth manifold with boundary (it is instead a smooth manifold with corners, see REF-TO-DEF). However, it can be shown that if $ \man_1 , \dots , \man_k $ are smooth manifolds without boundary and $ \man $ is a smooth manifold with boundary, then $ \man_1 \times \dots \times \man_k \times \man $ is a smooth manifold with boundary, and $ \pa (\man_1 \times \dots \times \man_k \times \man) = \man_1 \times \dots \times \man_k \times \pa \man $.

\begin{theorem}{Smooth invariance of boundary}{}
  Let $ \man $ be a smooth manifold with boundary, and let $ p \in \man $. If there exists a chart $ (U,\varphi) $ of $ \man $ such that $ \varphi(U) \ssq \U^n $ and $ \varphi(p) \in \pa \U^n $, then the same is true for every smooth chart of $ \man $ whose domain contains $ p $.
\end{theorem}

\section{Tangent vectors}

The central idea of calculus is \emph{linear approximation}: for example, the approximations of a single-variable function with its tangent line, of a parametrized curve with its velocity vector, or of a surface with its tangent plane. In order to generalize calculus on smooth manifolds, then, one needs to introduce the notion of a ``tangent space" to a manifold at each of its points.

\subsection{Derivations}

Before considering generic smooth manifolds, it is convenient to analyze the Euclidean case first.

\begin{definition}{Geometric tangent space}{}
  Given a point $ a \in \R^n $, the \bcdef{geometric tangent space} to $ \R^n $ at $ a $ is the set $ \R^n_a \defeq \{a\} \times \R^n = \{(a,\ve{v}) : \ve{v} \in \R^n\} $. A \bcdef{geometric tangent vector} in $ \R^n $ is an element $ \ve{v}_a \equiv (a,\ve{v}) \in \R^n_a $, for some $ a \in \R^n $.
\end{definition}

Geometric tangent vectors can be thought as ``arrows" with initial point at $ a $. The set $ \R^n_a $ is an $ \R $-vector space under the natural operations of pointwise addition and constant multiplication.

Given a basis $ \{e_i\}_{i = 1 , \dots , n} $ of $ \R^n $, then $ \{\ve{e}_i \vert_a\}_{i = 1 , \dots , n} $ is a basis of $ \R^n_a $: indeed, $ \R^n_a $ and $ \R^n $ are essentially the same space, with the index in $ \R^n_a $ only needed to distinguish between geometric tangent spaces at different points, since they are disjoint sets.

The same geometric construction does not generalize to smooth manifolds, which are in general not embedded in any Euclidean space. However, a suitable construction can be defined starting from geometric vectors. Indeed, any geometric tangent vector $ \ve{v}_a \in \R^n_a $ defines a map $ D_\ve{v}\vert_a : \cfk{\infty}{\R^n} \ra \R $ which takes the directional derivative along $ \ve{v} $ at $ a $:
\begin{equation}
  D_\ve{v} \vert_a f \equiv D_\ve{v} f(a) \defeq \frac{\dd}{\dd t} \bigg\vert_{t = 0} f(a + t \ve{v})
  \label{eq:directional-derivative}
\end{equation}
In terms of the standard basis, by the chain rule the directional derivative can be written more concretely as:
\begin{equation*}
  D_\ve{v} \vert_a f = v^i \frac{\pa f}{\pa x^i}(a)
\end{equation*}
where the Einstein's summation convention is assumed. With this construction in mind, it is possible to give a general definition.

\begin{definition}{Derivation}{derivation-at}
  Let $ \man $ be a smooth manifold and $ p \in \man $. Then, a linear map $ X_p : \cm \ra \R $ is a \bcdef{derivation at $ p $} if it satisfies the \emph{Leibniz rule}:
  \begin{equation}
    X_p (fg) = f(p) X_p(g) + g(p) X_p(f)
    \qquad
    \forall f,g \in \cm
  \end{equation}
\end{definition}

The set of all derivations of $ \cm $ at a point $ p \in \man $, denoted by $ T_p\man $, is a $ \R $-vector space called the \bctxt{tangent space} to $ \man $ at $ p $, and its elements are called \emph{tangent vectors} to $ \man $ at $ p $.

\begin{lemma}{Properties of derivations}{}
  Let $ \man $ be a smooth manifold and $ p \in \man $. Then, given $ X_p \in T_p\man $ and $ f,g \in \cm $:
  \begin{enumerate}
    \item \label{item:der-1} if $ f $ is a constant function, then $ X_p f = 0 $;
    \item \label{item:der-2} if $ f(p) = g(p) = 0 $, then $ X_p (fg) = 0 $;
  \end{enumerate}
\end{lemma}

\begin{proofbox}
  \begin{proof}
    \itemref{item:der-2} follows trivially from the Leibniz rule. To prove \itemref{item:der-1}, it suffices to check it for the constant function $ f_1(x) \equiv 1 $, since others follow from linearity. For $ f_1 $, the product rule gives:
    \begin{equation*}
      X_p(f_1) = X_p(f_1 f_1) = 2 f_1(p) X_p(f_1) = 2 X_p(f_1)
    \end{equation*}
    which implies $ X_p(f_1) = 0 $.
  \end{proof}
\end{proofbox}

\subsection{Differentials}

To relate the abstract definition of tangent spaces through derivations to the geometric tangent spaces on $ \R^n $, one needs to study the relation between smooth maps and derivations.

In the case of a smooth map between Euclidean spaces, the total derivative of the map at a point (represented by its Jacobian matrix) is a linear map that represents the linear approximation of the map near the given point. In the manifold case there is a similar map, but it makes no sense to talk about a linear map between manifolds: it will instead be a linear map between tangent spaces.

\begin{definition}{Differential}{differential}
  Let $ \man_1 , \man_2 $ be smooth manifolds and $ f : \man_1 \ra \man_2 $ a smooth map. The \bcdef{differential} of $ f $ at $ p \in \man_1 $ is the map $ \dd f_p : T_p\man_1 \ra T_{f(p)}\man_2 $ defined as follows: given $ X_p \in T_p\man_1 $, let $ \dd f_p(X_p) $ be the derivation at $ f(p) $ that acts on $ g \in \cfk{\infty}(\man_2) $ as:
  \begin{equation}
    \dd f_p(X_p)(g) \defeq X_p(g \circ f)
    \label{eq:differential-def}
  \end{equation}
\end{definition}

The nature of $ \dd f_p(X_p) $ as a derivation is naturally derived from the nature of $ X_p $ as a derivation:
\begin{equation*}
  \dd f_p(X_p)(gh) = X_p((gh) \circ f) = X_p((g \circ f) (h \circ f)) = g(f(p)) \dd f_p(X_p)(h) + h(f(p)) \dd f_p(X_p)(g)
\end{equation*}

\begin{lemma}{Properties of differentials}{differential-properties}
  Let $ \man , \man_1 , \man_2 , \man_3 $ be smooth manifolds and $ p \in \man_1 $. Then, given smooth maps $ f : \man_1 \ra \man_2 $ and $ g : \man_2 \ra \man_3 $:
  \begin{enumerate}
    \item \label{item:diff-1} $ \dd f_p : T_p\man_1 \ra T_{f(p)}\man_2 $ is linear;
    \item \label{item:diff-2} $ \dd (g \circ f)_p = \dd g_{f(p)} \circ \dd f_p : T_p\man_1 \ra T_{g \circ f(p)}\man_3 $;
    \item \label{item:diff-3} $ \dd (\id_\man) = \id_{T_p\man} $;
    \item \label{item:diff-4} if $ f $ is a diffeomorphism, then $ \dd f_p $ is an isomorphism, and $ (\dd f_p)^{-1} = \dd (f^{-1})_{f(p)} $.
  \end{enumerate}
\end{lemma}

The differential of a smooth map can be visualized as in \figref{fig:differential}. Its main application is to relate the tangent space to a smooth manifold with a Euclidean tangent space.

There is, however, an important issue to be addressed first: while tangent spaces are defined in terms of smooth functions on the whole manifold, coordinate charts are in general defined only on open subsets. The key point is that tangent vectors (i.e. derivations) act locally.

\begin{figure}
  \centering
  \includegraphics[width = 0.70 \textwidth]{differential.png}
  \caption{Visualization of the differential of a smooth map.}
  \label{fig:differential}
\end{figure}

\begin{proposition}{Tangent space to an open submanifold}{isomorphic-tangent-space}
  Let $ \man $ be a smooth manifold, $ U \in \tp(\man) $ and $ \inc : U \hookrightarrow \man $ the inclusion map. For every $ p \in U $, the differential $ \dd \inc_p : T_pU \ra T_p\man $ is an isomorphism.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    To prove injectivity, suppose $ X_p \in T_p\man $ and $ \dd \inc_p(X_p) = 0 \in T_p\man $. Let $ B \in \tp(\man) : p \in \man \land \overline{B} \ssq U $. If $ f \in \cfk{\infty} $ is arbitrary, it can be extended as $ \tilde{f} \in \cm : \tilde{f}\vert_{\bar{B}} \equiv f $.

    \begin{lemma}[borderline = {1pt}{0pt}{blue!35!black}]{Locality of derivations}{local-derivation}
      Let $ \man $ be a smooth manifold and $ p \in \man $. Given $ X_p \in T_p\man $, if $ f , g \in \cm $ agree on some neighbourhood of $ p $, then $ X_p(f) = X_p(g) $.
    \end{lemma}

    By \lref{lemma:local-derivation}, since $ f $ and $ \tilde{f}\vert_U $ are smooth functions on $ U $ which agree on a neighbourhood of $ p $:
    \begin{equation*}
      X_p(f) = X_p(\tilde{f}\vert_U) = X_p(\tilde{f} \circ \inc) = \dd \inc_p (X_p) (\tilde{f}) = 0
    \end{equation*}
    Since this holds for every $ f \in \cfk{\infty}(U) $, then $ X_p \equiv 0 $, so $ \dd \inc_p $ is injective as $ \ker \dd \inc_p = \{0\} $.

    To prove surjectivity, suppose $ U_p \in T_p\man $ is arbitrary. Define an operator $ X_p : \cfk{\infty}(U) \ra \R $ by setting $ X_p(f) = Y_p(\tilde{f}) $, where $ \tilde{f} $ is any smooth function on all $ \man $ which agrees with $ f $ on $ \overline{B} $. By \lref{lemma:local-derivation}, $ X_p(f) $ is independent of the choice of $ \tilde{f} $, hence $ X_p $ is well-defined, and it is easily checked to be a derivation of $ \cfk{\infty}(U) $ at $ p $. For any $ g \in \cm $, then:
    \begin{equation*}
      \dd \inc_p(X_p) (g) = X_p(g \circ \inc) = Y_p(\widetilde{g \circ \inc}) = Y_p(g)
    \end{equation*}
    where the last two equalities follow from the fact that $ g \circ \inc $, $ \widetilde{g \circ \inc} $ and $ g $ all agree on $ B $. Hence, $ \dd \inc_p $ is surjective, and therefore an isomorphism.
  \end{proof}
\end{proofbox}

Given an open set $ U \ssq \man $, the isomorphism $ \dd \inc_p : T_pU \ra T_p\man $ is canonically defined, independently on any choices: it is then possible to identify $ T_pU \cong T_p\man $ for any point $ p \in U $. This identification comes from the fact that $ \dd \inc_p(X_p) $ is the same derivation as $ X_p $, just acting on functions on the whole manifold instead of on $ U $ only: since by \lref{lemma:local-derivation} the action of a derivation on a function depends only on the values of the function on arbitrarily small neighbourhoods, this identification is well-posed. In particular, this means that any tangent vector $ X_p \in T_p\man $ can be unambiguously applied to functions defined only in a neighbourhood of $ p $, not necessarily on the whole $ \man $.

\begin{theorem}{Dimension of tangent spaces}{tangent-space}
  Let $ \man $ be a smooth $ n $-manifold and $ p \in \man $. Then $ T_p\man $ is an $ n $-dimensional $ \R $-vector space.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    Given $ p \in \man $, let $ (U,\varphi) $ be a smooth coordinate chart containing $ p $. Since $ \varphi : U \ra \varphi(U) \ssq \R^n $ is a diffeomorphism, by \itemref{item:diff-4} of \lref{lemma:differential-properties} it follows that $ \dd \varphi_p : T_pU \ra T_{\varphi(p)}\varphi(U) $ is an isomorphism. \pref{prop:isomorphic-tangent-space} guarantees that $ T_pU \cong T_p\man $ and $ T_{\varphi(p)}\varphi(U) \cong T_{\varphi(p)}\R^n $, hence $ \dim_\R T_p\man = \dim_\R T_{\varphi(p)}\R^n = n $.
  \end{proof}
\end{proofbox}

In the case of finite-dimensional vector spaces, a stronger result holds. Given an $ n $-dimensional $ \R $-vector field $ V $ (with its standard smooth structure) and a point $ a \in V $, for each $ \ve{v} \in V $ consider the map $ D_\ve{v}\vert_a : \cfk{\infty}(V) \ra \R $ defined as in \eref{eq:directional-derivative}. Then, it can be shown that the map $ \ve{v} \mapsto D_\ve{v}\vert_a $ is a canonical isomorphism $ V \cong T_aV $, so that given $ L \in \Hom_\R(V,W) $ the following diagram commutes:
\begin{equation}
  \begin{tikzcd}
    V \ar[r , "\cong"] \ar[d , "L" , swap] & T_aV \ar[d , "\dd L_a"] \\
    W \ar[r , "\cong" , swap] & T_{L(a)} W
  \end{tikzcd}
  \label{eq:canonical-isomorphism-vector-spaces}
\end{equation}
The fact that $ V \cong T_aV $ is a canonical isomorphism (i.e. independent of the choice of basis) allows to identify the tangent vectors to a finite-dimensional vector space with the elements of the space itself. In particular, since isomorphisms map bases to bases, given the canonical basis $ \{\ve{e}_i\}_{i = 1,\dots,n} $ of $ V $, the canonical basis of $ T_aV $ is formed by:
\begin{equation}
  D_{\ve{e}_i}\vert_a = \frac{\pa}{\pa x^i}\bigg\vert_a
  \label{eq:canonical-tangent-basis}
\end{equation}
More generally, if $ \man $ is an open submanifold of a vector space $ V $, then the isomorphisms $ T_p\man \cong T_pV \cong V $ allow to canonically identify each tangent space to $ \man $ with $ V $.

\begin{proposition}{Tangent space to a product manifold}{}
  Let $ \man_1 , \dots , \man_k $ be smooth manifolds, and for each $ j = 1 , \dots , k $ let $ \pi_j : \man_1 \times \dots \times \man_k \ra \man_j $ be the projection onto $ \man_j $. For any point $ p = (p_1 , \dots , p_k) \in \man_1 \times \dots \times \man_k $, the map $ \alpha : T_p(\man_1 \times \dots \times \man_k) \ra T_{p_1}\man_1 \oplus \dots \oplus T_{p_k}\man_k $ defined as:
  \begin{equation*}
    \alpha(\ve{v}) = (\dd (\pi_1)_p(\ve{v}) , \dots , \dd (\pi_k)_p(\ve{v}))
  \end{equation*}
  is a canonical isomorphism.
\end{proposition}

The canonical identification $ T_{p,q}(\man_1 \times \man_2) \cong T_p\man_1 \oplus T_q\man_2 $ is then justified.

\subsection{Coordinate bases}

To make tangent spaces less abstract, it is possible to introduce coordinates on them.

\subsubsection{Derivations}

Consider a smooth $ n $-manifold $ \man $ and let $ (U,\varphi) $ be a smooth chart on $ \man $. Since $ \varphi : U \ra \varphi(U) \ssq \R^n $ is a diffeomorphism, combining \itemref{item:diff-4} of \lref{lemma:differential-properties} and \pref{prop:isomorphic-tangent-space}, it is clear that $ \dd \varphi_p : T_p\man \ra T_{\varphi(p)}\R^n $ is an isomorphism. By \eref{eq:canonical-tangent-basis}, the canonical basis of $ T_{\varphi(p)}\R^n $ is $ \pa_1 \vert_{\varphi(p)} , \dots , \pa_n \vert_{\varphi(p)} $, therefore the preimages of these vectors under the isomorphism $ \dd \varphi_p $ form a basis of $ T_p\man $:
\begin{equation*}
  \frac{\pa}{\pa x^i} \bigg\vert_p \equiv (\dd \varphi_p)^{-1} \left( \frac{\pa}{\pa x^i} \bigg\vert_{\varphi(p)} \right) = \dd (\varphi^{-1})_{\varphi(p)} \left( \frac{\pa}{\pa x^i} \bigg\vert_{\varphi(p)} \right)
\end{equation*}
Then, by \dref{def:differential}, these basis vector act on $ \cfk{\infty}(U) $ as:
\begin{equation}
  \frac{\pa}{\pa x^i}\bigg\vert_p f = \frac{\pa}{\pa x^i}\bigg\vert_{\varphi(p)} (f \circ \varphi^{-1}) = \frac{\pa \hat{f}}{\pa x^i}(\hat{p})
\end{equation}
where the hat denotes the coordinate representation, as usual. Therefore, $ \pa_i\vert_p $ is the derivation that takes the $ i^\text{th} $ partial derivative of (the coordinate representation of) $ f \in \cfk{\infty}(U) $ at (the coordinate representation of) $ p \in U $: these are called the \bctxt{coordinate vectors} at $ p $, forming a \emph{coordinate basis} of $ T_p\man $ associated to the given coordinate system. In the special case $ \man = \R^n $, the coordinate vectors are precisely the partial derivative operators. Tangent vectors $ X_p \in T_p\man $ can thus be written uniquely as:
\begin{equation}
  X_p = X_p^i \frac{\pa}{\pa x^i}\bigg\vert_p
  \label{eq:tangent-vector-components}
\end{equation}
Since $ p_i\vert_p x^j = \delta^j_i $, the components of $ X_p $ can be easily computed by its action the coordinate functions: $ X_p^i = X_p(x^i) $.

\subsubsection{Differentials}

To see how differentials are expressed in coordinates, consider first the special case of a smooth map $ f : U \ra V $, with $ U \in \tp(\R^n) $ and $ V \in \tp(\R^m) $. Denoting the coordinates in the domain $ U $ as $ (x^1 , \dots , x^n) $ and the coordinates in the codomain $ V $ as $ (y^1 , \dots , y^m) $, by the chain rule:
\begin{equation*}
  \dd f_p \left( \frac{\pa}{\pa x^i} \bigg\vert_p \right)(g) = \frac{\pa}{\pa x^i} \bigg\vert_p (g \circ f) = \frac{\pa g}{\pa y^j} (f(p)) \frac{\pa f^j}{\pa x^i}(p) = \left( \frac{\pa f^j}{\pa x^i}(p) \frac{\pa}{\pa y^j} \bigg\vert_{(f(p))} \right) g
\end{equation*}
Therefore:
\begin{equation}
  \dd f_p \left( \frac{\pa}{\pa x^i} \bigg\vert_p \right) = \frac{\pa f^j}{\pa x^i}(p) \frac{\pa}{\pa y^j} \bigg\vert_{(f(p))}
  \label{eq:differential-jacobian}
\end{equation}
In other words, the matrix representation of the action of $ \dd f_p $ on $ T_pU $ is given by:
\begin{equation*}
  [\dd f_p] =
  \begin{bmatrix}
    \frac{\pa f^1}{\pa x^1}(p) & \dots & \frac{\pa f^1}{\pa x^n}(p) \\
    \vdots & \ddots & \vdots \\
    \frac{\pa f^m}{\pa x^1}(p) & \dots & \frac{\pa f^m}{\pa x^n}(p) \\
  \end{bmatrix}
\end{equation*}
This is precisely the Jacobian of $ f $ at $ p $, which is the matrix representation of the total derivative: hence, the differential $ \dd f_p : T_p\R^n \ra T_{f(p)}\R^m $ corresponds to the total derivative $ [\dd f_p] : \R^n \ra \R^m $, under the usual identification $ T_p\R^n \cong \R^n $.

\begin{figure}
  \centering
  \includegraphics[width = 0.60 \textwidth]{diff-coord.png}
  \caption{The differential of a smooth map in coordinates.}
  \label{fig:diff-coord}
\end{figure}

Now, consider the general case of a smooth map $ f : \man_1 \ra \man_2 $ between smooth manifolds. Choosing smooth charts $ (U,\varphi) : U \ni p $ for $ \man_1 $ and $ (V,\psi) : V \ni f(p) $ for $ \man_2 $, the coordinate representation of $ f $ reads $ \hat{f} = \psi \circ f \circ \varphi^{-1} : \varphi(U \cap f^{-1}(V)) \ra \psi(V) $ (see \figref{fig:diff-coord}). Denoting $ \hat{p} = \varphi(p) $, by \eref{eq:differential-jacobian} $ \dd \hat{f}_{\hat{p}} $ is represented with respect to the standard coordinate basis by the Jacobian matrix of $ \hat{f} $ at $ \hat{p} $. Using $ f \circ \varphi^{-1} = \psi^{-1} \circ \hat{f} $, then:
\begin{equation*}
  \dd f_p \left( \frac{\pa}{\pa x^i} \bigg\vert_p \right) = \dd f_p \left( \dd (\varphi^{-1})_{\hat{p}} \left( \frac{\pa}{\pa x^i} \bigg\vert_{\hat{p}} \right) \right) = \dd (\psi^{-1})_{\hat{f}(\hat{p})} \left( \dd \hat{f}_{\hat{p}} \left( \frac{\pa}{\pa x^i} \bigg\vert_{\hat{p}} \right) \right)
\end{equation*}
Using \eref{eq:differential-jacobian}:
\begin{equation}
  \dd f_p \left( \frac{\pa}{\pa x^i} \bigg\vert_p \right) = \dd (\psi^{-1})_{\hat{f}(\hat{p})} \left( \frac{\pa \hat{f}^j}{\pa x^i}(\hat{p}) \frac{\pa}{\pa y^j} \bigg\vert_{\hat{f}(\hat{p})} \right) = \frac{\pa \hat{f}^j}{\pa x^i}(\hat{p}) \frac{\pa}{\pa y^j}\bigg\vert_{f(p)}
\end{equation}
Therefore, the representation of the differential $ \dd f_p $ in the coordinate basis is the Jacobian matric of (the coordinate representation of) $ f $ at (the coordinate representation of) $ p $. Indeed, the differential is defined precisely to give a coordinate-independent meaning to the Jacobian. Moreover, since the differential ``pushes" tangent vectors forward from the domain manifold to the codomain manifold (as per \dref{def:differential}), it is also called the \emph{(pointwise) push-forward}.

\subsubsection{Change of coordinates}

Suppose $ (U,\varphi) $ and $ (V,\psi) $ are two smooth charts on $ \man $, and let $ p \in U \cap V $. Denoting the coordinate functions of $ \varphi $ by $ x^i $ and those of $ \psi $ by $ \tilde{x}^i $, any tangent vector at $ p $ can be expressed either on the basis $ \pa_i\vert_p $ or $ \tilde{\pa}_i\vert_p $. To see how these two representations are related, consider the transition map $ \psi \circ \varphi^{-1} : \varphi(U \cap V) \ra \psi(U \cap V) $, which is expressed as (with an abuse of notation):
\begin{equation*}
  \psi \circ \varphi^{-1} (x) = (\tilde{x}^1(x) , \dots , \tilde{x}^n(x))
\end{equation*}
Its differential $ \dd (\psi \circ \varphi^{-1})_{\varphi(p)} : T_{\varphi(p)}\R^n \ra T_{\psi(p)}\R^n $ can then be written using \eref{eq:differential-jacobian}:
\begin{equation*}
  \dd (\psi \circ \varphi^{-1})_{\varphi(p)} \left( \frac{\pa}{\pa x^i} \bigg\vert_{\varphi(p)} \right) = \frac{\pa \tilde{x}^j}{\pa x^i}(\varphi(p)) \frac{\pa}{\pa \tilde{x}^j} \bigg\vert_{\psi(p)}
\end{equation*}
A visualization of this map is pictured in \figref{fig:coo-ch}. Using the definition of coordinate vectors:
\begin{equation*}
  \begin{split}
    \frac{\pa}{\pa x^i} \bigg\vert_p
    & = \dd (\varphi^{-1})_{\varphi(p)} \left( \frac{\pa}{\pa x^i} \bigg\vert_{\varphi(p)} \right) \\
    & = \dd (\psi^{-1})_{\psi(p)} \circ \dd (\psi \circ \varphi^{-1})_{\varphi(p)} \left( \frac{\pa}{\pa x^i} \bigg\vert_{\varphi(p)} \right)=  \dd (\psi^{-1})_{\psi(p)} \left( \frac{\pa \tilde{x}^j}{\pa x^i}(\varphi(p)) \frac{\pa}{\pa \tilde{x}^j} \bigg\vert_{\psi(p)} \right)
  \end{split}
\end{equation*}
Using the chain rule and introducing the coordinate representation $ \hat{p} = \varphi(p) $:
\begin{equation}
  \frac{\pa}{\pa x^i} \bigg\vert_p = \frac{\pa \tilde{x}^j}{\pa x^i}(\hat{p}) \frac{\pa}{\pa \tilde{x}^j} \bigg\vert_p
  \label{eq:contravariant-alibi}
\end{equation}
Applying this equation to \eref{eq:tangent-vector-components}, the transformation rule for the components of a tangent vector is found:
\begin{equation}
  \tilde{X}_p^j = \frac{\pa \tilde{x}^j}{\pa x^i}(\hat{p}) X_p^i
  \label{eq:contravariant-alias}
\end{equation}
It is then clear that the components of tangent vectors transform inversely with respect to coordinate basis vectors: for this reason, they are known as \emph{contravariant vectors}. \eref{eq:contravariant-alibi} and \eref{eq:contravariant-alias} respectively give the \emph{alibi} (passive) and \emph{alias} (active) interpretation for the change of basis: either the components are left unchanged while the basis vector are transformed, or vice versa.

\end{figure}

\subsection{Tangent bundle}

\begin{definition}{Tangent bundle}{}
  Given a smooth manifold $ \man $, its \bcdef{tangent bundle} is defined by the following disjoint union:
  \begin{equation}
    T\man \defeq \bigsqcup_{p \in \man} T_p\man \equiv \bigcup_{p \in \man} \{p\} \times T_p\man = \{(p , X_p) : p \in \man , X_p \in T_p\man\}
  \end{equation}
\end{definition}
This set comes equipped with a natural projection map $ \pi : T\man \ra \man $ which sends each vector in $ T_p\man $ to the point $ p \in \man $ at which it is tangent: $ \pi(p , X_p) = p $. In the following, with an abuse of notation, the tangent space $ T_p\man $ is often identified with its image under the canonical injection $ T_p\man \hookrightarrow T\man : X_p \mapsto (p , X_p) $, so as to write simply $ X_p $ in place of $ (p , X_p) $.

Note that, in general, the tangent bundle can be identified as a Cartesian product only locally, while globally its structure may be more complex (see \secref{ssec:tangent-bundle}).

\begin{proposition}{Tangent bundle as smooth manifold}{tangent-bundle-smooth}
  Given a smooth $ n $-manifold $ \man $, its tangent bundle $ T\man $ has a natural structure of $ 2n $-dimensional smooth manifold, with respect to which the projection $ \pi : T\man \ra \man $ is smooth.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    First, show that $ T\man $ is a smooth $ 2n $-manifold. Since $ T\man $ does not come endowed with a topology, it is convenient to use \lref{lemma:smooth-chart-lemma}.

    Given any smooth chart $ (U , \varphi) $ over $ \man $, note that $ \pi^{-1}(U) \ssq T\man $ is the set of all tangent vectors to $ \man $ at all points of $ U $. Denote the coordinate functions of $ \varphi $ as $ (x^1 , \dots , x^n) $, and define the map $ \tilde{\varphi} : \pi^{-1}(U) \ra \R^{2n} $ as:
    \begin{equation*}
      \tilde{\varphi} \left( X^i \frac{\pa}{\pa x^i} \bigg\vert_p \right) \equiv (x^1(p) , \dots , x^n(p) , X^1 , \dots , X^n)
    \end{equation*}
    Clearly, $ \im \tilde{\varphi} = \varphi(U) \times \R^n \in \tp(\R^{2n}) $, hence it is continuous. Moreover, it is a bijection onto its image, since its inverse can be written explicitly as:
    \begin{equation*}
      \tilde{\varphi}^{-1}(x^1 , \dots , x^n , X^1 , \dots , X^n) = X^i \frac{\pa}{\pa x^i} \bigg\vert_{\varphi^{-1}(p)}
    \end{equation*}
    Therefore, $ \tilde{\varphi} $ is a homeomorphism, and $ (\pi^{-1}(U) , \tilde{\varphi}) $ is a chart over $ T\man $. Then, consider two smooth charts $ (U , \varphi) , (V , \psi) $ over $ \man $, and the associated charts $ (\pi^{-1}(U) , \tilde{\varphi}) , (\pi^{-1}(V) , \tilde{\psi}) $ over $ T\man $. The following sets:
    \begin{equation*}
      \tilde{\varphi}(\pi^{-1}(U) \cap \pi^{-1}(V)) = \varphi(U \cap V) \times \R^n
    \end{equation*}
    \begin{equation*}
      \tilde{\psi}(\pi^{-1}(U) \cap \pi^{-1}(V)) = \psi(U \cap V) \times \R^n
    \end{equation*}
    are in $ \tp(\R^{2n}) $, and the transition map $ \tilde{\psi} \circ \tilde{\varphi}^{-1} : \varphi(U \cap V) \times \R^n \ra \psi(U \cap V) \times \R^n $ between them can be written explicitly as:
    \begin{equation*}
      \tilde{\psi} \circ \tilde{\varphi}^{-1} (x^1 , \dots , x^n , X^1 , \dots , X^n) = \left( y^1(x) , \dots , y^n(x) , \frac{\pa y^1}{\pa x^i} X^i , \dots , \frac{\pa y^n}{\pa x^i} X^i \right)
    \end{equation*}
    where $ (y^1 , \dots , y^n) $ are the coordinate functions of $ \psi $. This is clearly smooth, hence the charts $ (\pi^{-1}(U) , \tilde{\varphi}) $ and $ (\pi^{-1}(V) , \tilde{\psi}) $ are smoothly compatible.

    Choosing a countable cover $ \{U_\alpha\} $ of $ \man $ from the smooth coordinate domains of $ \man $, a countable cover of $ T\man $ is obtained as $ \{\pi^{-1}(U_\alpha)\} $: this countable cover satisfies the conditions \itemref{item:scl-1}-\itemref{item:scl-4} of \lref{lemma:smooth-chart-lemma}. To check the Hausdorff condition \itemref{item:scl-5}, just note that any two points in same fiber of $ \pi $ lie in the same chart, while if $ (p,X) $ and $ (q,Y) $ lie in different fibers, there exist disjoint smooth coordinate domains $ U , V \ssq \man $ such that $ p \in U $ and $ q \in V $, and $ \pi^{-1}(U) \ni (p,X) $ and $ \pi^{-1}(V) \ni (q,Y) $ are disjoint coordinate neighbourhoods. Therefore, by \lref{lemma:smooth-chart-lemma}, $ T\man $ is a smooth $ 2n $-manifold.

    Now, to see that $ \pi : T\man \ra \man $ is smooth, note that, with respect to charts $ (U,\varphi) $ of $ \man $ and $ (\pi^{-1}(U) , \tilde{\varphi}) $ of $ T\man $, its coordinate representation is $ \pi(p , X) = (x^1 , \dots , x^n) $.
  \end{proof}
\end{proofbox}

The coordinates $ (x^i , X^i) $ are called the \emph{natural coordinates} on $ T\man $.

A particular case in which the tangent bundle can in fact be thought as a Cartesian product is that of a smooth $ n $-manifold $ \man $ which can be covered by a single smooth chart $ (\man,\varphi) $. In this case, $ \varphi $ is a diffeomorphism from $ \man $ to $ \varphi(\man) \ssq \R^n $: defining $ \tilde{\varphi} : \pi^{-1}(\man) = T\man \ra \man \times \R^n $ as in the previous proof, it retains the nature of diffeomorphism of $ \varphi $, hence $ T\man $ is diffeomorphic to $ \man \times \R^n $.

Given a smooth map $ f : \man_1 \ra \man_2 $, the introduction of the tangent bundle allows to define a \bctxt{global differential} $ \dd f : T\man_1 \ra T\man_2 $, which is just the map whose restriction to each $ T_p\man_1 \ssq T\man_1 $ is $ \dd f_p $. The global differential $ \dd f : T\man_1 \ra T\man_2 $ is a smooth map, since in terms of natural coordinates on $ T\man_1 $ it is expressed as:
\begin{equation*}
  \dd f(x^1 , \dots , x^n , X^1 , \dots , X^n) = \left( f^1(x) , \dots , f^m(x) , \frac{\pa f^1}{\pa x^i}(x) X^i , \dots , \frac{\pa f^m}{\pa x^i}(x) X^i \right)
\end{equation*}

\begin{lemma}{Properties of global differentials}{global-differential-properties}
  Let $ \man , \man_1 , \man_2 $ be smooth manifolds. Then, given smooth maps $ f : \man_1 \ra \man_2 $ and $ g : \man_2 \ra \man_3 $:
  \begin{enumerate}
    \item \label{item:glob-diff-1} $ \dd (g \circ f) = \dd g \circ \dd f $;
    \item \label{item:glob-diff-2} $ \dd (\id_\man) = \id_{T\man} $;
    \item \label{item:glob-diff-3} if $ f $ is a diffeomorphism, the $ \dd f $ is also a diffeomorphism, and $ (\dd f)^{-1} = \dd (f^{-1}) $.
  \end{enumerate}
\end{lemma}

\itemref{item:glob-diff-3} allows to set unambiguous $ \dd f^{-1} \equiv (\dd f)^{-1} = \dd (f^{-1}) $ when $ f $ is a diffeomorphism.

\subsection{Vector fields}

The notion of a vector field over $ U \in \tp(\R^n) $ as a continuous map $ U \mapsto \R^n $ which assigns vector to each point $ p \in U $ can be extended directly to smooth manifolds.

\begin{definition}{Vector field}{}
  Let $ \man $ be a smooth manifold. A \bcdef{vector field} over $ \man $ is a continuous map $ X : \man \ra T\man $ such that $ \pi \circ X = \id_\man $, where $ \pi : T\man \ra \man $ is the natural projection.
\end{definition}

The map which defines a vector field is usually written as $ p \mapsto X_p $, hence the condition $ \pi \circ X = \id_\man $ is equivalent to $ X_p \in T\man $. If this map is smooth, it is called a \bctxt{smooth vector field}: the set of all the smooth vector fields on a smooth manifold $ \man $ is denoted by $ \xm $. This is trivially a vector space under pointwise addition and constant multiplication.

Smooth vector fields can also be multiplied by smooth functions. Given $ f \in \cm $ and $ X \in \xm $, then $ fX : \man \ra T\man : (fX)_p = f(p) X_p $ is still a smooth vector field\footnotemark. Then, as per \eref{eq:tangent-vector-components}, a vector field can be expressed as:
\begin{equation*}
  X = X^i \frac{\pa}{\pa x^i}
\end{equation*}
where $ X^i : p \mapsto X_p^i $ are its \emph{component functions} in the given chart: a vector field is smooth if and only if its component functions are smooth.

\footnotetext{This means that $ \xm $ is a module over the ring $ \cm $.}

\begin{example}{Coordinate vector fields}{}
  If $ (U,\varphi) $ is a smooth chart on the smooth manifold $ \man $ and $ (x^1 , \dots , x^n) $ are the coordinate functions of $ \varphi $, then the map:
  \begin{equation*}
    p \mapsto \frac{\pa}{\pa x^i} \bigg\vert_p
  \end{equation*}
  determines a vector field on $ U $, called the $ i^\text{th} $ \bcex{coordinate vector field}, which is trivially smooth since its component functions are constants.
\end{example}

\begin{example}{Angle coordinates on tori}{}
  Let $ \theta $ be an angle coordinate on $ U \in \tp(\Sn{1}) $, and let $ \frac{\dd}{\dd \theta} $ denote the corresponding coordinate vector field. Since any other angle coordinate may differ from $ \theta $ only by an additive constant, by \eref{eq:contravariant-alibi} $ \frac{\dd}{\dd \tilde{\theta}} = \frac{\dd}{\dd \theta} $ on their common domain. Thus, there is a globally-defined smooth vector field on $ \Sn{1} $ whose coordinate representation is $ \frac{\dd}{\dd \theta} $ with respect to any angle coordinate.

  Generalizing to $ \Tn{n} = \Sn{1} \times \dots \times \Sn{1} $, the globally-defined smooth vector fields are $ \frac{\dd}{\dd \theta^i} $, for each $ i = 1 , \dots , n $.
\end{example}

\subsubsection{Frames}

Coordinate vector fields in a smooth chart are not the only way to represent vector fields.

Suppose $ \man $ is a smooth $ n $-manifold. An ordered $ k $-tuple $ (X_1 , \dots , X_k) $ of vector fields defined on some subset $ U \ssq \man $ is said to be \bctxt{linearly independent} if $ (X_1 \vert_p , \dots , X_k \vert_p) $ is a linearly independent $ k $-tuple in $ T_p\man $ for each $ p \in U $, and is said to \emph{span the tangent bundle} if $ (X_1 \vert_p , \dots , X_k \vert_p) $ spans $ T_p\man $ at each $ p \in U $.

\begin{definition}{Frame}{}
  Let $ \man $ be a smooth $ n $-manifold. A \bcdef{local frame} on $ \man $ is an ordered $ n $-tuple $ (E_1 , \dots , E_n) $ of vector fields on $ U \in \tp(\man) $ that is linearly independent and spans the tangent bundle $ T\man $. If $ U = \man $, it is a \bcdef{global frame}, and a \bcdef{smooth frame} if the vector fields $ E_i $ are smooth.
\end{definition}

If $ (E_1 , \dots , E_n) $ is a local frame, then the tangent vectors $ (E_1 \vert_p , \dots , E_n \vert_p) $ form a basis of $ T_p\man $ for each $ p \in U $. The coordinate vector fields of some chart $ (U,\varphi) $ form a \emph{coordinate frame}.

\begin{proposition}{Completion of local frames}{completion-local-frames}
  Let $ \man $ be a smooth $ n $-manifold. Then:
  \begin{enumerate}
    \item \label{item:locfr-1} if $ (X_1 , \dots , X_k) $ is a linearly independent $ k $-tuple of smooth vector fields on $ U \in \tp(\man) $, with $ 1 \leq k < n $, then for each $ p \in U $ there exist $ X_{k+1} , \dots , X_n $ smooth vector fields on a neighbourhood $ V \ni p $ such that $ (X_1 , \dots , X_n) $ is a smooth local frame on $ U \cap V $;
    \item \label{item:locfr-2} if $ (v_i , \dots , v_k) $ is a linearly independent $ k $-tuple of vectors in $ T_p\man $ for some $ p \in \man $, with $ 1 \leq k \leq n $, then there exists a smooth local frame $ (X_1 , \dots , X_n) $ on a neighbourhood of $ p $ such that $ X_i \vert_p = v_i \,\,\forall i = 1, \dots, k $;
  \end{enumerate}
\end{proposition}

For subsets of $ \R^n $, there is a special type of frames. A $ k $-tuple of vector fields $ (E_1 , \dots , E_k) $ on some $ A \ssq \R^n $ is \emph{orthonormal} if for each $ p \in A $ the vectors $ (E_1 \vert_p , \dots , E_k \vert_p) $ are orthonormal with respect to the Euclidean scalar product (with the identification $ T_p\R^n \cong \R^n $). A (local or global) frame consisting of orthonormal vector fields is called an \bctxt{orthonormal frame}.

Starting from a smooth local frame $ (X_1 , \dots , X_n) $ on $ T\R^n $ over $ U \in \tp(\R^n) $, it is possible to find a smooth orthonormal frame $ (E_1 , \dots , E_n) $ over $ U $, such that $ \lspan(E_1 \vert_p , \dots , E_j \vert_p) = \lspan(X_1 \vert_p , \dots , X_j \vert_p) $ for each $ j = 1 , \dots , n $ and each $ p \in U $, using the \emph{Gram--Schmidt algorithm}:
\begin{equation}
  E_j = \frac{X_j - \sum_{i = 1}^{j-1} (X_j \cdot E_i) E_i}{\norm{X_j - \sum_{i = 1}^{j-1} (X_j \cdot E_i) E_i}}
\end{equation}
This is a recursive definition, and it is well-posed since $ X_j \vert_p \notin \lspan(E_1 \vert_p , \dots , E_{j-1} \vert_p) $ for each $ j = 1 , \dots , n $ and each $ p \in U $, hence the denominator is a nowhere-vanishing smooth function on $ U $.

While smooth local frames are easily found, using \pref{prop:completion-local-frames}, global ones are not. Indeed, smoth manifolds which admit a smooth local frames are said to be \bctxt{parallelizable}: for example, $ \R^n $ and $ \Tn{n} $ are all parallelizable, while the only parallelizable spheres are $ \Sn{1} $, $ \Sn{3} $ and $ \Sn{7} $. In general, the parallelizability of a smooth $ n $-manifold $ \man $ is linked to whether its tangent bundle is diffeomorphic to $ \man \times \R^n $.

\subsubsection{Vector fields as derivations}

A fundamental property of vector fields is that they define linear operators on the space smooth real-valued functions. If $ X \in \xm $ and $ f \in \cfk{\infty}(U) $, with $ U \in \tp(\man) $, then $ Xf : U \ra \R $ defined by:
\begin{equation}
  (Xf)(p) = X_p(f)
\end{equation}
which can be shown to be smooth over $ U $. Since the action of a tangent vector on a function is determined by the values of the function on an arbitrarily small neighbourhood, it follows that $ Xf $ is locally determined: on any open subset $ V \ssq U $, it is $ (Xf)\vert_V = X(f\vert_V) $.

Since $ X \in \xm $ defines an endomorphism in $ \cm $ as $ f \mapsto Xf $ which obeys the Leibniz rule, it generalizes \dref{def:derivation-at}.

\begin{definition}{Derivation}{}
  Let $ \man $ be a smooth manifold. Then, $ X \in \End \cm $ is a \bcdef{derivation} of $ \cm $ if it satisfies the Leibniz rule:
  \begin{equation}
    X (fg) = f Xg + g Xf
    \qquad
    \forall f,g \in \cm
  \end{equation}
\end{definition}

In general, derivations of $ \cm $ can be identified with smooth vectors, since it can be shown that a map $ D : \cm \ra \cm $ is a derivation if and only if it is of the form $ D f = X f $, for some $ X \in \xm $. For this reason, the same symbol $ X $ denotes both maps $ X : \man \ra T\man $ and $ X : \cm \ra \cm $.

\subsubsection{Push-forward of vector fields}

If $ f : \man_1 \ra \man_2 $ is a smooth map and $ X $ is a vector field on $ \man_1 $, then for each point $ p \in \man_1 $ a tangent vector to $ \man_2 $ is obtained as $ \dd f_p(X_p) \in T_{f(p)}\man_2 $. However, this does not in general defined a vector field on $ \man_2 $: for example, if $ f $ is not surjective, there is no way to decide how to assign tangent vectors to points $ q \in \man_2 - f(\man_1) $, while if $ f $ is not injective there are some points $ q \in \man_2 $ where many tangent vectors can be assigned, one for each point in $ f^{-1}(q) $.

\begin{definition}{$ f $-relation}{}
  Let $ f : \man_1 \ra \man_2 $ be a smooth map, $ X \in \xm $ and $ Y \in \xm $. Then, $ X $ and $ Y $ are \bcdef{$ f $-related} if $ \dd f_p(X_p) = Y_{f(p)} \,\,\forall p \in \man_1 $.
\end{definition}

$ f $-related vector fields can be characterized by their action on smooth functions.

\begin{proposition}{}{f-relation-characterization}
  Let $ f : \man_1 \ra \man_2 $ be a smooth map, $ X \in \xm $ and $ Y \in \xm $. Then, $ X $ and $ Y $ are $ f $-related if and only if for every $ g \in \cfk{\infty}(U) $, with $ U \in \tp(\man_2) $:
  \begin{equation}
    X(g \circ f) = Y(g) \circ f
  \end{equation}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    For any $ p \in \man_1 $ and any $ g \in \cfk{\infty}(U) $, with $ U \in \tp(\man_2) : U \ni f(p) $:
    \begin{equation*}
      X(g \circ f)(p) = X_p(g \circ f) = \dd f_p(X_p) (g)
    \end{equation*}
    while:
    \begin{equation*}
      Y(g) \circ f(p) = (Y(g)) (f(p)) = Y_{f(p)} (g)
    \end{equation*}
    Therefore $ X(g \circ f) = Y(g) \circ f \iff \dd f_p(X_p) = Y_{f(p)} \,\,\forall p \in \man_1 $.
  \end{proof}
\end{proofbox}

Note that, for a given smooth map $ f : \man_1 \ra \man_2 $, there may be no $ f $-related vector fields. In a special case, however, the existence is guaranteed.

\begin{theorem}{Push-forward}{push-forward-vector-fields}
  Let $ \man_1 , \man_2 $ be smooth manifolds and $ f : \man_1 \ra \man_2 $ a diffeomorphism. For every $ X \in \xm $ there is a unique $ Y \in \xm $ that is $ f $-related to $ X $, called the \bcth{push-forward} of $ X $.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    Given $ X \in \xm $, defined the vector field $ Y_p \equiv \dd f_{f^{-1}(q)}(X_{f^{-1}(q)}) \,\,\forall q \in \man_2 $. Clearly, this is the unique vector that is $ f $-related to $ X $. To show that it is smooth, note that $ Y : \man_2 \ra T\man_2 $ is the following composition:
    \begin{equation*}
      \man_2 \xrightarrow{f^{-1}} \man_1 \xrightarrow{X} T\man_1 \xrightarrow{\dd f} T\man_2
    \end{equation*}
    which is a composition of smooth maps, hence it is smooth.
  \end{proof}
\end{proofbox}

The push-forward of $ X $ by a diffeomorphism $ f $ is usually denoted by $ f_* X $, thus by the preceding proof:
\begin{equation}
  (f_* X)_{f(p)} \defeq \dd f_p (X_p)
\end{equation}
An alternative characterization, in terms of vector fields directly, comes from \pref{prop:f-relation-characterization}. For any $ g \in \cfk{\infty}(\man_2) $:
\begin{equation}
  (f_* X)(g) \circ f = X(g \circ f)
\end{equation}

\subsubsection{Lie brakets}

Consider a smooth manifold $ \man $ and $ X,Y \in \xm $. Since smooth vector fields are derivations of $ \cm $, given $ f \in \cm $ it is possible to apply $ Y $ to $ X(f) $, obtaining another smooth function $ Y(X(f)) \equiv YX(f) \in \cm $. The operator $ f \mapsto YX(f) $, however, does not satisfy Leibniz rule, hence it does not define a vector field.

To obtain another vector field from two smooth vector fields, define the \bctxt{Lie bracket} of $ X $ and $ Y $ as $ [X,Y] : \cm \ra \cm $ such that:
\begin{equation}
  [X,Y](f) \defeq XY(f) - YX(f)
\end{equation}

\begin{lemma}{Lie bracket}{}
  Let $ X,Y \in \xm $. Then $ [X,Y] \in \xm $.
\end{lemma}

\begin{proofbox}
  \begin{proof}
    Since smooth vector fields are identified with derivations of $ \cm $, it suffices to show that $ [X,Y] $ is a derivation of $ \cm $. For arbitrary $ f,g \in \cm $:
    \begin{equation*}
      \begin{split}
        [X,Y](fg)
        & = X(Y(fg)) - Y(X(fg)) = X(f Y(g) - g Y(f)) - Y(f X(g) - g X(f)) \\
        & = X(f) Y(g) + f XY(g) - X(g) Y(f) - g XY(f) + \\
        & \quad - Y(f) X(g) - f YX(g) + Y(g) X(f) + g YX(f) \\
        & = f XY(g) + g XY(f) - f YX(g) - g YX(f) = f [X,Y](g) + g [X,Y](f)
      \end{split}
    \end{equation*}
    which concludes the proof.
  \end{proof}
\end{proofbox}

In general, the expression for $ [X,Y]_p(f) $ contains second derivatives of $ f $ at $ p \in \man $ that cancel out, leaving a simple expression. Since $ [X,Y] $ is a smooth vector field, its action on function is determined locally $ [X,Y](f) \vert_U = [X,Y](f\vert_U) $, thus its expression needs only to be computed in a single smooth chart $ (U,(x^i)) $:
\begin{equation*}
  [X,Y](f) = X^i \frac{\pa}{\pa x^i} \left( Y^j \frac{\pa f}{\pa x^j} \right) - Y^j \frac{\pa}{\pa x^j} \left( X^i \frac{\pa f}{\pa x^i} \right) = X^i \frac{\pa Y^j}{\pa x^i} \frac{\pa f}{\pa x^j} - Y^j \frac{\pa X^i}{\pa x^j} \frac{\pa f}{\pa x^i}
\end{equation*}
where the Schwarz lemma $ \pa_i \pa_j f = \pa_j \pa_i f $ has been used, as $ f \in \cm $. Renaming dummy indices, the coordinate expression of the Lie bracket is found:
\begin{equation}
  [X,Y] = \left( X^i \frac{\pa Y^j}{\pa x^i} - Y^i \frac{\pa X^j}{\pa x^i} \right) \frac{\pa}{\pa x^j}
  \label{eq:lie-bra-comp}
\end{equation}
It is then clear, using again the Schwarz lemma, that for coordinate vector fields:
\begin{equation*}
  \left[ \frac{\pa}{\pa x^i} , \frac{\pa}{\pa x^j} \right] \equiv 0
\end{equation*}

\begin{lemma}{Properties of Lie brackets}{lieb-properties}
  Let $ \man $ be a smooth manifold and $ X,Y,Z \in \xm $. Then:
  \begin{enumerate}
    \item \label{item:lieb-1} (bilinearity) for $ a,b \in \R $:
      \begin{equation*}
        [aX + bY , Z] = a [X,Z] + b [Y,Z]
        \qquad \qquad
        [Z , aX + bY] = a [Z,X] + b [Z,Y]
      \end{equation*}
    \item \label{item:lieb-2} (antisymmetry):
      \begin{equation*}
        [X,Y] = - [Y,X]
      \end{equation*}
    \item \label{item:lieb-3} (Jacobi identity):
      \begin{equation}
        [X,[Y,Z]] + [Y,[Z,X]] + [Z,[X,Y]] = 0
      \end{equation}
    \item \label{item:lieb-4} for $ f,g \in \cm $:
      \begin{equation}
        [fX , gY] = fg [X,Y] + (f X(g)) Y - (g Y(f)) X
        \label{eq:lied-aux}
      \end{equation}
  \end{enumerate}
\end{lemma}

\begin{proofbox}
  \begin{proof}
    Bilinearity and antisymmetry are trivial by definition. The Jacobi identity and \itemref{item:lieb-4} are direct computations.
  \end{proof}
\end{proofbox}

The importance of \eref{eq:lied-aux} will be clear once the Lie derivative is introduced (see \secref{ssec:lie-der}).

\begin{proposition}{Naturality of Lie brackets}{}
  Let $ f : \man_1 \ra \man_2 $ be a smooth map, $ X_1 , X_2 \in \xv(\man_1) $ and $ Y_1 , Y_2 \in \xv(\man_2) $ such that $ X_i $ is $ f $-related to $ Y_i $, for $ i = 1,2 $. Then, $ [X_1 , X_2] $ is $ f $-related to $ [Y_1 , Y_2] $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Using \pref{prop:f-relation-characterization} and the fact that $ X_i $ and $ Y_i $ are $ f $-related, given $ g \in \cfk{\infty}(\man_1) $:
    \begin{equation*}
      X_1 X_2 (g \circ f) = X_1 (Y_2(g) \circ f) = (Y_1 Y_2(g)) \circ f
    \end{equation*}
    and in a similar way $ X_2 X_1 (g \circ f) = (Y_2 Y_1(g)) \circ f $, therefore:
    \begin{equation*}
      [X_1 , X_2](g \circ f) = (Y_1 Y_2(g)) \circ f - (Y_2 Y_1(g)) \circ f = [Y_1 , Y_2](g) \circ f
    \end{equation*}
    which shows that $ [X_1 , X_2] $ and $ [Y_1 , Y_2] $ are $ f $-related.
  \end{proof}
\end{proofbox}

The special case of $ f : \man_1 \ra \man_2 $ diffeomorphism yields the push-forward of a Lie bracket:
\begin{equation}
  f_* [X_1 , X_2] = [f_* X_1 , f_* X_2]
\end{equation}

\section{Curves}

The familiar notion of curve in $ \R^n $ can be extended to generic manifolds.

\begin{definition}{Curve}{}
  Let $ \man $ be a topological manifold. A \bcdef{curve} on $ \man $ is a continuous map $ \gamma : J \ra \man $, where $ J \ssq \R $ is an interval.
\end{definition}

The definition of tangent spaces leads to a natural interpretation of velocity vectors of curves. Given a smooth curve $ \gamma : J \ra \man $ and $ t_0 \in J $, the \bctxt{velocity} of $ \gamma $ at $ t_0 $ is the vector:
\begin{equation}
  \gamma'(t_0) \defeq \dd \gamma \left( \frac{\dd}{\dd t} \bigg\vert_{t_0} \right) \in T_{\gamma(t_0)}\man
\end{equation}
where $ \frac{\dd}{\dd t} \vert_{t_0} $ is the standard coordinate basis vector in $ T_{t_0}\R $. This tangent vector acts on functions as:
\begin{equation*}
  \gamma'(t_0) (f) = \dd \gamma \left( \frac{\dd}{\dd t} \bigg\vert_{t_0} \right) f = \frac{\dd}{\dd t} \bigg\vert_{t_0} (f \circ \gamma) = (f \circ \gamma)'(t_0)
\end{equation*}
Therefore, $ \gamma'(t_0) $ is the derivation at $ \gamma(t_0) $ obtained by taking the derivative of a function along the curve\footnotemark. If $ (U,\varphi) $ is a smooth chart with coordinate functions $ (x^1 , \dots , x^n) $ and $ \gamma(t_0) \in U $, writing the coordinate representation of $ \gamma $ as $ \gamma(t) = (\gamma^1(t) , \dots , \gamma^n(t)) $, the coordinate expression for the velocity vector is:
\begin{equation}
  \gamma'(t_0) = \frac{\dd \gamma^i}{\dd t}(t_0) \frac{\pa}{\pa x^i} \bigg\vert_{\gamma(t_0)}
\end{equation}

\footnotetext{If $ J $ has one or two endpoints, the $ t $-derivative is interpreted as one-sided.}

\begin{proposition}{Tangent vectors and curves}{}
  Let $ \man $ be a smooth manifold and $ p \in \man $. Every $ X_p \in T_p\man $ is the velocity of some smooth curve on $ \man $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Let $ (U,\varphi) $ be a smooth chart centered at $ p \in \man $, and write $ X_p = X_p^i \pa_i \vert_p $. For a sufficiently small $ \epsilon > 0 $, let $ \gamma : (-\epsilon , \epsilon) \ra U $ be the curve whose coordinate representation is:
    \begin{equation*}
      \gamma(t) = (t X_p^1 , \dots , t X_p^n)
    \end{equation*}
    Clearly, this is a smooth curve with $ \gamma(0) = p $ and $ \gamma'(0) = X_p $.
  \end{proof}
\end{proofbox}

Given a smooth map $ f : \man_1 \ra \man_2 $ and a smooth curve $ \gamma : J \ra \man_1 $, the velocity of the composite curve $ f \circ \gamma : J \ra \man_2 $ if given, for any $ t_0 \in J $, by:
\begin{equation}
  (f \circ \gamma)'(t_0) = \dd (f \circ \gamma) \left( \frac{\dd}{\dd t} \bigg\vert_{t_0} \right) = \dd f \circ \dd \gamma \left( \frac{\dd}{\dd t} \bigg\vert_{t_0} \right) = \dd f(\gamma'(t_0))
  \label{eq:composite-curve}
\end{equation}
Since curves behave well under composition of smooth maps, it is possible to compute the action of the smooth map $ f : \man_1 \ra \man_2 $ on $ X_p \in T_p\man_1 $ as:
\begin{equation}
  \dd f_p(X_p) = (f \circ \gamma)'(0)
\end{equation}
for any smooth curve $ \gamma : J \ra \man_1 $ such that $ \gamma(0) = p $ and $ \gamma'(0) = X_p $.

\subsection{Integral curves}

A smooth curve $ \gamma : J \ra \man $ naturally determines a tangent vector $ \gamma'(t) \in T_{\gamma(t)}\man $ for each $ t \in J $. It is also possible to work backward, associating a curve to a vector field.

\begin{definition}{Integral curve}{integral-curve}
  Let $ \man $ be a smooth manifold and $ X $ be a vector field on $ \man $. An \bcdef{integral curve} of $ X $ is a differentiable curve $ \gamma : J \ra \man $ such that $ \gamma'(t) = X_{\gamma(t)} \,\,\forall t \in J $.
\end{definition}

Finding an integral curve generally reduces to solving a system of ordinary differential equations (ODEs) in a smooth chart. Indeed, on a smooth coordinate domain $ U \ssq \man $, writing $ \gamma(t) = (\gamma^1(t) , \dots , \gamma^n(t)) $, the condition for $ \gamma $ to be an integral curve of $ X \in \xm $ becomes:
\begin{equation*}
  \dot{\gamma}^i(t) \frac{\pa}{\pa x^i} \bigg\vert_{\gamma(t)} = X^i_{\gamma(t)} \frac{\pa}{\pa x^i} \bigg\vert_{\gamma(t)}
\end{equation*}
which results in the following autonomous\footnotemark system of ODEs (setting $ X^i_p \equiv X^i(p) $):
\begin{equation}
  \begin{cases}
    \dot{\gamma}^1(t) = X^1(\gamma^1(t) , \dots , \gamma^n(t)) \\
    \qquad \vdots \\
    \dot{\gamma}^n(t) = X^n(\gamma^1(t) , \dots , \gamma^n(t)) \\
  \end{cases}
  \label{eq:aut-sys}
\end{equation}

\footnotetext{A system $ \dot{\ve{x}}(t) = \ve{f}(\ve{x}(t)) $ is \emph{autonomous} if $ \ve{f} $ does not depend explicitly on $ t $.}

An important result of Real Analysis guarantees the existence of a solution to this system.

\begin{theorem}{Fundamental theorem of autonomous ODEs}{aut-odes}
  Let $ U \in \tp(\R^n) $ and $ \ve{f}  : U \ra \R^n $ is a smooth vector-valued function. Consider the following Cauchy problem:
  \begin{equation}
    \begin{cases}
      \dot{\ve{x}}(t) = \ve{f}(\ve{x}(t)) \\
      \ve{x}(t_0) = \ve{c}
    \end{cases}
    \label{eq:aut-ode-sys}
  \end{equation}
  for arbitrary $ t_0 \in \R $ and $ \ve{c} \in U $. Then:
  \begin{enumerate}
    \item \label{item:aut-ode-1} (existence) for any $ t_0 \in \R $ and $ \ve{x}_0 \in U $, there exist an open interval $ J_0 \ni t_0 $ and an open subset $ U_0 \ssq U : U_0 \ni \ve{x}_0 $ such that for each $ \ve{c} \in U $ there is a $ \cfk{1} $ map $ \ve{x} : J_0 \ra U $ which solves \eref{eq:aut-ode-sys};
    \item \label{item:aut-ode-2} (uniqueness) any two differentiable solutions agree on their common domain;
    \item \label{item:aut-ode-3} (smoothness) given $ J_0 $ and $ U_0 $ as in \itemref{item:aut-ode-1}, and given $ \theta_0 : J_0 \times U_0 \ra U $ defined as $ \theta(t,\ve{c}) = \ve{x}(t) $, where $ \ve{x} : J_0 \ra U $ is the unique solution to \eref{eq:aut-ode-sys} with $ \ve{x}(t_0) = \ve{c} $, then $ \theta $ is smooth.
  \end{enumerate}
\end{theorem}

Therefore, given $ X \in \xm $, for each $ p \in \man $ there exist $ \epsilon > 0 $ and a smooth curve $ \gamma : (-\epsilon , \epsilon) \ra \man $ which is an integral curve of $ X $.

\begin{proposition}{Affine reparametrization invariance}{}
  Let $ X \in \xm $ and $ \gamma : J \ra \man $ an integral curve of $ X $. For any $ a,b \in \R : a \neq 0 $, the curve $ \tilde{\gamma} : \tilde{J} \ra \man : \tilde{\gamma}(t) = a \gamma(t) + b $, with $ \tilde{J} = \{t \in \R : at + b \in J\} $, is still an integral curve of $ X $.
\end{proposition}

Moreover, it can be shown that, given a smooth map $ f : \man_1 \ra \man_2 $, then $ X \in \xv(\man_1) $ and $ Y \in \xv(\man_2) $ are $ f $-related if and only if $ f $ takes integral curves of $ X $ to integral curves of $ Y $, i.e. for each integral curve $ \gamma $ of $ X $, $ f \circ \gamma $ is an integral curve of $ Y $.

\subsection{Flows}

Another way to visualize the family of integral curves associated with a vector field is the following. Let $ \man $ be a smooth manifold and $ X \in \xm $, and suppose that, for each point $ p \in \man $, there is a unique integral curve $ \theta^{(p)} : \R \ra \man $ of $ X $ starting at $ p $ (i.e. with $ \gamma(0) = p $) and defined for all $ t \in \R $. For each $ t \in \R $, define $ \theta_t : \man \ra \man $ which sends each $ p \in \man $ to the point obtained by following for a time $ t $ the integral curve starting at $ p $, i.e.:
\begin{equation*}
  \theta_t(p) = \theta^{(p)}(t)
\end{equation*}
Therefore, each map $ \theta_t $ ``slides" the manifold along the integral curves of $ X $ for a time $ t $. By affine reparametrization invariance, for each $ s \in \R $ the map $ t \mapsto \theta^{(p)}(t + s) $ is still an integral curve of $ X $ starting at $ q = \theta^{(p)}(s) $: by the assumed uniqueness of integral curves, then, $ \theta^{(q)}(t) = \theta^{(p)}(s + t) $. Translating to the maps $ \theta_t $:
\begin{equation*}
  \theta_t \circ \theta_s = \theta_{t + s}
\end{equation*}
Together with $ \theta_0 = \id_\man $, which holds by definition, this implies that the map $ \theta : \R \times \man \ra \man $ is an action of the additive group $ \R_+ $ on $ \man $.

\begin{definition}{Global flow}{}
  Let $ \man $ be a topological manifold. A \bcdef{global flow} on $ \man $ is a continuous left $ \R_+ $-action on $ \man $, i.e. a continuous map $ \theta : \R \times \man \ra \man $ such that, for any $ s,t \in \R $ and $ p \in \man $:
  \begin{equation}
    \theta(t , \theta(s , p)) = \theta(t + s , p)
    \qquad \qquad
    \theta(0 , p) = p
  \end{equation}
\end{definition}

$ \theta : \R \times \man \ra \man $ is also called a \emph{one-parameter group action}. Given a global flow $ \theta $ on $ \man $, two collections of maps are defined:
\begin{itemize}
  \item for each $ t \in \R $, the map $ \theta_t : \man \ra \man $ is defined as $ \theta_t(p) \equiv \theta(t,p) $;
  \item for each $ p \in \man $, the map $ \theta^{(p)} : \R \ra \man $ is defined as $ \theta^{(q)}(t) = \theta(t,p) $.
\end{itemize}
As already noted, $ \theta_t $ follows the group laws, while the image of $ \theta^{(p)} $ is the orbit of $ p $ under the group action.
Moreover, on a smooth manifold it is possible to associate a vector field $ X $ to the global flow as $ p \mapsto X_p = {\theta^{(p)}}'(0) \in T_p\man $: this is called the \bctxt{infinitesimal generator} of $ \theta $.

\begin{proposition}{Smooth infinitesimal generator}{smooth-infinitesimal-generator}
  Let $ \theta : \R \times \man \ra \man $ be a smooth global flow on a smooth manifold $ \man $. The infinitesimal generator $ X $ of $ \theta $ is a smooth vector field on $ \man $, and each curve $ \theta^{(p)} $ is an integral curve of $ X $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    To show that $ X \in \xm $, it suffices to show that $ X(f) \in \cfk{\infty}(U) \,\,\forall f \in \cfk{\infty}(U) $, with $ U \in \tp(\man) $. For any such $ f $ and any $ p \in U $:
    \begin{equation*}
      X(f)(p) = X_p(f) = {\theta^{(p)}}'(0)(f) = \frac{\dd}{\dd t} \bigg\vert_{t = 0} f(\theta^{(p)}(t)) =  \frac{\pa}{\pa t} \bigg\vert_{(0,p)} f(\theta(t,p))
    \end{equation*}
    Since $ f(\theta(t,p)) $ is a smooth function of $ (t,p) $ by composition, $ X(f)(p) $ depends smoothly on $ p $, hence $ X \in \xm $.

    Next, show that $ \theta^{(p)} $ is an integral curve of $ X $ for all $ p \in \man $. This means that $ {\theta^{(p)}}'(t) = X_{\theta^{(p)}(t)} \,\,\forall p \in \man , t \in \R $. Let $ t_0 \in \R $ be arbitrary, and set $ q = \theta^{(p)}(t_0) = \theta_{t_0}(p) $, so WTS $ {\theta^{(p)}}'(t_0) = X_q $. By group law, for all $ t \in \R $:
    \begin{equation*}
      \theta^{(q)}(t) = \theta_t(q) = \theta_t(\theta_{t_0}(p)) = \theta_{t+t_0}(p) = \theta^{(p)}(t+t_0)
    \end{equation*}
    therefore, for any smooth real-valued function $ f $ defined on a neighbourhood of $ q $:
    \begin{equation*}
      X_q(f) = {\theta^{(q)}}'(0)(f) = \frac{\dd}{\dd t} \bigg\vert_{t=0} f(\theta^{(q)}(t)) = \frac{\dd}{\dd t} \bigg\vert_{t=0} f(\theta^{(p)}(t+t_0)) = {\theta^{(p)}}'(t_0)(f)
    \end{equation*}
    which completes the proof.
  \end{proof}
\end{proofbox}

While every smooth global flow gives rise to a smooth vector field whose integral curves are precisely the curves defined by the flow, the converse is not true: indeed, there are smooth vector fields whose integral curves are not defined for all $ t \in \R $.

\begin{definition}{Local flow}{}
  Let $ \man $ be a smooth manifold. A \emph{flow domain} is $ \dom \in \tp(\R \times \man) $ such that, for each $ p \in \man $, the set $ \dom^{(p)} \equiv \{t \in \R : (t,p) \in \dom\} $ is an open interval with $ 0 \in \dom^{(p)} $. A \bcdef{local flow} on $ \man $ is a continuous map $ \theta : \dom \ra \man $ such that, for all $ p \in \man $:
  \begin{enumerate}
    \item $ \theta(0,p) = p $;
    \item $ \theta(t,\theta(s,p)) = \theta(t+s,p) \,\,\forall s \in \dom^{(p)} , t \in \dom^{(\theta(s,p))} : s+t \in \dom^{(p)} $.
  \end{enumerate}
\end{definition}

Just as for a global flow, whenever $ (t,p) \in \dom $, the maps $ \theta_t(p) = \theta^{(p)}(t) = \theta(t,p) $ are defined. Moreover, for each $ t \in \R $, the set $ \man_t \equiv \{p \in \man : (t,p) \in \dom\} $ is defined, so that:
\begin{equation*}
  p \in \man_t
  \quad \iff \quad
  t \in \dom^{(p)}
  \quad \iff \quad
  (t,p) \in \dom
\end{equation*}
Given a smooth local flow $ \theta $, its infinitesimal generator $ X $ is defined by $ X_p = {\theta^{(p)}}'(0) $. Identically to \pref{prop:smooth-infinitesimal-generator}, if $ \theta : \dom \ra \man $ is a smooth local flow, then the infinitesimal generator $ X $ of $ \theta $ is a smooth vector field, and each curve $ \theta^{(p)} $ is an integral curve of $ X $. From now on, local flows are simply called ``flows".

A \bctxt{maximal integral curve} is an integral curve which cannot be extended to an integral curve on any larger open interval, while a \bctxt{maximal flow} is a flow which admits no extension to a flow on a larger flow domain.

\begin{theorem}{Fundamental theorem of flows}{fundamental-flows}
  Let $ X \in \xm $. There is a unique smooth maximal flow $ \theta : \dom \ra \man $, called the \bcth{flow of $ X $}, whose infinitesimal generator is $ X $ and such that, for each $ p \in \man $:
  \begin{enumerate}
    \item \label{item:ftf-1} $ \theta^{(p)} : \dom^{(p)} \ra \man $ is the unique maximal integral curve of $ X $ starting at $ p $;
    \item \label{item:ftf-2} if $ s \in \dom^{(p)} $, then $ \dom^{(\theta(s,p))} = \dom^{(p)} - s \equiv \{t-s : t \in \dom^{(p)}\} $.
  \end{enumerate}
\end{theorem}

It can be shown that, given a smooth map $ f : \man_1 \ra \man_2 $ and two vector fields $ X \in \xv(\man_1) , Y \in \xv(\man_2) $, denoting the flow of $ X $ as $ \theta $ and the flow of $ Y $ as $ \eta $, if $ X $ and $ Y $ are $ f $-related, then for each $ t \in \R $ the following diagram commutes:
\begin{equation}
  f(\man_{1,t}) \ssq \man_{2,t}
  \qquad \qquad
  \begin{tikzcd}
    \man_{1,t} \ar[r , "f"] \ar[d , "\theta_t" , swap] & \man_{2,t} \ar[d , "\eta_t"] \\
    \man_{1,-t} \ar[r , "f" , swap] & \man_{2,-t}
  \end{tikzcd}
\end{equation}
If $ f $ is a diffeomorphism, then clearly the flow of $ f_* X $ has the domain $ \mat_{2,t} = f(\man_{1,t}) $, for each $ t \in \R $, and is given by:
\begin{equation}
  \eta_t = f \circ \theta_t \circ f^{-1}
\end{equation}

A smooth vector field whose flow is a global flow is said to be \bctxt{complete}. Equivalently, the maximal integral curves of a complete vector field are defined for all $ t \in \R $. While it is difficult, in general, to determine a priori whether a vector field is complete or no, it turns out that a sufficient condition can be imposed on its \emph{support}, i.e. on the closure of the set $ \{p \in \man : X_p \neq 0\} $.

\begin{proposition}{Compactly support and completeness}{}
  Every compactly-supported smooth vector field on a smooth manifold is complete.
\end{proposition}

Trivially, then, on a compact smooth manifold every smooth vector field is complete.

\subsection{Lie derivative}
\label{ssec:lie-der}

In Euclidean space $ \R^n $, the directional derivative is trivially generalized to smooth vector fields. Indeed, given $ X \in \xv(\R^n) $ and $ \ve{v} \in T_p\R^n $, the directional derivative of $ X $ along $ \ve{v} $ is:
\begin{equation*}
  D_\ve{v} X(p) = \frac{\dd}{\dd t} \bigg\vert_{t = 0} X_{p + t \ve{v}} = \lim_{t \ra 0} \frac{X_{p + t \ve{v}} - X_p}{t} = D_\ve{v} X^i_p \frac{\pa}{\pa x^i} \bigg\vert_p
\end{equation*}
This definition, however, relies on the fact that $ T_{p + t\ve{v}}\R^n \cong \R^n \cong T_p\R^n $, being it a vector space, while for a general manifold $ T_p\man $ and $ T_q\man $ are entirely different vector spaces for $ p \neq q $. The solution is to consider a vector field $ V \in \xm $, in place of $ \ve{v} \in T_p\man $, so as to use the flow of $ V $ to push values of $ X $ to the same point.

\begin{definition}{Lie derivative of vector field}{lie-der-vec}
  Let $ \man $ be a smooth manifold and $ X,V \in \xm $. Denoting as $ \theta $ the flow of $ V $ on $ \man $, the \bcdef{Lie derivative} of $ X $ with respect to $ V $ is the vector field (provided it exists):
  \begin{equation}
    (\ld_V X)_p \defeq \frac{\dd}{\dd t} \bigg\vert_{t = 0} \dd (\theta_{-t})_{\theta_t(p)} (X_{\theta_t(p)})
  \end{equation}
\end{definition}

Since it is defined from smooth maps, it can be shown that $ \ld_V X \in \xm $. A visualization of the Lie derivative is pictured in \figref{fig:lie-der}.

\begin{figure}
  \centering
  \includegraphics[width = 0.70 \textwidth]{lie-der.png}
  \caption{Action of the Lie derivative on a vector field.}
  \label{fig:lie-der}
\end{figure}

\dref{def:lie-der-vec} is not particularly useful for computations, since it requires first to find the flow of $ V \in \xm $. Fortunately, a simple formula can be proved to hold for $ X,Y \in \xm $:
\begin{equation}
  \ld_V X = [V , X]
  \label{eq:lie-der-vec}
\end{equation}
which relates directly the Lie derivative to the two vector fields considered through their Lie bracket. This formula allows for an interpretation of the Lie bracket: it is the directional derivative of the second vector field along the flow of the first one.

Moreover, a number of non-obvious properties of Lie derivatives follow from the properties of Lie brackets.

\begin{lemma}{Properties of Lie derivative}{properties-lieder}
  Let $ \man $ be a smooth manifold and $ V,X,Y \in \xm $. Then:
  \begin{enumerate}
    \item \label{item:lieder-1} $ \ld_V X = - \ld_X V $;
    \item \label{item:lieder-2} $ \ld_V [X,Y] = [\ld_V X , Y] + [X , \ld_V Y] $;
    \item \label{item:lieder-3} $ \ld_{[V,X]} Y = \ld_V \ld_X Y - \ld_X \ld_V Y $;
    \item \label{item:lieder-4} if $ f \in \cm $, then $ \ld_V(fX) = V(f) X + f \ld_V X $;
    \item \label{item:lieder-5} if $ f : \man \ra \man' $ is a diffeomorphism, then $ f_*(\ld_V X) = \ld_{f_* V} f_* X $.
  \end{enumerate}
\end{lemma}

If $ V,X \in \xm $ and $ \theta $ if the flow of $ V $, the Lie derivative $ (\ld_V X)_p $ expressed, by definition, the $ t $-derivative of $ \dd (\theta_{-t})_{\theta_t(p)} (X_{\theta_t(p)}) $ at $ t = 0 $. To compute this $ t $-derivative at later times, the following formula is found:
\begin{equation}
  \frac{\dd}{\dd t} \bigg\vert_{t = t_0} \dd (\theta_{-t})_{\theta_t(p)} (X_{\theta_t(p)}) = \dd (\theta_{-t}) \left( (\ld_V X)_{\theta_{t_0}(p)} \right)
  \label{eq:vec-dir-der-aux}
\end{equation}

\subsubsection{Commuting vector fields}

Let $ \man $ be a smooth manifold. $ X,Y \in \xm $ are said to \bctxt{commute} if $ [X,Y] \equiv 0 $. Given a smooth flow $ \theta $ over $ \man $, a vector field if \emph{invariant under $ \theta $} if $ X $ is $ \theta_t $-related to itself for each $ t $: this means that $ X \vert_{\man_t} $ is $ \theta_t $-related to $ X \vert_{\man_{-t}} $ for each $ t $, or equivalently $ \dd (\theta_t)_p (X_p) = X_{\theta_t(p)} $ for all $ (t,p) $ in the domain of $ \theta $. These two concepts are closely related by the Lie derivative.

\begin{proposition}{Commutativity and flow invariance}{comm-flow}
  Let $ X,Y \in \xm $. Then the following conditions are equivalent:
  \begin{enumerate}
    \item \label{item:comm-flow-1} $ X $ and $ Y $ commute;
    \item \label{item:comm-flow-2} $ X $ is invariant under the flow of $ Y $;
    \item \label{item:comm-flow-3} $ Y $ is invariant under the flow of $ X $.
  \end{enumerate}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Denote the flow of $ Y $ with $ \theta $. If \itemref{item:comm-flow-2} holds, then $ X_{\theta_t(p)} = \dd (\theta_t)_p (X_p) $ for all $ (t,p) $ in the domain of $ \theta $. Applying $ \dd (\theta_{-t})_{\theta_t(p)} $ to both sides, $ X_p = \dd (\theta_{-t})_{\theta_t(p)}(X_{\theta_t(p)}) $ which directly implies $ [Y,X] = \ld_Y X = 0 $ from \dref{def:lie-der-vec}. The same argument shows that \itemref{item:comm-flow-3} $ \implies $ \itemref{item:comm-flow-1}.

    To prove that \itemref{item:comm-flow-1} $ \implies $ \itemref{item:comm-flow-2}, assume $ [Y,X] = \ld_Y X = 0 $. Let $ p \in \man $ be arbitrary, and let $ W(t) = \dd (\theta_{-t})_{\theta_t(p)}(X_{\theta_t(p)}) $ for $ t \in \dom^{(p)} $. By \eref{eq:vec-dir-der-aux} $ W'(t) \equiv 0 $, but since $ W(0) = X_p $, this implies $ W(t) = X_p \,\,\forall t \in \dom^{(p)} $. Applying $ \dd (\theta_t)_p $ to both sides shows that $ X $ is invariant under $ \theta $. The same argument shows that \itemref{item:comm-flow-1} $ \implies $ \itemref{item:comm-flow-3}.
  \end{proof}
\end{proofbox}

Trivially, then, every smooth vector field is invariant under its own flow, as $ [X,X] \equiv 0 $.

A deeper characterization of commuting vector fields is in terms of the relationship between their respective flows. Suppose $ X,Y \in \xm $, and let $ \theta $ and $ \psi $ denote their respective flows. If $ X $ and $ Y $ are complete, these flows are said to \emph{commute} if $ \theta_t \circ \psi_s = \psi_s \circ \theta_t \,\,\forall s,t \in \R $. In general, however, the naive imposition of this relation on the intersection of the flow domains does not give the right result (i.e. the flows might not commute in this naive sense, even though the vector field do). A generalization of the commutation condition is possible.

\begin{definition}{Commuting flows}{}
  Let $ \man $ be a smooth manifold and let $ \theta , \psi $ be smooth flows on $ \man $. Then, $ \theta $ and $ \psi $ are said to \bcdef{commute} if, for every $ p \in \man $, whenever $ J , K \ni 0 $ are open intervals such that one of the expressions $ \theta_t \circ \psi_s $ or $ \psi_s \circ \theta_t $ is defined for all $ (s,t) \in J \times K $, then both are defined and they are equal.
\end{definition}

\begin{theorem}{Commutativity}{}
  Smooth vector fields commute if and only if their flows commute.
\end{theorem}

An important application of commuting vector fields is in the study of frames on smooth manifolds. Given a smooth $ n $-manifold $ \man $, a smooth local frame $ (E_1 , \dots , E_n) $ is a \bctxt{commuting frame} (or \emph{holonomic frame}) if $ [E_i , E_j] = 0 \,\,\forall i,j = 1 , \dots , n $.

Since every coordinate frame is a commuting frame, given that Lie brackets are defined independently of local charts, it follows that a necessary condition for a smooth frame to be expressible as a coordinate frame in some smooth chart is that it must be a commuting frame. It turns out that this is, in fact, a sufficient condition too.

\begin{proposition}{Commuting and coordinate frames}{}
  Let $ \man $ be a smooth $ n $-manifold, and let $ (X_1 , \dots , X_k) $ be a linearly independent $ k $-tuple of smooth commuting vector fields on $ V \in \tp(\man) $. Then, for each $ p \in V $, there exists a smooth coordinate chart $ (U , (x^i)) $ centered at $ p $ such that $ V_i = \pa_i $ for $ i = 1 , \dots , k $.
\end{proposition}

\subsubsection{Time-dependent vector fields}

The analysis of integral curves has been so far limited to time-independent vector fields, for which the system of autonomous ODEs \eref{eq:aut-sys} is granted to have a solution by \tref{th:aut-odes}. Now, the case of non-autonomous ODEs is explored.

\begin{definition}{Time-dependent vector field}{}
  Let $ \man $ be a smooth manifold. A \bcdef{time-dependent vector field} is a continuous map $ X : J \times \man \ra T\man $, where $ J \ssq \R $ is an interval, such that $ X(t,p) \in T_p\man $ for each $ (t,p) \in J \times \man $.
\end{definition}

By this definition, the map $ X_t : \man \ra T\man : X_t(p) = X(t,p) $ is clearly a vector field on $ \man $ for each $ t \in J $. Then, given an interval $ J_0 \ssq J $, a curve $ \gamma : J_0 \ra \man $ is an integral curve of $ X $ if:
\begin{equation}
  \gamma'(t) = X(t, \gamma(t))
\end{equation}
for all $ t \in J_0 $. This is a natural extension of \dref{def:integral-curve}: indeed, note that any ordinary vector field $ X $ on $ \man $ determines a time-dependent vector field on $ \R \times \man $ just by setting $ X(t,p) \equiv X_p $.

A time-dependent vector field might not generate a flow, since two integral curves starting at the same point but at different times might follow a different path, while all integral curves of a flow through a given point must have the same image. In this context, the fundamental theorem of flows \tref{th:fundamental-flows} is generalized as follows.

\begin{theorem}{Fundamental theorem of time-dependent flows}{}
  Let $ \man $ be a smooth manifold, let $ J \ssq \R $ be an interval, and let $ X : J \times \man \ra T\man $ be a smooth time-dependent vector field on $ \man $. There exist $ \mathcal{E} \in \tp(J \times J \times \man) $ and a smooth map $ \psi : \mathcal{E} \ra \man $, called the \bcth{time-dependent flow of $ X $}, such that:
  \begin{enumerate}
    \item \label{item:fttdf-1} for each $ t_0 \in J $ and $ p \in \man $, the set $ \mathcal{E}^{(t_0,p)} = \{t \in J : (t,t_0,p) \in \mathcal{E}\} \ni t_0 $ is an open interval, and the smooth curve $ \psi^{(t_0,p)} : \mathcal{E}^{(t_0,p)} \ra \man $ defined by $ \psi^{(t_0,p)}(t) = \psi(t,t_0,p) $ is the unique maximal integral curve of $ X $ with initial condition $ \psi^{(t_0,p)}(t_0) = p $;
    \item \label{item:fttdf-2} if $ t_1 \in \mathcal{E}^{(t_0,p)} $ and $ q = \psi^{(t_0,p)}(t_1) $, then $ \mathcal{E}^{(t_1,p)} = \mathcal{E}^{(t_0,p)} $ and $ \psi^{(t_1,p)} = \psi^{(t_0,p)} $;
    \item \label{item:fttdf-3} for each $ (t_1,t_0) \in J \times J $, the set $ \man_{t_1,t_0} = \{p \in \man : (t_1,t_0,p) \in \mathcal{E}\} \in \tp(\man) $, and the map $ \psi_{t_1,t_0} : \man_{t_1,t_0} \ra \man $ defined by $ \psi_{t_1,t_0}(p) = \psi(t_1,t_0,p) $ is a diffeomorphism from $ \man_{t_1,t_0} $ to $ \man_{t_0,t_1} $ with inverse $ \psi_{t_0,t_1} $;
    \item \label{item:fttdf-4} if $ p \in \man_{t_1,t_0} $ and $ \psi_{t_1,t_0}(p) \in \man_{t_2,t_1} $, then $ p \in \man_{t_2,t_0} $ and $ \psi_{t_2,t_1} \circ \psi_{t_1,t_0}(p) = \psi_{t_2,t_0}(p) $.
  \end{enumerate}
\end{theorem}

\begin{example}{Euclidean example}{}
  Define a time-dependent vector field $ X $ on $ \R^n $ as:
  \begin{equation*}
    X(t,x) = \frac{1}{t} x^i \frac{\pa}{\pa x^i} \bigg\vert_x
    \qquad
    (t,x) \in \R^+ \times \R^n
  \end{equation*}
  Consider arbitrary $ (t_0,x_0) \in \R_+ \times \R^n $, and let $ \gamma(t) = (x^1(t) , \dots , x^n(t)) $ denote the integral curve of $ X $ with initial condition $ \gamma(t_0) = x_0 $, described by the following system of non-autonomous ODEs: $ \dot{x}^i(t) = x^i(t) / t $ and $ x^i(t_0) = x^i_0 $, for $ i = 1 , \dots , n $.

  The maximal solution is easily found as $ x^i(t) = t x^i_0 / t_0 $, for $ t \in \R^+ $, hence the time-dependent flow of $ X $ is given by $ \psi(t,t_0,x) = t x / t_0 $, for $ (t,t_0,x) \in \R^+ \times \R^+ \times \R^n $.
\end{example}

\section{Vector bundles}

It is possible to define additional structure on a topological manifold, in addition to a smooth structure.

\begin{definition}{$ \R $-vector bundle}{}
  Given a topological space $ M $, an \bcdef{$ \R $-vector bundle of rank $ k $} over $ M $ is a topological space $ E $ together with a surjective continuous map $ \pi : E \ra M $ such that:
  \begin{enumerate}
    \item $ \forall p \in M $ the \emph{fiber} $ E_p = \pi^{-1}(p) $ over $ p $ is endowed with the structure of a $ k $-dimensional $ \R $-vector space;
    \item $ \forall p \in M \, \exists U \ssq M : p \in U \land \chi : \pi^{-1}(U) \ra U \times \R^k $ homeomorphism, called \emph{local trivialization of $ E $ over $ U $}, such that:
      \begin{itemize}
        \item $ \pi_1 \circ \chi = \pi $, with the projector $ \pi_1 : U \times \R^k \ra U $;
        \item $ \forall q \in U $ the restriction of $ \chi $ to $ E_q $ is a vector-space isomorphism between $ E_q $ and $ \{q\} \times \R^k \cong \R^k $.
      \end{itemize}
  \end{enumerate}
\end{definition}

This definition generalizes trivially to complex vector bundle substituting $ \R $ with $ \C $. In the following, only real vector bundles are considered.

If $ M $ and $ E $ are smooth manifolds, $ \pi $ is a smooth map and the local trivializations can be chosen to be diffeomorphisms, then $ E $ is called a \bctxt{smooth vector bundle}. If there exists a local trivialization of $ E $ over the whole $ M $, i.e. $ \chi : E \ra M \times \R^k $, then $ E $ is said to be a \emph{trivial bundle}: clearly, any non-trivial bundle requires more than one trivialization.

\begin{lemma}{}{}
  Let $ \pi : E \ra \man $ be a smooth vector bundle of rank $ k $ over $ \man $, and suppose $ \chi : \pi^{-1}(U) \ra U \times \R^k $ and $ \psi : \pi^{-1}(V) \ra V \times \R^k $ are smooth local trivializations of $ E $ such that $ U \cap V \neq \emptyset $. Then, there exists a smooth map $ \tau : U \cap V \ra \GL{k,\R} $ such that $ \chi \circ \psi^{-1} : (U \cap V) \times \R^k \ra (U \cap V) \times \R^k $ has the form:
  \begin{equation}
    \chi \circ \psi^{-1}(p,v) = (p , \tau(p)v)
  \end{equation}
\end{lemma}

\begin{proofbox}
  \begin{proof}
    By hypothesis, the following diagram commutes:
    \begin{equation*}
      \begin{tikzcd}
        (U \cap V) \times \R^k \ar[dr , "\pi_1" , swap] & \pi^{-1}(U \cap V) \ar[l , "\psi" , swap] \ar[r , "\chi"] \ar[d , "\pi" , swap] & (U \cap V) \times \R^k \ar[dl , "\pi_1"] \\
                                                                       & U \cap V
      \end{tikzcd}
    \end{equation*}
    where the local trivialization are interpreted as the respective restrictions on $ \pi^{-1}(U \cap V) $. It follows that $ \pi_1 \circ (\chi \circ \psi^{-1}) = \pi_1 $, hence:
    \begin{equation*}
      \chi \circ \psi^{-1}(p,v) = (p , \sigma(p,v))
    \end{equation*}
    for some smooth map $ \sigma : (U \cap V) \times \R^k \ra \R^k $. Moreover, for each $ p \in U \cap V $, the map $ \R^k \ra \R^k $ defined as $ v \mapsto \sigma(p,v) $ is an invertible linear map, hence the thesis.
  \end{proof}
\end{proofbox}

The smooth map $ \tau : U \cap V \ra \GL{n,\R} $ is called the \bctxt{transition function} between $ \chi $ and $ \psi $.

A natural construction for a smooth vector bundle on a smooth manifold is to define a collection of vector spaces, one for each point of the manifold: this is the same idea as the tangent bundle, which is in fact a vector bundle (see \secref{ssec:tangent-bundle}). In order to make such a collection a smooth vector bundle, one first needs to define a manifold topology and a smooth structure on the total space of the bundle, i.e. on the disjoint union of said collection, and then construct the smooth local trivializations. However, these steps can be condensated, provided that local trivializations overlap with smooth transition functions.

\begin{theorem}{Vector bundle construction theorem}{}
  Let $ \man $ be a smooth manifold, and let $ \{U_\alpha\}_{\alpha \in \mathcal{I}} $ be an open cover of $ \man $. Suppose that, for each $ \alpha , \beta , \gamma \in \mathcal{I} $, there exists a smooth map $ \tau_{\alpha \beta} : U_\alpha \cap U_\beta \ra \GL{k,\R} $ such that:
  \begin{equation}
    \tau_{\alpha \beta}(p) \tau_{\beta \gamma}(p) = \tau_{\alpha \gamma}(p)
    \qquad
    \forall p \in U_\alpha \cap U_\beta \cap U_\beta
  \end{equation}
  Then, there exists a smooth rank-$ k $ vector bundle $ \pi : E \ra \man $ with smooth local trivializations $ \chi_\alpha : \pi^{-1}(U_\alpha) \ra U_\alpha \times \R^k $, whose transition functions are the given maps $ \tau_{\alpha \beta} $.
\end{theorem}

\subsection{Tangent bundle}
\label{ssec:tangent-bundle}

\begin{proposition}{Tangent bundle as vector bundle}{tangent-bundle-vector-bundle}
  Given a smooth $ n $-manifold $ \man $, then its tangent bundle $ T\man $ has the structure of smooth vector bundle of rank $ n $ over $ \man $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Since the fibers $ \pi^{-1}(p) = T_p\man $ are $ n $-dimensional $ \R $-vector spaces by \tref{th:tangent-space}, only the smooth local trivializations need to be constructed.

    Given any smooth chart $ (U,\varphi) $ of $ \man $ with coordinates $ (x^1 , \dots , x^n) $, define the map $ \chi : \pi^{-1}(U) \ra U \times \R^n $ as:
    \begin{equation*}
      \chi : X^i \frac{\pa}{\pa x^i} \bigg\vert_p \mapsto (p , X^1 , \dots , X^n)
    \end{equation*}
    This is linear on fibers and satisfies $ \pi_1 \circ \chi = \pi $, i.e. the following diagram is commutative:
    \begin{equation*}
      \begin{tikzcd}
        \pi^{-1}(U) \ar[r , "\chi"] \ar[dr, "\pi", swap] & U \times \R^n \ar[d , "\pi_1"] \\
        & U
      \end{tikzcd}
    \end{equation*}
    Now, note that the composite map $ (\varphi \times \id_{\R^n}) \circ \chi : \pi^{-1}(U) \varphi(U) \times \R^n $, i.e.:
    \begin{equation*}
      \pi^{-1}(U) \xrightarrow{\chi} U \times \R^n \xrightarrow{\varphi \times \id_{\R^n}} \varphi(U) \times \R^n \ssq \R^{2n}
    \end{equation*}
    is equal to the coordinate map $ \tilde{\varphi} : \pi^{-1}(U) \ra \R^{2n} $ defined above. Both $ \varphi \times \id_{\R^n} $ and $ \tilde{\varphi} $ are diffeomorphisms, hence $ \chi $ is a smooth local trivialization of $ T\man $ over $ U \ssq \man $.
  \end{proof}
\end{proofbox}

It can be shown that the smooth structure defined in \pref{prop:tangent-bundle-smooth} is the only smooth structure on $ T\man $ such that $ \pi : T\man \ra \man $ is smooth vector bundle. The coordinates $ (x^i , X^i) $ are called \emph{natural coordinates} on $ T\man $.

\subsubsection{Sections and frames}

The natural structure of $ T\man $ as a vector bundle over $ \man $ allows to import the notions of vector bundles on the tangent bundle.

\begin{definition}{Section of vector bundle}{}
  Let $ \pi : E \ra M $ be a vector bundle. A \bcdef{section} of $ E $ is a continuous map $ \sigma : M \ra E $ such that:
  \begin{equation}
    \pi \circ \sigma = \id_M
  \end{equation}
\end{definition}

This means that $ \sigma $ associate to each point $ p \in M $ an element of its fiber, i.e. $ \sigma : p \mapsto \sigma(p) \in E_p $. In general, it is possible to define \emph{local} sections of $ E $ as continuous maps $ \sigma : U \ra E $ defined on $ U \in \tp(M) $ and satisfying $ \pi \circ \sigma = \id_U $: sections of $ E $ defined over all $ M $ are then called \emph{global} sections (clearly a local section over $ U $ is a global section of the restricted bundle $ E \vert_U $). If the vector bundle considered is a smooth vector bundle and $ \sigma $ is a smooth map from its domain to $ E $, then it is a \emph{smooth} (local or global) section of $ E $.

The set of all smooth global sections of a smooth vector bundle $ \pi : E \ra M $, denoted by $ \Gamma(E) $, has the natural structure of $ \R $-vector space under pointwise addition and scalar multiplication:
\begin{equation*}
  (c_1 \sigma_1 + c_2 \sigma_2)(p) = c_1 \sigma_1(p) + c_2 \sigma_2(p)
  \qquad
  \forall c_1 , c_2 \in \R , \sigma_1 , \sigma_2 \in \Gamma(E) , p \in M
\end{equation*}
The multiplication of $ \sigma \in \Gamma(E) $ by $ f \in \mathcal{C}^\infty(M) $ yields a new section $ (f\sigma)(p) = f(p) \sigma(p) $.

The identification of sections of $ T\man $ with vector fields over $ \man $ is then clear, and in particular:
\begin{equation}
  \Gamma(T\man) = \xm
\end{equation}

To introduce frames on vector bundles, it is first necessary to import the notion of linear independence on sections. Given $ U \in \tp(M) $, an $ m $-tuple of local sections $ (\sigma_1 , \dots , \sigma_m) $ of $ E $ over $ U $ is linearly independent if $ (\sigma_1(p) , \dots , \sigma_m(p)) $ is a linearly independent $ m $-tuple in $ E_p $ for all $ p \in U $.

\begin{definition}{Frame for vector bundle}{vector-bundle-frame}
  Let $ \pi : E \ra M $ be a vector bundle of rank $ k $. A \bcdef{local frame} for $ E $ over $ U \in \tp(M) $ is an ordered $ k $-tuple $ (\sigma_1 , \dots , \sigma_k) $ of linearly independent local sections over $ U $ such that:
  \begin{equation*}
    E_p = \lspan \{\sigma_1(p) , \dots , \sigma_k(p)\}
    \qquad
    \forall p \in U
  \end{equation*}
  If $ U = M $, then this is a \bcdef{global frame}.
\end{definition}

If $ E \ra M $ is a smooth vector bundle and each $ \sigma_i $ is a smooth (local or global) section, then $ (\sigma_1 , \dots , \sigma_k) $ is a \emph{smooth} (local or global) frame.

Local frames are linked to local trivializations. To see this, consider a smooth vector bundle $ \pi : E \ra M $ of rank $ k $ and a smooth local trivializiation $ \chi : \pi^{-1}(U) \ra U \times \R^k $, with $ U \in \tp(M) $. Take the standard basis $ \{e_1 , \dots , e_k\} $ for $ \R^k $ and define $ \tilde{e}_i : U \ra U \times \R^k $ as $ \tilde{e}_i(p) = (p , e_i) $. Moreover, define the maps $ \sigma_1 , \dots , \sigma_k : U \ra E $ by:
\begin{equation*}
  \sigma_i(p) = \chi^{-1}(p,e_i) = \chi^{-1} \circ \tilde{e}_i(p)
  \qquad \qquad
  \begin{tikzcd}
    \pi^{-1}(U) \ar[rr , "\chi"] \ar[dr , "\pi"] & & U \times \R^k \ar[dl , "\pi_1" , swap] \\
                                                 & U \ar[ul, "\sigma_i", bend left = 40] \ar[ur, "\tilde{e}_i", bend right = 40] &
  \end{tikzcd}
\end{equation*}
Then, $ \sigma_i $ is smooth since $ \chi $ is a diffeomorphism, and it is a section of $ E $:
\begin{equation*}
  \pi_1 \circ \chi = \pi
  \quad \implies \quad
  \pi \circ \sigma_i(p) = \pi \circ \chi^{-1}(p , e_i) = \pi_1(p , e_i) = p
\end{equation*}
To see that $ (\sigma_1(p) , \dots , \sigma_k(p)) $ is a basis of $ E_p $, note that $ \chi $ is an isomorphism $ E_p \cong \{p\} \times \R^k $ such that $ \chi(\sigma_i(p)) = (p , e_i) $, hence $ \chi $ maps $ \{\sigma_1(p) , \dots , \sigma_k(p)\} $ to the standard basis of $ \{p\} \times \R^k \cong \R^k $. Therefore, $ (\sigma_1 , \dots , \sigma_k) $ is a \emph{local frame associated to $ \chi $}. It can be shown that every smooth local frame for a smooth vector bundle is associated to a smooth local trivialization in this way.

On $ T\man $, local frames determine local bases for tangent spaces. An important case is that of \emph{coordinate frames}: given a smooth chart $ (U , \varphi) $ with coordinate functions $ (x^1 , \dots , x^n) $, then the coordinate vector fields $ (\pa_1 , \dots , \pa_n) $ form a smooth local frame on $ U $.

\subsubsection{Bundle homomorphisms}

\begin{definition}{Bundle homomorphism}{bundle-homomorphism}
  Let $ \pi : E \ra \man $ and $ \pi' : E' \ra \man' $ be vector bundles. A continuous map $ f : E \ra E' $ is a \bcdef{bundle homomorphism} if there exists a map $ F : \man \ra \man' $ satisfying:
  \begin{equation}
    \pi' \circ f = F \circ \pi
    \qquad \qquad
    \begin{tikzcd}
      E \ar[r , "f"] \ar[d , "\pi" , swap] & E' \ar[d , "\pi'"] \\
      \man \ar[r , "F" , swap] & \man'
    \end{tikzcd}
  \end{equation}
  and such that, for each $ p \in \man $, the restriction $ f \vert_{E_p} : E_p \ra E'_{F(p)} $ is linear.
\end{definition}

The relation between $ f $ and $ F $ is expressed by saying that $ f $ \emph{covers} $ F $. The covered map $ F $ is continuous and uniquely by $ f $ as:
\begin{equation}
  F = \pi' \circ f \circ \zeta
\end{equation}
where $ \zeta : \man \ra E $ is the \bctxt{zero section}, defined as $ \zeta(p) = 0 \in E_p \,\,\forall p \in \man $. Clearly, if $ f $ is smooth between smooth bundles, then $ F $ is smooth too.
A bijective bundle homomorphism whose inverse is also a bundle homomorphism is called a \emph{bundle isomorphism}, and the two bundles are said to be isomorphic.

\begin{example}{Tangent bundle homomorphism}{}
  Let $ \man_1 $ and $ \man_2 $ be smooth manifolds, and let $ f : \man_1 \ra \man_2 $ be a smooth map. Then, the global differential $ \dd f : T\man_1 \ra T\man_2 $ is a smooth bundle homomorphism covering $ f $.
\end{example}

A case of particular interest is that of vector bundles with the same basis. In this case, if $ \pi : E \ra \man $ and $ \pi' : E' \ra \man $ are homomorphic, the bundle homomorphism $ f : E \ra E' $ is said to be \emph{over} $ \man $ if it covers $ \id_\man $, i.e. if the following diagram commutes:
\begin{equation*}
  \begin{tikzcd}
    E \ar[rd , "\pi" , swap] \ar[rr , "f"] & & E' \ar[dl , "\pi'"] \\
                                          & \man &
  \end{tikzcd}
\end{equation*}
and if the restriction of $ f $ to each fiber is linear.
Given smooth bundles $ \pi : E \ra \man $ and $ \pi' : E' \ra \man $, if $ f : E \ra E' $ is a smooth bundle homomorphism over $ \man $, then composition with $ f $ induces a map $ \tilde{f} : \Gamma(E) \ra \Gamma(E') $ defined as $ \tilde{f}(\sigma) = f \circ \sigma $. Since a bundle homomorphism is linear on fibers, the resulting map $ \tilde{f} $ is linear over $ \R $. In fact, it is linear over $ \cm $:
\begin{equation*}
  \tilde{f}(g_1 \sigma_1 + g_2 \sigma_2) = g_1 \tilde{f}(\sigma_1) + g_2 \tilde{f}(\sigma_2)
\end{equation*}
for any $ g_1 , g_2 \in \cm $ and $ \sigma_1 , \sigma_2 \in \Gamma(E) $. The converse is true as well.

\begin{lemma}{Characterization of bundle homomorphisms}{char-bundle-homo}
  Let $ \pi : E \ra \man $ and $ \pi' : E' \ra \man $ be smooth vector bundles. A map $ \tilde{f} : \Gamma(E) \ra \Gamma(E') $ is linear over $ \cm $ if and only if there is a smooth bundle homomorphism $ f : E \ra E' $ over $ \man $ such that $ \tilde{f} = f \circ \sigma \,\,\forall \sigma \in \Gamma(E) $.
\end{lemma}

Given this lemma, it is customary to identify $ \tilde{f} $ with $ f $, hence writing directly $ f : \Gamma(E) \ra \Gamma(E') $, since the action on sections is obtained simply applying the bundle homomorphism pointwise: for example, if $ a \in \R $, then the same notation $ X \mapsto a X $ denotes both the operation of multiplying vectors in each tangent space $ T_p\man $ by $ a $, and the operation of multiplying vector fields by $ a $ (note that multiplying by $ a $ in indeed a bundle homomorphism $ T\man \ra T\man $).

\subsection{Cotangent bundle}

While tangent vectors naturally give a coordinate-free interpretation of derivatives of curves, the same can be achieved for real-valued functions on manifolds introducing functionals on the tangent space at each point.

\subsubsection{Covectors}

\begin{definition}{Covector}{}
  Let $ V $ be a finite-dimensional $ \R $-vector space. A \bcdef{covector} on $ V $ is a linear map $ \omega : V \ra \R $.
\end{definition}

The space of all covectors is itself a real vector space under pointwise addition and scalar multiplication, and it is usually denoted by $ V^* \defeq \Hom_\R(V,\R) $.

\begin{proposition}{Dual basis}{dual-basis}
  Let $ V $ be a finite-dimensional $ \R $-vector space. Given any basis $ (E_1 , \dots , E_n) $ of $ V $, its \bcprop{dual basis} $ (\varepsilon^1 , \dots , \varepsilon^n) \ssq V^* $ is defined by:
  \begin{equation}
    \varepsilon^i(E_j) = \delta^i_j
  \end{equation}
  and it is a basis of $ V^* $. Therefore, $ \dim_\R V_* = \dim_\R V $.
\end{proposition}

If $ (e_1 , \dots , e_n) $ is the standard basis of $ \R^n $, the dual basis $ (e^1 , \dots , e^n) $ is called the \emph{standard dual basis} of $ V^* $. In general, if $ (E_i) $ is a basis of $ V $ and $ (\varepsilon^i) $ is its dual basis, then given $ v = v^j E_j \in V $:
\begin{equation*}
  \varepsilon^i(v) = v^j \varepsilon^i(E_j) = v^j \delta^i_j = v^i
\end{equation*}
By \pref{prop:dual-basis}, any covector can be expressed as $ \omega = \omega_i \varepsilon^i \in V_* $, hence its action on $ v = v^j E_j \in V $ is clearly determined:
\begin{equation}
  \omega(v) = \omega_i v^i
\end{equation}

Given two finite-dimensional $ \R $-vector spaces $ V,W $ and $ A \in \Hom_\R(V,W) $, it is possible to define a \bctxt{dual map} (or transposed map) $ A^* \in \Hom_\R(W^* , V^*) $ as:
\begin{equation}
  A^* \circ \omega = \omega \circ A
\end{equation}
The dual map clearly satisfies $ (A \circ B)^* = B^* \circ A^* $ and $ \id_V^* = \id_{V^*} $.

\begin{proposition}{Second dual space}{second-dual-basis}
  For any finite-dimensional $ \R $-vector space $ V $, there exists a canonical isomorphism $ \xi : V \ra V^{**} \equiv (V^*)^* $ such that, for each $ v \in V $, $ \xi(v) : V^* \ra \R $ acts on $ \omega \in V^* $ as:
  \begin{equation}
    \xi(v)(\omega) = \omega(v)
  \end{equation}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    By \pref{prop:dual-basis}, $ \dim_\R V^{**} = \dim_\R V $, hence it suffices to verify that $ \xi $ is injective. Suppose $ v \in V - \{0\} $, extend it to a basis $ (v \equiv E_1 , E_2 , \dots , E_n) $ of $ V $ and let $ (\varepsilon^1 , \dots , \varepsilon^n) $ be its dual basis. Then, $ \xi(v) \neq 0 $ since:
    \begin{equation*}
      \xi(v)(\varepsilon^1) = \varepsilon^1(v) = \varepsilon^1(E_1) = 1
    \end{equation*}
    Therefore, $ \ker \xi = \{0\} $, i.e. it is injective.
  \end{proof}
\end{proofbox}

The existence of this canonical isomorphism requires $ \dim_\R V $ to be finite: in the infinite-dimensional case, \pref{prop:dual-basis} does not hold in the first place, i.e. there exists in general no isomorphism $ V \cong V^* $, hence no canonical isomorphism $ V \cong V^{**} $ either. Moreover, note that the isomorphism $ V \cong V^* $ can never be canonical, since it depends explicitly on the choice of basis.

Given \pref{prop:second-dual-basis}, $ \omega(v) \in \R $ is usually denoted with a more symmetric notation $ \braket{\omega , v} = \braket{v , \omega} $, which can be thought of either as the action of $ \omega \in V^* $ on $ v \in V $, or as the action of $ \xi(v) \in V^{**} $ on $ \omega \in V^* $. In the following, the map $ \xi $ is usually omitted, and $ v \in V $ is thought of either as a vector or as a linear functional on $ V^* $, depending on the context.

\subsubsection{Tangent covectors and covector fields}

\begin{definition}{Cotangent space}{}
  Let $ \man $ be a smooth manifold. For each $ p \in \man $, the \bcdef{cotangent space} at $ p $ is $ T^*_p\man \defeq (T_p\man)^* $.
\end{definition}

The elements of $ T^*_p\man $ are called the \emph{tangent covectors} at $ p $. Given a smooth chart $ (U,(x^i)) $ of $ \man $, for each $ p \in U $ the coordinate basis $ (\pa_i \vert_p) $ of $ T_p\man $ determines a \emph{coordinate dual basis} $ (\lambda^i \vert_p) $ of $ T^*_p\man $. Hence, any covector $ \omega \in T^*_p\man $ can be written as:
\begin{equation*}
  \omega = \omega_i \lambda^i \vert_p
  \qquad \qquad
  \omega_i = \omega \left( \frac{\pa}{\pa x^i} \bigg\vert_p \right)
\end{equation*}
Now, let $ (V , (\tilde{x}^i)) $ be another smooth chart that contains $ p $, and let $ (\tilde{\lambda}^i \vert_p) $ be the dual basis of $ (\pa_i \vert_p) $. Applying \eref{eq:contravariant-alibi}, by the linearity of $ \omega \in V^* $, the transformation rule for covectors is found:
\begin{equation}
  \omega_i = \frac{\pa \tilde{x}^j}{\pa x^i}(\hat{p}) \tilde{\omega}_j
  \label{eq:covariant-alias}
\end{equation}
Since they transform as basis vectors, i.e. according to the Jacobian of the coordinate transformation, covectors are also called \emph{covariant vectors}.

As for tangent vectors, it is possible to define the \bctxt{cotangent bundle} of a smooth manifold $ \man $ as the disjoint union:
\begin{equation}
  T^*\man \defeq \bigsqcup_{p \in \man} T^*_p\man
\end{equation}
which comes with a natural projection $ \pi : T^*\man \ra \man $ which sends $ (p,\omega) \mapsto p $.

\begin{proposition}{Cotangent bundle as vector bundle}{}
  Given a smooth $ n $-manifold $ \man $, then its cotangent bundle $ T^*\man $ has the structure of smooth vector bundle of rank $ n $ over $ \man $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    The proof is analogous to that of \pref{prop:tangent-bundle-vector-bundle}.
  \end{proof}
\end{proofbox}

As in the case of the tangent bundle, the smooth structure on $ T^*\man $ determined by that of $ T\man $ is the unique one which gives to $ \pi : T^*\man \ra \man $ the structure of smooth rank-$ n $ vector bundle on $ \man $. Moreover, smooth local coordinates on $ \man $ yield smooth local coordinates on $ T^*\man $: indeed, if $ (x^i) $ are smooth coordinates on $ U \in \tp(\man) $, then the map $ \pi^{-1}(U) \ra \R^{2n} $ defined as:
\begin{equation}
  \xi_i \lambda^i \vert_p \mapsto (x^1(p) , \dots , x^n(p) , \xi_1 , \dots , \xi_n)
\end{equation}
is a smooth local coordinate map on $ \pi^{-1}(U) \in \tp(T^*\man) $. These are called the \emph{natural coordinates} on $ T^*\man $ associated to $ (x^i) $.

As for vector fields, a (local or global) section of $ T^*\man $ is called a \bctxt{covector field} (or \emph{(differential) 1-form}). In any smooth local coordinates on $ U \in \tp(\man) $, a covector field $ \omega $ can be written in terms of coordinate covector fields $ \lambda^i $ as $ \omega = \omega_i \lambda^i $, with the $ n $ functions $ \omega_i : U \ra \R $ being the \emph{component functions} of $ \omega $:
\begin{equation*}
  \omega_i(p) = \omega_p \left( \frac{\pa}{\pa x^i} \bigg\vert_p \right)
\end{equation*}
Given a vector field $ X $ and a covector field $ \omega $ over $ \man $, it is possible to form a function $ \omega(X) : \man \ra \R $ such that, for any $ p \in \man $:
\begin{equation}
  \omega(X)(p) = \omega_p(X_p)
\end{equation}
In terms of the respective component functions, this function has the coordinate representation $ \omega(X) = \omega_i X^i $.

\begin{lemma}{Smoothness criteria for covector fields}{smooth-covector-fields}
  Let $ \man $ be a smooth manifold, and let $ \omega : \man \ra T^*\man $ be a covector field. Then the following conditions are equivalent:
  \begin{enumerate}
    \item \label{item:sccf-1} $ \omega $ is smooth;
    \item \label{item:sccf-2} in every coordinate chart, the component functions of $ \omega $ are smooth;
    \item \label{item:sccf-3} each point of $ \man $ is contained in some coordinate chart in which $ \omega $ has smooth component functions;
    \item \label{item:sccf-4} for every $ X \in \xm $, the function $ \omega(X) $ is smooth on $ \man $;
    \item \label{item:sccf-5} for every $ U \in \tp(\man) $ and $ X \in \xv(U) $, the function $ \omega(X) : U \ra \R $ is smooth on $ U $.
  \end{enumerate}
\end{lemma}

Since any open subset of a smooth manifold is again a smooth manifold, this lemma applies equally well on covector fields defined only on some open subset of $ \man $.

\begin{definition}{Coframe}{}
  Let $ \man $ be a smooth manifold, and let $ U \in \tp(\man) $. A \bcdef{local coframe} over $ U $ is an ordered $ n $-tuple of covector fields $ (\varepsilon^1 , \dots , \varepsilon^n) $ defined on $ U $ such that $ (\varepsilon^i \vert_p) $ forms a basis of $ T^*_p\man $ at each $ p \in U $. If $ U = \man $, this is a \bcdef{global coframe}.
\end{definition}

Clearly, a local coframe of $ \man $ is just a local frame of the vector bundle $ T^*\man $, as per \dref{def:vector-bundle-frame}. Given a local frame $ (E_1 , \dots , E_n) $ for $ T\man $ over $ U \in \tp(\man) $, there is a uniquely determined local coframe $ (\varepsilon^1 , \dots , \varepsilon^n) $ for $ T^*\man $ over $ U $ defined by $ \varepsilon^i(E_j) = \delta^i_j $: this coframe is the \emph{dual coframe}.

\begin{lemma}{}{}
  Let $ \man $ be a smooth manifold. If $ (E_i) $ is a local frame over $ U \in \tp(\man) $ and $ (\varepsilon^i) $ is its dual coframe, then $ (E^i) $ is smooth if and only if $ (\varepsilon^i) $ is smooth.
\end{lemma}

Given a local coframe $ (\varepsilon^i) $ over $ U \in \tp(\man) $, every covector field $ \omega $ over $ U $ can be expressed as $ \omega = \omega_i \varepsilon^i $, where the $ n $-functions $ \omega_i : U \ra \R $ are the component functions of $ \omega $ with respect to the given coframe and are determined by $ \omega_i = \omega(E_i) $, where $ (E_i) $ is the dual frame of $ (\omega^i) $.

The $ \R $-vector space of all smooth covector fields on $ \man $ is denoted by $ \xms \equiv \Gamma(T^*\man) $. As smooth sections of a vector bundle, elements of $ \xms $ can be multiplied by $ \R $-valued functions, yielding new elements\footnotemark of $ \xms $: given $ f \in \cm $ and $ \omega \in \xms $, then $ f\omega \in \xms $ is defined as $ (f\omega)_p = f(p) \omega_p $.

\footnotetext{This means that $ \xms $ is a module over the ring $ \cm $.}

\subsubsection{Differential of functions}

The partial derivatives of a smooth function cannot be interpreted in a coordinate-free way as the components of a vector fields, but they can be as the components of a covector field.

\begin{definition}{Differential}{differential-function}
  Let $ f \in \cm $. The \bcdef{differential of $ f $} is the covector field $ \dd f : \man \ra T^*\man $ defined by:
  \begin{equation}
    \dd f_p(X_p) = X_p(f)
    \qquad
    \forall X_p \in T_p\man
  \end{equation}
\end{definition}

By \itemref{item:sccf-4} of \lref{lemma:smooth-covector-fields}, the differential of a smooth function is a smooth covector field. To compute the coordinate expression of $ \dd f $, consider a smooth map $ (U,(x^i)) $ on $ \man $: on the coordinate basis, the components of $ \dd f_p $ are $ \dd f_p (\pa_i \vert_p) = \pa_i \vert_p f $, hence:
\begin{equation}
  \dd f_p = \frac{\pa f}{\pa x^i} (p) \lambda^i \vert_p
\end{equation}
The differential can then be though of as the analogue of the gradient, reinterpreted in a coordinate-free way. In the special case $ f = x^j : U \ra \R $, the differential is:
\begin{equation*}
  \dd x^j \vert_p = \frac{\pa x^j}{\pa x^i} (p) \lambda^i \vert_p = \delta^j_i \lambda^i \vert_p = \lambda^j \vert_p
\end{equation*}
This shows that the coordinate dual basis $ \lambda^i \vert_p $ is composed just by the differentials of the coordinate functions. Hence, in terms of covector fields:
\begin{equation}
  \dd f = \frac{\pa f}{\pa x^i} \dd x^i
  \label{eq:func-diff}
\end{equation}
which is the usual expression, reinterpreted in terms of covector fields. The so-defined differential has the usual properties expected from Analysis.

\begin{lemma}{Properties of the differential}{}
  Let $ \man $ be a smooth manifold, and let $ f,g \in \cm $. Then:
  \begin{enumerate}
    \item $ \dd (a f + b g) = a \dd f + b \dd g \,\,\forall a,b \in \R $;
    \item $ \dd (fg) = f \dd g + g \dd f $;
    \item if $ h : J \ra \R $ is smooth over the interval $ J \ssq \R : J \supseteq \im f $, then $ \dd (h \circ f) = (h' \circ f) \dd f $.
  \end{enumerate}
\end{lemma}

The differential can also be used to express derivatives along curves. Consider a smooth manifold $ \man $, a curve $ \gamma : J \ra \man $ and $ f \in \cm $. Then, the derivative of the function $ f \circ \gamma : J \ra \R $ is (from definitions):
\begin{equation}
  (f \circ \gamma)'(t) = \dd f_{\gamma(t)}(\gamma'(t))
  \label{eq:composite-curve-function}
\end{equation}
Moreover, the notions of differential of a function $ f \in \cm $ given by \dref{def:differential} and by \dref{def:differential-function} are truly the same object, once the canonical identification $ \R \cong T_{f(p)}\R $ is taken into account, and they are both represented by a row vector whose components are the partial derivatives of $ f $. Similarly, given a smooth curve $ \gamma : J \ra \man $, the expression $ (f \circ \gamma)'(t) $ has two different meanings. On the one hand, $ f \circ \gamma $ can be interpreted as a smooth curve over $ \R $, and thus $ (f \circ \gamma)'(t) $ is the velocity at the point $ f \circ \gamma(t) $, which is an element of $ T_{f \circ \gamma(t)}\R $ and equal to $ \dd f_{\gamma(t)}(\gamma'(t)) $ as per \eref{eq:composite-curve}. On the other hand, $ f \circ \gamma $ can also be interpreted simply as a single-variable $ \R $-valued function, and $ (f \circ \gamma)'(t) $ is thus its ordinary derivative and equal to $ \dd f_{\gamma(t)}(\gamma'(t)) $ as per \eref{eq:composite-curve-function}, now seen as an ordinary real number. The two interpretations are again equivalent as $ \R \cong T_{f \circ \gamma(t)}\R $.

\subsubsection{Pull-back of covector fields}

Since the differential is a linear map on tangent vectors, its dual is a linear map on covectors in the opposite direction.

\begin{definition}{Pull-back}{pull-back}
  Let $ f : \man_1 \ra \man_2 $ be smooth map, and let $ p \in \man_1 $. The differential $ \dd f_p : T_p\man_1 \ra T_{f(p)}\man_2 $ yields a dual linear map $ \dd f_p^* : T_{f(p)}^*\man_2 \ra T_p^*\man_1 $, called the \bcdef{(pointwise) pull-back} by $ f $ at $ p $, defined as:
  \begin{equation}
    \dd f_p^*(\omega_{f(p)})(v) = \omega_{f(p)}(\dd f_p(v))
    \qquad
    \forall \omega_{f(p)} \in T_{f(p)}^*\man_2 , v \in T_p\man_1
    \label{eq:pull-back-def}
  \end{equation}
\end{definition}

Contrary to vector fields, whose push-forwards are only defined in presence of diffeomorphisms (\tref{th:push-forward-vector-fields}), covector fields are always pulled-back to covector fields. Given a smooth map $ f : \man_1 \ra \man_2 $ and a covector field $ \omega $ on $ \man_2 $, the pull-back of $ \omega $ by $ f $ is defined as:
\begin{equation}
  (f^*\omega)_p \defeq \dd f^*_p(\omega_{f(p)})
\end{equation}
which is a covector in $ T_p^*\man_1 $ and, as per \dref{def:pull-back}, acts on $ v \in T_p\man_1 $ as:
\begin{equation*}
  (f^*\omega)_p(v) = \omega_{f(p)}(\dd f_p(v))
\end{equation*}

\begin{proposition}{Smoothness of pull-back}{}
  Let $ f : \man_1 \ra \man_2 $ be a smooth map, and let $ \omega $ be a covector field on $ \man_2 $. Then, $ f^*\omega $ is a (continuous) covector field on $ \man_1 $. Moreover, if $ \omega $ is smooth, then $ f^*\omega $ is smooth.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Let $ p \in \man_1 $ arbitrary, and choose a smooth chart $ (V,(y^i)) $ on $ \man_2 $ which contains $ f(p) $, so that $ U = f^{-1}(V) $ is a neighbourhood of $ p $.

    \begin{lemma}[borderline = {1pt}{0pt}{blue!35!black}]{}{pb-cont-aux}
      Let $ f : \man_1 \ra \man_2 $ be a smooth map, and suppose that $ u : \man_2 \ra \R $ is a continuous function and that $ \omega $ is a covector field on $ \man_2 $. Then:
      \begin{equation}
        f^*(u\omega) = (u \circ f) f^*\omega
      \end{equation}
      In addition, if $ u $ is smooth:
      \begin{equation}
        f^* \dd u = \dd (u \circ f)
      \end{equation}
    \end{lemma}

    \begin{proofbox}
      \begin{proof}
        By direct computation, using the definitions:
        \begin{equation*}
          \begin{split}
            (f^*(u\omega))_p = \dd f_p^* ((u\omega)_{f(p)})
            & = \dd f_p^*(u(f(p)) \omega_{f(p)}) \\
            & = u(f(p)) \dd f_p^*(\omega_{f(p)}) = u(f(p)) (f^*\omega)_p = ((u \circ f) f^*\omega)_p
          \end{split}
        \end{equation*}
        Letting $ v \in T_p\man_1 $, then:
        \begin{equation*}
          (f^* \dd u)_p(v) = (\dd f_p^*(\dd u_{f(p)}))(v) = \dd u_{f(p)} (\dd f_p(v)) = \dd f_p(v) u = v(u \circ f) = \dd (u \circ f)_p(v)
        \end{equation*}
        This completes the proof.
      \end{proof}
    \end{proofbox}

    Writing $ \omega $ in coordinate as $ \omega = \omega_j \dd y^j $, with continuous component functions $ \omega_j $ on $ V $, and using \lref{lemma:pb-cont-aux} twice (applied to $ f \vert_U $):
    \begin{equation*}
      f^* \omega = f^* (\omega_j \dd y^j) = (\omega_j \circ f) f^* \dd y^j = (\omega_j \circ f) \dd (y^j \circ f)
    \end{equation*}
    which is clearly continuous, and smooth if $ \omega $ is smooth.
  \end{proof}
\end{proofbox}

The formula derived in this proof provides a straightforward expression to compute the pull-back of a covector field, using the fact that $ y^j \circ f = f^j $ is the $ j^\text{th} $ component function of $ f $ in the $ (y^j) $ coordinates:
\begin{equation}
  f^* \omega = (\omega_j \circ f) \dd f^j
  \label{eq:pull-back-simple}
\end{equation}

\begin{example}{Euclidean spaces}{}
  Let $ f : \R^3 \ra \R^2 $ be the map:
  \begin{equation*}
    (u,v) = f(x,y,z) = (x^2y , y \sin z)
  \end{equation*}
  and let $ \omega \in \xvs(\R^2) $ be the covector field:
  \begin{equation*}
    \omega = v \, \dd u + u \, \dd v
  \end{equation*}
  Using \eref{eq:pull-back-simple}, the pull-back of $ \omega $ by $ f $ reads:
  \begin{equation*}
    \begin{split}
      f^*\omega
      & = (v \circ f) \dd (u \circ f) + (u \circ f) \dd (v \circ f) \\
      & = (x^2 y) \dd (y \sin z) + (y \sin z) \dd (x^2 y) = 2xy^2 \sin z \, \dd x + 2 x^2 y \sin z \, \dd y + x^2 y^2 \cos z \, \dd z
    \end{split}
  \end{equation*}
\end{example}

\section{Tensors}

So far, the analysis of smooth manifold has been carried by importing concepts of Linear Algebra to them. Now, this approach is extended by considering multilinear maps.

\subsection{Multilinear algebra}

Let $ V_1 , \dots , V_k , W $ be $ \R $-vector spaces. A map $ f : V_1 \times \dots \times V_k \ra W $ is said to be \bcdef{multilinear} if it is linear in each variable, i.e. for each $ i = 1 , \dots , k $ and arbitrary $ a,b \in \R $:
\begin{equation*}
  f(v_1 , \dots , a v_i + b v'_i , \dots , v_k) = a f(v_1 , \dots , v_i , \dots , v_k) + b f(v_1 , \dots , v'_i , \dots , v_k)
\end{equation*}
The set of multilinear maps from $ V_1 \times \dots \times V_k $ to $ W $ is denoted by $ \Mult(V_1 , \dots , V_k ; W) $: it clearly is a $ \R $-vector space under pointwise addition and scalar multiplication.

\begin{example}{Familiar multilinear maps}{}
  Both the dot product and the determinant are multilinear functions on $ \R^n $: the first one is a bilinear function, while the second one is an $ n $-linear function.
  In the special case of $ \R^3 $, another bilinear map is encountered: the cross product, which has $ \R^3 $ itself as codomain.
\end{example}

An extremely important example of multilinear map is the following one.

\begin{definition}{Tensor product}{}
  Let $ V_1 , \dots , V_k , W_1 , \dots , W_\ell $ be $ \R $-vector spaces, and suppose $ f \in \Mult(V_1 , \dots , V_k ; \R) $ and $ g \in \Mult(W_1 , \dots , W_\ell ; \R) $. The \bcdef{tensor product} of $ f $ and $ g $ is a function $ f \otimes g : V_1 \times \dots \times V_k \times W_1 \times \dots \times W_\ell \ra \R $ defined as:
  \begin{equation}
    f \otimes g (v_1 , \dots , v_k , w_1 , \dots , w_\ell) \defeq f(v_1 , \dots , v_k) g(w_1 , \dots , w_\ell)
  \end{equation}
  which is an element of $ \Mult(V_1 , \dots , V_k , W_1 , \dots , W_\ell ; \R) $.
\end{definition}

It is easy to check that the tensor product is bilinear and associative, hence tensor products of three or more multilinear maps can be written anambiguously without parantheses.

Since linear maps on vector spaces are covectors, a natural description of multilinear maps is through tensor products of covectors.

\begin{proposition}{Basis for multilinear functions}{mult-basis}
  Let $ V_1 , \dots , V_k $ be $ \R $-vector spaces of dimensions $ n_1 , \dots , n_k $, respectively, and, for each $ j = 1 , \dots , k $, let $ (E_1^{(j)} , \dots , E_{n_j}^{(j)}) $ be a basis for $ V_j $, with $ (\varepsilon_{(j)}^1 , \dots , \varepsilon_{(j)}^{n_j}) $ the corresponding dual basis of $ V_j^* $. Then, the set:
  \begin{equation*}
    \bas \defeq \{ \varepsilon_{(1)}^{i_1} \otimes \dots \otimes \varepsilon_{(k)}^{i_k} : 1 \leq i_1 \leq n_1 , \dots , 1 \leq i_k \leq n_k \}
  \end{equation*}
  is a basis of $ \Mult(V_1 , \dots , V_k ; \R) $, which therefore has dimension $ n_1 \dots n_k $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Show that $ \bas $ is linearly independent and spans $ \Mult(V_1 , \dots , V_k ; \R) $. Suppose $ f \in \Mult(V_1 , \dots , V_k ; \R) $ is arbitrary. For each ordered $ k $-tuple $ (i_1 , \dots , i_k) $ of integers such that $ 1 \leq i_j \leq n_j $, define the real number $ f_{i_1 \dots i_k} = f(E_{i_1}^{(1)} , \dots , E_{i_k}^{(k)}) $. WTS $ f = f_{i_1 \dots i_k} \varepsilon_{(1)}^{i_1} \otimes \dots \otimes \varepsilon_{(k)}^{i_k} $.
    For any $ k $-tuple $ (v_1 , \dots , v_k) \in V_1 \times \dots \times V_k $, write $ v_j = v_j^{i_j} E_{i_j}^{(j)} $:
    \begin{equation*}
      f_{i_1 \dots i_k} \varepsilon_{(1)}^{i_1} \otimes \dots \otimes \varepsilon_{(k)}^{i_k}(v_1 , \dots , v_k) = f_{i_1 \dots i_k} \varepsilon_{(1)}^{i_1}(v_1) \dots \varepsilon_{(k)}^{i_k}(v_k) = f_{i_1 \dots i_k} v^1 \dots v^k
    \end{equation*}
    which is just $ f(v_1 , \dots , v_k) $ by multilinearity. This proves that $ \bas $ spans $ \Mult(V_1 , \dots , V_k ; \R) $.

    To show that $ \bas $ is linearly independent, suppos some linear combination vanishes:
    \begin{equation*}
      f_{i_1 \dots i_k} \varepsilon_{(1)}^{i_1} \otimes \dots \otimes \varepsilon_{(k)}^{i_k} = 0
    \end{equation*}
    Apply this to any ordered $ k $-tuple of basis vectors $ (E_{j_1}^{(1)} , \dots , E_{j_k}^{(k)}) $. By the same computation above, this implies that all the coefficients vanish: $ f_{i_1 \dots i_k} \equiv 0 $. Thus, $ \bas $ is linearly independent.
  \end{proof}
\end{proofbox}

This proof shows that the components of $ f \in \Mult(V_1 , \dots , V_k ; \R) $ in the basis $ \bas $ are:
\begin{equation}
  f_{i_1 \dots i_k} = f(E_{i_1}^{(1)} , \dots , E_{i_k}^{(k)})
\end{equation}
Thus, $ f $ is completely determined by its action on all possible sequences of basis vectors.

\subsubsection{Abstract tensor products}

The idea that $ \Mult(V_1 , \dots , V_k ; \R) $ can be viewed as the set of all linear combinations of objects of the for $ \omega^1 \otimes \dots \otimes \omega^k $, where $ \omega^1 , \dots , \omega^k $ are covectors, can be extended with the introduction of ``formal linear combinations".

Let $ S $ be a set. A \emph{formal linear combination} of elements of $ S $ is an expression of the form $ \sum_{i = 1}^m a_i x_i $, where $ a_1 , \dots , a_m \in \R $ and $ x_1 , \dots , x_m \in S $. Since no algebraic structure is being imposed on $ S $, its elements cannot literally be added or multiplied by numbers, however such an expression is completely determined by which elements of $ S $ appear in the sum and with which coefficients, hence this definition can in fact be formalized.

\begin{definition}{Free real vector space}{}
  Let $ S $ be a set. A \bcdef{formal linear combination} of elements of $ S $ is a function $ f : S \ra \R $ such that $ f(s) = 0 $ for all but finitely many $ s \in S $. The \bcdef{free $ \R $-vector space} on $ S $, denoted by $ \free(S) $, is the set of all formal linear combinations of elements of $ S $.
\end{definition}

$ \free(S) $ is indeed a $ \R $-vector space under pointwise addition and scalar multiplication. For each $ x \in S $, there is a function $ \delta_x \in \free(S) $ that takes the value $ 1 $ on $ x $ and vanishes on $ S - \{x\} $: typically, the element $ x $ is identified with $ \delta_x $, hence $ S $ can be thought as a subset of $ \free(S) $, and every $ f \in \free(S) $ can be written uniquely in the form $ f = \sum_{i = 1}^m a_i x_i $, where $ x_1 , \dots , x_m $ are the elements of $ S $ such that $ a_i = f(x_i) \neq 0 $. Thus, $ S $ is a basis of $ \free(S) $, which is therefore finite-dimensional if and only if $ S $ is a finite set.

\begin{proposition}{Characteristic property of free vector spaces}{char-prop-free-vector-spaces}
  For any set $ S $ and any $ \R $-vector space $ W $, every map $ A : S \ra W $ has a unique extension to a linear map $ \bar{A} : \free(S) \ra W $.
\end{proposition}

It is now possible to generalize the tensor product. Let $ V_1 , \dots , V_k $ be $ \R $-vector spaces, and let $ \free(V_1 \times \dots \times V_k) $ be the set of all finite formal combinations of $ k $-tuples $ (v_1 , \dots , v_k) $, with $ v_i \in V_i $ for $ i = 1 , \dots , k $. Then, define $ \free_0 $ as the subspace of $ \free(V_1 \times \dots \times V_k) $ spanned by all elements of the forms:
\begin{equation*}
  (v_1 , \dots , a v_i , \dots , v_k) - a (v_1 , \dots , v_i , \dots , v_k)
\end{equation*}
\begin{equation*}
  (v_1 , \dots , v_i + v_i' , \dots , v_k) - (v_1 , \dots , v_i , \dots , v_k) - (v_1 , \dots , v_i' , \dots , v_k)
\end{equation*}
with $ v_j , v_j' \in V_j $ for $ j = 1 , \dots , k $ and $ a \in \R $. The \emph{tensor product} of the spaces $ V_1 , \dots , V_k $, denoted as $ V_1 \otimes \dots \otimes V_k $, is defined as the following quotient vector space:
\begin{equation}
  V_1 \otimes \dots \otimes V_k \defeq \free(V_1 \times \dots \times V_k) / \free_0
\end{equation}
Moreover, define the natural projection $ \Pi : \free(V_1 \times \dots \times V_k) \ra V_1 \otimes \dots \otimes V_k $, so that the equivalence class of an element $ (v_1 , \dots , v_k) \in V_1 \times \dots \times V_k $ is $ V_1 \otimes \dots \otimes V_k $ is denoted by:
\begin{equation}
  v_1 \otimes \dots \otimes v_k \equiv \Pi(v_1 , \dots , v_k)
\end{equation}
This is called the \bctxt{(abstract) tensor product} of $ v_1 , \dots , v_k $. It follows from definition that abstract tensor products satisfy:
\begin{equation*}
  v_1 \otimes \dots \otimes (a v_i + b v_i') \otimes \dots \otimes v_k = a (v_1 \otimes \dots \otimes v_i \otimes \dots \otimes v_k) + b (v_1 \otimes \dots \otimes v_i' \otimes \dots \otimes v_k)
\end{equation*}

Note that, although by definition every element of $ V_1 \otimes \dots \otimes V_k $ can be expressed as a linear combination of elements of the form $ v_1 \otimes \dots \otimes v_k $, it is not true in general that every element of the tensor product space is of the form $ v_1 \otimes \dots \otimes v_k $.

\begin{theorem}{Characteristic property of tensor product spaces}{char-prop-tensor-prod}
  Let $ V_1 , \dots , V_k $ be finite-dimensional $ \R $-vector spaces. If $ A : V_1 \times \dots \times V_k \ra X $ is any multilinear map into a vector space $ X $, then there is a unique linear map $ \tilde{A} : V_1 \otimes \dots \otimes V_k \ra X $ such that the following diagram commutes:
  \begin{equation}
    \begin{tikzcd}
      V_1 \times \dots \times V_k \ar[r , "A"] \ar[d , "\pi" , swap] & X \\
      V_1 \otimes \dots \otimes V_k \ar[ur , "\tilde{A}" , swap , dashed] &
    \end{tikzcd}
  \end{equation}
  where $ \pi : V_1 \times \dots \times V_k \ra V_1 \otimes \dots \otimes V_k $ is the map $ \pi(v_1 , \dots , v_k) = v_1 \otimes \dots \otimes v_k $.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    First, note that any map $ A : V_1 \times \dots \times V_k \ra X $ extends uniquely to a linear map $ \bar{A} : \free(V_1 \times \dots \times V_k) \ra X $ by \pref{prop:char-prop-free-vector-spaces}, which is characterized by $ \tilde{A}(v_1 , \dots , v_k) = A(v_1 , \dots , v_k) $ whenever $ (v_1 , \dots , v_k) \in V_1 \times \dots \times V_k \ssq \free(V_1 \times \dots \times V_k) $. Since $ A $ is multilinear, the subspace $ \free_0 $ is contained in $ \ker \tilde{A} $, therefore $ \bar{A} $ determines a linear map $ \tilde{A} : V_1 \otimes \dots \otimes V_k = \free(V_1 \times \dots \times V_k) / \free_0 \ra X $ satisfying $ \tilde{A} \circ \Pi = \bar{A} $. Since $ \pi $ is equal to the inclusion $ V_1 \times \dots \times V_k \hookrightarrow \free(V_1 \times \dots \times V_k) $ followed by $ \Pi $, this implies $ \tilde{A} \circ \pi = A $, which is the thesis.

    Uniqueness follows from the fact that every element of $ V_1 \otimes \dots \otimes V_k $ can be written as a linear combination of elements of the form $ v_1 \otimes \dots \otimes v_k $, and $ \tilde{A} $ is uniquely determined on such elements by $ \tilde{A}(v_1 \otimes \dots \otimes v_k) = \bar{A}(v_1 , \dots , v_k) = A(v_1 , \dots , v_k) $.
  \end{proof}
\end{proofbox}

This is called the ``characteristic property" of the tensor product since it uniquely characterizes the tensor product up to isomorphism. Using this characteristic property, it is possible to show that the tensor product is associative. i.e. there are unique isomorphisms:
\begin{equation*}
  V_1 \otimes (V_2 \otimes V_3) \cong V_1 \otimes V_2 \otimes V_3 \cong (V_1 \otimes V_2) \otimes V_3
\end{equation*}
under which elements of the forms $ v_1 \otimes (v_2 \otimes v_3) $, $ v_1 \otimes v_2 \otimes v_3 $ and $ (v_1 \otimes v_2) \otimes v_3 $ all correspond.

\begin{proposition}{Basis of tensor product space}{tensor-prod-basis}
  Let $ V_1 , \dots , V_k $ be $ \R $-vector spaces of dimensions $ n_1 , \dots , n_k $, respectively, and, for each $ j = 1 , \dots , k $, let $ (E_1^{(j)} , \dots , E_{n_j}^{(j)}) $ be a basis for $ V_j $. Then, the set:
  \begin{equation*}
    \bas \defeq \{ E_{i_1}^{(1)} \otimes \dots \otimes E_{i_k}^{(k)} : i \leq i_1 \leq n_1 , \dots , 1 \leq i_k \leq n_k \}
  \end{equation*}
  is a basis for $ V_1 \otimes \dots \otimes V_k $, which therefore has dimension equal to $ n_1 \dots n_k $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Elements of the form $ v_1 \otimes \dots \otimes v_k $ span the tensor space by definition: expanding each $ v_i $ in such an expression in terms of its basis representation shows that $ \bas $ spans $ V_1 \otimes \dots \otimes V_k $.

    To prove that $ \bas $ is linearly independent, assume that some linear combination vanishes:
    \begin{equation*}
      a^{i_1 \dots i_k} E_{i_1}^{(1)} \otimes \dots \otimes E_{i_k}^{(k)} = 0
    \end{equation*}
    For each ordered $ k $-tuple of indices $ (m_1 , \dots , m_k) $, define a multilinear map $ \tau^{m_1 \dots m_k} : V_1 \times \dots \times V_k \ra \R $ by:
    \begin{equation*}
      \tau^{m_1 \dots m_k}(v_1 , \dots , v_k) = \varepsilon_{(1)}^{m_1}(v_1) \dots \varepsilon_{(k)}^{m_k}(v_k)
    \end{equation*}
    where $ (\varepsilon_{(j)}^j) $ is the basis of $ V_j^* $ dual to $ (E_j^{(j)}) $. Since $ \tau^{m_1 \dots m_k} $ is multilinear, it determines a linear map $ \tilde{\tau}^{m_1 \dots m_k} : V_1 \otimes \dots \otimes V_k \ra \R $ by \tref{th:char-prop-tensor-prod}. It follows that:
    \begin{equation*}
      0 = \tilde{\tau}^{m_1 \dots m_k}(a^{i_1 \dots i_k} E_{i_1}^{(1)} \otimes \dots \otimes E_{i_k}^{(k)}) = a^{i_1 \dots i_k} \tau^{m_1 \dots m_k}(E_{i_1}^{(1)} , \dots , E_{i_k}^{(k)}) = a^{m_1 \dots m_k}
    \end{equation*}
    which shows that $ \bas $ is linearly independent.
  \end{proof}
\end{proofbox}

It is now possible to show the connection between ``abstract" tensor products and ``concrete" tensor products.

\begin{lemma}
  If $ V_1 , \dots , V_k $ are finite-dimensional vector spaces, there is a canonical isomorphism:
  \begin{equation}
    V_1^* \otimes \dots \otimes V_k^* \cong \Mult(V_1 , \dots , V_k ; \R)
  \end{equation}
  under which the $ \omega^1 \otimes \dots \otimes \omega^k \in \Mult(V_1 , \dots , V_k ; \R) $, defined by $ \omega^1 \otimes \dots \otimes \omega^k (v_1 , \dots , v_k) = \omega^1(v_1) \dots \omega^k(v_k) $, and $ \omega^1 \otimes \dots \otimes \omega^k \in V_1^* \otimes \dots \otimes V_k^* $, defined by $ \omega^1 \otimes \dots \otimes \omega^k = \Pi(\omega^1 , \dots , \omega^k) $, correspond.
\end{lemma}

\begin{proofbox}
  \begin{proof}
    Define the map $ \Phi : V_1^* \times \dots \times V_k^* \ra \Mult(V_1 , \dots , V_k ; \R) $ as:
    \begin{equation*}
      \Phi(\omega^1 , \dots , \omega^k)(v_1 , \dots , v_k) = \omega^1(v_1) \dots \omega^k(v_k)
    \end{equation*}
    which is clearly multilinear in $ (v_1 , \dots , v_k) $. Moreover, $ \Phi(\omega^1 , \dots , \omega^k) $ is multilinear in $ (\omega^1 , \dots , \omega^k) $, hence by the characteristic property of the tensor product it determined a unique linear map $ \tilde{\Phi} : V_1^* \otimes \dots \otimes V_k^* \ra \Mult(V_1 , \dots , V_k ; \R) $ which satisfies:
    \begin{equation*}
      \tilde{\Phi}(\omega^1 \otimes \dots \otimes \omega^k)(v_1 , \dots , v_k) = \omega^1(v_1) \dots \omega^k(v_k)
    \end{equation*}
    By this defintion, $ \tilde{\Phi} $ clearly maps abstract tensor products to tensor products of covectors, and it also maps the basis of $ V_1^* \otimes \dots \otimes V_k^* $ given by \pref{prop:tensor-prod-basis} to the basis of $ \Mult(V_1 , \dots , V_k ; \R) $ given by \pref{prop:mult-basis}, therefore it is an isomorphism.
  \end{proof}
\end{proofbox}

In presence of this canonical isomorphism, $ \Mult(V_1 , \dots , V_k ; \R) $ is henceforth identified with $ V_1^* \otimes \dots \otimes V_k^* $. Moreover, given that the vector spaces considered are finite-dimensional, the canonical isomorphism $ V \cong V^{**} $ allows for another canonical identification:
\begin{equation}
  V_1 \otimes \dots \otimes V_k \cong \Mult(V_1^* , \dots , V_k^* ; \R)
\end{equation}
which emphasizes again the symmetry between $ V $ and $ V^* $ for finite-dimensional vector spaces.

\subsubsection{Mixed tensors}

Having defined tensor products of vector spaces, it is now possible to characterize multilinear maps on a $ \R $-vector space $ V $. Elements of the $ k $-fold tensor product $ V^* \otimes \dots \otimes V^* $ are called \emph{covariant $ k $-tensors} on $ V $: these are just multilinear maps $ \alpha : V \times \dots \times V \ra \R $. Similarly, elemnts of the $ k $-fold product $ V \otimes \dots \otimes V $ are called \emph{contravariant $ k $-tensors} on $ V $, which are multilinear maps $ \alpha : V^* \times \dots \times V^* \ra \R $.

\begin{definition}{Tensor}{}
  Let $ V $ be a $ \R $-vector space. Given $ k,l \in \N_0 $, the space of \bcdef{tensors of rank $ (k,l) $} is defined as:
  \begin{equation}
    T^k_l(V) \defeq \underbrace{V \otimes \dots \otimes V}_{k \text{ times}} \otimes \underbrace{V^* \otimes \dots \otimes V^*}_{l \text{ times}}
  \end{equation}
\end{definition}

Clearly, $ T^0_0(V) = \R $, $ T^1_0(V) = V $ and $ T^0_1(V) = V^* $, hence scalars, vectors and covectors can be thought as tensors. Moreover, by \pref{prop:tensor-prod-basis}, when $ V $ is finite-dimensional, any choice of basis $ (E_i) $ of $ V $ yields automatically a basis of $ T^k_l(V) $:
\begin{equation}
  \bas = \{E_{i_1} \otimes \dots \otimes E_{i_k} \otimes \varepsilon^{j_1} \otimes \dots \otimes \varepsilon^{j_l} \in T^k_l(V) : 1 \leq i_1 , \dots , i_k , j_1 , \dots , j_l \leq n\}
\end{equation}
where $ (\varepsilon^i) $ is the dual basis of $ V^* $ and $ n \equiv \dim_\R V $. Therefore, $ \dim_\R T^k_l(V) = n^{k+l} $, and given $ \alpha \in T^k_l(V) $, its $ n^{k+l} $ components are found as:
\begin{equation}
  \tensor{\alpha}{^{i_1 \dots i_k}_{j_1 \dots j_l}} = \alpha(\varepsilon^{i_1} , \dots , \varepsilon^{i_k} , E_{j_1} , \dots , E_{j_l})
  \label{eq:tensor-components}
\end{equation}

\subsection{Tensor fields}

\begin{definition}{Tensor bundles}{tensor-bundle}
  Given a smooth manifold $ \man $, its \bcdef{tensor bundle of rank $ (k,l) $} is defined as:
  \begin{equation}
    T^k_l\man \defeq \bigsqcup_{p \in \man} T^k_l(T_p\man)
  \end{equation}
\end{definition}

There are some natural identifications: $ T^0_0\man = \man \times \R $, $ T^1_0\man = T\man $ and $ T^0_1\man = T^*\man $. As for vector fields, sections of $ T^k_l\man $ are \bctxt{tensor fields of rank $ (k,l) $} over $ \man $, and in any smooth chart $ (x^i) $ on $ \man $ they can be written as:
\begin{equation}
  A = \tensor{A}{^{i_1 \dots i_k}_{j_1 \dots j_l}} \frac{\pa}{\pa x^{i_1}} \otimes \dots \otimes \frac{\pa}{\pa x^{i_k}} \otimes \dd x^{j_1} \otimes \dots \otimes \dd x^{j_l}
  \label{eq:tensor-comp}
\end{equation}
 This expression clearly determined the alias transformation rule for tensor fields. Consider two overlapping charts $ (U,(x^i)) , (V,(\tilde{x}^i)) $, and suppose that $ \tensor{A}{^{i_1 \dots i_k}_{j_1 \dots j_l}} $ are the components of a rank-$ (k,l) $ tensor field $ A $ over $ \man $ in the first chart. Then, the components of $ A $ in the second chart are:
\begin{equation}
  \tensor{\tilde{A}}{^{i_1 \dots i_k}_{j_1 \dots j_l}} = \frac{\pa \tilde{x}^{i_1}}{\pa x^{\alpha_1}} \dots \frac{\pa \tilde{x}^{i_k}}{\pa x^{\alpha_k}} \frac{\pa x^{\beta_1}}{\pa \tilde{x}^{j_1}} \dots \frac{\pa x^{\beta_l}}{\pa \tilde{x}^{j_l}} \tensor{A}{^{\alpha_1 \dots \alpha_k}_{\beta_1 \dots \beta_l}}
  \label{eq:tensor-trans}
\end{equation}

The space of smooth sections of $ T^k_l\man $ is an infinite-dimensional $ \R $-vector space and a module over $ \cm $, denoted by:
\begin{equation}
  \tns{k}{l}(\man) \equiv \Gamma(T^k_l\man)
\end{equation}
Note that, given $ A \in \tns{k}{l}(\man) $ and $ B \in \tns{p}{q}(\man) $, then trivially $ A \otimes B \in \tns{k+p}{l+q}(\man) $. Moreover, if $ A \in \tns{k}{l}(\man) $, $ \omega^1 , \dots , \omega^k \in \xms $ and $ X_1 , \dots , X_l \in \xm $, then $ A(\omega^1 , \dots , \omega^k , X_1 , \dots , X_l) \in \cm $, hence a rank-$ (k,l) $ tensor field $ A \in \tns{k}{l}(\man) $ induces a map:
\begin{equation}
  A : \underbrace{\xms \times \dots \times \xms}_{k \text{ times}} \times \underbrace{\xm \times \dots \times \xm}_{l \text{ times}} \ra \R
  \label{eq:tensor-linear}
\end{equation}
which is multilinear\footnotemark over $ \cm $.

\footnotetext{A map $ A : \xm \ra \R $ is linear over $ \cm $ if $ A(fX + gY) = f A(X) + g A(Y) \,\,\forall f,g \in \cm , X,Y \in \xm $, with obvious generalization to $ \xms $ and to the multilinear case.}

\subsubsection{Pull-back of tensor fields}

Recall that a diffeomorphism $ f : \man_1 \ra \man_2 $ induces two maps: the push-forward of vector fields through the differential $ \dd f_p : T_p\man_1 \ra T_{f(p)}\man_2 $ and the pull-back of covector fields through the dual of the differential $ \dd f_p^* : T_{f(p)}^*\man_2 \ra T_p^*\man_1 $. These are respectively defined by \ceref{eq:differential-def}{eq:pull-back-def}, and are denoted as $ (f_*X)_{f(p)} \defeq \dd f_p(X_p) $ and $ (f^*\omega)_p \defeq \dd f_p^*(\omega_{f(p)}) $, respectively.

\begin{definition}{Pull-back of tensor field}{}
  Let $ f : \man_1 \ra \man_2 $ be a diffeomorphism. Then, the \bcdef{pull-back} by $ f $ at $ p $ is defined as the map $ f^*_p : T^k_l(T_{f(p)}\man_2) \ra T^k_l(T_p\man_1) $ which acts on $ A \in T^k_l(T_{f(p)}\man_2) $ as:
  \begin{equation}
    (f^*A)_p(\omega^1 , \dots , \omega^k , X_1 , \dots , X_l) \defeq A((f^*\omega^1)_p , \dots , (f^*\omega^k)_p , (f^{-1}_* X_1)_p , \dots , (f^{-1}_* X_l)_p)
    \label{eq:tens-pb}
  \end{equation}
  with $ \omega^1 , \dots , \omega^k \in T^*\man_2 $, $ X_1 , \dots , X_l \in T\man_2 $ and $ f^{-1}_* \equiv (f^{-1})_* $.
\end{definition}

Note that the vector fields are ``pushed-back" by the inverse diffeomorphism $ f^{-1} : \man_2 \ra \man_1 $, so that all fields are mapped from $ \man_2 $ to $ \man_1 $.

The definition of a pull-back on tensor fields allows for the extension of the Lie derivative on them (which was defined in \dref{def:lie-der-vec} on vector fields through their ``push-back").

\begin{definition}{Lie derivative of tensor field}{}
  Let $ \man $ be a smooth manifold, and let $ V \in \xm , A \in \tns{k}{l}(\man) $. Denoting as $ \theta $ the flow of $ V $ on $ \man $, the \bcdef{Lie derivative} of $ A $ with respect to $ V $ is the tensor field (provided it exists):
  \begin{equation}
    (\ld_V A)_p \defeq \frac{\dd}{\dd t} \bigg\vert_{t = 0} (\theta_t^* A)_p
  \end{equation}
\end{definition}

Note that the expression being differentiated lies in $ T^k_l(T_p\man) $ for all $ t $, hence correctly $ (\ld_V A)_p \in T^k_l(T_p\man) $. Moreover, since it is defined from smooth maps, it can be shown that indeed $ \ld_V A \in \tns{k}{l}(\man) $.

\begin{lemma}{Properties of Lie derivative}{}
  Let $ V \in \xm $, $ f \in \cm $, $ A \in \tns{k}{l}(\man) $ and $ B \in \tns{p}{q}(\man) $. Then:
  \begin{enumerate}
    \item \label{item:ld-prop-1} $ \ld_V f = V(f) $;
    \item \label{item:ld-prop-2} $ \ld_V (fA) = \ld_V(f) A + f \ld_V A $;
    \item \label{item:ld-prop-3} $ \ld_V (A \otimes B) = (\ld_V A) \otimes B + A \otimes (\ld_V B) $;
  \end{enumerate}
\end{lemma}

A fundamental property of the Lie derivative is that it obeys a convenient product rule.

\begin{proposition}{Product rule}{}
  Let $ V \in \xm $, $ A \in \tns{k}{l}(\man) $ and $ \omega^1 , \dots , \omega^k \in \xms $, $ X_1 , \dots , X_l \in \xm $. Then:
  \begin{equation}
    \begin{split}
      \ld_V (A(\omega^1 , \dots , \omega^k , X_1 , \dots , X_l))
      & = (\ld_V A)(\omega^1 , \dots , \omega^k , X_1 , \dots , X_l) + \\
      & \quad + \sum_{i = 1}^k A(\omega^1 , \dots , \ld_V \omega^i , \dots , \omega^k , X_1 , \dots , X_l) + \\
      & \quad + \sum_{j = 1}^l A(\omega^1 , \dots , \omega^k , X_1 , \dots , \ld_V X_j , \dots , X_l)
    \end{split}
    \label{eq:tensor-prod-rule}
  \end{equation}
\end{proposition}

\begin{propcorollary}{Components of Lie derivative}{}
  Let $ V \in \xm $ and $ A \in \tns{k}{l}(\man) $. In a smooth chart $ (U,(x^i)) $ on $ \man $, the components of $ \ld_V A \in \tns{k}{l}(\man) $ are:
  \begin{equation}
    \tensor{(\ld_V A)}{^{i_1 \dots i_k}_{j_1 \dots j_l}} = V^\ell \pa_\ell \tensor{A}{^{i_1 \dots i_k}_{j_1 \dots j_l}} - \sum_{i = 1}^k \tensor{A}{^{i_1 \dots \ell \dots i_k}_{j_1 \dots j_l}} \pa_\ell V^{i_i} + \sum_{j = 1}^l \tensor{A}{^{i_1 \dots i_k}_{j_1 \dots \ell \dots j_l}} \pa_{j_j} V^\ell
    \label{eq:tensor-lie-der}
  \end{equation}
\end{propcorollary}

\begin{proofbox}
  \begin{proof}
    Recall \eref{eq:tensor-components}. By \eref{eq:tensor-prod-rule}, it is necessary to compute $ \ld_V \pa_i $ and $ \ld_V \dd x^i $. The first one is straighforward, using $ \ld_V \pa_i = [V , \pa_i] $ and \eref{eq:lie-bra-comp}:
    \begin{equation}
      \ld_V \pa_i = - \pa_i V^\ell \pa_\ell
    \end{equation}
    The second one needs more care.

    \begin{lemma}[borderline = {1pt}{0pt}{blue!35!black}]{Lie derivative and differential}{lie-der-diff}
      If $ V \in \xm $ and $ f \in \cm $, then:
      \begin{equation}
        \ld_V (\dd f) = \dd (\ld_V f)
        \label{eq:lie-der-vec-basis}
      \end{equation}
    \end{lemma}

    \begin{proofbox}
      \begin{proof}
        By \eref{eq:tensor-prod-rule}, for any $ X \in \xm $:
        \begin{equation*}
          \begin{split}
            (\ld_V (\dd f))(X)
            & = V(\dd f(X)) - \dd f([V,X]) = VX(f) - [V,X](f) = XV(f)
          \end{split}
        \end{equation*}
        Since $ XV(f) = \dd (V(f))(X) = \dd (\ld_V f)(X) $, the proof is complete.
      \end{proof}
    \end{proofbox}

    By \lref{lemma:lie-der-diff}, then, $ \ld_V \dd x^i = \dd (\ld_V x^i) = \dd (V^j \pa_j x^i) = \dd V^i $, so using \eref{eq:func-diff}:
    \begin{equation}
      \ld_V \dd x^i = \pa_\ell V^i \dd x^\ell
      \label{eq:lie-der-covec-basis}
    \end{equation}
    Inserting \eeref{eq:lie-der-vec-basis}{eq:lie-der-covec-basis} into \eref{eq:tensor-prod-rule} yields the thesis.
  \end{proof}
\end{proofbox}

In particular, the Lie derivative of a covector field $ \omega \in \xms $ reads (in components):
\begin{equation}
  \ld_V \omega = (V^\ell \pa_\ell \omega_i + \omega_\ell \pa_i V^\ell) \dd x^i
  \label{eq:lie-der-covec}
\end{equation}

Finally, it is possible to characterize tensor fields which remain unchanged under certain flows. Given a flow $ \theta $ on $ \man $, a tensor field $ A $ over $ \man $ is said to be \bctxt{invariant under $ \theta $} if, for any $ (t,p) $ in the domain of $ \theta $:
\begin{equation*}
  (\theta_t^* A)_p = A_p
\end{equation*}

\begin{theorem}{Invariance under flow}{}
  Let $ \man $ be a smooth manifold, and let $ V \in \xms $ and $ A \in \tns{k}{l}(\man) $. Then, $ A $ is invariant under the flow of $ V $ if and only if $ \ld_V A = 0 $.
\end{theorem}

This theorem gives a clear interpretation to the Lie derivative.
