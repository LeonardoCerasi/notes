\section{de Rham cohomology}

\begin{definition}{Closed form}{}
  Let $ \man $ be a smooth manifold. Then, $ \omega \in \df{k} $ is \bcdef{closed} if $ \defcol{\dd \omega = 0} $.
\end{definition}

\begin{definition}{Exact form}{}
  Let $ \man $ be a smooth manifold. Then, $ \omega \in \df{k} $ is \bcdef{exact} if $ \exists \eta \in \df{k-1} : \defcol{\omega = \dd \eta} $, known as the \emph{potential} of $ \omega $.
\end{definition}

Since $ \dd^2 = 0 $, every exact for is closed. The converse, however, is in general not true: every closed form is exact only on manifolds with particular topological properties. By Stokes' theorem, it is clear that the integral of an exact form over any compact submanifold without boundary vanishes.

\subsection{Conservative covector fields}

Given a smooth manifold $ \man $, a smooth covector field $ \omega \in \df{1} $ is \bctxt{conservative} if its line integral over every piecewise smooth closed curve segment vanishes: it can be shown that this condition is equivalent to the requirement that the line integrals of $ \gamma $ are path-independent once the endpoints are fixed.

\begin{theorem}{Conservative forms and exact forms}{conservative-exact}
  Let $ \man $ be a smooth manifold. Then, $ \omega \in \df{1} $ is conservative if and only if it is exact.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    If $ \omega \in \df{1} $ is exact, then it is conservative by \ref{eq:fundamental-th-line-integrals} it is conservative.

    Suppose $ \omega \in \df{1} $ is conservative, and assume $ \man $ to be connected. Since the line integrals of $ \omega $ are path-independent, denote by $ \int_p^q \omega $ the line integral of $ \omega $ along any piecewise smooth curve segment from $ p \in \man $ to $ q \in \man $. As a backward reparametrization of a path from $ p $ to $ q $ is a path to $ q $ to $ p $, then \eref{eq:line-reparam} implies $ \int_q^p \omega = - \int_p^q \omega $, while \eref{eq:line-piece} implies:
    \begin{equation*}
      \int_{p_1}^{p_2} \omega + \int_{p_2}^{p_3} \omega = \int_{p_1}^{p_3} \omega
    \end{equation*}
    Choose any $ p_0 \in \man $ and define $ f : \man \ra \R $ as $ f(q) = \int_{p_0}^q \man $: WTS $ \dd f = \omega $. Let $ q_0 \in \man $ arbitrary, let $ (U,(x^i)) $ be a smooth chart centered at $ q_0 $ and write the coordinate representation $ \omega = \omega_i \dd x^i $ on $ U $. $ \dd f_{q_0} = \omega_{q_0} $ is then equivalent to:
    \begin{equation*}
      \frac{\pa f}{\pa x^j}(q_0) = \omega_j(q_0)
    \end{equation*}
    for $ j = 1 , \dots , n $. First, suppose $ q_0 \in \Int \man $, Fix $ j $ and let $ \gamma : [-\varepsilon , \varepsilon] \ra U $ be the smooth curve segment $ \gamma(t) = (0, \dots , t , \dots , 0) $, with $ t $ in the $ j^\text{th} $ place, and with $ \varepsilon \in \R^+ $ small enough that $ \gamma([-\varepsilon , \varepsilon]) \ssq U $. Let $ p_1 = \gamma(-\varepsilon) $ and define $ \tilde{f} : \man \ra \R $ as $ \tilde{f}(q) = \int_{p_1}^q \omega $. Then, for all $ q \in \man $:
    \begin{equation*}
      f(q) - \tilde{f}(q) = \int_{p_0}^q \omega - \int_{p_1}^q \omega = \int_{p_0}^q \omega + \int_q^{p_1} = \int_{p_0}^{p_1} \omega
    \end{equation*}
    which is independent of $ q $, thus $ \tilde{f} $ and $ f $ differ only by a constant, so it suffices to show that $ \pa_j \tilde{f}(q_0) = \omega_j(q_0) $. Since $ \gamma'(t) = \pa_j \vert_{\gamma(t)} $ by construction:
    \begin{equation*}
      \omega_{\gamma(t)}(\gamma'(t)) = \omega_i(\gamma(t)) \dd x^i \left( \frac{\pa}{\pa x^j} \bigg\vert_{\gamma(t)} \right) = \omega_j(\gamma(t))
    \end{equation*}
    Since the restriction of $ \gamma $ to $ [-\varepsilon , t] $ is a smooth curve from $ p_1 $ to $ \gamma(t) $:
    \begin{equation*}
      \tilde{f} \circ \gamma(t) = \int_{p_1}^{\gamma(t)} \omega = \int_{-\varepsilon}^t \dd s \, \omega_{\gamma(s)}(\gamma'(s)) = \int_{-\varepsilon}^t \dd s \, \omega_j(\gamma(s))
    \end{equation*}
    Thus, by the fundamental theorem of calculus:
    \begin{equation*}
      \frac{\pa \tilde{f}}{\pa x^j}(q_0) = \gamma'(0)(\tilde{f}) = \frac{\dd}{\dd t} \bigg\vert_{t = 0} \tilde{f} \circ \gamma(t) = \frac{\dd}{\dd t} \bigg\vert_{t = 0} \int_{-\varepsilon}^t \dd s \, \omega_j(\gamma(s)) = \omega_j(\gamma(0)) = \omega_j(q_0)
    \end{equation*}
    Therefore $ \dd f_{q_0} = \omega_{q_0} $ for $ q_0 \in \Int \man $. Now, consider $ q_0 \in \pa \man $ with a boundary chart $ (U,(x^i)) $ centered at $ q_0 $. The proof above still works, except for $ j = n \equiv \dim_\R \man $, since $ x^n $ can only take non-negative values in this case. Let $ p_1 = \gamma(0) $ (instead of $ \gamma(-\varepsilon) $), and proceed as before: this shows that $ \dd f = \omega $ on $ \man $ connected.

    Finally, if $ \man $ is not connected, let $ \{\man_i\} $ be the connected components of $ \man $. By the argument above, on each component there is a function $ f_i \in \cfk{\infty}(\man_i) $ such that $ \dd f_i = \omega $ on $ \man_i $. Letting $ f : \man \ra \R $ be the function that is equal to $ f_i $ on $ \man_i $ for each $ i $, then $ \dd f = \omega $, completing the proof.
  \end{proof}
\end{proofbox}

While not all closed forms are exact on a generic smooth manifold, since it depends on its shape, it is possible to show that \emph{locally} this is always true. Given a finite-dimensional $ \R $-vector space $ V $, then $ U \ssq V $ is said to be \bctxt{star-shaped} if there is a point $ x \in U $ such that for every $ x \in U $ the line segment from $ c $ to $ x $ is entirely contained in $ U $. For example, every convex subset if star-shaped.

\begin{theorem}{Poincaré lemma}{poincare-1}
  If $ U \in \tp(\R^n) $ (or $ \U^n $) is star-shaped, then every closed $ \omega \in \dfm{1}(U) $ is exact.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    Suppose $ U $ is star-shaped with respect to $ c \in U $, and let $ \omega = \omega_i \dd x^i \in \dfm{1}(U) $ closed. Since diffeomorphisms take closed forms to closed forms and exact forms to exact forms, translate $ U $ so that $ c = 0 $. Then, for any $ x \in U $, let $ \gamma_x : [0,1] \ra U $ denote the line segment from $ 0 $ to $ x $ parametrized as $ \gamma_x(t) = tx $: by hypothesis $ \gamma_x([0,1]) \ssq U \,\,\forall x \in U $. Define $ f : U \ra \R $ as:
    \begin{equation*}
      f(x) = \int_{\gamma_x} \omega
    \end{equation*}
    WTS $ \dd f = \omega $, or equivalently $ \pa_j f = \omega_j $ for $ j = 1 , \dots , n $. First, compute:
    \begin{equation*}
      f(x) = \int_0^1 \dd t \, \omega_{\gamma_x(t)}(\gamma'_x(t)) = \int_0^1 \dd t \, \omega_i(tx) x^i
    \end{equation*}
    Since the integrand is smooth in all variables, it is permissible to differentiate under the integral sign:
    \begin{equation*}
      \frac{\pa f}{\pa x^j}(x) = \int_0^1 \dd t \left[ t \frac{\pa \omega_i}{\pa x^j}(tx) x^i + \omega_j(tx) \right]
    \end{equation*}
    Since $ \omega $ is closed, $ \pa_i \omega_j = \pa_j \omega_i $, hence:
    \begin{equation*}
      \frac{\pa f}{\pa x^j}(x) = \int_0^1 \dd t \left[ t \frac{\pa \omega_j}{\pa x^i}(tx) x^i + \omega_j(tx) \right] = \int_0^1 \dd t \frac{\dd}{\dd t} \left[ t \omega_j(tx) \right] = \left[ t \omega_j(tx) \right]_{t = 0}^{t = 1} = \omega_j(x)
    \end{equation*}
    which completes the proof.
  \end{proof}
\end{proofbox}

\begin{corollary}{Local exactness of closed forms}{}
  Let $ \man $ be a smooth manifold and $ \omega \in \df{1} $ closed. Then, every point of $ \man $ has a neighbourhood on which $ \omega $ is exact.
\end{corollary}

\begin{proofbox}
  \begin{proof}
    Let $ p \in \man $ be arbitrary. By hypothesis $ \pa_i \omega_j = \pa_j \omega_i $ in some smooth coordinate ball or half-ball $ (U,\varphi) $ containing $ p $. Balls and half-balls are convex and hence star-shaped, therefore applying \tref{th:poincare} to the coordinate representation of $ \omega $ there exists $ f \in \cfk{\infty}(U) : \omega \vert_U = \dd f $.
  \end{proof}
\end{proofbox}

The key feature in constructing the potential function in \tref{th:poincare} is that every point of $ x \in U $ can be reached by a definite path from $ c \in U $ to $ x $, chosen to vary smoothly as $ x $ varies. This fails, for example, on a punctured plane, i.e. a plane with a hole.

\begin{example}{Swirl form on the plane}{}
  Consider $ \man = \R^2 - \{0\} $ and the closed $ \omega \in \df{1} $ defined as:
  \begin{equation*}
    \omega = \frac{x \, \dd y - y \, \dd x}{x^2 + y^2}
  \end{equation*}
  Choosing the smooth closed curve segment $ \gamma : [0,2\pi] \ra \man $ defined as $ \gamma(\vartheta) = (\cos \vartheta , \sin \vartheta) $:
  \begin{equation*}
    \int_\gamma \omega = \int_{[0,2\pi]} \frac{\cos \vartheta (\cos \vartheta \, \dd \vartheta) - \sin \vartheta (\sin \vartheta \, \dd \vartheta)}{\sin^2 \vartheta + \cos^2 \vartheta} = \int_0^{2\pi} \dd \vartheta = 2\pi
  \end{equation*}
  which shows, by \tref{th:conservative-exact} that $ \omega $ is not exact on $ \R^2 - \{0\} $. However, restricting the domain to $ U = \{(x,y) \in \R^2 : x > 0\} $, by direct computation $ \omega = \dd f $ with potential function $ f : U \ra \R $ given by:
  \begin{equation*}
    f(x) = \arctan \frac{y}{x}
  \end{equation*}
  or, more clearly, in polar coordinates $ \omega = \dd \vartheta $. The problem is then that there is no smooth (or even continuous) angle function on all of $ \R^2 - \{0\} $, which is a consequence of the hole in the center.
\end{example}

\subsection{de Rham cohomology groups}
\label{ssec:de-rham-cohomology-groups}

The aim of de Rham cohomology is to study whether closed forms are exact on a smooth manifold by means of its global topological properties.

Let $ \man $ be a smooth $ n $-manifold. Since $ \dd : \df{k} \ra \df{k+1} $ is defined for all $ k = 0 , \dots , n-1 $, given $ p \in \N_0 $ denote $ \dd_p : \df{p} \ra \df{p+1} $, with the convention $ \df{p} = \{0\} \equiv 0 $ for $ p < 0 $ and $ p > n $. Then, define:
\begin{equation}
  \zp{p}(\man) \defeq \ker \dd_p
\end{equation}
\begin{equation}
  \bp{p}(\man) \defeq \im \dd_{p-1}
\end{equation}
which are respectively the $ \R $-vector spaces of closed $ p $-forms and exact $ p $-forms on $ \man $. By the above conventions, $ \bp{0}(\man) = 0 $ and $ \zp{n}(\man) = \df{n} $.

\begin{definition}{de Rham cohomology groups}{}
  Let $ \man $ be a smooth manifold. The \bcdef{de Rham cohomology group of degree $ p $}, with $ p \in \N_0 $, is defined as:
  \begin{equation}
    \defcol{\hp{p}(\man)} \defeq \frac{\zp{p}(\man)}{\bp{p}(\man)}
  \end{equation}
\end{definition}

This quotient vector space is well-defined for all $ p \in \N_0 $, since every exact form is closed, i.e. $ \bp{p}(\man) \ssq \zp{p}(\man) $, and it is a group under vector addition. Clearly, $ \hp{p}(\man) = 0 $ for $ p < 0 $ and $ p > n $, while for $ 0 \leq p \leq n $ the condition $ \hp{p}(\man) = 0 $ holds if and only if every closed $ p $-form on $ \man $ is exact.

For any $ \omega \in \df{p} $, let $ [\omega] \in \hp{p}(\man) $ denote the equivalence class of $ \omega $, known its \emph{cohomology class}: if $ [\omega] = [\omega'] $, i.e. if $ \omega $ and $ \omega' $ differ by an exact form, then $ \omega $ and $ \omega' $ are said to be \emph{cohomologous}.

\begin{proposition}{Induced cohomology map}{induced-cohomology}
  For any smooth map $ f : \man_1 \ra \man_2 $, the pull-back $ f^* : \dfm{p}(\man_2) \ra \dfm{p}(\man_1) $ maps $ \zp{p}(\man_2) \mapsto \zp{p}(\man_1) $ and $ \bp{p}(\man_2) \mapsto \bp{p}(\man_1) $. It therefore descends to a linear map $ \propcol{f^* : \hp{p}(\man_2) \ra \hp{p}(\man_1)} $, called the \bcprop{induced cohomology map}, such that $ \id_\man^* = \id_{\hp{p}(\man)} $ and, if $ g : \man_2 \ra \man_3 $ is another smooth map, then:
  \begin{equation*}
    (g \circ f)^* = f^* \circ g^* : \hp{p}(\man_3) \ra \hp{p}(\man_1)
  \end{equation*}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    If $ \omega $ is closed, then $ \dd (f^*\omega) = f^*(\dd \omega) = 0 $, so $ f^*\omega $ is also closed. If $ \omega = \dd \eta $ is exact, then $ f^*\omega = f^*(\dd \eta) = \dd (f^*\eta) $, which is also exact. Therefore, $ f^* $ maps $ \zp{p}(\man_2) \mapsto \zp{p}(\man_1) $ and $ \bp{p}(\man_2) \mapsto \bp{p}(\man_1) $.

    The induced cohomology map $ f^* : \hp{p}(\man_2) \ra \hp{p}(\man_1) $ is defined as follows: for $ \omega \in \df{p} $ closed, let $ f^*[\omega] = [f^*\omega] $, so if $ \omega' = \omega + \dd \eta $, then $ [f^*\omega'] = [f^*\omega + \dd (f^*\eta)] = [f^*\omega] = f^*[\omega] $, hence this map is well-defined. Its properties follow immediately from the analogous properties of pull-back maps on forms.
  \end{proof}
\end{proofbox}

\begin{propcorollary}{Diffeomorphism-invariance of de Rham cohomology}{dr-diff-inv}
  Diffeomorphic smooth manifolds have isomorphic de Rham cohomology groups.
\end{propcorollary}

Although in general the direct computation of de Rham groups is not easy, this process can be simplified with the following result.

\begin{proposition}{Cohomology of disjoint unions}{}
  Let $ \{\man_j\} $ be a countable collection of smooth $ n $-manifolds, and let $ \man = \bigsqcup_j \man_j $. The inclusion maps $ \inc_j : \man_j \hookrightarrow \man $ induce an isomorphism:
  \begin{equation*}
    \hp{p}(\man) \cong \prod_j \hp{p}(\man_j)
  \end{equation*}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    The pullback maps $ \inc_j^* : \df{p} \ra \dfm{p}(\man_j) $ alread induce an isomorphism $ \df{p} \cong \prod_j \dfm{p}(\man_j) $ as:
    \begin{equation*}
      \omega \mapsto (\inc_1^* \omega , \inc_2^* \omega , \dots) = (\omega \vert_{\man_1} , \omega \vert_{\man_2} , \dots)
    \end{equation*}
    since it is injective, as any $ \omega \in \df{p} $ whose restriction to each $ \man_j $ is zero must itself be zero, and surjective, since defining an arbitrary $ \omega_j \in \dfm{p}(\man_j) $ on each $ \man_j $ defines a global $ \omega \in \df{p} $.
  \end{proof}
\end{proofbox}

Therefore, each de Rham group of a disconnected manifold is just the direct product of the corresponding groups of its components. The main objective is thus to compute the de Rham groups of connected manifolds.

\begin{example}{Cohomology of degree zero}{cohomology-zero}
  Let $ \man $ be a connected smooth manifolds. First, note that $ \bp{0}(\man) = 0 $, since there are no $ (-1) $-forms. Then, observe that a closed $ 0 $-form is $ f \in \cm : \dd f = 0 $, and since $ \man $ is connected, this is true if and only if $ f $ is constant: therefore, $ \hp{0}(\man) = \zp{0}(\man) $ is just the space of constant functions, which is $ 1 $-dimensiona.

  If $ \man $ is a $ 0 $-manifold, then $ \hp{0}(\man) $ is a direct product of $ 1 $-dimensional vector spaces, one for each point of $ \man $, and the cohomology groups with non-zero degree vanish for dimensional reasons.
\end{example}

\subsection{Homotopy invariance}

\cref{cor:dr-diff-inv} has a profound generalization: de Rham cohomology groups are topological invariants (actually, they are homotopy invariants, see \secref{sec:homotopy} for homotopy equivalence).

\begin{definition}{Cochain homotopy}{}
  Let $ f , g : \man_1 \ra \man_2 $ be smooth maps. A collection of linear maps $ \defcol{h_p : \dfm{p}(\man_2) \ra \dfm{p-1}(\man_1)} $ is a \bcdef{cochain homotopy} between $ f^* $ and $ g^* $ if, for all $ \omega \in \dfm{p}(\man_2) $:
  \begin{equation}
    (\dd_{p-1} \circ h_p \omega + h_{p+1} \circ \dd_p) \omega = g^* \omega - f^* \omega
    \label{eq:cochain-homotopy-def}
  \end{equation}
\end{definition}

\begin{proposition}{Cochain homotopy and induced cohomology maps}{}
  Let $ f , g : \man_1 \ra \man_2 $ be smooth maps. If there exists a cochain homotopy between $ f^* $ and $ g^* $, then the induced cohomology maps $ f^* , g^* : \hp{p}(\man_2) \ra \hp{p}(\man_1) $ are equal.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Consider $ \omega \in \df{p}(\man_2) $ closed. Then, by \eref{eq:cochain-homotopy-def}:
    \begin{equation*}
      \dd_{p-1} (h_p \omega) = g^* \omega - f^* \omega
      \quad \implies \quad
      g^* [\omega] - f^* [\omega] = [\dd \eta] = 0
    \end{equation*}
    where $ \eta \equiv h_p \omega \in \dfm{p-1}(\man_1) $.
  \end{proof}
\end{proofbox}

Consider a smooth manifold $ \man $ and, for each $ t \in [0,1] $, let $ \inc_t : \man \ra \man \times [0,1] $ be the map $ \inc_t(x) = (x,t) $. If $ \pa \man = \emptyset $, then $ \man \times I $ is a smooth manifold with boundary, while if $ \pa \man \neq \emptyset $, then $ \man \times I $ is a smooth manifold with corners: it is straightforward to check that the definitions of de Rham groups and induced cohomology maps remain unchanged in the context of smooth manifolds with corners.

\begin{lemma}{Existence of cochain homotopy}{cochain-homotopy-existence}
  For any smooth manifold $ \man $, there exists a cochain homotopy between the two maps $ \inc_0^* , \inc_1^* : \dfv(\man \times [0,1]) \ra \dft $.
\end{lemma}

\begin{proofbox}
  \begin{proof}
    For each $ p \in \N_0 $, WTS there exists a linear map $ h_p : \dfm{p}(\man \times I) \ra \dfm{p-1}(\man) $ such that, for all $ \omega \in \dfm{p}(\man \times I) $:
    \begin{equation*}
      h_{p+1}(\dd_p \omega) + \dd_{p-1} (h_p \omega) = \inc_1^* \omega - \inc_0^* \omega
    \end{equation*}
    Let $ s $ denote the standard coordinate on $ \R $, and let $ S $ be the vector field on $ \man \times \R $ defined as $ S_{q,s} = (0 , \pa/\pa s \vert_s) $ under the usual identification $ T_{(q,s)}\man \cong T_q\man \times T_s\R $. Given $ \omega \in \dfm{p}(\man \times [0,1]) $, define $ h_p \omega \in \df{p-1} $ as:
    \begin{equation*}
      h_p \omega = \int_0^1 \dd t \, \inc_t^* (\ip_S \omega)
    \end{equation*}
    which, for any $ q \in \man $, is expressed as:
    \begin{equation*}
      (h_p \omega)_p = \int_0^1 \dd t \, \inc_t^* ((\ip_S \omega)_{(q,t)})
    \end{equation*}
    where the integrand is a map $ t \mapsto \lv{p-1}(T_q\man) $. On any smooth coordinate domain $ U \ssq \man $, the components of the integrand are smooth function of $ (q,t) \in U \times I $, so correctly $ h_p \omega \in \df{p-1} $. Then, $ \dd_{p-1} (h_p \omega) $ can be computed at any point by differentiating under the integral sign in local coordinates:
    \begin{equation*}
      \dd_{p-1} (h_p \omega) = \int_0^1 \dd t \, \dd (\inc_t^* (\ip_S \omega))
    \end{equation*}
    Therefore, using \tref{th:cartan}:
    \begin{equation*}
      \begin{split}
        h_{p+1} (\dd_p \omega) + \dd_{p-1} (h_p \omega)
        & = \int_0^1 \dd t \left[ \inc_t^* (\ip_S \dd \omega) + \dd (\inc_t^* (\ip_S \omega)) \right] \\
        & = \int_0^1 \dd t \, \inc_t^* \left( \ip_S (\dd \omega) + \dd (\ip_S \omega) \right) = \int_0^1 \dd t \, \inc_t^* \ld_S \omega
      \end{split}
    \end{equation*}
    To simplify this expression, use the flow of $ S $ over $ \man \times \R $, which is given explicitly by $ \theta_t(q,s) = (q , t+s) $. Then, using \eref{eq:tens-flow-aux}:
    \begin{equation*}
      \inc_t^* (\ld_S \omega) = \inc_0^* (\theta_t^* (\ld_S \omega)) = \inc_0^* \left( \frac{\dd}{\dd t} (\theta_t^* \omega) \right) = \frac{\dd}{\dd t} \inc_0^* (\theta_t^* \omega) = \frac{\dd}{\dd t} \inc_t^* \omega
    \end{equation*}
    Inserting this into the integral and applying the fundamental theorem of calculus yields the thesis.
  \end{proof}
\end{proofbox}

This lemma allows to prove the homotopy invariance of de Rham cohomology.

\begin{theorem}{Homotopy invariance of de Rham cohomology}{dr-homo-inv}
  Let $ \man_1 $ and $ \man_2 $ be homotopy equivalent smooth manifolds. Then, any smooth homotopy equivalence $ f : \man_1 \ra \man_2 $ induces an isomorphism $ \hp{p}(\man_1) \cong \hp{p}(\man_2) $ for each $ p \in \N_0 $.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    Let $ f : \man_1 \ra \man_2 $ be a homotopy equivalence, with homotopy inverse $ g : \man_2 \ra \man_1 $: by the Whiteny approximation theorem \tref{th:whitney-approx}, there are smooth maps $ \tilde{f} : \man_1 \ra \man_2 $ homotopic to $ f $ and $ \tilde{g} : \man_2 \ra \man_1 $ homotopic to $ g $. Since homotopy is preserved by composition, it follows that $ \tilde{f} \circ \tilde{g} \homo f \circ g \homo \id_{\man_2} $ and $ \tilde{g} \circ \tilde{f} \homo g \circ f \homo \id_{\man_1} $, hence $ \tilde{f} $ and $ \tilde{g} $ are homotopy inverses of each other.

    \begin{lemma}{}{homo-inv-aux}
      If $ \man_1 , \man_2 $ are smooth manifolds and $ f,g : \man_1 \ra \man_2 $ are homotopic smooth maps, then the induced cohomology maps $ f^* , g^* : \hp{p}(\man_2) \ra \hp{p}(\man_1) $ are equal for every $ p \in \N_0 $.
    \end{lemma}

    \begin{proofbox}
      \begin{proof}
        By \lref{lemma:cochain-homotopy-existence} the two chomology maps $ \inc_0^* , \inc_1^* : \hp{p}(\man_1 \times [0,1]) \ra \hp{p}(\man_1) $ are equal. Moreover, it can be shown that if $ f , g : \man_1 \ra \man_2 $ are homotopic smooth maps, then they are smoothly homotopic, i.e. there is a smooth homotopy $ H : \man_1 \times [0,1] \ra \man_2 $ from $ f $ to $ g $. Since $ f = H \circ \inc_0 $ and $ g = H \circ \inc_1 $ (see \figref{fig:homo-maps}), by \pref{prop:induced-cohomology}:
        \begin{equation*}
          f^* = (H \circ \inc_0)^* = \inc_0^* \circ H^* = \inc_1^* \circ H^* = (H \circ \inc_1)^* = g^*
        \end{equation*}
        which is the thesis.
      \end{proof}
    \end{proofbox}

    Using \lref{lemma:homo-inv-aux}, then, on the cohomology:
    \begin{equation*}
      \tilde{f}^* \circ \tilde{g}^* = (\tilde{g} \circ \tilde{g})^* = \id_{\man_1}^* = \id_{\hp{p}(\man_1)}
    \end{equation*}
    and equivalently for $ \tilde{g}^* \circ \tilde{f}^* = \id_{\hp{p}(\man_2)} $. Therefore, $ \tilde{f}^* : \hp{p}(\man_2) \ra \hp{p}(\man_1) $ is an isomorphism.
  \end{proof}
\end{proofbox}

\begin{corollary}{Topological invariance of de Rham cohomology}{dr-top-inv}
  Let $ \man_1 $ and $ \man_2 $ are homeomorphic smooth manifolds, then their de Rham cohomology groups are isomorphic.
\end{corollary}

\begin{proofbox}
  \begin{proof}
    Every homeomorphism is a homotopy equivalence, thus use \tref{th:dr-homo-inv}.
  \end{proof}
\end{proofbox}

Homotopy invariance can be used to compute some de Rham groups.

\begin{proposition}{Cohomology of contractible manifolds}{cohomology-contractible}
  Let $ \man $ be a contractible\footnotemark smooth manifold. Then, $ \hp{p}(\man) = 0 $ for $ p \geq 1 $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    By the hypothesis, there is some $ q \in \man $ such that $ \id_\man \homo c_q : \man \ra \man $ which sends all of $ \man $ to $ q $. Denoting the inclusion map as $ \inc_q : \{q\} \hookrightarrow \man $, it follows that $ c_q \circ \inc_q = \id_{\{q\}} $ and $ \inc_q \circ c_q \homo \id_\man $, so $ \inc_q $ is a homotopy equivalence between $ \{q\} $ and $ \man $. Then, by homotopy invariance, the de Rham groups of $ \man $ are isomorphic to those of a $ 0 $-manifold, which vanish for $ p \geq 1 $ as per \exref{ex:cohomology-zero}.
  \end{proof}
\end{proofbox}

\footnotetext{A topological space $ X $ is \emph{contractible} if $ \id_X $ is homotopic to a constant map.}

The Poincaré lemma \tref{th:poincare-1} can be generalized can be generalized to forms of all degrees.

\begin{theorem}{Poincaré lemma}{poincare}
  If $ U \in \tp(\R^n) $ (or $ \U^n $) is star-shaped, then $ \hp{p}(U) = 0 $ for $ p \geq 1 $.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    If $ U $ is star-shaped with respect to $ c \in \R^n $, then it is contractible by the followinf straight-line homotopy:
    \begin{equation*}
      H(x,t) = c + t (x - c)
    \end{equation*}
    The thesis follows from \pref{prop:cohomology-contractible}.
  \end{proof}
\end{proofbox}

\begin{corollary}{Local exactness of closed forms}{}
  Each point of a smooth manifold has a neighbourhood on which every closed form is exact.
\end{corollary}

\begin{proofbox}
  \begin{proof}
    Every point of a smooth manifold has a neighbourhood diffeomorphic ot an open ball in $ \R^n $ or half-ball in $ \U^n $, each of which are star-shaped. The result follows from \tref{th:poincare}.
  \end{proof}
\end{proofbox}

Since $ \R^n $ and $ \U^n $ are both star-shaped for $ n \in \N_0 $, $ \hp{p}(\R^n) = 0 $ and $ \hp{p}(\U^n) = 0 $ for $ p \geq 1 $.

Another case in which de Rham cohomology can be studied quite extensively is in degree $ 1 $. Let $ \man $ be a connected smooth manifold and $ q \in \man $. Let $ \Hom(\pi_1(\man,q) , \R) $ denote the set of group homomorphisms from $ \pi_1(\man,q) $ to the additive group $ \R(+) $: this is a $ \R $-vector space under pointwise addition of homomorphisms and multiplication by constants. Define a linear map $ \Phi : \hp{1}(\man) \ra \Hom{\pi_1(\man,q),\R} $ as follows: given a cohomology class $ [\omega] \in \hp{1}(\man) $, define $ \Phi[\omega] : \pi_1(\man,q) \ra \R $ as:
\begin{equation*}
  \Phi[\omega][\gamma] \equiv \int_{\tilde{\gamma}} \omega
\end{equation*}
where $ \gamma $ is any piecewise smooth curve representing the path homotopy class $ [\gamma] \in \pi_1(\man,q) $.

\begin{theorem}{First cohomology and fundamental group}{cohomology-fundamental-group}
  Let $ \man $ be a connected smooth manifold. For each $ q \in \man $, the linear map $ \Phi : \hp{1}(\man) \to \Hom(\pi_1(\man,q),\R) $ is an isomorphism.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    The proof that $ \Phi $ is surjective requires more advanced instruments, so WTS $ \Phi $ is well-defined and injective. Given $ [\gamma] \in \pi_1(\man,q) $, by the Whitney approximation theorem \tref{th:whitney-approx} there is a smooth closed curve segment $ \tilde{\gamma} $ in the same path class of $ \gamma $, and by \tref{th:integral-path-homotopy}, $ \int_{\tilde{\gamma}} \omega $ gives the same result for every piecewise smooth curve $ \tilde{\gamma} $ in the given class. Moreover, if $ \tilde{\omega} \in \df{1} $ is in the same cohomology class of $ \omega $, then $ \tilde{\omega} - \omega = \dd f $ for some $ f \in \cm $, so:
    \begin{equation*}
      \int_{\tilde{\gamma}} \tilde{\omega} - \int_{\tilde{\gamma}} \omega = \int_{\tilde{\gamma}} \dd f = f(q) - f(q) = 0
    \end{equation*}
    Thus $ \Phi $ is well-defined. By \eref{eq:line-piece}, $ \Phi[\omega] $ is a group homomorphism from $ \pi_1(\man,q) $ to $ \R $, and from linearity of the line integral $ \Phi $ is itself a linear map.

    To see that $ \Phi $ is injective, suppose $ \Phi[\omega] \equiv 0 $, which means that $ \int_{\tilde{\gamma}} \omega = 0 $ for every piecewise smooth closed curve starting at $ q $. If $ \sigma $ is a piecewise smooth closed curve starting at some other point $ q' \in \man $, then a piecewise smooth curve $ \alpha $ from $ q $ to $ q' $ can be chosen so that the path product $ \alpha \cdot \sigma \cdot \bar{\alpha} $ is a closed curve based at $ q $, where $ \bar{\alpha} $ is a backward reparametrization of $ \alpha $. Then:
    \begin{equation*}
      0 = \int_{\alpha \cdot \sigma \cdot \bar{\alpha}} \omega = \int_\alpha \omega + \int_\sigma \omega - \int_\alpha \omega = \int_\sigma \omega
    \end{equation*}
    Therefore, $ \omega $ is conservative and thus exact.
  \end{proof}
\end{proofbox}

By \tref{th:integral-path-homotopy}, on simply connected smooth manifolds every closed $ 1 $-form is exact, therefore $ \hp{1}(\man) = 0 $. This result can be generalized.

\begin{corollary}{}{connected-cohomology}
  If $ \man $ is a connected smooth manifold with finite fundamental group, then $ \hp{1}(\man) = 0 $.
\end{corollary}

\begin{proofbox}
  \begin{proof}
    There are no non-trivial homomorphisms from a finite group to $ \R $.
  \end{proof}
\end{proofbox}

The same result holds if the fundamental group of a connected smooth manifold is a torsion group\footnotemark.

\footnotetext{A group $ \Gamma $ is a \emph{torsion group} if $ \forall g \in \Gamma \, \exists k \in \N : g^k = e $.}

\subsection{Mayer--Vietoris theorem}

\subsubsection{Concepts of algebraic topology}
\label{sssec:alg-top}

Let $ \ring $ be a commutative ring\footnotemark, and consider the following sequence of $ \ring $-modules and $ \ring $-linear maps:
\begin{equation}
  \dots \longrightarrow A^{p-1} \xrightarrow{\ \dd_{p-1} \ } A^p \xrightarrow{\ \dd_p \ } A^{p+1} \longrightarrow \dots
  \label{eq:sequence}
\end{equation}

\footnotetext{A \emph{ring} is a set $ \ring $ equipped with two binary operations $ + , \cdot : \ring \times \ring \ra \ring $ (addition and multiplication) which satisfy the following ring axioms:
\begin{enumerate}
  \item $ (\ring,+) $ is an abelian group;
  \item $ (\ring,\cdot) $ is a monoid (i.e. multiplication is associative and has an identity);
  \item multiplication is distributive (from left and right) with respect to addition.
\end{enumerate}
The identities of addition and multiplication are usually denoted by $ \zz $ and $ \uz $, respectively.
If multiplication is commutative, $ \ring $ is a commutative ring.

A \emph{left-module} over a ring is an abelian group $ (M,+) $ with an operation $ \odot : \ring \times M \ra M $ (scalar multiplication) such that, for all $ r,s \in \ring $ and $ x,y \in M $:
\begin{enumerate}
  \item $ r \odot (x + y) = r \odot x + r \odot y $ and $ (r + s) \odot x = r \odot x + s \odot x $;
  \item $ (r \cdot s) \odot x = r \odot (s \odot x) $ and $ \uz \odot x = x $.
\end{enumerate}
A \emph{righ-module} is defined analogously with $ \odot : M \times \ring \ra M $. If $ \ring $ is a commutative ring, the two definitions coincide and $ (M,+) $ is just a \emph{module} over $ \ring $. In practice, $ \cdot $ is omitted for multiplication and used in place of $ \odot $.}

denoted by $ A^* $. Such a sequence is called a \bctxt{(cochain) complex}\footnotemark if the composition of any successive applications is the zero map, i.e. if for each $ p \in \N_0 $:
\begin{equation}
  \dd_p \circ \dd_{p+1} \equiv 0 : A^p \ra A^{p+2}
\end{equation}

\footnotetext{A \emph{chain complex} is defined analogously, but with maps $ \pa_p : A^p \ra A^{p-1} $ going in the opposite direction. In this case, the term \emph{homology} is used in place of cohomology. The conclusions of this section hold for chain complexes too: in particular, chain maps induce linear maps on homologies.}

If $ \im \dd_p = \ker \dd_{p+1} $, the sequence is said to be an \bctxt{exact sequence}. Clearly, every exact sequence is a complex, while the converse is not true in general.

If $ A^* $ is a cochain complex, it is possible to define the \emph{$ p^\textit{th} $ cohomology group} of $ A^* $ as the quotient module which gives a quantitative measure of the failure of exactness at $ A^p $:
\begin{equation}
  \hp{p}(A^*) \defeq \frac{\ker \dd_p}{\im \dd_{p-1}}
\end{equation}
If $ A^* $ and $ B^* $ are cochain complexes, a \emph{cochain map} from $ A^* $ to $ B^* $, denoted by $ f : A^* \ra B^* $, is a collection of linear maps $ f_p : A^p \ra B^p $ such that the following diagram commutes for each $ p \in \N_0 $:
\begin{equation*}
  \begin{tikzcd}
    \dots \ar[r] & A^p \ar[r , "\dd_p"] \ar[d , "f_p" , swap] & A^{p+1} \ar[d , "f_{p+1}"] \ar[r] & \dots \\
    \dots \ar[r] & B^p \ar[r , "\dd_p" , swap] & B^{p+1} \ar[r] & \dots \\
  \end{tikzcd}
\end{equation*}
Suppressing indices, this condition can be written as $ f \circ \dd = \dd \circ f $, which means that any cochain map $ f : A^* \ra B^* $ induces a linear cohomology map $ f_p : \hp{p}(A^*) \ra \hp{p}(B^*) $ for each $ p \in \N_0 $. A \bctxt{short exact sequence} of complexes consists of three complexes $ A^* , B^* , C^* $ together with cochain maps:
\begin{equation}
  0 \longrightarrow A^* \xrightarrow{\ f \ } B^* \xrightarrow{\ g \ } C^* \longrightarrow 0
  \label{eq:short-exact-sequence}
\end{equation}
such that each sequence:
\begin{equation*}
  0 \longrightarrow A^p \xrightarrow{\ f_p \ } B^p \xrightarrow{\ g_p \ } C^p \longrightarrow 0
\end{equation*}
is exact. This means that $ f_p $ is injective, $ g_p $ is surjective and $ \im f_p = \ker g_p $ for each $ p \in \N_0 $.

\begin{lemma}{Zigzag lemma}{zigzag}
  Given a short exact sequence of complexes as in \eref{eq:short-exact-sequence}, for each $ p \in \N_0 $ there is a linear map $ \lemmacol{\delta_p : \hp{p}(C^*) \ra \hp{p+1}(A^*)} $, called the \bclemma{connecting homomorphism}, such that the following sequence is exact:
  \begin{equation*}
    \dots \xrightarrow{\ \delta_{p-1} \ } \hp{p}(A^*) \xrightarrow{\ f_p^* \ } \hp{p}(B^*) \xrightarrow{\ g_p^* \ } \hp{p}(C^*) \xrightarrow{\ \delta_p \ } \hp{p+1} \xrightarrow{\ f_{p+1}^* \ } \dots
  \end{equation*}
\end{lemma}

\subsubsection{Proof of the Mayer--Vietoris theorem}

The discussion of cochain complexes in \secref{sssec:alg-top} applies equally well e.g. to $ \ring = \Z $, in which case the sequence is composed of abelian groups and homomorphisms, and to $ \ring = \R $, in which case it is composed of vector spaces and linear maps. The case of interest here is $ \ring = \R $.

Let $ \man $ be a smooth manifold, and let $ U,V \in \tp(\man) : \man = U \cup V $. There are then four inclusion maps and four induced pull-back maps:
\begin{equation*}
  \begin{tikzcd}
    & U \ar[rd , "k"] & \\
    U \cup V \ar[ru , "i"] \ar[rd , "j" , swap] & & \man \\
                                                & V \ar[ru , "l" , swap] &
  \end{tikzcd}
  \qquad \qquad
  \begin{tikzcd}
    & \dfm{p}(U) \ar[ld , "i^*" , swap] & \\
    \dfm{p}(U \cup V)  & & \df{p} \ar[lu , "k^*" , swap] \ar[ld , "l^*"] \\
                       & \dfm{p}(V) \ar[lu , "j^*"] &
  \end{tikzcd}
\end{equation*}
There are also corresponding cohomology maps. Note that these pull-back maps are just restrictions, e.g. $ k^*\omega = \omega \vert_U $. Consider the following sequence of maps:
\begin{equation}
  0 \longrightarrow \df{p} \xrightarrow{k^* \oplus \, l^*} \dfm{p}(U) \oplus \dfm{p}(V) \xrightarrow{i^* - j^*} \dfm{p}(U \cap V) \longrightarrow 0
  \label{eq:mv-aux}
\end{equation}
with:
\begin{equation}
  (k^* \oplus l^*) \omega \equiv (k^* \omega , l^* \omega)
  \qquad \qquad
  (i^* - j^*) (\omega , \eta) \equiv i^* \omega - j^* \eta
\end{equation}
Since pull-backs commute with exterior derivatives, these maps descend to linear maps on the corresponding de Rham cohomology groups.

\begin{theorem}{Mayer--Vietoris theorem}{mayer-vietoris}
  Let $ \man $ be a smooth manifold and $ U,V \in \tp(\man) : U \cup V $. For each $ p \in \N_0 $, there is a linear map $ \delta_p : \hp{p}(U \cap V) \ra \hp{p+1}(\man) $ such that the following sequence, called the \bcth{Mayer--Vietoris sequence} for the open cover $ \{U,V\} $, is exact:
  \begin{equation*}
    \dots \xrightarrow{\ \delta_p \ } \hp{p}(\man) \xrightarrow{k^* \oplus \, l^*} \hp{p}(U) \oplus \hp{p}(V) \xrightarrow{i^* - j^*} \hp{p}(U \cap V) \xrightarrow{\ \delta_{p+1} \ } \hp{p+1}(\man) \xrightarrow{k^* \oplus \, l^*} \dots
  \end{equation*}
\end{theorem}

\begin{proofbox}
  \begin{proof}
    It suffices to show that the sequence \eref{eq:mv-aux} is exact for each $ p $, as then, because then, since pull-back maps commute with exterior derivatives, it defines a short exact sequence of cochain maps, and the Mayer--Vietoris sequence is exact by \lref{lemma:zigzag}.

    First, prove exactness at $ \df{p} $: WTS $ k^* \oplus l^* $ is injective. Let $ \sigma \in \df{p} : (k^* \oplus l^*) \sigma = (\sigma \vert_U , \sigma \vert_V) = (0,0) $: since $ \{U,V\} $ is an open cover of $ \man $, this implies $ \sigma = 0 $, hence $ k^* \oplus l^* $ is injective.

    Now, prove exactness at $ \dfm{p}(U) \oplus \dfm{p}(V) $. Observe that:
    \begin{equation*}
      (i^* - j^*) \circ (k^* \oplus l^*) (\sigma) = (i^* - j^*) (\sigma \vert_U , \sigma \vert_V)  = \sigma \vert_{U \cap V} - \sigma \vert_{U \cap V} = 0
    \end{equation*}
    which shows $ \im(k^* \oplus l^*) \ssq \ker (i^* - j^*) $. Conversely, let $ (\eta , \eta') \in \dfm{p}(U) \oplus \dfm{p}(V) : (i^* - j^*)(\eta , \eta') = 0 $, which means that $ \eta \vert_{U \cap V} = \eta' \vert_{U \cap V} $, so there is a global $ \sigma \in \df{p} $ defined by:
    \begin{equation*}
      \sigma =
      \begin{cases}
        \eta & \text{on $ U $} \\
        \eta' & \text{on $ V $}
      \end{cases}
    \end{equation*}
    Clearly $ (\eta , \eta') = (k^* \oplus l^*) \sigma $, hence $ \ker(i^* - j^*) \ssq \im(k^* \oplus l^*) $, i.e. $ \ker(i^* - j^*) = \im(k^* \oplus l^*) $.

    Finally, prove exactness at $ \dfm{p}(U \cap V) $. Let $ \omega \in \dfm{p}(U \cap V) $ arbitrary: WTS there exist $ \eta \in \dfm{p}(U) $ and $ \eta' \in \dfm{p}(V) $ such that (see \figref{fig:mv-th}):
    \begin{equation*}
      \omega = (i^* - j^*) (\eta , \eta') = i^* \eta - j^* \eta' = \eta \vert_{U \cap V} - \eta' \vert_{U \cap V}
    \end{equation*}
    Let $ \{\varphi , \psi\} $ be a smooth partition of unity subordinate to the open cover $ \{U,V\} $, and define $ \eta \in \dfm{p}(U) $ and $ \eta' \in \dfm{p}(V) $ as:
    \begin{equation*}
      \eta =
      \begin{cases}
        \psi \omega & \text{on $ U \cap V $} \\
        0 & \text{on $ U - \supp \psi $}
      \end{cases}
      \qquad \qquad
      \eta' =
      \begin{cases}
        -\varphi \omega & \text{on $ U \cap V $} \\
        0 & \text{on $ V - \supp \varphi $}
      \end{cases}
    \end{equation*}
    In each form, on the sets where the two definitions overlap, they both vanish, so $ \eta \in \dfm{p}(U) $ and $ \eta' \in \dfm{p}(V) $ are well-defined. Then:
    \begin{equation*}
      \eta \vert_{U \cap V} - \eta' \vert_{U \cap V} = \psi \omega - (- \varphi \omega) = (\psi + \varphi) \omega = \omega
    \end{equation*}
    which concludes the proof.
  \end{proof}
\end{proofbox}

  \includegraphics[width = 0.70 \textwidth]{mv-th.png}
  \caption{Surjectivity of $ i^* - j^* $.}
  \label{fig:mv-th}
\end{figure}

\begin{corollary}{Characterization of connecting homomorphism}{}
  The \bcthcol{connecting homomorphism} $ \thcorcol{\delta_p : \hp{p}(U \cap V) \ra \hp{p+1}(\man)} $ of the Mayer--Veitoris sequence is defined as follows. For each $ \omega \in \zp{p}(U \cap V) $, there are $ \eta \in \dfm{p}(U) , \eta' \in \dfm{p}(V) $ such that $ \omega = \eta \vert_{U \cap V} - \eta' \vert_{U \cap V} $: then, $ \delta_p [\omega] = [\sigma] $, where $ \sigma \in \df{p+1} $ such that $ \sigma \vert_U = \dd \eta $ and $ \sigma \vert_V = \dd \eta' $.

  If $ \{\varphi , \psi\} $ is a smooth partition of unity subordinate to $ \{U,V\} $, then one can take $ \eta = \psi \omega $ and $ \eta' = - \varphi \omega $, both extended to zero outside of the supports of $ \psi $ and $ \varphi $.
\end{corollary}

\subsubsection{Applications of the Mayer--Vietoris theorem}

The Mayer--Vietoris theorem allows to compute all of the de Rham cohomology groups of spheres.

\begin{proposition}{Cohomology of spheres}{cohomology-sphere}
  For $ n \in \N $, the de Rham cohomology groups of $ \Sn{n} $ are:
  \begin{equation}
    \hp{p}(\Sn{n}) \cong
    \begin{cases}
      \R & p = 0,n \\
      0 & 0 < p < n
    \end{cases}
  \end{equation}
  The cohomology class of any smooth orientation form is a basis for $ \hp{p}(\Sn{n}) $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    By \exref{ex:cohomology-zero}, $ \hp{0}(\Sn{n}) \cong \R $, so consider $ p \geq 1 $. The proof is by induction on $ n \in \N $.

    $ (n = 1) $ Any orientation for on $ \Sn{1} $ has non-zero integral, so it is not exact by Stokes' theorem, hence $ \dim_\R \hp{1}(\Sn{1}) \geq 1 $. On the other hand, by \tref{th:cohomology-fundamental-group} there is an injective linear map $ \hp{1}(\Sn{1}) \ra \Hom(\pi_1(\Sn{1},1) , \R) $, but $ \dim_\R \Hom(\pi_1(\Sn{1},1) , \R) $, thus $ \dim_\R \hp{1}(\Sn{1}) = 1 $, and is spanned by the cohomology class of any smooth orientation form.

    $ (n \geq 2) $ Assume that the theorem is true for $ \Sn{n-1} $. Since $ \Sn{n} $ is simply connected (for $ n \geq 1 $), $ \hp{1}(\Sn{n}) = 0 $ by \cref{cor:connected-cohomology}. For $ p > 1 $, use \tref{th:mayer-vietoris} as follows: let $ N $ and $ S $ be the north and south poles in $ \Sn{n} $, respectively, and let $ U = \Sn{n} - \{S\} $ and $ V = \Sn{n} - \{N\} $, which are both diffeomorphic to $ \R^n $ (through the stereographic projection, see \figref{fig:coho-sph}), thus $ U \cap V $ is diffeomorphic to $ \R^n - \{0\} $. Part of the Mayer--Vietoris sequence for $ \{U,V\} $ reads:
    \begin{equation*}
      \hp{p-1}(U) \oplus \hp{p-1}(V) \ra \hp{p-1}(U \cap V) \ra \hp{p}(\Sn{n}) \ra \hp{p}(U) \oplus \hp{p}(V)
    \end{equation*}
    Since $ U , V $ are diffeomorphic to $ \R^n $, the groups at both ends are trivial when $ p > 1 $, hence $ \hp{p}(\Sn{n}) \cong \hp{p-1}(U \cap V) $. Moreover, $ U \cap V $ is diffeomorphic to $ \R^n - \{0\} $ and thus homotopy equivalent to $ \Sn{n-1} $ by \exref{ex:homotopy-spheres}, so in the end $ \hp{p}(\Sn{n}) \cong \hp{p-1}(\Sn{n-1}) $ for $ p > 1 $. The thesis then follows by induction. As in the $ n = 1 $ case, any smooth orientation on $ \Sn{n} $ determines a non-zero cohomology class, which therefore spans $ \hp{n}(\Sn{n}) $.
  \end{proof}
\end{proofbox}

  \centering
  \includegraphics[width = 0.45 \textwidth]{coho-sph.png}
  \caption{Stereographic projections on $ \Sn{n} $.}
  \label{fig:coho-sph}
\end{figure}

It can be shown that $ \eta \in \dfm{n}(\Sn{n}) $ is exact if and only if $ \int_{\Sn{n}} \eta = 0 $.

\begin{propcorollary}{Cohomology of punctured Euclidean spaces}{cohomology-punctured}
  Let $ x \in \R^n $ and $ \man = \R^n - \{x\} $, with $ n \geq 2 $. The only non-trivial de Rham groups of $ \man $ are $ \hp{0}(\man) \cong \R $ and $ \hp{n-1}(\man) \cong \R $.
  A closed $ \eta \in \df{n-1} $ is exact if and only if $ \int_S \eta = 0 $ for some (and hence every) $ (n-1) $-dimensional sphere $ S \ssq \man $ centered at $ x $.
\end{propcorollary}

\begin{proofbox}
  \begin{proof}
    Let $ S \ssq \man $ be any $ (n-1) $-dimensional sphere centered at $ x $. The inclusion map $ \inc : S \hookrightarrow \man $ is a homotopy equivalence, thus $ \inc^* : \hp{p}(\man) \ra \hp{p}(S) $ is an isomorphism for each $ p \in \N_0 $ and the result on de Rham groups of $ \man $ follows from \pref{prop:cohomology-sphere}.

    If $ \eta \in \zp{n-1}(\man) $, then $ \eta \in \bp{n-1}(\man) $ if and only if $ \inc^* \eta \in \bp{n-1}(S) $, which is true if and only if $ \int_S \eta = \int_S \inc^* \eta = 0 $ by the observation above.
  \end{proof}
\end{proofbox}

This result remains true even when $ \man = \R^n - \overline{B} $, for some closed ball $ \overline{B} \ssq \R^n $.

\begin{propcorollary}{}{cohomology-open-subset}
  Let $ U \in \tp(\R^n) $, with $ n \geq 2 $, and $ x \in U $. Then $ \hp{n-1}(U - \{x\}) \neq 0 $.
\end{propcorollary}

\begin{proofbox}
  \begin{proof}
    Since $ U $ is open, there is an $ (n-1) $-dimensional sphere $ S $ centered at $ x $ such that $ S \ssq U - \{x\} $. Let $ \inc : S \hookrightarrow U - \{x\} $ be the inclusion map and $ r : U - \{x\} \ra S $ be the radial projection onto $ S $. Then, $ r $ and $ \inc $ are smooth with $ r \circ \inc = \id_S $, which implies $ \inc^* \circ r^* = \id_{\hp{n-1}(S)} $, and therefore $ r^* : \hp{n-1}(S) \ra \hp{n-1}(U - \{x\}) $ is injective: since $ \hp{n-1}(S) \neq 0 $ by \pref{prop:cohomology-sphere}, the thesis follows.
  \end{proof}
\end{proofbox}

An important application of the topological invariance of the de Rham cohomology groups is the proof of the topological invariance of dimension.

\begin{theorem}{Topological invariance of dimension}{topological-invariance-dimension-proof}
  A non-empty $ n $-dimensional topological manifold cannot be homeomorphic to an $ m $-dimensional topoplogical manifold, unless $ m = n $.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    If $ \man $ is a topological $ n $-mnaifold that is homeomorphic to an $ m $-manifold, then $ \man $ is itself both an $ n $-manifold and an $ m $-manifold. The case in which $ m = 0 $ or $ n = 0 $ is trivial, hence WLOG $ m > n \geq 1 $. Since $ \man $ is an $ m $-manifold, there is $ V \in \tp(\man) $ that is homeomorphic to $ \R^m $, but an open subset of an $ n $-manifold is itself an $ n $-manifold, hence any point $ x \in V $ has a neighbourhood $ U \ssq V $ that is homeomorphic to $ \R^n $. Since $ U $ is homeomorphic to $ \R^n $, the homeomorphism can be used to define a smooth structure on $ U $, and then $ \hp{m-1}(U - \{x\}) = 0 $ by \cref{cor:cohomology-punctured}. On the other hand, since $ U $ is homeomorphic to an open set of $ \R^m $, this homeomorphism can be used to define another smooth structure on $ U $, and \cref{cor:cohomology-open-subset} implies $ \hp{m-1}(U - \{x\}) \neq 0 $, which contradicts the topological invariance of de Rham cohomology $ \absurd $
  \end{proof}
\end{proofbox}

\subsection{Degree theory}

Top-degree cohomology can be used to prove some properties of smooth functions between certain compact manifolds of the same dimension.

It is useful to introduce a generalization of the de Rham cohomology groups using compactly supported forms. Let $ \man $ be a smooth manifold: the $ \R $-vector space of compactly supported smooth $ p $-forms on $ \man $ is denoted by $ \dfcm{p}(\man) \ssq \df{p} $. Then, restricting to compactly supported smooth $ p $-forms the spaces $ \zpc{p}(\man) \ssq \zp{p}(\man) $ and $ \bpc{p}(\man) \ssq \bp{p}(\man) $, the \emph{$ p^\textit{th} $ compactly supported de Rham cohomology group} of $ \man $ is the quotient space:
\begin{equation}
  \hpc{p}(\man) \defeq \frac{\zpc{p}(\man)}{\bpc{p}(\man)}
\end{equation}
This reduces to the ordinary de Rham cohomology when $ \man $ is compact. For non-compact manifolds, however, these are different groups: for example, when $ n \geq 1 $:
\begin{equation}
  \hpc{p}(\R^n) \cong
  \begin{cases}
    \R & p = n \\
    0 & 0 \leq p < n
  \end{cases}
\end{equation}
contrary to $ \hp{p}(\R^n) = 0 $ for $ p \geq 1 $. In general, smooth maps need not pull-back compactly supported forms to compactly supported forms, so they don't induce maps on compactly supported cohomology. However, a proper map does pull-back compactly supported forms to compactly supported forms, so a proper map $ f : \man_1 \ra \man_2 $ induces a cohomology map $ f^* : \hpc{p}(\man_2) \ra \hpc{p}(\man_1) $ for each $ p \in \N_0 $.

\subsubsection{Top-degree cohomology}

Let $ \man $ be an oriented smooth $ n $-manifold. There is a natural linear map $ I : \dfc{n} \ra \R $ given by integration over $ \man $:
\begin{equation*}
  I(\omega) \defeq \int_\man \omega
\end{equation*}
Since the integral of the exterior derivative of a compactly supported $ (n-1) $-form vanishes, $ I $ descends to a linear map $ I : \hpc{n}(\man) \ra \R $ (every smooth $ n $-form on an $ n $-manifold is closed).

\begin{proposition}{Top cohomology, orientable compact support case}{top-or-comp-supp}
  If $ \man $ is a connected oriented $ n $-manifold, then the integration map $ I : \hpc{n}(\man) \ra \R $ is an isomorphism, so $ \dim_\R \hpc{n}(\man) = 1 $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    If $ n = 0 $, the thesis is trivial by \exref{ex:cohomology-zero}, so assume $ n \geq 1 $. Let $ (U,(x^i)) $ be an oriented smooth coordinate chart on $ \man $, and let $ f $ be a smooth bump function with compact support in $ U $: then, the $ n $-form defined as $ \theta_0 = f \, \dd x^1 \wedge \dots \wedge \dd x^n $ in $ U $ and $ 0 $ outside $ U $ is smooth and compactly supported on $ \man $, and satisfies $ I(\theta_0) > 0 $. Therefore, $ I $ is surjective, so WTS $ I $ is injective, which is equivalent to showing that if $ \omega \in \dfc{n} $ such that $ \int_\man \omega = 0 $, then $ \exists \eta \in \dfc{n-1} : \omega = \dd \eta $.

    Let $ \{U_i\}_{i \in \N} $ be a countable open cover of $ \man $ such that each $ U_i $ is diffeomorphic to $ \R^n $, and let $ \man_k \equiv \bigcup_{i = 1}^k U_i $ for each $ k \in \N $. Since $ \man $ is connected, it is possible to arrange the series so that $ \man_k \cap U_{k+1} \neq \emptyset \,\,\forall k \in \N $. Since every compactly supported $ n $-form is supported in $ \man_k $ for some $ k \in \N $, it suffices to prove that if $ \omega \in \dfcm{n}(\man_k) $ has $ \int_{\man_k} \omega = 0 $, then $ \omega = \dd \eta $ for some $ \eta \in \dfcm{n-1}(\man_k) $, which can then be extended by zero to a compactly supported form on all of $ \man $. The proof is by induction on $ k \in \N $.

    $ (k = 1) $ Since $ \man_1 = U_1 $ is diffeomorphic to $ \R^n $, the claim is proved by the following lemma.

    \begin{lemma}[borderline = {1pt}{0pt}{blue!35!black}]{Poincaré lemma with compact support}{poincare-compact}
      Let $ n \geq p \geq 1 $ and let $ \omega $ be a compactly supported $ p $-form on $ \R^n $. If $ p = n $, suppose $ \int_{\R^n} \omega = 0 $. Then, there exists $ \eta \in \dfcm{p-1}(\R^n) $ such that $ \omega = \dd \eta $.
    \end{lemma}

    $ (k \geq 1) $ Assume the claim to be true for some $ k \geq 1 $, and suppose $ \omega \in \dfcm{n}(\man_{k+1}) $, with $ \man_{k+1} = \man_k \cup U_{k+1} $ that satisfies $ \int_{\man_{k+1}} \omega = 0 $. Let $ \theta \in \dfcm{n}(\man_{k+1}) $ be supported in $ \man_k \cap U_{k+1} $ and usch that $ \int_{\man_{k+1}} \theta = 1 $. Let $ \{\varphi , \psi\} $ be a smooth partition of unity for $ \man_{k+1} $ subordinate to the cover $ \{\man_k , U_{k+1}\} $, as in \figref{fig:top-deg-coho}, and let $ c \equiv \int_{\man_{k+1}} \varphi \omega $. Since $ \varphi \omega - c \theta $ is compactly supported in $ \man_k $ and its integral vanished by definition of $ c $, by the inductive hypothesis there is $ \alpha \in \dfcm{n-1}(\man_k) $ such that $ \dd \alpha = \varphi \omega - c \theta $. Similarly, $ \psi \omega + c \theta $ is compactly supported in $ U_{k+1} $ and:
    \begin{equation*}
      \int_{U_{k+1}} (\psi \omega + c \theta) = \int_{\man_{k+1}} (1 - \vartheta) \omega + c \int_{\man_{k+1}} \theta = \int_{\man_{k+1}} \omega - \int_{\man_{k+1}} \varphi \omega + c = 0
    \end{equation*}
    Thus, by \lref{lemma:poincare-compact}, there exists $ \beta \in \dfcm{n-1}(U_{k+1}) $ such that $ \dd \beta = \psi \omega + c \theta $. Both $ \alpha $ and $ \beta $ can be extended by zero to smooth compactly supported forms on $ \man_{k+1} $, and:
    \begin{equation*}
      \dd (\theta + \beta) = (\varphi \omega - c \theta) + (\psi \omega + c \theta) = (\varphi + \psi) \omega = \omega
    \end{equation*}
    which completes the proof.
  \end{proof}
\end{proofbox}

\begin{figure}
  \centering
  \includegraphics[width = 0.80 \textwidth]{top-def-coho.png}
  \caption{Computation of top-degree cohomology.}
  \label{fig:top-deg-coho}
\end{figure}

\begin{propcorollary}{Top cohomology, orientable compact case}{top-or-comp}
  If $ \man $ is a compact connected orientable smooth $ n $-manifold, then $ \dim_\R \hp{n}(\man) = 1 $ and is spanned by the cohomology class of any smooth orientation form.
\end{propcorollary}

\begin{proofbox}
  \begin{proof}
    For compact manifolds $ \hpc{p}(\man) = \hp{p}(\man) $, hence use \pref{prop:top-or-comp-supp} and the fact that any orientation form has non-vanishing integral.
  \end{proof}
\end{proofbox}


\begin{proposition}{Top cohomology, orientable non-compact case}{}
  If $ \man $ is a non-compact connected orientable smooth $ n $-manifold, then $ \hp{n}(\man) = 0 $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    First, choose an orientation on $ \man $. Let $ f \in \cm $ be a smooth exhaustion function: by adding a constant, it is possible to set $ \inf_\man f = 0 $, and then connectedness and non-compactness of $ \man $ imply $ f(\man) = [0 , \infty) $. For each $ i \in \N $, let $ V_i = f^{-1}((i - 1 , i)) $: clearly, $ \{V_i\}_{i \in \N} $ is a cover of $ \man $ by non-empty precompact\footnotemark open sets, with $ V_i \cap V_j \neq \emptyset $ if and only if $ j = i-1,i,i+1 $. Let $ \{\psi_i\} $ be a smooth partition of unity subordinate to this cover and, for each $ i \in \N $, let $ \theta_i \in \dfc{n} $ be compactly supported in $ V_i \cap V_{i+1} $ with $ \int_\man \theta_i = 1 $.

    Let $ \omega \in \df{n} $ arbitrary, and let $ \omega_i = \psi_i \omega \in \dfcm{n}(V_i) $ for each $ i \in \N $. Let $ c_1 = \int_{V_1} \omega_1 $, so that $ \omega_1 - c_1 \theta_1 $ is compactly supported in $ V_1 $ and has non-zero integral. It follows form \pref{prop:top-or-comp-supp} that there exists $ \eta_1 \in \dfcm{n}(V_1) $ such that $ \dd \eta_1 = \omega_1 - c_1 \theta_1 $. Next, choose $ c_2 \in \R $ such that $ \int_{V_2} (\omega_2 + c_1 \theta_1 - c_2 \theta_2) = 0 $, so there exists $ \eta_2 \in \dfcm{n}(V_2) $ such that $ \dd \eta_2 = \omega_2 + c_1 \theta_1 - c_2 \theta_2 $. Continuining by induction, choose $ c_j \in \R $ and $ \eta_j \in \dfcm{n}(V_j) $ such that $ \dd \eta_j = \omega_j + c_{j-1} \theta_{j-1} - c_j \theta_j $.

    Now, set $ \eta \equiv \sum_{j \in \N} \eta_j $, with each $ \eta_j $ extended by zero on $ \man - V_j $. Since at most three terms in this sum are non-zero on each $ V_i $, this is $ \eta \in \df{n} $. When taking the exterior derivative, the $ c_j \theta_j $ terms cancel, so $ \dd \eta = \sum_{j \in \N} \omega_j = \omega $.
  \end{proof}
\end{proofbox}

\footnotetext{Given a topological space $ X $, $ U \ssq X $ is \emph{precompact} if $ \overline{U} $ is compact.}

An analogous result for non-orientable manifolds can be found in a similar way.

\begin{proposition}{Top cohomology, non-orientable case}{}
  If $ \man $ is a connected non-orientable smooth $ n $-manifold, then $ \hpc{n}(\man) = 0 $ and $ \hp{n}(\man) = 0 $.
\end{proposition}

\subsubsection{Smooth maps}

Once the top-degree cohomology groups of all compact manifolds are found, smooth maps between them can be studied by assigning an integer to them in such a way that homotopic maps have the same integer.

\begin{theorem}{Degree of smooth map}{deg}
  Let $ \man_1 , \man_2 $ be compact oriented smooth $ n $-manifolds, and let $ f : \man_1 \ra \man_2 $ be a smooth map. There exists a unique $ \thcol{k_f \in \Z} $, called the \bcth{degree of $ f $}, that satisfies:
  \begin{enumerate}
    \item \label{item:deg-1} for every $ \omega \in \dfm{n}(\man_2) $:
      \begin{equation*}
        \int_{\man_1} f^* \omega = k_f \int_{\man_2} \omega
      \end{equation*}
    \item \label{item:deg-2} if $ q \in \man_2 $ is a regular value of $ f $, then:
      \begin{equation*}
        k_f = \sum_{x \in f^{-1}(q)} \sgn(x)
      \end{equation*}
      where $ \sgn(x) = +1 $ if $ \dd f_x $ is orientation-preserving, and $ -1 $ otherwise.
  \end{enumerate}
\end{theorem}

\begin{proofbox}
  \begin{proof}
    By \cref{cor:top-or-comp}, two smooth $ n $-forms on either $ \man_1 $ or $ \man_2 $ are cohomologous if and only if they have the same integral. Let $ \theta \in \dfm{n}(\man_2) : \int_{\man_2} \theta = 1 $, and let $ k_f = \int_{\man_1} f^* \theta $. If $ \omega \in \dfm{n}(\man_2) $ is arbitrary, then $ \omega $ is cohomologous to $ a \theta $, where $ a \in \int_{\man_2} \omega $, and therefore $ f^* \omega $ is cohomologous to $ a f^* \omega $. Then:
    \begin{equation*}
      \int_{\man_1} f^* \omega = a \int_{\man_1} f^* \theta = a k_f = k_f \int_{\man_2} \omega
    \end{equation*}
    Thus $ k_f $ satisfies \itemref{item:deg-1} and is clearly the only number that does so.

    Next WTS \itemref{item:deg-2}, from which it also follows that $ k_f \in \Z $. Let $ q \in \man_2 $ be an arbitrary regular value of $ f $: since $ f^{-1}(q) $ is a properly embedded $ 0 $-submanifold of $ \man $, it is finite. Suppose first that $ f^{-1}(q) $ is not empty, i.e. $ f^{-1}(q) = \{x_1 , \dots , x_m\} $. By the inverse function theorem, for each $ i = 1 , \dots , m $ there is a neighbourhood $ U_i $ of $ x_i $ such that $ f $ is a diffeomorphism from $ U_i $ to a neighbourhood $ W_i $ of $ q $, and, by shrinking the $ U_i $ sets if necessary, they can be assumed to be disjoint. Then, $ K = \man - (U_1 \cup \dots \cup U_m) $ is closed in $ \man $ and thus compact, so $ f(K) $ is closed in $ \man_2 $ and disjoint from $ q $. Let $ W $ be the connected component of $ W_1 \cap \dots \cap W_m \cap (\man_2 - f(K)) $ containing $ q $, and let $ V_i = f^{-1}(W) \cap U_i $: it follows that $ W $ is a connected neighbourhood of $ q $ such that $ f^{-1}(W) = V_1 \sqcup \dots \sqcup V_m $, and $ f $ restricts to a diffeomorphism from each $ V_i $ to $ W $. Since each $ V_i $ is connected, the restriction of $ f $ to $ V_i $ must be either orientation-preserving or reversing.

    Let $ \omega \in \dfm{n}(\man_2) $ be compactly supported in $ W $ and satisfying $ \int_{\man_2} \omega = \int_W \omega = 1 $. From \itemref{item:deg-1} it follows that $ \int_{\man_1} f^* \omega = k_f $ and, since $ f^*\omega $ is compactly supported in $ f^{-1}(W) $, $ \int_{\man_1} f^* \omega = \sum_{i = 1}^m \int_{V_i} f^* \omega $. From \itemref{item:pif-4} of \lref{lemma:pif}, $ \int_{V_i} f^* \omega = \pm \int_W \omega = \pm 1 $ for each $ i = 1 , \dots , m $, with positive sign if $ f $ is orientation-preserving on $ V_i $ and negative otherwise. This proves \itemref{item:deg-2} when $ f^{-1}(q) \neq \emptyset $.

    If $ f^{-1}(q) = \emptyset $, then $ q $ has a neighbourhood $ W \ssq \man_2 - f(\man_1) $ since $ f(\man_1) $ is compact and thus closed. Given $ \omega \in \dfm{n}(\man_2) $ compactly supported in $ W $, then $ \int_{\man_1} f^* \omega = 0 $, so $ k_f = 0 $. This proves \itemref{item:deg-2}.
  \end{proof}
\end{proofbox}

The importance of degree theory comes from the fact that these two characterizations are independent of each other: for example, it is often easy to compute the degree of a map by simply counting the points in the preimage of a regular value, and then use the characterization in terms of differential forms to prove important properties.

\begin{lemma}{Properties of degree}{pd}
  Let $ \man_1 , \man_2 , \man_3 $ be compact connected oriented smooth $ n $-manifolds. Then:
  \begin{enumerate}
    \item \label{item:pd-1} if $ f : \man_1 \ra \man_2 $ and $ g : \man_2 \ra \man_3 $ are smooth maps, then $ k_{g \circ f} = k_g k_f $;
    \item \label{item:pd-2} if $ f : \man_1 \ra \man-2 $ is a diffeomorphism, the $ k_f = + 1 $ if $ f $ is orientation-preserving and $ -1 $ otherwise;
    \item \label{item:pd-3} if two smooth maps $ f_0 , f_1 : \man_1 \ra \man_2 $ are homotopic, then $ k_{f_0} = k_{f_1} $.
  \end{enumerate}
\end{lemma}

By \lref{lemma:pd}, it is possible to defined the degree of a continuous map $ f : \man_1 \ra \man_2 $ between compact connected oriented smooth $ n $-manifolds as the degree of any smooth map that is homotopic to $ f $: the Whitney approximation theorem \tref{th:whitney-approx} guarantees the existence of such a map.

\begin{proposition}{}{brouwer-aux}
  Let $ \man $ be a compact connected oriented smooth $ n $-manifold, and $ X $ a compact oriented smooth $ (n+1) $-manifold with boundary. If $ f : \pa X \ra \man $ is a continuous map that has a continuous extension to $ X $, then $ k_f = 0 $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Let $ F : X \ra \man $ be the continuous extension of $ f $ on $ X $. By \tref{th:whitney-approx}, there is a smooth map $ \tilde{f} : X \ra \man $ that is homotopic to $ F $. Replacing $ F $ by $ \tilde{F} $ and $ f $ by $ \tilde{F} \vert_{\pa X} $, it is possible to assume that both $ f $ and $ F $ are smooth.

    Let $ \omega \in \dfm{n}(\man) $: then, $ \dd \omega = 0 $ since $ \omega $ is a top form. From Stokes' theorem:
    \begin{equation*}
      \int_{\pa X} f^* \omega = \int_{\pa X} F^* \omega = \int_X \dd (F^* \omega) = \int_X F^* \dd \omega = 0
    \end{equation*}
    By \itemref{item:deg-1} of \tref{th:deg}, then, $ k_f = 0 $.
  \end{proof}
\end{proofbox}

\begin{theorem}{Brouwer fixed-point theorem}{brouwer}
  Every continuous map $ \Dn{n} \ra \Dn{n} $ has a fixed point.
\end{theorem}

\begin{proofbox}
  \begin{proof}
    Suppose that $ F : \Dn{n} \ra \Dn{n} $ is continuous and has no fixed points. It is then possible to define a continuous map $ G : \Dn{n} \ra \Sn{n-1} $ as:
    \begin{equation*}
      G(x) = \frac{x - F(x)}{\norm{x - F(x)}}
    \end{equation*}
    and let $ g = G \vert_{\Sn{n-1}} : \Sn{n-1} \ra \Sn{n-1} $. By \pref{prop:brouwer-aux} $ k_g = 0 $. On the other hand, consider $ H : \Sn{n-1} \times [0,1] \ra \Sn{n-1} $ defined as:
    \begin{equation*}
      H(x,t) = \frac{x - t F(x)}{\norm{x - t F(x)}}
    \end{equation*}
    which is well-defined since the denominator never vanishes when $ t = 1 $, as $ F $ has no fixed points, and when $ t < 1 $, as $ \norm{x} = 1 $ while $ \norm{t F(x)} \leq t < 1 $. Therefore, $ H $ is continuous and a homotopy from $ \id_{\Sn{n-1}} $ to $ g $: this implies $ k_g = 1 $ by \itemref{item:pd-3} of \lref{lemma:pd} $ \absurd $
  \end{proof}
\end{proofbox}

\section{Hodge theory}

The study of the de Rham cohomology of a metric manifold makes use of the additional structure given by the metric.

\subsection{Inner product on differential forms}

Recall \secref{ssec:tan-cot-isomorphism}: on a metric manifold $ (\man,g) $, the canonical smooth bundle isomorphism $ T\man \cong T^*\man $ allows the metric to define an inner product on $ T_p^*\man $ as:
\begin{equation*}
  \braket{\omega , \eta}_g \equiv \braket{\omega^\sharp , \eta^\sharp}_g
\end{equation*}
where the inner product on the right-hand side is that on $ T_p\man $, i.e. $ \braket{v,w}_g \defeq g_p(v,w) $, which is positive-definite on Riemannian manifolds. Clearly, then, $ \braket{\omega , \eta}_g = \omega(\eta^\sharp) = \eta(\omega^\sharp) $ by definition of the sharp isomorphism.

This construction is easily generalized to decomposable differential forms. Indeed, given two decomposable $ p $-forms $ \alpha = \alpha_1 \wedge \dots \alpha_p $ and $ \beta = \beta_1 \wedge \dots \wedge \beta_p $, their metric-induced inner product can be defined through the \bctxt{Gram determinant}:
\begin{equation}
  \braket{\alpha , \beta}_g \defeq \det [\braket{\alpha_i , \beta_j}_g]
  \label{eq:gram-determinant}
\end{equation}
This is a natural choice since the determinant is alternating under the exchange of columns, thus preserving the alternating nature of the exterior product, and it reduces to the inner product on $ 1 $-forms defined above. This definition clearly generalizes bilinearly to arbitrary $ p $-forms, since any $ p $-form can be expressed as a linear combination of decomposable forms (\eref{eq:diff-form-basis}).

\begin{proposition}{Inner product of $ p $-forms}{inner-product-forms}
  Let $ (\man,g) $ be a metric manifold, and let $ \omega , \eta $ be two $ p $-forms on $ \man $. Then, their inner product is:
  \begin{equation}
    \braket{\omega , \eta}_g = \frac{1}{p!} \omega_{i_1 \dots i_p} \eta_{j_1 \dots j_p} g^{i_1 j_1} \dots g^{i_p j_p}
  \end{equation}
  on every local oriented orthonormal frame on $ \man $.
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Let $ (x^i) $ be a smooth chart on $ \man $, so that $ \omega $ and $ \eta $ can be expressed as in \eref{eq:diff-form-basis}. Then, extending bilinearly \eref{eq:gram-determinant}:
    \begin{equation*}
      \begin{split}
        \braket{\omega , \eta}_g
        & = \frac{1}{(p!)^2} \omega_{i_1 \dots i_p} \eta_{j_1 \dots j_p} \det[\braket{\dd x^{i_a} , \dd x^{j_b}}_g] = \frac{1}{(p!)^2} \omega_{i_1 \dots i_p} \eta_{j_1 \dots j_p} \det[g^{i_a j_b}] \\
        & = \frac{1}{(p!)^2} \omega_{i_1 \dots i_p} \eta_{j_1 \dots j_p} \sum_{\sigma \in \sy{p}} \sgn(\sigma) g^{i_1 j_{\sigma(1)}} \dots g^{i_p j_{\sigma(p)}} = \frac{1}{p!} \omega_{i_1 \dots i_p} \eta_{j_1 \dots j_p} g^{i_1 j_1} \dots g^{i_p j_p}
      \end{split}
    \end{equation*}
    where in the last step the total antisymmetry of $ \omega_{i_1 \dots i_p} $ was used. To show that this expression is basis-independent, consider two oriented orthonormal frames $ (E_1 , \dots , E_n) $ and $ (\tilde{E}_1 , \dots , \tilde{E_n}) $ on $ \man $ such that:
    \begin{equation*}
      \tilde{E}_i = A^j_i E_j
    \end{equation*}
    for some smooth functions $ [A^j_i] $. Since these frames are consistently oriented, $ [A^j_i](p) \in \SOn{n} $ for each $ p \in \man $. Then, the components of $ \omega $ in the two bases are linked by (\eref{eq:covariant-alias}):
    \begin{equation*}
      \tilde{\omega}_{i_1 \dots i_p} = A^{j_1}_{i_1} \dots A^{j_p}_{i_p} \omega_{j_1 \dots j_p}
    \end{equation*}
    and similarly for $ \eta $. Then, given the expression for the metric $ g^{ij} = \braket{\varepsilon^i , \varepsilon^j}_g $ and the transformation law $ \tilde{\varepsilon}^i = [A^{-1}]^i_j \varepsilon^j $ for the dual coframes, the inner product in the transformed frame is:
    \begin{equation*}
      \begin{split}
        \braket{\omega , \eta}_g
        & = \tilde{\omega}_{i_1 \dots i_p} \tilde{\eta}_{j_1 \dots j_p} \tilde{g}^{i_1 j_1} \dots \tilde{g}^{i_p j_p} \\
        & = A^{\ell_1}_{i_1} \dots A^{\ell_p}_{i_p} \omega_{\ell_1 \dots \ell_p} A^{m_1}_{j_1} \dots A^{m_p}_{j_p} \eta_{m_1 \dots m_p} [A^{-1}]^{i_1}_{a_1} [A^{-1}]^{j_1}_{b_1} \dots [A^{-1}]^{i_p}_{a_p} [A^{-1}]^{j_p}_{b_p} g^{a_1 b_1} \dots g^{a_p b_p} \\
        & = \delta^{\ell_1}_{a_1} \delta^{m_1}_{b_1} \dots \delta^{\ell_p}_{a_p} \delta^{m_p}_{b_p} \omega_{\ell_1 \dots \ell_p} \eta_{m_1 \dots m_p} g^{a_1 b_1} \dots g^{a_p b_p} = \omega_{i_1 \dots i_p} \eta_{j_1 \dots j_p} g^{i_1 j_1} \dots g^{i_p j_p}
      \end{split}
    \end{equation*}
    which indeed shows that the above expression for the inner product is independent of the choice of oriented orthonormal frame.
  \end{proof}
\end{proofbox}

\subsubsection{\texorpdfstring{$ L^2 $}{L²}-inner product}

Another usefuld inner product is the \emph{$ \sqi $-inner product} of compactly supported $ p $-forms, which can be defined on orientable metric manifolds:
\begin{equation}
  (\omega , \eta) \defeq \int_\man \braket{\omega , \eta}_g \volfg
\end{equation}
where $ \omega_g $ is the volume form on $ \man $ induced by $ g $ (\eref{eq:volume-form}). To be precise, this inner product is well-defined on \emph{square-integrable $ p $-forms}:
\begin{equation}
  \dfs{p} \defeq \{\omega \in \df{p} : \norm{\omega}_{L^2} < \infty\}
\end{equation}
where $ \norm{\omega}_{L^2} = (\omega , \omega) $. Clearly, $ \dfc{p} \subsetneq \dfs{p} \subsetneq \df{p} $ in general (if $ \man $ is compact, they are all equal), and it can be shown that $ \dfc{p} $ is dense\footnotemark in $ \dfs{p} $. Note that $ \dfs{p} $ is a pre-Hilbert space, since it is not complete: it is possible to construct a Cauchy sequence of forms in $ \dfs{p} $ whose limit exists in a $ L^2 $ sense, but is not smooth.

\footnotetext{Given a topological space $ X $ and a subset $ A \ssq X $, a \emph{limit point} of $ A $ in $ X $ is a point $ p \in X $ such that every neighbourhood of $ x $ contains at least one point of $ A $ different from $ x $ itself. Then, $ A $ is said to be \emph{dense} in $ X $ if every point of $ X $ either is in $ A $ or is a limit point of $ A $.}

\subsection{Hodge dual}

\begin{definition}{Hodge operator}{hodge-dual}
  Let $ (\man,g) $ be a metric manifold. The \bcdef{Hodge operator} $ \star : \df{p} \ra \df{n-p} $ is defined such that, given $ \omega , \eta \in \df{p} $:
  \begin{equation}
    \omega \wedge \star \eta = \braket{\omega , \eta}_g \volfg
    \label{eq:hodge-def}
  \end{equation}
\end{definition}

The image $ \star \eta \in \df{n-p} $ is often called the \emph{Hodge dual} of $ \eta \in \df{p} $. It is trivial to show that the Hodge operator is a linear operator, hence it is completely determined by its action on the basis $ p $-forms (in each smooth map on $ \man $):
\begin{equation}
  \star ( \dd x^{i_1} \wedge \dots \wedge \dd x^{i_p} ) = \frac{\sqg}{(n-p)!} g^{i_1 j_1} \dots g^{i_p j_p} \epsilon_{j_1 \dots j_n} \dd x^{j_{p+1}} \wedge \dots \wedge \dd x^{j_n}
  \label{eq:hodge-alibi}
\end{equation}
where the factorial accounts for the contraction of $ n-p $ indices between two totally antisymmetric objects. This expression makes it clear that $ \volfg = \star 1 $. Equivalently, the Hodge dual of $ \omega \in \df{p} $ can be expressed as:
\begin{equation*}
  \star \omega = \frac{1}{(n-p)!} (\star \omega)_{i_1 \dots i_{n-p}} \dd x^{i_1} \wedge \dots \wedge \dd x^{i_{n-p}}
\end{equation*}
with components:
\begin{equation}
  (\star \omega)_{i_1 \dots i_{n-p}} = \frac{\sqg}{n!} \omega_{j_1 \dots j_p} g^{j_1 \ell_1} \dots g^{j_p \ell_p} \epsilon_{\ell_1 \dots \ell_k i_1 \dots i_{n-p}}
  \label{eq:hodge-alias}
\end{equation}

\begin{lemma}{Invariance of the Hodge dual}{}
  The Hodge dual is independent of the choice of oriented orthonormal frame.
\end{lemma}

\begin{proofbox}
  \begin{proof}
    This invariance is trivially derived from \eref{eq:hodge-def}, since both the inner product (\pref{prop:inner-product-forms}) and the volume form (\pref{prop:riemannian-volume-form}) are independent of the choice of oriented orthonromal frame.

    This can also be seen directly from \eref{eq:hodge-alibi} (alibi interpretation) or \eref{eq:hodge-alias} (alias interpretation). For example, given two smooth charts $ (U,x^i) , (V,y^i) : U \cap V \neq \emptyset $, then on $ U \cap V $:
    \begin{equation*}
      \begin{split}
        \star (\dd y^{i_1} \wedge \dots \wedge \dd y^{i_p}) \hspace{-0.15em}
        & = \frac{1}{(n-p)!} \sqrt{\abs{\det [\tilde{g}_{ij}]}} \, \tilde{g}^{i_1 j_1} \dots \tilde{g}^{i_p j_p} \epsilon_{j_1 \dots j_n} \dd y^{j_{p+1}} \wedge \dots \wedge \dd y^{j_n} \\
        & = \frac{1}{(n-p)!} \sqrt{\abs{\det \left[ \frac{\pa x^s}{\pa y^i} \right] \det \left[ \frac{\pa x^t}{\pa y^j} \right] \det [g_{st}]}} \times \\
        & \quad \times \frac{\pa y^{i_1}}{\pa x^{a_1}} \frac{\pa y^{j_1}}{\pa x^{b_1}} g^{a_1 b_1} \dots \frac{\pa y^{i_p}}{\pa x^{a_p}} \frac{\pa y^{j_p}}{\pa x^{b_p}} g^{a_p b_p} \epsilon_{j_1 \dots j_n} \frac{\pa y^{j_{p+1}}}{\pa x^{b_{p+1}}} \dd x^{b_{p+1}} \wedge \dots \wedge \frac{\pa y^{j_n}}{\pa x^{b_n}} \dd x^{b_n} \\
        & = \abs{\det \left[ \frac{\pa x}{\pa y} \right]} \frac{\pa y^{i_1}}{\pa x^{a_1}} \frac{\pa y^{j_1}}{\pa x^{b_1}} \dots \frac{\pa y^{i_p}}{\pa x^{a_p}} \frac{\pa y^{j_p}}{\pa x^{b_p}} \frac{\pa y^{j_{p+1}}}{\pa x^{b_{p+1}}} \dots \frac{\pa y^{j_n}}{\pa x^{b_n}} \epsilon_{j_1 \dots j_n} \times \\
        & \qquad \qquad \qquad \qquad \qquad \qquad \qquad \times \frac{\sqg}{(n-p)!} g^{a_1 b_1} \dots g^{a_p b_p} \dd x^{b_{p+1}} \wedge \dots \dd x^{b_n} \\
        & = \abs{\det \left[ \frac{\pa x}{\pa y} \right]} \frac{\pa y^{i_1}}{\pa x^{a_1}} \dots \frac{\pa y^{i_p}}{\pa x^{a_p}} \epsilon_{b_1 \dots b_n} \det \left[ \frac{\pa y}{\pa x} \right] \\
        & \qquad \qquad \qquad \qquad \qquad \qquad \qquad \times \frac{\sqg}{(n-p)!} g^{a_1 b_1} \dots g^{a_p b_p} \dd x^{b_{p+1}} \wedge \dots \dd x^{b_n} \\
        & = \sgn \left( \det \left[ \frac{\pa y}{\pa x} \right] \right) \frac{\pa y^{i_1}}{\pa x^{j_1}} \dots \frac{\pa y^{i_p}}{\pa x^{j_p}} \star (\dd x^{j_1} \wedge \dots \wedge \dd x^{j_p})
      \end{split}
    \end{equation*}
    where the following identity for $ \mt{M} \in \R^{n \times n} $ has been used:
    \begin{equation}
      \epsilon^{i_1 \dots i_n} M_{i_1 j_1} \dots M_{i_n j_n} = \epsilon_{j_1 \dots j_n} \det \mt{M}
      \label{eq:matrix-identity}
    \end{equation}
    Since the two charts are assumed to be have the same orientation, the sign function returns $ +1 $, and the relation found clearly shows that the factors coming from the transformation of $ \star (\dd y^{i_1} \wedge \dots \wedge \dd y^{i_p}) $ exactly cancel those in the transformation of $ \omega_{i_1 \dots i_p} $, leaving $ \star \omega $ unchanged.
  \end{proof}
\end{proofbox}

Therefore, the Hodge operator is a well-defined operator.

\begin{lemma}{Nilpotence of the Hodge operator}{}
  Let $ (\man,g) $ be a metric manifold, and let $ \omega \in \df{p} $. Then:
  \begin{equation}
    \star (\star \omega) = \sigma (-1)^{k (n-p)} \omega
    \label{eq:hodge-nilpotence}
  \end{equation}
  where $ \sigma \defeq \sgn \det [g_{ij}] $ is the \bclemma{parity} of the signature of $ g $ (basis-independent).
\end{lemma}

\begin{proofbox}
  \begin{proof}
    By linearity, it suffices to show the result only for basis $ p $-forms:
    \begin{equation*}
      \begin{split}
        \star (\star (\dd x^{i_1} \wedge \dots \wedge \dd x^{i_p}))
        & = \frac{\sqg}{(n-p)!} g^{i_1 j_1} \dots g^{i_p j_p} \epsilon_{j_1 \dots j_n} \star (\dd x^{j_{p+1}} \wedge \dots \wedge \dd x^{j_n}) \\
        & = \frac{\abs{\det [g_{ij}]}}{(n-p)! p!} g^{i_1 j_1} \dots g^{i_p j_p} g^{j_{p+1} a_1} \dots g^{j_n a_{n-p}} \epsilon_{j_1 \dots j_n} \epsilon_{a_1 \dots a_n} \times \\
        & \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad \times \dd x^{a_{n-p+1}} \wedge \dots \wedge \dd x^{a_n} \\
        & = \frac{\abs{\det [g_{ij}]}}{(n-p)! p!} \det [g_{ij}] \epsilon^{i_1 \dots i_p a_1 \dots a_{n-p}} \epsilon_{a_1 \dots a_n} \dd x^{a_{n-p+1}} \wedge \dots \wedge \dd x^{a_n} \\
        & = \frac{\sigma}{(n-p)! p!} (-1)^{p(n-p)} \epsilon^{a_1 \dots a_{n-p} i_1 \dots i_p} \epsilon_{a_1 \dots a_{n-p} j_1 \dots j_p} \dd x^{j_1} \wedge \dots \wedge \dd x^{j_p}
      \end{split}
    \end{equation*}
    where \eref{eq:matrix-identity} has been used. Now, the following identity for the generalized Kroncker delta can be used:
    \begin{equation}
      \delta^{i_1 \dots i_p}_{j_1 \dots j_p} = \frac{1}{m!} \epsilon^{a_1 \dots a_m i_1 \dots i_p} \epsilon_{a_1 \dots a_m j_1 \dots j_p}
    \end{equation}
    yielding:
    \begin{equation*}
      \begin{split}
        \star (\star (\dd x^{i_1} \wedge \dots \wedge \dd x^{i_p}))
        & = \frac{\sigma}{p!} (-1)^{p(n-p)} \delta^{i_1 \dots i_p}_{j_1 \dots j_p} \dd x^{j_1} \wedge \dots \wedge \dd x^{j_p} \\
        & = \sigma (-1)^{p(n-p)} \delta^{i_1}_{[j_1} \dots \delta^{i_p}_{j_p]} \dd x^{j_1} \wedge \dots \wedge \dd x^{j_p} \\
        & = \sigma (-1)^{p(n-p)} \dd x^{i_1} \wedge \dots \wedge \dd x^{i_p}
      \end{split}
    \end{equation*}
    which is the thesis.
  \end{proof}
\end{proofbox}

The nilpotence of the Hodge operator allows to define a simple expression for its inverse $ \star^{-1} : \df{p} \ra \df{n-p} $:
\begin{equation}
  \star^{-1} = \sigma (-1)^{p(n-p)} \star
\end{equation}

\subsubsection{Codifferential}

Using the Hodge operator, the $ L^2 $-inner product of $ \omega , \eta \in \dfs{p} $ can be written as:
\begin{equation}
  (\omega,\eta) = \int_\man \omega \wedge \star \eta
\end{equation}

\begin{proposition}{Codifferential}{}
  Let $ (\man,g) $ be a metric manifold without boundary. Then, the \bcprop{codifferential} is the unique operator $ \cdd : \df{p} \ra \df{p-1} $ such that, for any $ \omega \in \dfs{p-1} , \eta \in \dfs{p} $:
  \begin{equation}
    (\dd \omega , \eta) = (\omega , \cdd \eta)
  \end{equation}
  Its expression is:
  \begin{equation}
    \cdd = \sigma (-1)^{n(p+1)+1} \star \dd \star
  \end{equation}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    It suffices to consider $ \omega \in \dfs{p-1} , \eta \in \dfs{p} $. Then, $ \omega \wedge \star \eta \in \df{n-1} $, and by \eref{eq:ext-der-prod-rule}:
    \begin{equation*}
      \dd (\omega \wedge \star \eta) = \dd \omega \wedge \star \eta + (-1)^{p-1} \omega \wedge \dd \star \eta \in \df{n}
    \end{equation*}
    By Stokes' theorem, then, since $ \pa \man = \emptyset $:
    \begin{equation*}
      0 = \int_\man \dd (\omega \wedge \star \eta) = \int_\man \dd \omega \wedge \star \eta + (-1)^{p-1} \int_\man \omega \wedge \dd \star \eta
    \end{equation*}
    The first term is just $ (\dd \omega , \eta) $, while the second term is proportional to $ (\omega , \star \dd \star \eta) $. To determine the coefficient, note that $ \dd \star \eta \in \df{n-p+1} $, so using \eref{eq:hodge-nilpotence}:
    \begin{equation*}
      (\omega , \star \dd \star \eta) = \int_\man \omega \wedge \star (\star \dd \star \eta) = \sigma (-1)^{(n-p+1)(p-1)} \int_\man \omega \wedge \dd \star \eta
    \end{equation*}
    Thus, combining the two relations:
    \begin{equation*}
      (\omega , \star \dd \star \eta) = - \sigma (-1)^{(n-p)(p-1)} (\dd \omega , \eta)
    \end{equation*}
    Noting that $ \sigma^2 = 1 $ and:
    \begin{equation*}
      (n-p)(p-1) + 1 = n (p-1) - p (p - 1) + 1 \equiv n (p + 1) + 1 \quad (\mod 2)
    \end{equation*}
    yields the expression for $ \cdd $.
  \end{proof}
\end{proofbox}

For metric manifolds with boundary, the $ \cdd $ is not the adjoint of $ \dd $ in the operator-theoretic sense, since $ (\dd \omega , \eta) $ and $ (\omega , \cdd \eta) $ differ by a boundary term proportional to $ \int_{\pa \man} \omega \wedge \star \eta $.

\begin{definition}{Laplace--de Rham operator}{}
  Let $ (\man,g) $ be a metric manifold. The \bcdef{Laplace--Beltrami operator} is the operator $ \lapd : \df{p} \ra \df{p} $ defined as:
  \begin{equation}
    \lapd \defeq (\dd + \cdd)^2
  \end{equation}
\end{definition}

To simplify the expression for the Laplace--de Rham operator, it is crucial to extend the nilpotence $ \dd^2 = 0 $ to the codifferential.

\begin{lemma}[before upper = {\tcbtitle}]{Nilpotence of the codifferential}{}
  \begin{equation}
    \cdd \circ \cdd \equiv 0
  \end{equation}
\end{lemma}

\begin{proofbox}
  \begin{proof}
    By direct computation, given $ \omega \in \df{p} $, then $ \dd \star \omega \in \df{n-p+1} $:
    \begin{equation*}
      \begin{split}
        \cdd (\cdd \omega)
        & = \star \dd \star \star \dd \star \omega = \sigma (-1)^{(n-p+1)(p-1)} \star \dd \circ \dd \star \omega
      \end{split}
    \end{equation*}
    Using $ \dd \circ \dd \equiv 0 $ completes the proof.
  \end{proof}
\end{proofbox}

By the nilpotence of both $ \dd $ and $ \cdd $, the Laplace--de Rham operator can be expressed as:
\begin{equation}
  \lapd = \dd \cdd + \cdd \dd
\end{equation}

\begin{example}{Laplacian of functions}{}
  Let $ (\man,g) $ be an oriented metric manifold. Trivially, $ \cdd f = 0 $ for any $ f \in \cm = \dd{0} $, since $ \star f \in \df{n} $ is a top form, hence in a smooth chart $ (U,x^i) $ on $ \man $:
  \begin{equation*}
    \begin{split}
      \lapd f
      & = \cdd \dd f = \sigma (-1)^{2n+1} \star \dd \star (\pa_i f \dd x^i) = - \sigma \star \dd (\pa_i f \star \dd x^i) \\
      & = - \frac{\sigma}{(n-1)!} \star \dd \left( \pa_i f \sqg \, g^{i j_1} \epsilon_{j_1 \dots j_n} \dd x^{j_2} \wedge \dots \wedge \dd x^{j_n} \right) \\
      & = - \frac{\sigma}{(n-1)!} \star \left[ \pa_\ell (\sqg \, g^{ij_1} \pa_i f) \epsilon_{j_1 \dots j_n} \dd x^\ell \wedge \dd x^{j_2} \wedge \dots \wedge \dd x^{j_n} \right] = - \sigma \pa_\ell (\sqg \, g^{i\ell} \pa_i f) \star \volf
    \end{split}
  \end{equation*}
  Now, since $ \volfg = \star 1 $, by \eref{eq:hodge-nilpotence} it follows that $ \star \volfg = \sigma $, so inserting $ \volfg = \sqg \volf $ the final expression for the Laplacian of a function is found:
  \begin{equation}
    \lapd f = - \frac{1}{\sqg} \frac{\pa}{\pa x^i} \left( \sqg \, g^{ij} \frac{\pa f}{\pa x^j} \right)
    \label{eq:laplacian-function}
  \end{equation}
\end{example}

As hinted by \eref{eq:laplacian-function}, the Laplace--de Rham operator is the opposite of the ordinary Laplace--Beltrami operator: $ \lapd = - \lap $.

\subsubsection{Harmonic forms and de Rham cohomology}

The Laplace--de Rham operator is the link connecting Hodge theory to the de Rham cohomology. In particular, it is possible to define a new space of differential forms.

Let $ (\man,g) $ be a metric manifold: $ \omega \in \df{p} $ is said to be \bctxt{harmonic} if $ \lapd \omega = 0 $. The space of such harmonic $ p $-forms is denoted by $ \hrm{p} \ssq \df{p} $. Note that, on oriented Riemannian manifolds, harmonic forms are both closed and co-closed, since:
\begin{equation*}
  0 = (\omega , \lapd \omega) = (\dd \omega , \dd \omega) + (\cdd \omega , \cdd \omega)
\end{equation*}
As the $ L^2 $-inner product is positive-definite in the Riemannian case, this relation holds if and only if both $ \dd \omega = 0 $ and $ \cdd \omega = 0 $. The Riemannian case also has additional properties.

\begin{theorem}{Harmonic decomposition of forms}{harmonic-decomposition}
  Let $ (\man,g) $ be a compact Riemannian manifold. Then, for all $ p \in \N_0 $:
  \begin{equation}
    \df{k} = \im \dd_{p-1} \oplus \im \cdd_{p+1} \oplus \hrm{p}
  \end{equation}
\end{theorem}

\begin{proofbox}
  \begin{proof}
    See \S 5 of \cite{voisin}.
  \end{proof}
\end{proofbox}

As in \secref{ssec:de-rham-cohomology-groups}, the form-degree of $ \dd_p : \df{p} \ra \df{p+1} $ and $ \cdd_p : \df{p} \ra \df{p-1} $ has been made explicit for convenience. With this direct sum, on a compact Riemannian manifold $ \man $, any $ \omega \in \df{p} $ can be uniquely decomposed as $ \omega = \dd \alpha + \cdd \beta + \gamma $, with $ \alpha \in \df{p-1} $, $ \beta \in \df{p+1} $ and $ \gamma \in \df{p} $.

\begin{theorem}{Hodge's theorem}{}
  Let $ (\man,g) $ be a compact Riemannian manifold. Then, for all $ p \in \N_0 $:
  \begin{equation}
    \hp{p}(\man) \cong \hrm{p}
  \end{equation}
\end{theorem}

\begin{proofbox}
  \begin{proof}
    Since harmonic $ p $-forms are closed, $ \hrm{p} \ssq \zp{p}(\man) $. Moreover, \tref{th:harmonic-decomposition} implies that $ \hrm{p} \cap \im \dd_{p-1} = \emptyset $, hence $ \hrm{p} \ssq \hp{p}(\man) $.

    WTS any equivalence class $ [\omega] \in \hp{p}(\man) $ can be represented by a harmonic $ p $-form. By \tref{th:harmonic-decomposition}, let $ \omega = \dd \alpha + \cdd \beta + \gamma $ be the unique decomposition of $ \omega \in \zp{p}(\man) $, with $ \alpha \in \df{p-1} $, $ \beta \in \df{p+1} $ and $ \gamma \in \hrm{p} $. By definition $ \dd \omega = 0 $, but $ \dd^2 = 0 $ and $ \dd \gamma = 0 $ since harmonic forms are closed, hence $ \dd \omega = \dd \cdd \beta = 0 $ and:
    \begin{equation*}
      0 = (\dd \omega , \beta) = (\dd \cdd \beta , \beta) = (\cdd \beta , \cdd \beta)
    \end{equation*}
    Since the $ L^2 $-inner product is positive-definite on Riemannian manifolds, this implies $ \cdd \beta = 0 $, therefore $ \omega = \dd \alpha + \gamma $. By definition $ \hp{p}(\man) \defeq \zp{p}(\man) / \bp{p}(\man) $, so $ [\omega] = \gamma $ as sought.
  \end{proof}
\end{proofbox}

This is the fundamental result of Hodge theory: each de Rham cohomology class contains exactly one harmonic form.

\section{Cartan formalism}

Although typically calculations are carried on a coordinate basis $ \{e_\mu\} = \{\pa_\mu\} $, it is possible to choose bases without such an interpretation. For example, a linear combination of a coordinate basis is not in general a coordinate basis itself:
\begin{equation}
  e_a = \tensor{e}{_a^\mu} \pa_\mu
\end{equation}
with $ a = 1, \dots, n $ like $ \mu = 1, \dots, n $.
A particularly useful non-coordinate basis is one such that:
\begin{equation}
  g(e_a, e_b) = g_{\mu \nu} \tensor{e}{_a^\mu} \tensor{e}{_b^\nu} = \eta_{ab}
  \label{eq:vielbeins-def}
\end{equation}
The components $ \tensor{e}{_a^\mu} $ are called \bctxt{vielbeins} or \textit{tetrads}. In this non-coordinate system, the manifold looks flat (or, at least, its patch covered by the given chart). In the following computations, Greek indices are raised/lowered by the metric $ g_{\mu \nu} $, while Latin indices by the metric $ \eta_{ab} $.

The vielbeins are not unique: given $ \{\tensor{e}{_a^\mu}\} $, it is always possible to find a new vielbeins like:
\begin{equation}
  \tensor{\tilde{e}}{_a^\mu} = \tensor{e}{_b^\mu} \tensor{(\Lambda^{-1})}{^b_a}
  \label{eq:vielbeins-trans}
\end{equation}
The transformation matrix must satisfy the condition imposed by \eref{eq:vielbeins-def}, i.e.:
\begin{equation}
  \tensor{\Lambda}{^c_a} \tensor{\Lambda}{^d_b} \eta_{cd} = \eta_{ab}
\end{equation}
These are \bctxt{local Lorentz transformations}, because this condition defines an $ \lorgn $ transformation\footnotemark, but $ \Lambda $ is now allowed to vary over the manifold. The dual basis of 1-forms $ \{\theta^a\} $ is defined by $ \theta^a(e_b) = \delta^a_b $. The relation to the coordinate basis is:
\begin{equation}
  \theta^a = \tensor{e}{^a_\mu} \dd x^\mu
\end{equation}
where the coefficients satisfy $ \tensor{e}{^a_\mu} \tensor{e}{_b^\mu} = \delta^a_b $ and $ \tensor{e}{^a_\mu} \tensor{e}{_a^\nu} = \delta_\mu^\nu $.
The metric is a tensor, therefore $ g = g_{\mu \nu} \dd x^\mu \otimes \dd x^\nu = \eta_{ab} \theta^a \otimes \theta^b $, thus it is related to the vielbeins by:
\begin{equation}
  g_{\mu \nu} = \tensor{e}{^a_\mu} \tensor{e}{^b_\nu} \eta_{ab}
\end{equation}

\footnotetext{To be precise, this condition defines a $ \On{1,n} $ transformation on $ \R^{1,n} $, but this orthogonal group has four disconnected components: the $ \lorgn $ component is the one defined by $ \det \Lambda = 1 $ and $ \tensor{\Lambda}{^0_0} \ge 1 $, while the other three components can be obtained by composition of $ \lorgn $ with the possible combinations of parity reversal and time reversal. For a more detailed account on the Lorentz group, see \S 1.2 of \cite{leonardo}.}

\subsection{Connection 1-form}

On a vielbein basis $ \{e_a\} $, connection components are computed in the usual way:
\begin{equation}
  \na_{e_c} e_b \equiv \omega^a_{cb} e_a
  \label{eq:connection-vielbeins}
\end{equation}
where the $ n^3 $ functions $ \omega^a_{cb} $ are just the connection coefficients $ \Gamma^a_{cb} $ on the vielbein basis.

\begin{definition}{Connection 1-form}{}
  On a metric manifold with vielbeins, the \bcdef{connection 1-form} is defined as:
  \begin{equation}
    \tensor{\omega}{^a_b} \defeq \omega^a_{cb} \theta^c
    \label{eq:connection-1-form}
  \end{equation}
\end{definition}

Note that these are really $ n^2 $ 1-forms, according to values of $ a,b = 1, \dots, n $. This is also known as the \textit{spin connection}, due to its relationship to spinors in curved spacetime.
Then, the connection in the vielbein basis is easily defined:
\begin{equation*}
  \na_X Y = X(Y^a) e_a + Y^a X^b \na_{e_b} e_a = X^b [e_b(Y^a) + \omega^a_{bc} Y^c] e_a
\end{equation*}
Hence, setting the notation $ \na_b Y^a \equiv (\na_{e_b} Y)^a $, it is clear that the connection takes the form:
\begin{equation}
  \na_b Y^a = e_b(Y^a) + \omega^a_{bc} Y^c
  \label{eq:vielbein-cov-der}
\end{equation}
In general, $ e_b(Y^a) = \tensor{e}{_b^\mu} \pa_\mu Y^a $. Using \eref{eq:cov-der-form-aux}, the covariant derivative of forms is found:
\begin{equation}
  \na_b \zeta_a = e_b(\zeta_a) - \omega^c_{ba} \zeta_c
\end{equation}
Finally, for tensors of general rank:
\begin{equation}
  \na_c \tensor{T}{^{a_1 \dots a_r}_{b_1 \dots b_s}} = e_c(\tensor{T}{^{a_1 \dots a_r}_{b_1 \dots b_s}}) + \sum_{j = 1}^r \omega^{a_j}_{c d_j} \tensor{T}{^{a_1 \dots d_j \dots a_n}_{b_1 \dots b_s}} - \sum_{j = 1}^s \omega^{d_j}_{c b_j} \tensor{T}{^{a_1 \dots a_n}_{b_1 \dots d_j \dots b_n}}
\end{equation}

\begin{lemma}{Transformation law}{}
  Given a local Lorentzian transformation $ \Lambda $ acting as in \eref{eq:vielbeins-trans}:
  \begin{equation}
    \tensor{\tilde{\omega}}{^a_b} = \tensor{\Lambda}{^a_c} \tensor{\omega}{^c_d} \tensor{(\Lambda^{-1})}{^d_b} + \tensor{\Lambda}{^a_c} \dd \tensor{(\Lambda^{-1})}{^c_b}
    \label{eq:con-1-form-trans}
  \end{equation}
\end{lemma}

\begin{proofbox}
  \begin{proof}
    For ease of notation, rename the transformation in \eref{eq:vielbeins-trans} as $ \Lambda^{-1} \equiv \mt{L} $. Then, from the orthogonality condition $ \theta^a (e_b) = \delta^a_b $, it is clear that:
    \begin{equation*}
      \tilde{e}_a = \tensor{\mt{L}}{^b_a} e_b
      \qquad \qquad
      \tilde{\theta}^a = \tensor{(\mt{L}^{-1})}{^a_b} \theta^b
    \end{equation*}
    Then, the connection coefficients can be computed as:
    \begin{equation*}
      \begin{split}
        \tilde{\omega}^a_{cb} \tilde{e}_a
        & = \na_{\tilde{e}_c} \tilde{e}_b = \tensor{\mt{L}}{^d_c} \na_{e_d} (\tensor{\mt{L}}{^f_b} e_f) = \tensor{\mt{L}}{^d_c} \left[ \na_{e_d} \tensor{\mt{L}}{^f_b} e_f + \tensor{\mt{L}}{^f_b} \na_{e_d} e_f \right] \\
        & = \tensor{\mt{L}}{^d_c} \left[ \na_{e_d} \tensor{\mt{L}}{^f_b} e_f + \tensor{\mt{L}}{^f_b} \omega^g_{df} e_g \right] = \tensor{\mt{L}}{^d_c} \left[ \na_{e_d} \tensor{\mt{L}}{^f_b} + \tensor{\mt{L}}{^g_b} \omega^f_{dg} \right] \tensor{(\mt{L}^{-1})}{^a_f} \tilde{e}_a
      \end{split}
    \end{equation*}
    Finally, the transformed connection 1-form is:
    \begin{equation*}
      \begin{split}
        \tensor{\tilde{\omega}}{^a_b}
        & = \tilde{\omega}^a_{cb} \tilde{\theta^c} = \tensor{\mt{L}}{^d_c} \left[ \na_{e_d} \tensor{\mt{L}}{^f_b} + \tensor{\mt{L}}{^g_b} \omega^f_{dg} \right] \tensor{(\mt{L}^{-1})}{^a_f} \tensor{(\mt{L}^{-1})}{^c_h} \theta^h \\
        & = \left[ \na_{e_d} \tensor{\mt{L}}{^f_b} + \tensor{\mt{L}}{^g_b} \omega^f_{dg} \right] \tensor{(\mt{L}^{-1})}{^a_f} \theta^d = \tensor{(\mt{L}^{-1})}{^a_c} \left[ \dd x^\mu \na_\mu \tensor{\mt{L}}{^c_b} + \tensor{\omega}{^c_d} \tensor{\mt{L}}{^d_b} \right]
      \end{split}
    \end{equation*}
    Recalling that $ \dd f = \dd x^\mu \na_\mu f $, with $ f \in \cm $, and that $ \Lambda = \mt{L}^{-1} $ yields the thesis.
  \end{proof}
\end{proofbox}

The second term reflects the second term in \pref{prop:gamma-non-tens}, which involves the derivative of the coordinate transformation.

As for the connection coefficients, it is possible to compute the torsion tensor on a vielbein basis as $ \tensor{T}{^a_{bc}} = \tors(\theta^a , e_b , e_c) $, which retains the antisimmetry $ \tensor{T}{^a_{bc}} = - \tensor{T}{^a_{cb}} $.

\begin{definition}{Torsion 2-form}{}
  On a metric manifold with vielbeins, the \bcdef{torsion 2-form} is defined as:
  \begin{equation}
    T^a \defeq \frac{1}{2} \tensor{T}{^a_{bc}} \theta^b \wedge \theta^c
  \end{equation}
\end{definition}

This is a vector-valued 2-form, thus $ T^a $ has $ n $ independent 2-forms. Like \eref{eq:torsion-comp}, the torsion 2-form is linked to the connection 1-form by the first Cartan structure equation.

\begin{theorem}[before upper = {\tcbtitle}]{First Cartan structure equation}{}
  \begin{equation}
    T^a = \dd\theta^a + \tensor{\omega}{^a_b} \wedge \theta^b
  \end{equation}
\end{theorem}

\begin{proofbox}
  \begin{proof}
    From \eref{eq:connection-vielbeins} $ \omega^a_{cb} = \tensor{e}{^a_\rho} \tensor{e}{_c^\mu} \na_\mu \tensor{e}{_b^\rho} $, thus inserting \eref{eq:covariant-derivative-vector} and \eref{eq:connection-1-form}:
    \begin{equation*}
      \begin{split}
        \tensor{\omega}{^a_b} \wedge \theta^b
        & = \omega^a_{cb} \left( \tensor{e}{^c_\mu} \dd x^\mu \right) \wedge \left( \tensor{e}{^b_\nu} \dd x^\nu \right) = \tensor{e}{^a_\rho} \tensor{e}{_c^\lambda} \left( \pa_\lambda \tensor{e}{_b^\rho} + \tensor{e}{_b^\sigma} \Gamma^\rho_{\lambda \sigma} \right) \left( \tensor{e}{^c_\mu} \dd x^\mu \right) \wedge \left( \tensor{e}{^b_\nu} \dd x^\nu \right) \\
        & = \tensor{e}{^a_\rho} \tensor{e}{_c^\lambda} \tensor{e}{^c_\mu} \tensor{e}{^b_\nu} \left( \pa_\lambda \tensor{e}{_b^\rho} + \tensor{e}{_b^\sigma} \Gamma^\rho_{\lambda \sigma} \right) \dd x^\mu \wedge \dd x^\nu = \tensor{e}{^a_\rho} \tensor{e}{^b_\nu} \left( \pa_\mu \tensor{e}{_b^\rho} + \tensor{e}{_b^\sigma} \Gamma^\rho_{\mu \sigma} \right) \dd x^\mu \wedge \dd x^\nu \\
      \end{split}
    \end{equation*}
    Now, $ \tensor{e}{^b_\nu} \tensor{e}{_b^\rho} = \delta_\nu^\rho $ implies $ \tensor{e}{^b_\nu} \pa_\mu \tensor{e}{_b^\rho} = - \tensor{e}{_b^\rho} \pa_\mu \tensor{e}{^n_\nu} $, hence:
    \begin{equation*}
      \begin{split}
        \tensor{\omega}{^a_b} \wedge \theta^b
        & = \left[ - \tensor{e}{^a_\rho} \tensor{e}{_b^\rho} \pa_\mu \tensor{e}{^b_\nu} + \tensor{e}{^a_\rho} \tensor{e}{^b_\nu} \tensor{e}{_b^\sigma} \Gamma^\rho_{\mu \sigma} \right] \dd x^\mu \wedge \dd x^\nu \\
        & = - \pa_\mu \tensor{e}{^a_\nu} \dd x^\mu \wedge \dd x^\nu + \tensor{e}{^a_\rho} \Gamma^\rho_{\mu \nu} \dd x^\mu \wedge \dd x^\nu = -\dd\theta^a + \tensor{e}{^a_\rho} \Gamma^\rho_{[\mu \nu]} \dd x^\mu \wedge \dd x^\nu
      \end{split}
    \end{equation*}
    Now, from the multilinearity of the torsion tensor:
    \begin{equation*}
      \begin{split}
        T^a
        & = \frac{1}{2} \tensor{T}{^a_{bc}} \theta^b \wedge \theta^c = \frac{1}{2} \tensor{e}{^a_\rho} \tensor{e}{_b^\alpha} \tensor{e}{_c^\beta} \tensor{e}{^b_\mu} \tensor{e}{^c_\nu} \tensor{T}{^\rho_{\beta \gamma}} \dd x^\mu \wedge \dd x^\nu = \frac{1}{2} \tensor{e}{^a_\rho} \tensor{T}{^\rho_{\mu \nu}} \dd x^\mu \wedge \dd x^\nu
      \end{split}
    \end{equation*}
    Writing $ \tensor{T}{^\rho_{\mu \nu}} = 2 \Gamma^\rho_{[\mu \nu]} $ completes the proof.
  \end{proof}
\end{proofbox}

This construction can be generalized to arbitrary vector-valued 1-forms, defining the so-called \bctxt{covariant exterior derivative}:
\begin{equation}
  \dcov \zeta^a \defeq \dd \zeta^a + \tensor{\omega}{^a_b} \wedge \zeta^b
\end{equation}
This is a map $ \dcov : \df{p} \ra \df{p+1} $, like the exterior derivative, but it has an important invariance property.

\begin{lemma}{Lorentz invariance}{form-lor-inv}
  The covariant exterior derivative transformations under local Lorentz as:
  \begin{equation}
    \zeta^a \mapsto \tensor{\Lambda}{^a_b} \zeta^b
    \quad \implies \quad
    \dcov \zeta^a \mapsto \tensor{\Lambda}{^a_b} \dcov \zeta^b
  \end{equation}
\end{lemma}

\begin{proofbox}
  \begin{proof}
    Denoting transformed quantities with a tilde and recalling \eref{eq:con-1-form-trans}:
    \begin{equation*}
      \begin{split}
        \tilde{\dcov} \tilde{\zeta^a}
        & = \dd \tilde{\zeta^a} + \tensor{\tilde{\omega}}{^a_b} \wedge \tilde{\zeta}^a \\
        & = \dd \tensor{\Lambda}{^a_b} \wedge \zeta^b + \tensor{\Lambda}{^a_b} \dd \zeta^b + \tensor{\Lambda}{^a_c} \tensor{(\Lambda^{-1})}{^d_b} \tensor{\Lambda}{^b_f} \tensor{\omega}{^c_d} \wedge \zeta^f + \tensor{\Lambda}{^a_c} \tensor{\Lambda}{^b_d} \dd \tensor{(\Lambda^{-1})}{^c_b} \wedge \zeta^d \\
        & = \dd \tensor{\Lambda}{^a_b} \wedge \zeta^b + \tensor{\Lambda}{^a_b} \dd \zeta^b + \tensor{\Lambda}{^a_c} \tensor{\omega}{^c_d} \wedge \zeta^d - \tensor{\Lambda}{^a_c} \tensor{(\Lambda^{-1})}{^c_b} \dd \tensor{\Lambda}{^b_d} \wedge \zeta^d = \tensor{\Lambda}{^a_b} \left[ \dd \zeta^b + \tensor{\omega}{^b_c} \zeta^c \right]
      \end{split}
    \end{equation*}
    where $ \tensor{(\Lambda^{-1})}{^c_b} \tensor{\Lambda}{^b_d} = \delta^c_d \implies \tensor{\Lambda}{^b_d} \dd \tensor{(\Lambda^{-1})}{^c_b} = - \tensor{(\Lambda^{-1})}{^c_b} \dd \tensor{\Lambda}{^b_d} $ has been used.
  \end{proof}
\end{proofbox}

The exterior covariant derivative can be used to extend the covariant derivative to vector-valued 1-forms (using \eref{eq:vielbein-cov-der}):
\begin{equation}
  \dcov \zeta^a = \pa_\mu \zeta^a \tensor{e}{_c^\mu} \wedge \theta^c + \omega^a_{cb} \zeta^b \wedge \theta^c = \left[ e_c(\zeta^a) + \omega^a_{cb} \zeta^b \right] \wedge \theta^c \equiv \tensor{\zeta}{^a_{;c}} \wedge \theta^c
\end{equation}

The same construction for the exterior covariant derivative can be generalized to tensor-valued $ p $-forms as:
\begin{equation}
  \dcov \tensor{\zeta}{^{a_1 \dots a_r}_{b_1 \dots b_s}} = \dd \tensor{\zeta}{^{a_1 \dots a_r}_{b_1 \dots b_s}} + \sum_{j = 1}^r \tensor{\omega}{^{a_j}_c} \wedge \tensor{\zeta}{^{a_1 \dots c \dots a_r}_{b_1 \dots b_s}} - \sum_{j = 1}^s \tensor{\omega}{^c_{b_j}} \wedge \tensor{\zeta}{^{a_1 \dots a_r}_{b_1 \dots c \dots b_s}}
  \label{eq:dcov}
\end{equation}
It is then trivial to show, using \lref{lemma:form-lor-inv}, that a local Lorentz transformation acts as:
\begin{equation}
  \tensor{\zeta}{^{a_1 \dots a_r}_{b_1 \dots b_s}} \mapsto \tensor{\Lambda}{^{a_1}_{c_1}} \dots \tensor{\Lambda}{^{a_r}_{c_r}} \tensor{\zeta}{^{c_1 \dots c_r}_{d_1 \dots d_s}} \tensor{(\Lambda^{-1})}{^{d_1}_{b_1}} \dots \tensor{(\Lambda^{-1})}{^{d_s}_{b_s}}
\end{equation}
\begin{equation}
  \dcov\tensor{\zeta}{^{a_1 \dots a_r}_{b_1 \dots b_s}} \mapsto \tensor{\Lambda}{^{a_1}_{c_1}} \dots \tensor{\Lambda}{^{a_r}_{c_r}} \dcov\tensor{\zeta}{^{c_1 \dots c_r}_{d_1 \dots d_s}} \tensor{(\Lambda^{-1})}{^{d_1}_{b_1}} \dots \tensor{(\Lambda^{-1})}{^{d_s}_{b_s}}
\end{equation}
where the transformation rule $ \zeta_a \mapsto \zeta_b \tensor{(\Lambda^{-1})}{^b_a} $ follows from the requirement of $ \zeta^a \xi_a $ being Lorentz-invariant (as for usual Lorentz transformation).

\subsubsection{Levi--Civita spin connection}

If the connection is torsionless, the first structure equation reduces to:
\begin{equation}
  \dcov \theta^a = 0
  \label{eq:first-cartan-stru-eq}
\end{equation}
For a Levi--Civita connection, a stronger result holds.

\begin{proposition}{}{}
  On a metric manifold with a Levi--Civita connection:
  \begin{equation}
    \omega_{ab} = -\omega_{ba}
    \label{eq:connection-form-symm}
  \end{equation}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Consider the vielbein condition \eref{eq:vielbeins-def}. For a Levi--Civita connection, compatibility with the metric implies:
    \begin{equation*}
      \begin{split}
        0
        & = \na_X (g_{\mu \nu} \tensor{e}{_a^\mu} \tensor{e}{_b^\nu}) = X^c g_{\mu \nu} \left( \na_{e_c} \tensor{e}{_a^\mu} \tensor{e}{_b^\nu} + \tensor{e}{_a^\mu} \na_{e_c} \tensor{e}{_b^\nu} \right) \\
        & = X^c g_{\mu \nu} \left( \omega^d_{ca} \tensor{e}{_d^\mu} \tensor{e}{_b^\nu} + \omega^d_{cb} \tensor{e}{_a^\mu} \tensor{e}{_d^\nu} \right) = X^c \left( \omega^d_{ca} \eta_{db} + \omega^d_{cb} \eta_{ad} \right)
      \end{split}
    \end{equation*}
    Note that $ \tensor{\omega}{^a_b}(X) = \omega^a_{cb} \theta^c(X) = \omega^a_{cb} X^c $, so that:
    \begin{equation*}
      0 = \tensor{\omega}{^d_a}(X) \eta_{db} + \tensor{\omega}{^d_b}(X) \eta_{ad} = \omega_{ba}(X) + \omega_{ab}(X)
    \end{equation*}
    By the arbitrariness of $ X $, it follows the thesis.
  \end{proof}
\end{proofbox}

\eeref{eq:first-cartan-stru-eq}{eq:connection-form-symm} allow to quickly compute the spin connection of a Levi--Civita connection, as they uniquely define it. Indeed, by antisymmetry $ \omega_{ab} $ has $ \frac{1}{2} n (n-1) $ independent 1-forms, i.e. $ \frac{1}{2} n^2 (n-1) $ independent components. On the other hand, the first structure equation constraints a vector-valued 2-form, which has $ n $ 2-forms with $ \frac{1}{2} n (n-1) $ independent components each, thus posing $ \frac{1}{2} n^2 (n-1) $ constraints: this is the same number as the independent components of $ \omega_{ab} $, therefore the spin connection is uniquely determined.

The Levi--Civita connection is the unique connection such that \eeref{eq:first-cartan-stru-eq}{eq:connection-form-symm} both hold. Moreover, for a Levi--Civita connection $ \dcov (\zeta^a \xi_a) = \dd (\zeta^a \xi_a) $ and $ \dcov \eta_{ab} = 0 $, since $ \tensor{\omega}{_a^b} = - \tensor{\omega}{^b_a} $.

\subsection{Curvature 2-form}

As for the torsion tensor, it is possible to compute the curvature tensor on a vielbein basis as $ \tensor{R}{^a_{bcd}} = \riem(\theta^a , e_b , e_c , e_d) $, which retains the symmetries \pref{prop:riemann-symmetries}.

\begin{definition}{Curvature 2-form}{}
  On a metric manifold with vielbeins, the \bcdef{curvature 2-form} is defined as:
  \begin{equation}
    \tensor{\Omega}{^a_b} \defeq \frac{1}{2} \tensor{R}{^a_{bcd}} \theta^c \wedge \theta^d
    \label{eq:curvature-2-def}
  \end{equation}
\end{definition}

The curvature 2-form is a tensor-valued 2-form, hence $ \tensor{\Omega}{^a_b} $ has $ n^2 $ 2-forms, i.e. $ \frac{1}{2} n^3 (n-1) $ independent components (the same number as the curvature tensor, as expected). It is possible to derive a structure equation for the curvature 2-form too.

\begin{theorem}[before upper = {\tcbtitle}]{Second Cartan structure equation}{}
  \begin{equation}
    \tensor{\Omega}{^a_b} = \dd \tensor{\omega}{^a_b} + \tensor{\omega}{^a_c} \wedge \tensor{\omega}{^c_b}
    \label{eq:second-cartan-stru-eq}
  \end{equation}
\end{theorem}

\begin{proofbox}
  \begin{proof}
    Recalling \eref{eq:riemann-operator-def}, it is possible to write:
    \begin{equation*}
      \tensor{R}{^a_{bcd}} e_a = \riem(e_c , e_d) e_b = \na_{e_c} \na_{e_d} e_b - \na_{e_d} \na_{e_c} e_b - \na_{[e_c , e_d]} e_b
    \end{equation*}
    In general, by linearity of the connection, $ \na_X e_b = \tensor{\omega}{^c_b}(X) e_c $, hence (using $ \na_X f = X(f) $):
    \begin{equation*}
      \begin{split}
        \tensor{R}{^a_{bcd}} e_a
        & = \na_{e_c} [\omega^a_{db} e_a] - \na_{e_d} [\omega^a_{cb} e_a] - \tensor{\omega}{^a_b}([e_c , e_d]) e_a \\
        & = e_c(\omega^a_{db}) e_a + \omega^f_{db} \omega^a_{cf} e_a - e_d(\omega^a_{cb}) e_a - \omega^f_{cb} \omega^a_{df} e_a - \tensor{\omega}{^a_b}([e_c , e_d]) e_a \\
      \end{split}
    \end{equation*}

    \begin{lemma}[before upper = {\tcbtitle}, borderline = {1pt}{0pt}{blue!35!black}]{}{connection-ext-der}
      \begin{equation}
        \dd \tensor{\omega}{^m_n}(e_a , e_b) = e_a(\omega^m_{bn}) - e_b(\omega^m_{an}) - \tensor{\omega}{^m_n}([e_a , e_b])
      \end{equation}
    \end{lemma}

    \begin{proofbox}
      \begin{proof}
        Writing $ \pa_\mu = \tensor{e}{^a_\mu} e_a $ and $ \dd x^\mu = \tensor{e}{_a^\mu} \theta^a $, by linearity of the connection:
        \begin{equation*}
          \dd \tensor{\omega}{^m_n} = \na_{e_a} (\omega^m_{cn} \tensor{e}{^c_\mu}) \tensor{e}{_b^\mu} \theta^a \wedge \theta^b = \frac{1}{2} \left[ \na_{e_a} (\omega^m_{cn} \tensor{e}{^c_\mu}) \tensor{e}{_b^\mu} - \na_{e_b} (\omega^m_{cn} \tensor{e}{^c_\mu}) \tensor{e}{_a^\mu} \right] \theta^a \wedge \theta^b
        \end{equation*}
        Since $ \omega^m_{cn} \tensor{e}{^c_\mu} $ is a scalar function, using the Leibniz rule:
        \begin{equation*}
          \dd \tensor{\omega}{^m_n}(e_a , e_b) = e_a(\omega^m_bn) - e_b(\omega^m_an) + \omega^m_{cn} \left[ e_a(\tensor{e}{^c_\mu}) \tensor{e}{_b^\mu} - e_b(\tensor{e}{^c_\mu}) \tensor{e}{_a^\mu} \right]
        \end{equation*}
        Now, consider the action of the commutator of vielbeins on $ \cm $:
        \begin{equation*}
          \begin{split}
            [e_a , e_b](f)
            & \defeq e_a(e_b(f)) - e_b(e_a(f)) = \tensor{e}{_a^\mu} \pa_\mu (\tensor{e}{_b^\nu} \pa_\nu f) - \tensor{e}{_b^\nu} \pa_\nu (\tensor{e}{_a^\mu} \pa_\mu f) \\
            & = \tensor{e}{_a^\mu} \pa_\mu \tensor{e}{_b^\nu} \pa_\nu f - \tensor{e}{_b^\nu} \pa_\nu \tensor{e}{_a^\mu} \pa_\mu f = \left[ e_a(\tensor{e}{_b^\mu}) \tensor{e}{^c_\mu} - e_b(\tensor{e}{_a^\mu}) \tensor{e}{^c_\mu} \right] e_c(f) \\
          \end{split}
        \end{equation*}
        As $ \tensor{e}{_b^\mu} \tensor{e}{^c_\mu} = \delta^c_b $, by the Leibniz rule $ e_a(\tensor{e}{_b^\mu}) \tensor{e}{^c_\mu} + \tensor{e}{_b^\mu} e_a(\tensor{e}{^c_\mu}) = 0 $, hence:
        \begin{equation}
          ([e_a , e_b])^c = \tensor{e}{_a^\mu} e_b(\tensor{e}{^c_\mu}) - \tensor{e}{_b^\mu} e_a(\tensor{e}{^c_\mu})
        \end{equation}
        which, inserted in the expression for $ \dd \tensor{\omega}{^m_n}(e_a , e_b) $, yields the thesis.
      \end{proof}
    \end{proofbox}

    Using \lref{lemma:connection-ext-der}, the above expression becomes:
    \begin{equation*}
      \tensor{R}{^a_{bcd}} = \dd \tensor{\omega}{^a_b}(e_c , e_d) + \omega^f_{db} \omega^a_{cf} - \omega^f_{cb} \omega^a_{df}
    \end{equation*}
    But $ \tensor{R}{^a_{bcd}} = \tensor{\Omega}{^a_b}(e_c , e_d) $, while:
    \begin{equation*}
      \tensor{\omega}{^a_f} \wedge \tensor{\omega}{^f_b} = \omega^a_{cf} \omega^f_{db} \theta^c \wedge \theta^d = \frac{1}{2} \left[ \omega^f_{db} \omega^a_{cf} - \omega^f_{cb} \omega^a_{df} \right] \theta^c \wedge \theta^d
    \end{equation*}
    thus the proof is complete.
  \end{proof}
\end{proofbox}

Combining the two structure equations, it is possible to find simple expressions for the exterior covariant derivative of the torsion and curvature 2-forms.

\begin{proposition}[before upper = {\tcbtitle}]{}{}
  \begin{equation}
    \dcov T^a = \tensor{\Omega}{^a_b} \wedge e^b
    \qquad \qquad
    \dcov \tensor{\Omega}{^a_b} = 0
    \label{eq:cov-ext-der-torsion-curvature}
  \end{equation}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    From \eref{eq:first-cartan-stru-eq}, taking the exterior derivative and using \eref{eq:ext-der-prod-rule}:
    \begin{equation*}
      \dd T^a = \dd \tensor{\omega}{^a_b} \wedge \theta^b - \tensor{\omega}{^a_b} \wedge \dd \theta^b = \dd \tensor{\omega}{^a_b} \wedge \theta^b - \tensor{\omega}{^a_b} \wedge (T^b - \tensor{\omega}{^b_c} \wedge \theta^c)
    \end{equation*}
    Rearranging the terms and inserting \eref{eq:second-cartan-stru-eq}:
    \begin{equation*}
      \dd T^a + \tensor{\omega}{^a_b} \wedge T^a = \left( \dd \tensor{\omega}{^a_b} + \tensor{\omega}{^a_c} \wedge \tensor{\omega}{^c_b} \right) \wedge \theta^b
      \quad \implies \quad
      \dcov T^a = \tensor{\Omega}{^a_b} \wedge e^b
    \end{equation*}
    Now, take the exterior derivative of \eref{eq:second-cartan-stru-eq} and apply \eref{eq:ext-der-prod-rule}:
    \begin{equation*}
      \dd \tensor{\Omega}{^a_b} = \dd \tensor{\omega}{^a_c} \wedge \tensor{\omega}{^c_b} - \tensor{\omega}{^a_c} \wedge \dd \tensor{\omega}{^c_b} = \left( \tensor{\Omega}{^a_c} - \tensor{\omega}{^a_d} \wedge \tensor{\omega}{^d_c} \right) \wedge \tensor{\omega}{^c_b} - \tensor{\omega}{^a_c} \wedge \left( \tensor{\Omega}{^c_b} - \tensor{\omega}{^c_d} \wedge \tensor{\omega}{^d_b} \right)
    \end{equation*}
    The second and fourth terms cancel by renaming the mute indices, thus:
    \begin{equation*}
      0 = \dd \tensor{\Omega}{^a_b} + \tensor{\omega}{^a_c} \wedge \tensor{\Omega}{^c_b} - \tensor{\Omega}{^a_c} \wedge \tensor{\omega}{^c_b} = \dcov \tensor{\Omega}{^a_b}
    \end{equation*}
    since $ \tensor{\Omega}{^a_c} \wedge \tensor{\omega}{^c_b} = \tensor{\omega}{^c_b} \wedge \tensor{\Omega}{^a_c} $ by \eref{eq:form-skew-symmetry}.
  \end{proof}
\end{proofbox}

\subsubsection{Bianchi identities}

The Bianchi identities are useful identities which hold for a torsionless connection and put symmetry constraints on the curvature tensor.

\begin{theorem}{Bianchi identities}{}
  For a torsionless connection:
  \begin{equation}
    \tensor{R}{^a_{[bcd]}} = 0
    \qquad \qquad
    \tensor{R}{^a_{b[cd;k]}} = 0
    \label{eq:bianchi-vielbein}
  \end{equation}
\end{theorem}

\begin{proofbox}
  \begin{proof}
    Setting $ T^a \equiv 0 $ in the first \eref{eq:cov-ext-der-torsion-curvature}:
    \begin{equation*}
      0 = \tensor{\Omega}{^a_b} \wedge \theta^b = \frac{1}{2} \tensor{R}{^a_{bcd}} \theta^c \wedge \theta^d \wedge \theta^b = \frac{1}{2} \tensor{R}{^a_{[bcd]}} \theta^b \wedge \theta^c \wedge \theta^d
    \end{equation*}
    which is the first Bianchi identity. Then, from the second \eref{eq:cov-ext-der-torsion-curvature}:
    \begin{equation*}
      0 = \dcov \tensor{\Omega}{^a_b} = \frac{1}{2} \dcov \left[ \tensor{R}{^a_{bcd}} \theta^c \wedge \theta^d \right] = \frac{1}{2} \left[ \tensor{R}{^a_{bcd;k}} \theta^k \wedge \theta^c \wedge \theta^d + \tensor{R}{^a_{bcd}} \left( \dcov \theta^c \wedge \theta^d - \theta^c \wedge \dcov \theta^d \right) \right]
    \end{equation*}
    But $ \dcov \theta^a = T^a \equiv 0 $, hence this equation is equivalent to the second Bianchi identity.
  \end{proof}
\end{proofbox}

In the vielbein basis, the Ricci tensor is defined as the contraction $ R_{bd} \equiv \tensor{R}{^a_{bad}} $, while the Ricci scalar is the trace $ R \equiv \eta^{ab} R_{ab} $. In the same way, the Einstein tensor is defined as $ G_{ab} \equiv R_{ab} - \frac{1}{2} \eta_{ab} R $.

\begin{corollary}{Einstein--Bianchi identity}{}
  For a Levi--Civita connection:
  \begin{equation}
    \tensor{G}{_{ab}^{;b}} = 0
  \end{equation}
\end{corollary}

\begin{proofbox}
  \begin{proof}
    From the second Bianchi identity \eref{eq:bianchi-vielbein}, using the antisymmetry $ \tensor{R}{^a_{bcd}} = \tensor{R}{^a_{b[cd]}} $:
    \begin{equation*}
      R_{bd;k} + \tensor{R}{^a_{bka;d}} + \tensor{R}{^a_{bdk;a}} = 0
      \quad \implies \quad
      R_{bd;k} - R_{bk;d} + \tensor{R}{^a_{bdk;a}} = 0
    \end{equation*}
    Now, contract with $ \eta^{bd} $:
    \begin{equation*}
      R_{;k} - \tensor{R}{^b_{k;b}} + \tensor{R}{^{ad}_{dk;a}} = 0
      \quad \implies \quad
    \end{equation*}

    \begin{lemma}[borderline = {1pt}{0pt}{blue!35!black}]{Symmetries of Riemann tensor}{riemann-symmetry-vielbein}
      For a Levi--Civita connection:
      \begin{equation}
        R_{abcd} = R_{[ab][cd]}
      \end{equation}
    \end{lemma}

    \begin{proofbox}
      \begin{proof}
        By \eref{eq:curvature-2-def} $ \tensor{R}{^a_{bcd}} = \tensor{\Omega}{^a_b}(e_c , e_d) $, thus to compute this expression first consider:
        \begin{equation*}
          \dd \tensor{\omega}{^a_b} = \dd \omega^a_{mb} \wedge \theta^m + \omega^a_{mb} \dd \theta^m = \dd \omega^a_{mb} \wedge \theta^m - \omega^a_{mb} \tensor{\omega}{^m_k} \wedge \theta^k
        \end{equation*}
        where \eref{eq:first-cartan-stru-eq} was used. Then, since $ \dd f = \pa_\mu f \tensor{e}{_a^\mu} \theta^a = e_a(f) \theta^a \,\, \forall f \in \cm $:
        \begin{equation*}
          \dd \tensor{\omega}{^a_b} = e_k(\omega^a_{mb}) \theta^k \wedge \theta^m - \omega^a_{mb} \tensor{\omega}{^m_k} \wedge \theta^k = \left[ e_m(\omega^a_{kb}) - \omega^a_{cb} \omega^c_{mk} \right] \theta^m \wedge \theta^k
        \end{equation*}
        Therefore, form \eref{eq:second-cartan-stru-eq}:
        \begin{equation*}
          \tensor{\Omega}{^a_b} = \left[ e_m(\omega^a_{kb}) - \omega^a_{cb} \omega^c_{mk} + \omega^a_{mc} \omega^c_{kb} \right] \theta^m \wedge \theta^k
        \end{equation*}
        Now, to evaluate $ \tensor{\Omega}{^a_b}(e_c , e_d) $, use:
        \begin{equation*}
          \theta^m \wedge \theta^k (e_c , e_d) \equiv
          \begin{vmatrix}
            \theta^m(e_c) & \theta^m(e_d) \\ \theta^k(e_c) & \theta^k(e_d)
          \end{vmatrix}
          = \delta^m_c \delta^k_d - \delta^m_d \delta^k_c \eqdef \delta^{mk}_{cd}
        \end{equation*}
        Hence, defining $ \omega_{abc} \equiv \eta_{ak} \omega^k_{bc} $, the curvature tensor is found to be:
        \begin{equation*}
          R_{abcd} = \left[ e_m(\omega_{akb}) - \omega_{agb} \omega^g_{mk} + \omega_{amg} \omega^g_{kb} \right] \delta^{mk}_{cd}
        \end{equation*}
        Then, since $ \delta^{mk}_{cd} = \delta^{[mk]}_{[cd]} $ and, recalling \eref{eq:connection-form-symm} for a Levi--Civita connection, $ \omega_{ab} = - \omega_{ba} \implies \omega_{acb} = \omega_{[a|c|b]} $, this is equivalente to the thesis.
      \end{proof}
    \end{proofbox}

    Using \lref{lemma:riemann-symmetry-vielbein}, the above expression reduces to:
    \begin{equation*}
      R_{;k} = 2 \tensor{R}{^a_{k;a}}
    \end{equation*}
    Then, the thesis is trivial from the metric-compatibility of the connection.
  \end{proof}
\end{proofbox}

Note that \lref{lemma:riemann-symmetry-vielbein} allows to explicitly compute the curvature tensor from connection coefficients:
\begin{equation}
  R_{abcd} = e_c(\omega_{adb}) - e_d(\omega_{acb}) - \omega_{agb} \left( \omega^g_{cd} - \omega^g_{dc} \right) + \omega_{acg} \omega^g_{db} - \omega_{adg} \omega^g_{cb}
\end{equation}
Moreover, by multilinearity, the curvature tensor in the vielbein basis retains the symmetry $ R_{abcd} = R_{cdab} $. This, together with $ R_{abcd} = R_{[ab][cd]} $, means that the Riemann tensor can be thought as having two symmetrized multi-indices $ [ab] $ and $ [cd] $, hence the number of independent components is $ \frac{1}{2} N (N+1) $, where $ N $ is the number of possible values for a single multi-index, i.e. $ N = \frac{1}{2} n (n-1) $. However, one has to account for the constraint posed by the first Bianchi identity: since there is one such constraint for each choise of four distinc indices, the total number of independent components of the Riemann tensor is:
\begin{equation*}
  f_\text{Riemann} = \frac{1}{2} \frac{n (n-1)}{2} \left( \frac{n(n-1)}{2} + 1 \right) - \binom{n}{4} = \frac{1}{8} (n-1) (n (n-1) + 2) - \frac{1}{24} n (n-1) (n-2) (n-3)
\end{equation*}
Simplifying this expression yields the classical result:
\begin{equation}
  f_\text{Riemann} = \frac{n^2 (n^2-1)}{12}
\end{equation}
In particular, on $ 4 $-dimensional manifolds this analysis yields the separation of the $ 20 $ degrees of freedom of the Riemann tensor into the $ 10 $ degrees of freedom of the Ricci tensor (``symmetric" part) and the $ 10 $ of the Weyl tensor (``antisymmetric" part).

Note that, if the curvature tensor is computed on a non-coordinate and non-tetrad basis, then it is not possible to impose the symmetry $ R_{abcd} = R_{cdab} $ and the unconstrained components rise to $ 36 $ (i.e. $ N^2 $): the spurious $ 16 $ degrees of freedom thus need to be eliminated with additional constraints, depending on the formalism employed.
