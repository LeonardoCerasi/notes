Although the Einstein field equations, being second-order non-linear inhomogeneuos PDEs, are highly non-trivial to solve, they admit wave-like solutions, which produce what are known as \emph{gravitational waves} (GWs). To study GWs, three approaches are possible: perturbative solutions, exact solutions and asymptotic solutions of radiating systems.

\section{Perturbations of spacetime}

The basic perturbative approach for the study of GWs is to consider a background metric $ \bar{g} $ and introduce a perturbation $ h $ describing the GWs, so to obtain a spacetime metric like:
\begin{equation}
  g_{\mu \nu} = \bar{g}_{\mu \nu} + h_{\mu \nu}
  \label{eq:gw-pert-exp}
\end{equation}
A key property of GWs is the time-domain splitting between the background metric and the perturbations: indeed, observable GWs have frequencies $ \sim 1 \,\text{Hz} $, so the variations of $ h $ have timescale of $ \sim 1 \,\text{sec} $, while those of $ \bar{g} $ have cosmological timescale $ \sim H_0^{-1} $, i.e. in the order of billions of years. Note that this time-domain splitting is not gauge invariant, since it depends on the explicit coordinates used.

\subsection{Weak gravity}
\label{ssec:weak-gravity}

To generalize the perturbative expansion in \eref{eq:gw-pert-exp}, consider a 1-parameter family of metrics $ q(\lambda) $ analytic in $ \lambda \in \R $:
\begin{equation*}
  q(0) = \bar{g}
  \qquad \qquad
  \frac{\dd q}{\dd \lambda} \bigg\vert_{\lambda = 0} = h
\end{equation*}
Then, as $ \lambda \ra 0 $ the metric reduces to $ q(\lambda) = \bar{g} + \lambda h + \smo(\lambda^2) $, which is just the above expression. Now, consider the two Levi--Civita connections associated to $ q(\lambda) $ and $ \bar{g} $ respectively:
\begin{equation}
  \bar{\na}_\alpha \bar{g}_{\mu \nu} = 0
  \qquad \qquad
  \tilde{\na}_\alpha q_{\mu \nu} = 0
  \label{eq:perturbation-levi-civita}
\end{equation}

\begin{lemma}{Difference tensor}{}
  Let $ \man $ be a metric manifold, and let $ \tilde{\na} , \bar{\na} $ be two Koszul connections in $ \man $. Then, given $ \omega \in \df{1} $:
  \begin{equation}
    \tilde{\na}_\mu \omega_\alpha - \bar{\na}_\mu \omega_\alpha = - C^\rho_{\mu \alpha} \omega_\rho
    \label{eq:difference-tensor}
  \end{equation}
  with $ C \in \tns{1}{2}(\man) $. If both families of connections are torsion-free, $ C^\rho_{\mu \alpha} = C^\rho_{\alpha \mu} $.
\end{lemma}

\begin{proofbox}
  \begin{proof}
    To show that $ C \in \tns{1}{2}(\man) $, recall that any connection acts on $ T^0_0\man = \cm $ as $ \na_\mu f = \pa_\mu f $, hence all connections in $ \man $ agree on $ T^0_0\man $. Now, let $ f \in \cm , \omega \in \df{1} $:
    \begin{equation*}
      \tilde{\na}_\mu (f \omega_\alpha) - \bar{\na}_\mu (f \omega_\alpha) = (\tilde{\na}_\mu f) \omega_\alpha + f (\tilde{\na}_\mu \omega_\alpha) - (\bar{\na}_\mu f) \omega_\alpha - f (\bar{\na}_\mu \omega_\alpha) = f (\tilde{\na}_\mu \omega_\alpha - \bar{\na}_\mu \omega_\alpha)
    \end{equation*}
    Now, consider $ \eta \in \df{1} : \eta \vert_p = \omega \vert_p $: it can be shown that there exist $ \{f_a\} \ssq \cm : f_a(p) = 0 $ and $ \{\zeta^a\} \ssq \df{1} $ such that:
    \begin{equation*}
      \eta - \omega = \sum_a f_a \zeta^a
    \end{equation*}
    Then:
    \begin{equation*}
      \tilde{\na}_\mu (\eta_\alpha - \omega_\alpha) - \bar{\na}_\mu (\eta_\alpha - \omega_\alpha) = \sum_a \left[ \tilde{\na}_\mu (f_a \zeta^a_\alpha) - \bar{\na}_\mu (f_a \zeta^a_\alpha) \right] = \sum_a f_a \left[ \tilde{\na}_\mu \zeta^a_\alpha - \bar{\na}_\mu \zeta^a_\alpha \right]
    \end{equation*}
    which vanishes at $ p $ since $ f_a(p) = 0 $. Therefore:
    \begin{equation*}
      \left[ \tilde{\na}_\mu \eta_\alpha - \bar{\na}_\mu \eta_\alpha \right]_p = \left[ \tilde{\na}_\mu \omega_\alpha - \bar{\na}_\mu \omega_\alpha \right]_p
    \end{equation*}
    which shows that the difference $ \tilde{\na}_\mu \omega_\alpha - \bar{\na}_\mu \omega_\alpha $ only depends on the value of $ \omega_\alpha \vert_p $. This means that $ \tilde{\na}_\mu - \bar{\na}_\mu $ defines a linear map $ T_p^*\man \ra T^0_2(T_p\man) $, and so a $ (1,2) $-tensor field $ C $ with components \eref{eq:difference-tensor}.

    Now, consider $ \omega_\alpha = \na_\alpha f $ with $ f \in \cm $:
    \begin{equation*}
      \tilde{\na}_\mu \tilde{\na}_\nu f - \bar{\na}_\mu \bar{\na}_\nu f = - C^\rho_{\mu \nu} f
    \end{equation*}
    However, for a torsion-free connection $ \na $:
    \begin{equation*}
      \na_\mu \na_\nu f = \na_\mu (\pa_\nu f) = \pa_\mu \pa_\nu f - \Gamma^\rho_{\mu \nu} \pa_\rho f = \pa_\nu \pa_\mu - \Gamma^\rho_{\nu \mu} \pa_\rho f = \na_\nu \na_\mu f
    \end{equation*}
    Therefore, if both $ \tilde{\na} $ and $ \bar{\na} $ are torsion-free, $ C^\rho_{\mu \nu} = C^\rho_{\nu \mu} $.
  \end{proof}
\end{proofbox}

The difference tensor can be used to characterize the curvature tensors of the perturbed metric.

\begin{proposition}{Difference tensor of perturbed metric}{}
  Given the two Levi--Civita connections defined in \eref{eq:perturbation-levi-civita}, the difference tensor defined by ($ \omega \in \df{1} $):
  \begin{equation*}
    \tilde{\na}_\mu \omega_\alpha - \bar{\na}_\mu \omega_\alpha = - C^\rho_{\mu \alpha} \omega_\rho
  \end{equation*}
  is expressed as:
  \begin{equation}
    C^\rho_{\mu \nu}(\lambda) = \frac{1}{2} q^{\rho \alpha} (\bar{\na}_\mu q_{\nu \alpha} - \bar{\na}_\nu q_{\mu \alpha} - \bar{\na}_\alpha q_{\mu \nu})
    \label{eq:perturbation-difference-tensor}
  \end{equation}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    By the metric-compatibility of the Levi--Civita connection:
    \begin{equation*}
      0 = \tilde{\na}_\alpha q_{\mu \nu} = \bar{\na}_\alpha q_{\mu \nu} - C^\rho_{\alpha \mu} q_{\rho \nu} - C^\rho_{\alpha \nu} q_{\mu \rho}
    \end{equation*}
    Exchanging $ \alpha \lra \mu $ and then $ \mu \lra \nu $, a set of three equations is found:
    \begin{equation*}
      \bar{\na}_\alpha q_{\mu \nu} - C^\rho_{\alpha \mu} q_{\rho \nu} - C^\rho_{\alpha \nu} q_{\mu \rho} = 0
    \end{equation*}
    \begin{equation*}
      \bar{\na}_\mu q_{\alpha \nu} - C^\rho_{\mu \alpha} q_{\rho \nu} - C^\rho_{\mu \nu} q_{\alpha \rho} = 0
    \end{equation*}
    \begin{equation*}
      \bar{\na}_\nu q_{\alpha \mu} - C^\rho_{\nu \alpha} q_{\rho \mu} - C^\rho_{\nu \mu} q_{\alpha \rho} = 0
    \end{equation*}
    Summing the last two equations and subtracting the first one:
    \begin{equation*}
      \bar{\na}_\mu q_{\alpha \nu} + \bar{\na}_\nu q_{\alpha \mu} - \bar{\na}_\alpha q_{\mu \nu} - 2 C^\rho_{\mu \nu} q_{\alpha \rho} = 0
    \end{equation*}
    which yields the thesis.
  \end{proof}
\end{proofbox}

The Riemann curvature tensor can now be computed directly from definition (\eref{eq:riemann-operator-def}).

\begin{lemma}{Difference of Riemann tensors}{}
  Given the two Levi--Civita connections defined in \eref{eq:perturbation-levi-civita}, the difference of the associated Riemann tensors is:
  \begin{equation}
    \tensor{\tilde{\rr}}{_{\mu \nu \alpha}^\rho} - \tensor{\bar{\rr}}{_{\mu \nu \alpha}^\rho} = 2 \bar{\na}_{[\nu} C^\rho_{\mu]\alpha} + C^\rho_{\nu \beta} C^\beta_{\mu \alpha} - C^\rho_{\nu \alpha} C^\rho_{\mu \beta}
    \label{eq:perturbation-riemann}
  \end{equation}
\end{lemma}

\begin{proofbox}
  \begin{proof}
    By \eref{eq:riemann-operator-def}:
    \begin{equation*}
      \begin{split}
        \tensor{\tilde{\rr}}{_{\mu \nu \alpha}^\rho} \omega_\rho
        & = \tilde{\na}_\mu \tilde{\na}_\nu \omega_\alpha - \tilde{\na}_\nu \tilde{\na}_\mu \omega_\alpha \\
        & = \tilde{\na}_\mu ( \bar{\na}_\nu \omega_\alpha - C^\rho_{\nu \alpha} \omega_\rho ) - \tilde{\na}_\nu ( \bar{\na}_\mu \omega_\alpha - C^\rho_{\mu \alpha} \omega_\rho ) \omega_\rho \\
        & = \bar{\na}_\mu \bar{\na}_\nu \omega_\alpha - C^\rho_{\mu \nu} \bar{\na}_\rho \omega_\alpha - C^\rho_{\mu \alpha} \bar{\na}_\nu \omega_\rho - \tilde{\na}_\mu (C^\rho_{\nu \alpha} \omega_\rho) \\
        & \quad - \bar{\na}_\nu \bar{\na}_\mu \omega_\alpha + C^\rho_{\nu \mu} \bar{\na}_\rho \omega_\alpha + C^\rho_{\nu \alpha} \bar{\na}_\mu \omega_\rho + \tilde{\na}_\nu (C^\rho_{\mu \alpha} \omega_\rho) \\
        & = \tensor{\bar{\rr}}{_{\mu \nu \alpha}^\rho} \omega_\rho - C^\rho_{\mu \alpha} \bar{\na}_\nu \omega_\rho + C^\rho_{\nu \alpha} \bar{\na}_\mu \omega_\rho - \tilde{\na}_\mu (C^\rho_{\nu \alpha} \omega_\rho) + \tilde{\na}_\nu (C^\rho_{\mu \alpha} \omega_\rho)
      \end{split}
    \end{equation*}
    Now, note that:
    \begin{equation*}
      \begin{split}
        \tilde{\na}_\mu (C^\rho_{\nu \alpha} \omega_\rho)
        & = \na_\mu (C^\rho_{\nu \alpha}) \omega_\rho + C^\rho_{\nu \alpha} \tilde{\na}_\mu \omega_\rho \\
        & = \bar{\na}_\mu C^\rho_{\nu \alpha} \omega_\rho + C^\rho_{\mu \beta} C^\beta_{\nu \alpha} \omega_\rho - C^\beta_{\mu \nu} C^\rho_{\beta \alpha} \omega_\rho - C^\beta_{\mu \alpha} C^\rho_{\nu \beta} \omega_\rho + C^\rho_{\nu \alpha} \bar{\na}_\mu \omega_\rho - C^\rho_{\nu \alpha} C^\beta_{\mu \rho} \omega_\beta
      \end{split}
    \end{equation*}
    Thus:
    \begin{equation*}
      \begin{split}
        \tensor{\tilde{\rr}}{_{\mu \nu \alpha}^\rho} \omega_\rho
        & = \tensor{\bar{\rr}}{_{\mu \nu \alpha}^\rho} \omega_\rho - C^\rho_{\mu \alpha} \bar{\na}_\nu \omega_\rho + C^\rho_{\nu \alpha} \bar{\na}_\mu \omega_\rho - \tilde{\na}_\mu (C^\rho_{\nu \alpha} \omega_\rho) + \tilde{\na}_\nu (C^\rho_{\mu \alpha} \omega_\rho) \\
        & \quad - \bar{\na}_\mu C^\rho_{\nu \alpha} \omega_\rho - C^\rho_{\mu \beta} C^\beta_{\nu \alpha} \omega_\rho + C^\beta_{\mu \nu} C^\rho_{\beta \alpha} \omega_\rho + C^\beta_{\mu \alpha} C^\rho_{\nu \beta} \omega_\rho - C^\rho_{\nu \alpha} \bar{\na}_\mu \omega_\rho + C^\rho_{\nu \alpha} C^\beta_{\mu \rho} \omega_\beta \\
        & \quad + \bar{\na}_\nu C^\rho_{\mu \alpha} \omega_\rho + C^\rho_{\nu \beta} C^\beta_{\mu \alpha} \omega_\rho - C^\beta_{\nu \mu} C^\rho_{\beta \alpha} \omega_\rho - C^\beta_{\nu \alpha} C^\rho_{\mu \beta} \omega_\rho + C^\rho_{\mu \alpha} \bar{\na}_\nu \omega_\rho - C^\rho_{\mu \alpha} C^\beta_{\nu \rho} \omega_\beta \\
        & = \tensor{\bar{\rr}}{_{\mu \nu \alpha}^\rho} \omega_\rho + 2 \bar{\na}_{[\nu} C^\rho_{\mu]\alpha} \omega_\rho - 2 C^\beta_{\nu \alpha} C^\rho_{\mu \beta} \omega_\rho + 2 C^\rho_{\nu \beta} C^\beta_{\mu \alpha} \omega_\rho + C^\rho_{\nu \alpha} C^\beta_{\mu \rho} \omega_\beta - C^\rho_{\mu \alpha} C^\beta_{\nu \rho} \omega_\beta
      \end{split}
    \end{equation*}
    Renaming dummy indices yields the thesis.
  \end{proof}
\end{proofbox}

\eref{eq:perturbation-difference-tensor} and \eref{eq:perturbation-riemann} are valid at all orders of the analytic parameter $ \lambda \in \R $, hence they are well suited for a perturbative expansion. At NLO, the inverse metric tensor reads:
\begin{equation}
  q^{\mu \nu} = \bar{g}^{\mu \nu} - \lambda \bar{g}^{\mu \alpha} \bar{g}^{\nu \beta} h_{\alpha \beta} + \smo(\lambda^2)
\end{equation}
hence indices of quantities proportional to $ h $ are raised/lowered by $ \bar{g} $ only: $ h^{\mu \nu} = \bar{g}^{\mu \alpha} \bar{g}^{\nu \beta} h_{\alpha \beta} + \smo(\lambda^2) $.
Then, the difference tensor and the Riemann tensor at linear order become:
\begin{equation}
  C^\rho_{\mu \nu}(\lambda) = \frac{1}{2} \bar{g}^{\rho \alpha} ( \bar{\na}_\mu h_{\nu \alpha} + \bar{\na}_\nu h_{\mu \alpha} - \bar{\na}_\alpha h_{\mu \nu} ) \lambda + \smo(\lambda^2)
\end{equation}
\begin{equation}
  \tensor{\tilde{\rr}}{_{\mu \nu \alpha}^\rho} = \tensor{\bar{\rr}}{_{\mu \nu \alpha}^\rho} + 2 \bar{\na}_{[\nu} [C\od1]^\rho_{\mu]\alpha} \lambda + \smo(\lambda^2)
\end{equation}
Therefore, $ \tensor{[\tilde{\rr}\od1]}{_{\mu \nu \alpha}^\rho} = 2 \bar{\na}_{[\nu} [C\od1]^\rho_{\mu]\alpha} $. With this formalism, it is possible to analytically expand the Einstein field equations in powers of $ \lambda \in \R $:
\begin{equation*}
  \bar{G}_{\mu \nu} + [G\od1]_{\mu \nu} \lambda + \smo(\lambda^2) = 8 \pi \gd \left( \bar{T}_{\mu \nu} + [T\od1]_{\mu \nu} \lambda + \smo(\lambda^2) \right)
\end{equation*}
There is an order-by-order split of the field equations into background and N$ ^n $LO equations:
\begin{equation}
  \bar{G}_{\mu \nu} = 8 \pi \gd \bar{T}_{\mu \nu}
  \qquad \qquad
  [G\od n]_{\mu \nu} = 8 \pi \gd [T\od n]_{\mu \nu}
  \label{eq:perturbation-einstein}
\end{equation}

\begin{observation}{Gauge invariance of analytic expansion}{gauge-exp}
  Under an infinitesimal diffeomorphism generated by $ Z \in \xm : x^\mu \mapsto x^\mu + Z^\mu $, the metric changes as $ q_{\mu \nu} \mapsto q_{\mu \nu} + \delta q_{\mu \nu} $ with:
  \begin{equation*}
    \delta q = \ld_Z q = \sum_{n \in \N_0} \ld_Z q\od n \lambda^n = \ld_Z \bar{g} + \lambda \ld_Z h + \smo(\lambda^2)
  \end{equation*}
  However, the perturbative expasion requires $ \lambda \ra 0 $, hence the second term is a second order one (as both $ Z $ and $ \lambda $ are infinitesimal) and the variation $ \delta q $ only has beyond-leading-order terms. Therefore, under an infinitesimal diffeomorphism, the perturbed metric at first-order transforms as:
  \begin{equation}
    \bar{g} \mapsto \bar{g}
    \qquad \qquad
    h \mapsto = h + \ld_Z \bar{g}
  \end{equation}
  i.e. $ \delta h_{\mu \nu} = 2\bar{\na}_{(\mu} Z_{\nu)} $, with $ Z_\mu = \bar{g}_{\mu \nu} Z^\nu $. In the same way, $ \delta \tensor{[\tilde{\rr}\od1]}{_{\mu \nu \alpha}^\rho} = \ld_Z \tensor{\bar{\rr}}{_{\mu \nu \alpha}^\rho} $.
\end{observation}

The perturbed Einstein field equations \eref{eq:perturbation-einstein} are most conveniently expressed in terms of a modified perturbation metric:
\begin{equation}
  \bar{h}_{\mu \nu} \defeq h_{\mu \nu} - \frac{1}{2} h \bar{g}_{\mu \nu}
  \qquad \qquad
  h \equiv \bar{g}^{\mu \nu} h_{\mu \nu}
\end{equation}
The bar notation in $ \bar{h}_{\mu \nu} $ has nothing to do with the background metric. Then, setting $ \bar{\Box} \equiv \bar{g}^{\mu \nu} \bar{\na}_\mu \bar{\na}_\nu $, the Einstein tensor at NLO can be shown to be:
\begin{equation}
  [G\od1]_{\mu \nu} = - \frac{1}{2} \bar{\Box} \bar{h}_{\mu \nu} + \bar{h}_{\mu \nu} \bar{\rr} + \bar{\na}_{(\nu} \bar{\na}^\alpha \bar{h}_{\mu)\alpha} + \tensor{\bar{\rr}}{_{(\mu}^\alpha} \bar{h}_{\nu)\alpha} + \bar{\rr}_{\alpha \nu \mu \beta} \bar{h}^{\alpha \beta} - \frac{1}{2} \bar{g}_{\mu \nu} \left[ \bar{\rr}_{\alpha \beta} \bar{h}^{\alpha \beta} + \bar{\na}^\alpha \bar{\na}^\beta \bar{h}_{\alpha \beta} \right]
  \label{eq:nlo-einstein}
\end{equation}
This expression is greatly simplified by imposing a gauge-fixing condition (by \obsref{obs:gauge-exp}). In particular, a convenient choice is the \bctxt{Lorentz gauge}:
\begin{equation}
  \bar{\na}^\alpha \bar{h}_{\alpha \mu } = 0
  \label{eq:lorentz-gauge}
\end{equation}
In this gauge, the NLO Einstein tensor reduces to:
\begin{equation*}
  [G\od1]_{\mu \nu} = - \frac{1}{2} \bar{\Box} \bar{h}_{\mu \nu} + \bar{h}_{\mu \nu} \bar{\rr} + \tensor{\bar{\rr}}{_{(\mu}^\alpha} \bar{h}_{\nu)\alpha} + \bar{\rr}_{\alpha \nu \mu \beta} \bar{h}^{\alpha \beta} - \frac{1}{2} \bar{g}_{\mu \nu} \bar{\rr}_{\alpha \beta} \bar{h}^{\alpha \beta}
\end{equation*}
The time-domain splitting between the background metric and the perturbation in \eref{eq:gw-pert-exp} can be generally represented by an inequality between the spacetime scale $ L_\text{b} $ of the background and that $ \lambda $ of the perturbation: indeed, by dimensional analysis, the background curvature scales like $ \bar{\rr} \sim L_\text{b}^{-2} $, while the perturbation like $ \pa h \sim \lambda^{-1} $, hence by the split $ \lambda \ll L_\text{b} $ it is clear that the first perturbation term dominates over the other curvature terms. The NLO field equations thus have the schematic form of a wave equation:
\begin{equation*}
  \bar{\Box} \bar{h}_{\mu \nu} = - 16 \pi \gn [T\od1]_{\mu \nu} + (\text{background curvature terms})
\end{equation*}

\begin{observation}{Quantum gravity}{}
  The NLO Einstein tensor can be found by the variation of the linearized Einstein--Hilber action, also known as the \bcobs{Fierz--Pauli action}, which turns out to be:
  \begin{equation}
    \act_\text{FP}\od1 = \frac{1}{8\pi \gd} \int \dd^4 x \left[ - \frac{1}{4} \pa_\rho h_{\mu \nu} \pa^\rho h^{\mu \nu} + \frac{1}{2} \pa_\rho h_{\mu \nu} \pa^\nu h^{\rho \mu} + \frac{1}{4} \pa_\mu h \pa^\mu h - \frac{1}{2} \pa_\nu h^{\mu \nu} \pa_\mu h \right]
  \end{equation}
  An effective free theory of the graviton is given by the Fierz--Pauli lagrangian, which can then be generalized to an interacting theory introducing an interaction Lagrangian $ \lag_\text{int} $. Note that $ h_{\mu \nu} $ is dimensionless, hence to describe the graviton it is customary to define a canonical field $ \hat{h}_{\mu \nu} \equiv (\gd)^{-1/2} h_{\mu \nu} $: recall by \obsref{obs:dimensional-analysis} that $ [\gd] = d - 2 $, hence $ [\hat{h}] = \frac{2 - d}{2} $ and correctly $ [\lag_\text{FP}] = -d $. Since the Fierz--Pauli action must be at least of order $ (\pa h)^2 $, the lowest-order interacting term has schematic structure:
  \begin{equation*}
    \lag_\text{int} \sim \frac{1}{\gd} (\pa h)^2 h \sim \sqrt{\gd} (\pa \hat{h})^2 \hat{h} \sim \frac{1}{\pl^2} (\pa \hat{h}) \hat{h}
  \end{equation*}
  where the Plank mass is define (in natural units) as $ \pl^2 \equiv 1 / 8 \pi \gd $.
  The coupling has mass dimension $ \frac{d - 2}{2} $, hence this quantum field theory of the graviton is non-renormalizable for $ d > 2 $. This was to be expected from an effective theory: indeed, the inverse metric $ q^{\mu \nu} $ truly has infinite powers of $ h $, hence the radiative corrections given by virtual gravitons are extremely small, and they only become non-negligible when the perturbative expansion diverges (for $ E \gtrsim \pl $), with consequent loss of meaning of the effective theory.
\end{observation}

\subsection{Linear theory of GWs}

Consider a $ d = 4 $ background metric $ \bar{g}_{\mu \nu} = \eta_{\mu \nu} $: this is, of course, only an approximation, since the metric perturbation would have a back-propagation effect which curves the background, nontheless it can be seen as restricting to a spacetime region which is small when compared to the curvature scale of the background. The Lorentz gauge-fixing condition then reduces to:
\begin{equation*}
  \pa^\mu \bar{h}_{\mu \nu} = 0
\end{equation*}
Under an infinitesimall diffeomorphism generated by $ Z \in \xm $:
\begin{equation*}
  \delta h_{\mu \nu} = (\ld_Z \eta)_{\mu \nu} = \pa_\mu Z_\nu + \pa_\nu Z_\mu
  \qquad \qquad
  \delta h = 2 \pa^\mu Z_\mu
\end{equation*}
Then, setting $ \Box = \eta^{\mu \nu} \pa_\mu \pa_\nu $ (the bar notation is suppressed for objects of the flat background):
\begin{equation*}
  \delta \bar{h}_{\mu \nu} = \pa_\mu Z_\nu + \pa_\nu Z_\mu - \eta_{\mu \nu} \pa^\alpha Z_\alpha
  \quad \implies \quad
  \delta (\pa^\mu \bar{h}_{\mu \nu}) = \Box Z_\nu
\end{equation*}
The Lorentz gauge thus presents a residual gauge freedom for $ Z \in \xm $, such that:
\begin{equation}
  \Box Z^\mu = 0
\end{equation}
For this reason, the Lorentz gauge is also known as the \emph{harmonic gauge} (or \emph{de Donder gauge}), since it preserves the condition $ \Box x^\mu = 0 $ (as $ x^\mu \mapsto x^\mu + Z^\mu $).

In $ d \ge 4 $ it makes physical sense\footnotemark to consider the propagation of GWs in vacuum $ [T\od1]_{\mu \nu} = 0 $:
\begin{equation}
  \Box \bar{h}_{\mu \nu} = 0
  \label{eq:vacuum-gw}
\end{equation}

\footnotetext{In $ d = 2 , 3 $ the Riemann tensor has respectively $ 1 $ and $ 6 $ independent components (recall \eref{eq:riemann-deg-freedom}), so the vacuum field equations (which are $ 10 $ independent equations) completely nullify the Riemann tensor, making it identically vanish.}

$ \bar{h}_{\mu \nu} $ can then be seen as a massless spin-$ 2 $ field propagating in Minkowski spacetime: by symmetry, it would have $ 10 $ independent components, but the Lorentz gauge eliminates $ 4 $ degrees of freedom, and the residual gauge eliminates additional $ 4 $ of them, leaving only $ 2 $ degrees of freedom for the propagating GW.

To explicitly see this, consider the general ansatz for the vacuum wave equation in the homogeneous and isotropic spacetime $ \R^{1,3} $:
\begin{equation*}
  \bar{h}_{\mu \nu}(x) = \int_{\R^{1,3}} \frac{\dd^4k}{(2\pi)^4} e^{\img k \cdot x} \bar{\varepsilon}_{\mu \nu}(k)
\end{equation*}
for some polarization matrix $ \bar{\varepsilon}_{\mu \nu}(k) \in \C^{4 \times 4} $ (the physical perturbation is the real part of this Fourier transform). In order to be a solution, the on-shell condition given by the equations of motion \eref{eq:vacuum-gw} must be satisfied:
\begin{equation*}
  \eta^{\alpha \beta} k_\alpha k_\beta \bar{\varepsilon}_{\mu \nu}(k) = 0
  \quad \implies \quad
  k^2 = 0
\end{equation*}
confirming that $ \bar{h}_{\mu \nu} $ is a massless field\footnotemark. Then, the solution to the linearized field equations reads:

\footnotetext{For a massless field there is no mass gap, since the energy $ k^0 = \pm \sqrt{\ve{k}^2} = \pm \omega $ can be made arbitrarily small.}

\begin{equation}
  \bar{h}_{\mu \nu}(x) = \int_{\R^{1,3}} \frac{\dd^4k}{(2\pi)^4} \delta^{(4)}(k^2) e^{\img k \cdot x} \bar{\varepsilon}_{\mu \nu}(k)
\end{equation}
Imposing the Lorentz gauge-fixing condition:
\begin{equation*}
  \pa^\mu \bar{h}_{\mu \nu} = 0
  \quad \implies \quad
  k^\mu \bar{\varepsilon}_{\mu \nu} = 0
\end{equation*}
which means that the polarization matrix must be transverse to the direction of propagation of the GW. The residual gauge freedom can be exploited writing $ Z \in \xm $ as:
\begin{equation*}
  Z^\mu(x) = \int_{\R^{1,3}} \frac{\dd^4k}{(2\pi)^4} \hat{Z}^\mu(k) e^{\img k \cdot x}
\end{equation*}
Then, the gauge condition $ \Box Z^\mu = 0 $ simply becomes $ k^2 \hat{Z}^\mu = 0 $, i.e. $ k^2 = 0 $. This means that on-shell it is possible to freely choose the component functions $ Z^\mu $. First, observe that the infinitesimal gauge transformation $ x^\mu \mapsto x^\mu + Z^\mu $ acts as:
\begin{equation*}
  h_{\mu \nu}(x) \mapsto \bar{h}_{\mu \nu}(x) + 2 \pa_{(\mu} Z_{\nu)}(x) = \int_{\R^{1,3}} \frac{\dd^4k}{(2\pi)^4} \delta^{(4)}(k^2) [ \varepsilon_{\mu \nu} + 2 \img k_{(\mu} \hat{Z}_{\nu)} ] e^{\img k \cdot x}
\end{equation*}
which corresponds to a transformation of the polarization matrix:
\begin{equation}
  \varepsilon_{\mu \nu} \mapsto \varepsilon_{\mu \nu} + \img ( k_\mu \hat{Z}_\nu + k_\nu \hat{Z}_\mu )
\end{equation}
The component $ \hat{Z}^0 $ is chosen so that $ h = \eta^{\mu \nu} h_{\mu \nu} = 0 $, which means that:
\begin{equation*}
  \eta^{\mu \nu} \varepsilon_{\mu \nu} + 2 \img k_\mu \hat{Z}^\mu = 0
  \quad \implies \quad
  \hat{Z}^0 = \frac{1}{k^0} \left( \eta^{\mu \nu} \varepsilon_{\mu \nu} + \ve{k} \cdot \ve{Z} \right)
\end{equation*}
Once this condition is imposed, $ \bar{h}_{\mu \nu} = h_{\mu \nu} $ and $ \bar{\varepsilon}_{\mu \nu} = \varepsilon_{\mu \nu} $. Then, the remaining components $ \hat{Z}^i $ are chosen so that $ \bar{h}_{0i} = h_{0i} = 0 $, which means that:
\begin{equation*}
  \varepsilon_{0i} + \img ( k_0 \hat{Z}_i + k_i \hat{Z}_0 ) = 0
\end{equation*}
The complete gauge fixing then results into:
\begin{equation*}
  k^\mu \varepsilon_{\mu \nu} = 0
  \qquad
  \eta^{\mu \nu} \varepsilon_{\mu \nu} = 0
  \qquad
  \varepsilon_{0i} = 0
\end{equation*}
This are indeed $ 8 $ conditions on the polarization matrix, corresponding to the $ 4 $ Lorentz gauge-fixing conditions and the $ 4 $ residual gauge-fixing conditions. It is trivial to check that these conditions are equivalent to:
\begin{equation}
  \varepsilon_{0\mu} = 0
  \qquad
  k^i \varepsilon_{ij} = 0
  \qquad
  \delta^{ij} \varepsilon_{ij} = 0
\end{equation}
For this reason, this is known as the \bctxt{transverse-traceless gauge} (TT gauge). In the TT gauge, then:
\begin{equation*}
  \varepsilon_{\mu \nu}^\text{TT} =
  \begin{bmatrix}
    0 & \ve{0} \\
    \ve{0} & \varepsilon_{ij}
  \end{bmatrix}
\end{equation*}
where $ \varepsilon_{ij} $ is a transverse-traceless spatial polarization matrix.

\begin{observation}{Transverse and traceless matrices}{tt-projector}
  In general, any $ \R^{d \times d} $ matrix can be made traceless and transverse with respect to some versor $ \ve{n} \in \R^d $. Indeed, defining the transverse projector:
  \begin{equation}
    \mathcal{P}_{ij} = \delta_{ij} - n_i n_j
  \end{equation}
  it is clear that $ n_i \mathcal{P}_{ij} = n_j - n_j = 0 $. On the other hand, any $ q_{ij} \in \R^{d \times d} $ can be made traceless by:
  \begin{equation*}
    q_{ij} \mapsto q_{ij} - \frac{1}{d} q_{mn} \delta_{mn} \delta_{ij}
  \end{equation*}
  Hence, it is possible to define a transverse-traceless projector as:
  \begin{equation}
    q_{ij}^\text{TT} = \Lambda_{imjn} q_{mn}
    \qquad \qquad
    \Lambda_{imjn} = \mathcal{P}_{im} \mathcal{P}_{jn} - \frac{1}{d-1} \mathcal{P}_{ij} \mathcal{P}_{mn}
  \end{equation}
  which trivially satisfies $ n_i \Lambda_{imjn} = n_j \Lambda_{imjn} = \delta_{ij} \Lambda_{imjn} = 0 $.
\end{observation}

By the construction outlined in \obsref{obs:tt-projector}, assuming WLOG that the GW propagates along $ k^\mu = (\omega , 0 , 0 , \omega) $, the polarization matrix is found to be:
\begin{equation}
  \varepsilon_{\mu \nu}^\text{TT} =
  \begin{bmatrix}
    0 & 0 & 0 & 0 \\
    0 & H_+ & H_\times & 0 \\
    0 & H_\times & - H_+ & 0 \\
    0 & 0 & 0 & 0
  \end{bmatrix}
\end{equation}
which shows the two linearly-independent physical polarizations:
\begin{equation*}
  \varepsilon_{ij}^\text{TT} = H_+
  \begin{bmatrix}
    1 & 0 & 0 \\
    0 & -1 & 0 \\
    0 & 0 & 0
  \end{bmatrix}
  + H_\times
  \begin{bmatrix}
    0 & 1 & 0 \\
    1 & 0 & 0 \\
    0 & 0 & 0
  \end{bmatrix}
\end{equation*}

\subsection{Geodesic deviation}

To understand how GWs can be detected, it is necessary to study how the passage of a GW affects the distance between neighbourhing geodesics.

\begin{definition}{Geodesic congruence}{}
  Let $ (\man , g) $ be a spacetime. Given a nowhere-vanishing $ X \in \xm $, the set of its integral curves $ \{\gamma_s(\tau)\} $ is called a \bcdef{congruence}. If it is possible to define a tangent vector field $ u = \frac{\pa \gamma_s}{\pa \tau} $ such that $ \na_u u = 0 $, it is called a \bcdef{geodesic congruence}.
\end{definition}

A congruence is characterized by its nowhere-vanishing vector field $ X \in \xm $: if $ X $ is timelike/spacelike/null, the congruence is called a timelike/spacelike/null congruence. For example, a timelike geodesic congruence can be interpreted as a family of free-falling test particles (i.e. not affected by mutual interactions), while a null geodesic congruence can be seen as a family of freely propagating light rays.

Consider a timelike geodesic congruence $ \{\gamma_s(\tau)\} $, and define the tangent vector field and the deviation vector field:
\begin{equation}
  u = \frac{\pa}{\pa \tau} \gamma_s(\tau)
  \qquad \qquad
  \xi = \frac{\pa}{\pa s} \gamma_s(\tau)
\end{equation}
In particular, $ \xi $ is the vector which, at fixed affine parameter, points from a given geodesic to the next one in the congruence. In general, $ u $ and $ \xi $ need not be orthogonal, but they can be chosen so. Indeed, the angle between them is constant:
\begin{equation*}
  \frac{\dd}{\dd \tau} \braket{\xi , u}_g = \na_u \braket{\xi , u}_g = \braket{\na_u \xi , u}_g = u_\alpha u^\beta \na_\beta \xi^\alpha
\end{equation*}
Choosing $ u $ and $ \xi $ such that $ [u , \xi] = 0 $ (e.g. by choosing them among local coordinates) yields $ \xi^\alpha \na_\alpha u^\mu - u^\alpha \na_\alpha \xi^\mu = 0 $, hence:
\begin{equation*}
  \frac{\dd}{\dd \tau} \braket{\xi , u}_g = u_\alpha \xi^\beta \na_\beta u^\alpha = \frac{1}{2} \na_\xi u^2 = 0
\end{equation*}
since for a timelike congruence $ u $ can be chosen so that $ u^2 = -1 $. This allows to fix $ \braket{u , \xi}_g = 0 $ for a particular value of the affine parameter.

The relative deviation of the geodesics is encoded in the variation of $ \xi $. One defines the relative velocity and acceleration of the geodesic deviation along the congruence as:
\begin{equation*}
  v = \na_u \xi
  \qquad \qquad
  a = \na_u v
\end{equation*}

\begin{lemma}{Geodesic deviation acceleration}{}
  Let $ (\man,g) $ be a spacetime. Given a geodesic congruence, the geodesic deviation acceleration is:
  \begin{equation}
    a^\mu = \tensor{\rr}{^\mu_{\rho \alpha \beta}} u^\rho u^\alpha \xi^\beta
    \label{eq:geodesic-acceleration}
  \end{equation}
\end{lemma}

\begin{proofbox}
  \begin{proof}
    By definition:
    \begin{equation*}
      a^\mu = \na_u v^\mu = u^\alpha \na_\alpha (u^\beta \na_\beta \xi^\mu) = u^\alpha \na_\alpha (\xi^\beta \na_\beta u^\mu) = u^\alpha \xi^\beta \na_\alpha \na_\beta u^\mu + u^\alpha \na_\alpha \xi^\beta \na_\beta u^\mu
    \end{equation*}
    where in the second equality $ [\xi , u] = 0 $ was used. Inserting the Ricci identity \eref{eq:ricci-id} for a torsion-free connection and using $ [\xi , u] = 0 $ again:
    \begin{equation*}
      \begin{split}
        a^\mu
        & = u^\alpha \xi^\beta (\na_\beta \na_\alpha u^\mu + \tensor{\rr}{^\mu_{\rho \alpha \beta}} u^\rho) + \xi^\alpha \na_\alpha u^\beta \na_\beta u^\mu \\
        & = u^\alpha \xi^\beta \tensor{\rr}{^\mu_{\rho \alpha \beta}} u^\rho + \xi^\rho (u^\alpha \na_\beta \na_\alpha u^\mu + \na_\beta u^\alpha \na_\alpha u^\mu) \\
        & = u^\alpha \xi^\beta \tensor{\rr}{^\mu_{\rho \alpha \beta}} u^\rho + \xi^\beta \na_\beta (\na_u u^\mu) = u^\alpha \xi^\beta \tensor{\rr}{^\mu_{\rho \alpha \beta}} u^\rho
      \end{split}
    \end{equation*}
    which is the thesis.
  \end{proof}
\end{proofbox}

\eref{eq:geodesic-acceleration} shows that in flat spacetime the elements of a congruence have no relative non-uniform motion, and the only relative acceleration is provided by the curvature of spacetime: this makes the geodesic deviation a primary experimental probe of the passage of GWs.

Consider two test masses displaced (at constant time) by a spatial vector $ \bs{\xi} $. Both test masses are assumed t
o be in free fall in a flat region of spacetime, so that the background metric is $ \bar{g}_{\mu \nu} = \eta_{\mu \nu} $. Then, upon the passage of a GW, the metric is perturbed (in the TT frame) as:
\begin{equation*}
  \dd s^2 = - \dd t^2 + (\delta_{ij} + h_{ij}^\text{TT}) \dd x^i \dd x^j
\end{equation*}
Since the two test masses move along timelike geodesics, their trajectories are described by the geodesic equation:
\begin{equation*}
  \frac{\dd^2 x^i}{\dd \tau^2} + \Gamma^i_{\alpha \beta} \frac{\dd x^\alpha}{\dd \tau} \frac{\dd x^\beta}{\dd \tau} = 0
  \qquad \qquad
  \frac{\dd x^i}{\dd \tau} \bigg\vert_{\tau = 0} = 0
\end{equation*}
where the initial condition ensures that they are initially at rest. Inserting the initial condition in the geodesic equation:
\begin{equation*}
  \begin{split}
    \frac{\dd^2 x^i}{\dd \tau^2} \bigg\vert_{\tau = 0} = - \Gamma^i_{00} \left( \frac{\dd x^0}{\dd \tau} \right)^2 \bigg\vert_{\tau = 0} = - \frac{1}{2} g^{i \alpha} (2 \pa_0 g_{0 \alpha} - \pa_\alpha g_{00}) \left( \frac{\dd x^0}{\dd \tau} \right)^2 \bigg\vert_{\tau = 0} = 0
  \end{split}
\end{equation*}
This means that the test masses remain at rest if they are initially at rest. However, their proper distance is indeed affected by the passage of the GW:
\begin{equation*}
  \bs{\xi}^2 = g_{\mu \nu} \xi^\mu \xi^\nu = (\delta_{ij} + h_{ij}^\text{TT}) \xi^i \xi^j = \dist_0^2 + h_{ij}^\text{TT} \xi^i \xi^j
\end{equation*}
So, the proper distance between the two test masses oscillates with the GW:
\begin{equation*}
  \dist(t) = \norm{\bs{\xi}} = \dist_0 \sqrt{1 + \frac{1}{\dist_0^2} h_{ij}^\text{TT} \xi^i \xi^j} \simeq \dist_0 + \frac{1}{2 \dist_0} h_{ij}^\text{TT} \xi^i \xi^j
\end{equation*}

\begin{observation}{Free-falling frame and detector frame}{}
  Regardless of the motion of an observer along a curve $ \gamma(\tau) $, it is always possible to choose suitable local coordinates, known as \emph{Fermi normal coordinates}, such that:
  \begin{equation*}
    g_{\mu \nu} \vert_{\gamma(\tau)} = \eta_{\mu \nu}
    \qquad \qquad
    \Gamma^\mu_{\alpha \beta} \vert_{\gamma(\tau)} = 0
  \end{equation*}
  The metric then differs from the Lorentz--Minkowski metric only by terms proportional to the spacetime curvature and to the acceleration of the observer. In the case of a free-falling observer $ \na_u u = 0 $, and the metric can be shown to be:
  \begin{equation*}
    \dd s^2 = - \dd t^2 \left[ 1 + \rr_{i0j0} x^i x^j \right] - 2 \dd t \dd x^i \frac{2}{3} \rr_{0jik} x^j x^k + \dd x^i \dd x^j \left[ \delta_{ij} - \frac{1}{3} \rr_{ijk\ell} x^k x^\ell \right] + \dots
  \end{equation*}
  On Earth, observers comoving with the detector are not in free fall, hence acceleration effects should be taken into consideration.
\end{observation}

Now, apply the geodesic deviation equation \eref{eq:geodesic-acceleration} to the geodesics of the two test masses. Clearly, $ u = \frac{\dd}{\dd \tau} $, hence it becomes:
\begin{equation*}
  \frac{\dd^2 \xi^\mu}{\dd \tau^2} = - \tensor{\rr}{_{\beta \alpha}^\mu_\rho} u^\rho u^\alpha \xi^\beta
\end{equation*}
In the free-fall frame $ u^\mu = (1 , \ve{0}) $, so $ \braket{u , \xi}_g = 0 $ implies that $ \xi^\mu = (0 , \bs{\xi}) $ and:
\begin{equation*}
  \frac{\dd^2 \xi^i}{\dd \tau^2} = - \tensor{\rr}{_{j0}^i_0} \xi^j
\end{equation*}
For non-relativistic motion the proper time $ \tau $ can be replaced with the coordinate time $ t $. Moreover, at linear order the Riemann tensor is gauge-invariant, hence it can be computed in the preferred gauge. In the TT gauge at linear order:
\begin{equation*}
  \tensor{\rr}{_{j0}^i_0} = \tensor{[\rr\od1]}{_{j0}^i_0} = \rr_{i0j0}^\text{TT} = \frac{1}{2} \pa_0^2 h_{ij}^\text{TT}
\end{equation*}
The geodesic deviation is then described at linear order by a simple ODE:
\begin{equation}
  \frac{\dd^2 \xi^i}{\dd t^2} = \frac{1}{2} \pa_0^2 h_{ij}^\text{TT} \xi^j
\end{equation}
Note that this approximation is only valid if the detector frame can be taken to be in free fall: this is valid if the characteristic dimension of the detector $ x $ is much smaller than the scale of spacetime curvature $ L $, i.e. $ x \ll L $. In this case $ L^{-2} \sim \rr \sim \pa_0^2 h^\text{TT} \sim \lambda_\text{GW}^{-2} $, hence $ x \ll \lambda_\text{GW} $, where $ \lambda_\text{GW} $ is the wavelength of the GW: taking $ \lambda_\text{GW} \sim 10^5 \,\text{km} $ as a reference scale\footnotemark, Earth-based interferometers (e.g. LIGO, VIRGO, KAGRA) with $ x \sim 10 \,\text{km} $ are well described by this approximation, while space-based detectors (e.g. LISA) with $ x \sim 10^6 \,\text{km} $ require a more careful analysis.

\footnotetext{$ \lambda_\text{GW} \sim 10^5 \,\text{km} $ corresponds to $ \nu_\text{GW} \sim 1 \,\text{Hz} $, which is the typical frequency scale of GWs generated by compact binaries.}

\section{Gravitational energy}

The equivalence principle forbids the localization of gravitational energy and momentum, as already noted in \secref{ssec:energy-conservation}. Nonetheless, non-local objects describing these quantities do exist.

In classical Electromagnetism, given a system of electric charges described by a charge density $ \varrho = \varrho(\ve{x}) $, it is possible to define the total electric charge inside a spatial region $ V $ as:
\begin{equation*}
  q \equiv \int_V \dd^3x \, \varrho(\ve{x})
\end{equation*}
Using the first Maxwell equation $ \dive \ve{E} = 4\pi \varrho $, it is possible to express $ q $ in terms of $ \ve{E} $:
\begin{equation*}
  q = \frac{1}{4\pi} \int_V \dd^3x \, \dive \ve{E} = \frac{1}{4\pi} \int_{\pa V} \dd\ve{S} \cdot \ve{E}
\end{equation*}
Therefore, only the flux of the electric field through $ \pa V $ modifies the total charge inside $ V $. The same construction can be applied to Newtonian gravity, where a generic gravitational system is described by a mass density $ \varrho = \varrho(\ve{x}) $, so that the total mass (gravitational charge) inside a spatial region $ V $ is:
\begin{equation*}
  M \equiv \int_V \dd^3x \, \varrho(\ve{x})
\end{equation*}
In this case, use the Poisson equation $ \lap \Phi = 4 \pi \gn \varrho $ to rewrite:
\begin{equation*}
  M = \frac{1}{4\pi\gn} \int_V \dd^3x \, \dive \grad \Phi = \frac{1}{4\pi\gn} \int_{\pa V} \dd\ve{S} \cdot \grad \Phi
\end{equation*}
Again, only the flux of the gravitational field through $ \pa V $ modifies the total mass inside $ V $. A similar argument can be extended to the general relativistic case.

\subsection{Energy-momentum pseudotensor}
\label{ssec:pseudo-emt}

Since any gravitational field can be locally nullified with the choice of a local inertial frame (through Fermi normal coordinates), the energy of a gravitational field is not suitably described by a tensor, since a tensor which vanishes in a particular frame does so in all of them.

For this reason, it is necessary to find a non-tensorial object $ t^{\mu \nu} $, constructed from the metric, which is symmetric and such that $ \pa_\mu [ \gdet (T^{\mu \nu} + t^{\mu \nu}) ] = 0 $, so that the total energy-momentum distribution of the gravitational field together with matter fields is conserved in the usual sense\footnotemark.

\footnotetext{Add clarification on conservation of currents.}

\subsubsection{Weak gravity}
\label{sssec:pseudo-emt-weak-grav}

Consider first the simplified case of a perturbation $ h_{\mu \nu} $ propagating on a $ d = 4 $ Minkowski background. In this case, the background Einstein equations are identically satisfied (since the EMT of matter must vanish at leading order, for gravity to be weak), hence the energy of the gravitational field needs to by studied at NLO.

\begin{proposition}{Linearized Einstein equations}{}
  The NLO field equation in \eref{eq:perturbation-einstein} can be written as:
  \begin{equation}
    \pa^\alpha \pa^\beta H_{\mu \alpha \nu \beta} = 16 \pi \gn [T\od1]_{\mu \nu}
    \label{eq:linear-einstein}
  \end{equation}
  where the \bcprop{Lorentz tensor} is defined as:
  \begin{equation}
    H_{\mu \alpha \nu \beta} \equiv - \bar{h}_{\mu \nu} \eta_{\alpha \beta} - \bar{h}_{\alpha \beta} \eta_{\mu \nu} + \bar{h}_{\alpha \nu} \eta_{\mu \beta} + \bar{h}_{\mu \beta} \eta_{\alpha \nu}
  \end{equation}
\end{proposition}

\begin{proofbox}
  \begin{proof}
    Given the metric tensor $ g_{\mu \nu} = \eta_{\mu \nu} + h_{\mu \nu} $, at NLO the inverse metric is $ g^{\mu \nu} = \eta^{\mu \nu} - h^{\mu \nu} $, thus:
    \begin{equation*}
      [\Gamma\od1]^\sigma_{\nu \rho} = \frac{1}{2} \eta^{\sigma \lambda} \left( \pa_\nu h_{\lambda \rho} + \pa_\rho h_{\nu \lambda} - \pa_\lambda h_{\nu \rho} \right)
    \end{equation*}
    Recalling \eref{eq:riemann-comp}, the $ \Gamma\od1 \Gamma\od1 \sim h^2 $ terms of the Riemann tensor are NNLO and hence negligible, so:
    \begin{equation*}
      \tensor{[\rr\od1]}{^\sigma_{\rho \mu \nu}} = \frac{1}{2} \eta^{\sigma \lambda} \left( \pa_\mu \pa_\rho h_{\nu \lambda} - \pa_\mu \pa_\lambda h_{\nu \rho} - \pa_\nu \pa_\rho h_{\mu \lambda} + \pa_\nu \pa_\lambda h_{\mu \rho} \right)
    \end{equation*}
    Contracting $ (\sigma,\rho) $, the Ricci tensor is:
    \begin{equation*}
      [\rr\od1]_{\mu \nu} = \frac{1}{2} \left( \pa^\rho \pa_\mu h_{\nu \rho} + \pa^\rho \pa_\nu h_{\mu \rho} - \dal h_{\mu \nu} - \pa_\mu \pa_\nu h \right)
    \end{equation*}
    where $ \dal \equiv \pa^\mu \pa_\mu $. The Ricci scalar then is:
    \begin{equation*}
      \rr\od1 = \pa^\mu \pa^\nu h_{\mu \nu} - \Box h
    \end{equation*}
    Finally, the Einstein tensor can be expressed as:
    \begin{equation*}
      [G\od1]_{\mu \nu} = \frac{1}{2} \left[ \pa^\rho \pa_\mu h_{\nu \rho} + \pa^\rho \pa_\nu h_{\mu \rho} - \dal h_{\mu \nu} - \pa_\mu \pa_\nu h - \left( \pa^\rho \pa^\sigma h_{\rho \sigma} - \dal h \right) \eta_{\mu \nu} \right]
    \end{equation*}
    It suffices to show that $ [G\od1]_{\mu \nu} = \frac{1}{2} \pa^\alpha \pa^\beta H_{\mu \alpha \nu \beta} $. By direct computation:
    \begin{equation*}
      \begin{split}
        \pa^\alpha \pa^\beta H_{\mu \alpha \nu \beta}
        & = - \dal \bar{h}_{\mu \nu} - \eta^{\mu \nu} \pa^\alpha \pa^\beta \bar{h}_{\alpha \beta} + \pa_\mu \pa^\alpha \bar{h}_{\alpha \nu} + \pa_\nu \pa^\alpha \bar{h}_{\mu \alpha} \\
        & = - \dal \left( h_{\mu \nu} - \frac{1}{2} \eta_{\mu \nu} h \right) - \eta_{\mu \nu} \left( \pa^\alpha \pa^\beta h_{\alpha \beta} - \frac{1}{2} \dal h \right) \\
        & \qquad \qquad \qquad \qquad + \pa_\mu \pa^\alpha \left( h_{\alpha \nu} - \frac{1}{2} \eta_{\alpha \nu} h \right) + \pa_\nu \pa^\alpha \left( h_{\mu \alpha} - \frac{1}{2} \eta_{\mu \alpha} h \right) \\
        & = - \dal h_{\mu \nu} + \pa_\mu \pa^\alpha h_{\alpha \nu} + \pa_\nu \pa^\alpha h_{\mu \alpha} - \pa_\mu \pa_\nu h - \eta_{\mu \nu} \left( \pa^\alpha \pa^\beta h_{\alpha \beta} - \dal h \right)
      \end{split}
    \end{equation*}
    Renaming dummy indices yields the thesis.
  \end{proof}
\end{proofbox}

The Lorentz tensor has the same symmetries of the Riemann tensor, namely:
\begin{equation}
  H_{\mu \alpha \nu \beta} = H_{\nu \beta \mu \alpha} = H_{[\mu \alpha] [\nu \beta]}
  \qquad \qquad
  H_{\mu [\alpha \nu \beta]} = 0
  \label{eq:lorentz-symmetries}
\end{equation}
With \eref{eq:linear-einstein}, it is possible to define the total $ 4 $-momentum in a spatial region $ V $ in the usual way:
\begin{equation*}
  \begin{split}
    P^\mu \equiv \int_V \dd^3x \, T^{\mu 0}
    & = \frac{1}{16\pi \gn} \int_V \dd^3x \, \pa_\alpha \pa_\beta H^{\mu \alpha 0 \beta} \\
    & = \frac{1}{16\pi \gn} \int_V \dd^3x \, \pa_\alpha \pa_i H^{\mu \alpha 0 i} = \frac{1}{16\pi \gn} \int_{\pa V} \dd^2S \, n_i \pa_\alpha H^{\mu \alpha 0 i}
  \end{split}
\end{equation*}
where the index $ \beta $ is restricted to be a spatial index by the total antisymmetry of $ H^{\mu \alpha \nu \beta} $ in the last three indices.
Similarly, for the total angular momentum:
\begin{equation*}
  \begin{split}
    J^{\mu \nu}
    & \equiv \int_V \dd^3x \, ( x^\mu T^{\nu 0} - x^\nu T^{\mu 0} ) = \frac{1}{16\pi \gn} \int_V \dd^3x \, ( x^\mu \pa_\alpha \pa_\beta H^{\nu \alpha 0 \beta} - x^\nu \pa_\alpha \pa_\beta H^{\mu \alpha 0 \beta} ) \\
    & = \frac{1}{8\pi \gn} \int_V \dd^3x (\pa_\beta (x^{[\mu} \pa_\alpha H^{\nu] \alpha 0 \beta}) - \delta^{[\mu}_\beta \pa_\alpha H^{\nu] \alpha 0 \beta}) \\
    & = \frac{1}{8\pi \gn} \int_V \dd^3x (\pa_i (x^{[\mu} \pa_\alpha H^{\nu] \alpha 0 i}) - \pa_\alpha H^{[\nu | \alpha 0 | \mu]}) \\
    & = \frac{1}{8\pi \gn} \int_V \dd^3x \, \pa_i (x^{[\mu} \pa_\alpha H^{\nu]\alpha 0 i} + H^{[\mu | i 0 | \nu]}) + \frac{1}{8\pi \gn} \int_V \dd^3x \, \pa_0 H^{[\mu | 0 0 | \nu]}
  \end{split}
\end{equation*}
The second term vanishes since $ H^{\nu 0 0 \mu} = H^{0 \mu \nu 0} = H^{\nu 0 0 \mu} $, hence the total angular momentum in a spatial region $ V $ is given by yet another flux integral:
\begin{equation*}
  J^{\mu \nu} = \frac{1}{8\pi \gn} \int_{\pa V} \dd^2S \, n_i (x^{[\mu} H^{\nu] \alpha 0 i} + H^{[\mu | i 0 | \nu]})
\end{equation*}
The integrands in these fluxes are not gauge invariant, since it vanishes in any local free-fall frame. On the contrary, assuming $ \pa V $ to be far from any source, the total fluxes are indeed gauge-invariant, and only asymptotic properties are relevant for the computation of total momentum and angular momentum.

\subsubsection{Landau--Lifshitz pseudotensor}

In the general case, the Einstein field equations can be rewritten in a form similar to \eref{eq:linear-einstein}:
\begin{equation}
  \pa_\alpha \pa_\beta H^{\mu \alpha \nu \beta} = 16 \pi \gn \gdet (T^{\mu \nu} + t^{\mu \nu})
\end{equation}
which requires the Einstein tensor to split like:
\begin{equation}
  G^{\mu \nu} = \frac{1}{2 \gdet} \pa_\alpha \pa_\beta H^{\mu \alpha \nu \beta} - 8 \pi \gn \gdet t^{\mu \nu}
  \label{eq:ll-splitting}
\end{equation}
Now, $ H^{\mu \alpha \nu \beta} $ is no longer a tensor, but retains the same symmetries \eref{eq:lorentz-symmetries}. Then, by total antisymmetry of the last three indices, $ \pa_\nu \pa_\alpha \pa_\beta H^{\mu \alpha \nu \beta} = 0 $, meaning that the right-hand side satisfies an ordinary conservation law:
\begin{equation}
  \pa_\nu [\gdet (T^{\mu \nu} + t^{\mu \nu})] = 0
  \label{eq:pseudo-emt-cons}
\end{equation}
The main problem is that the construction of $ t^{\mu \nu} $ by the splitting \eref{eq:ll-splitting} is far from unique. The most widespread choice is the one used by Landau and Lifshitz in \cite{landau-2}:
\begin{equation}
  H_\text{LL}^{\mu \alpha \nu \beta} = \gdens^{\mu \nu} \gdens^{\alpha \beta} - \gdens^{\alpha \nu} \gdens^{\mu \beta}
  \label{eq:ll-split}
\end{equation}
where  the \emph{metric density} $ \gdens^{\mu \nu} \equiv \sqg \, g^{\mu \nu} $ is not a tensor. The \bctxt{Landau--Lifshitz pseudo-EMT} is then found to be:
\begin{multline}
  t_\text{LL}^{\alpha \beta} = \frac{1}{16 \pi \gn} \frac{1}{\gdet} \bigg[ \pa_\lambda \gdens^{\alpha \beta} \pa_\mu \gdens^{\lambda \mu} - \pa_\lambda \gdens^{\alpha \lambda} \pa_\mu \gdens^{\beta \mu} + \frac{1}{2} g^{\alpha \beta} g_{\lambda \mu} \pa_\rho \gdens{\lambda \mu} \pa_\lambda \gdens^{\rho \mu} - 2 g^{(\alpha | \lambda} g_{\mu \nu} \pa_\rho \gdens^{\beta) \nu} \pa_\lambda \gdens^{\mu \rho} + \\
  + g_{\mu \nu} g^{\nu \rho} \pa_\nu \gdens^{\alpha \lambda} \pa_\rho \gdens^{\beta \mu} + \frac{1}{8} (2 g^{\alpha \lambda} g^{\beta \mu} - g^{\alpha \beta} g^{\lambda \mu}) (2 g_{\nu \rho} g_{\tau \sigma} - g_{\rho \sigma} g_{\nu \tau}) \pa_\lambda \gdens^{\nu \rho} \pa_\mu g^{\rho \sigma} \bigg]
\end{multline}
This expression is symmetric in $ \alpha \beta $ and quadratic in $ \pa \gdens $, and indeed the Landau--Lifshitz splitting is the unique choice for which the pseudo-EMT has these properties. Moreover, it has no dependence on second derivatives of the metric, so that the pseudo-EMT vanishes in local inertial frames as expected from the nature of the gravitational field.

By \eref{eq:pseudo-emt-cons}, it is possible to define an \emph{effective EMT} (which is not a tensor, however) containing information about the energy-momentum distribution of both the gravitational field and the matter fields:
\begin{equation}
  T_\text{eff}^{\mu \nu} \equiv \gdet (T^{\mu \nu} + t_\text{LL}^{\mu \nu})
  \qquad \qquad
  \pa_\mu T_\text{eff}^{\mu \nu} = 0
\end{equation}
Then, the effective EMT plays the same role as the standard EMT in flat spacetime, since it is symmetric and $ \pa_\mu T_\text{eff}^{\mu \nu} = 0 $. In flat spacetime, the total 4-momentum is defined as:
\begin{equation*}
  P^\mu \equiv \int_V \dd^3x \, T^{\mu 0}
  \quad \implies \quad
  \frac{\dd P^\mu}{\dd t} = \int_V \dd^3x \, \pa_0 T^{\mu 0} = - \int_V \dd^3x \, \pa_i T^{\mu i} = - \int_{\pa V} \dd^2S \, n_i T^{\mu i}
\end{equation*}
which is the usual flux integral. The same equations hold for the effective EMT, hence the total 4-momentum obeys:
\begin{equation}
  \frac{\dd P^\mu}{\dd t} = - \int_{\pa V} \dd^2S \, n_i \gdet (T^{\mu i} + t_\text{LL}^{\mu i})
\end{equation}
which correctly reduces to the above flat-spacetime expression for $ g_{\mu \nu} = \eta_{\mu \nu} $, thus confirming the interpretation as the total 4-momentum of the gravitational fields and the matter fields. Assuming that the spatial region $ V $ contains $ N $ sources of the gravitational field, each occupying a spatial region $ V_i \ssq V $, then:
\begin{equation*}
  P^\mu = \int_V \dd^3x \, T_\text{eff}^{\mu 0} = \sum_{i = 1}^N \int_{V_i} \dd^3x \, T_\text{eff}^{\mu 0} + \int_{\bar{V}} \dd^3x \, T_\text{eff}^{\mu 0} = \sum_{i = 1}^N P^\mu_i + \int_{\bar{V}} \dd^3x \, T_\text{eff}^{\mu 0}
\end{equation*}
where $ P^\mu_i $ is the total 4-momentum of the $ i^\text{th} $ source and $ \bar{V} \equiv V - \bigcup_{i = 1}^N V_i $ is the interbody region between sources. In the majority of cases, the gravitational field is only strong near the sources, hence the contribution of the interbody term is negligible.

\begin{observation}{Energy in asymptotically-flat spacetimes}{}
  In an asymptotically-flat spacetime, taking $ V $ to be asymptotically large means that $ g_{\mu \nu} \vert_{\pa V} \simeq \eta_{\mu \nu} $, hence:
  \begin{equation*}
    T_\text{eff}^{\mu \nu} \vert_{\pa V} = \gdet (T^{\mu \nu} + t_\text{LL}^{\mu \nu}) \vert_{\pa V} \simeq T^{\mu \nu} \vert_{\pa V} + t_\text{LL}^{\mu \nu} \vert_{\pa V}
  \end{equation*}
  Furthermore, assuming the absence of matter at infinity, i.e. $ T^{\mu \nu} \vert_{\pa V} \simeq 0 $, reduces the effective EMT to the Landau--Lifshitz pseudo-EMT. Now, note that schematically (working in simil-spherical coordinates):
  \begin{equation*}
    \gdens \sim r^{-n}
    \quad \implies \quad
    \pa \gdens \sim r^{-n-1}
    \quad \implies \quad
    t_\text{LL} \sim r^{-2n -2}
    \quad \implies \quad
    \frac{\dd P}{\dd t} \sim r^2 t_\text{LL} \sim r^{-2n}
  \end{equation*}
  Therefore, the only non-negligible contribution to the flux integral comes from metric terms with $ n \leq 0 $: for example, GWs propagating at infinity have $ n = 0 $, hence their case needs to be studied carefully.
\end{observation}

\subsection{ADM formalism}

The Arnowitt--Deser--Misner (ADM) formalism consists of a Hamiltonian formulation of General Relativity built on a decomposition of spacetime into spacelike hypersurfaces.

\begin{definition}{Foliation}{}
  Let $ \man $ be a smooth $ n $-manifold, and let $ \rsscript{F} $ be any collection of $ k $-dimensional submanifolds of $ \man $. A smooth chart $ (U , \varphi) $ on $ \man $ is said to be \emph{flat for $ \rsscript{F} $} if $ \varphi(U) \ssq \R^n $ is a cube and each submanifold in $ \rsscript{F} $ intersects $ U $ in either the empty set or a countable union of $ k $-dimensional slices of the form $ x^{k+1} = c^{k+1} , \dots , x^n = c^n $.

  Then, $ \rsscript{F} $ is a \bcdef{$ k $-dimensional foliation} of $ \man $ if it consists of disjoint connected non-empty immersed $ k $-dimensional submanifolds of $ \man $, called the \emph{leaves of the foliation}, whose union is $ \man $ and such that in a neighbourhood of each point $ p \in \man $ there exists a flat chart for $ \rsscript{F} $.
\end{definition}

It is convenient to work not with arbitrary foliations of spacetime, but with particular foliations best-suited to express initial-value problems: indeed, the Einstein field equations are hyperbolic, hence fields at any point $ p \in \man $ can be influenced only by events in its causal past.

\begin{definition}{Cauchy hypersurface}{}
  Let $ (\man , g) $ be a spacetime. A \emph{causal curve} is a piecewise smooth curve whose tangent is everywhere timelike or null and future-directed (or past-directed): such a curve is \emph{inextendible} if it admits no extension to another causal curve\footnote{Intuitively, this means that the curve has no past or future endpoints in $ \man $.}.

  Then, a hypersurface $ \Sigma \ssq \man $ is called a \bcdef{Cauchy hypersurface} if every inextendible causal curve in $ \man $ intersects $ \Sigma $ exactly once.
\end{definition}

If $ \Sigma $ is a Cauchy hypersurface of a spacetime $ (\man , g) $, the causal past or future of every point $ p \in \man $ intersects $ \Sigma $ exactly once, thus initial data for the metric and matter fields on $ \Sigma $ uniquely determine the solution to the Einstein field equations in the entire spacetime. For this reason, a spacetime which admits a Cauchy hypersurface is called \bctxt{globally hyperbolic}.

Consider a globally hyperbolic spacetime $ (\man , g) $ and assume it admits a $ (d-1) $-dimensional foliation into spacelike achronal\footnotemark hypersurfaces $ \{\Sigma_t\} $, where $ t \in \cm $ is a smooth time function whose level sets are Cauchy hypersurfaces. On each slice $ \Sigma_t $ introduce spatial coordinates $ x^1 , \dots , x^{d-1} $ so that $ x^\mu = (t , x^1 , \dots x^{d-1}) $ defines a coordinate system adapted to the foliation. Let $ n^\mu $ be the future-directed unit normal to the hypersurface $ \Sigma_t $, so that $ n_\mu n^\mu = -1 $ and $ n_\mu \propto \na_\mu t $, and decompose the time evolution vector field $ t^\mu \defeq (\pa_t)^\mu $ as:

\footnotetext{An hypersurface is \emph{achronal} it no pair of its points can be connected by a timelike curve.}

\begin{equation*}
  t^\mu = N n^\mu + N^\mu
\end{equation*}
where $ N $ is the \emph{lapse function} and $ N^\mu $ is the \emph{lapse vector} tangent to $ \Sigma_t $. Geometrically, $ N $ measures the proper time separation between adjacent slices along the normal, while $ N^i $ encodes the relative displacement of spatial coordinates from one slice to the next. Then, the metric can be shown to take the form:
\begin{equation}
  \dd s^2 = - N^2 \dd t^2 + \gamma_{ij} (\dd x^i + N^i \dd t) (\dd x^j + N^j \dd t)
  \label{eq:adm-metric}
\end{equation}
where $ \gamma_{ij} $ is the induced metric on the spacelike slices. This metric is completely parametrized by $ N $, $ N^i $ and $ \gamma_{ij} $, hence it retains the correct number of degrees of freedom $ 1 + (d-1) + \frac{1}{2} d (d-1) = \frac{1}{2} d (d+1) $.

This formalism naturally provides definitions for the total energy, momentum and angular momentum of spacetime in terms of boundary intergrals at spatial infinity. In particular, given an asymptotically-flat spacetime with ADM decomposition \eref{eq:adm-metric} such that $ \gamma_{ij} \ra \delta_{ij} $ sufficiently fast at spatial infinity, its \bctxt{ADM mass} is defined as:
\begin{equation}
  M_\text{ADM} \defeq \frac{1}{16 \pi \gd} \int_{\Sn{d-2}_\infty} \dd^{d-2}S \, n^i (\pa^j \gamma_{ij} - \pa_i \gamma)
  \label{eq:adm-mass}
\end{equation}
where $ \gamma \equiv \tensor{\gamma}{^i_i} $. In the case of stationary spacetimes, the ADM mass coincides with the Komar mass, i.e. with the Komar charge \eref{eq:komar-charge} associated to the timelike Killing vector field (which exists by \dref{def:stationary-spacetime}).

\subsubsection{Weak gravity}

In the case of a small perturbation $ h_{\mu \nu} $ propagating on a $ d = 4 $ Minkowski background, the ADM mass \eref{eq:adm-mass} can be obtained extending the analysis in \secref{sssec:pseudo-emt-weak-grav}.

Given the perturbed metric $ g_{\mu \nu} = \eta_{\mu \nu} + h_{\mu \nu} $, at NLO $ g^{\mu \nu} = \eta^{\mu \nu} - h^{\mu \nu} + \smo(h^2) $ and:
\begin{equation*}
  \sqg = 1 + \frac{h}{2} + \smo(h^2)
  \quad \implies \quad
  \gdens^{\mu \nu} = \eta^{\mu \nu} - h^{\mu \nu} + \frac{h}{2} \eta^{\mu \nu} + \smo(h^2) = \eta^{\mu \nu} - \bar{h}^{\mu \nu} + \smo(h^2)
\end{equation*}
Then, the Landau--Lifshitz split \eref{eq:ll-split} becomes:
\begin{equation*}
  \pa_\alpha H^{\mu \alpha \nu \beta}_\text{LL} = - \pa^\beta \bar{h}^{\mu \nu} - \eta^{\mu \nu} \pa_\alpha \bar{h}^{\alpha \beta} + \pa^\nu \bar{h}^{\mu \beta} + \eta^{\mu \beta} \pa_\alpha \bar{h}^{\alpha \nu} + \smo(h^2)
\end{equation*}
This expression allows to explicitly compute the total energy and momentum of spacetime by direct calculation.

\begin{lemma}[before upper = {\tcbtitle}]{ADM energy and momentum in weak gravity}{}
  \begin{equation}
    P^0 = \frac{1}{16\pi \gn} \int_{\Sn{2}_\infty} \dd^2S \, n_i (\pa_j h_{ij} - \pa_i h)
  \end{equation}
  \begin{equation}
    P^j = \frac{1}{16\pi \gn} \int_{\Sn{2}_\infty} \dd^2S \, n_i [\delta_i^j (\pa_0 h - \pa_k h_{k0}) + \pa_i h_{j0} - \pa_0 h_{ij}]
  \end{equation}
\end{lemma}

\section{Propagation of GWs}

The perturbative study of GWs is based on the separation of the waves from the background, as in \eref{eq:gw-pert-exp}: the problem is that, in a general setting, there is no unambiguous way to perform this separation, and on top of that it is not gauge invariant either.

A natural splitting between the spacetime background and the gravitational waves arises when there is a clear separation of scales. This is the case, for example, when in some coordinate system the metric can be written as in \eref{eq:gw-pert-exp}, where $ \bar{g} $ has a typical scale of spatial variation $ L_\text{b} $, while $ h $ is characterized by a wavelength\footnotemark $ \lambda $ such that:

\footnotetext{For a function $ f(x) $ oscillating as $ e^{\img k x} $, with $ k = 2\pi / \lambda \equiv \lbar^{-1} $, the typical length-scale is $ \lbar $ rather than $ \lambda $, in the sense that $ \abs{f'} = \lbar^{-1} \abs{f} $.}

\begin{equation}
  \lbar \ll L_\text{b}
  \label{eq:l-condition}
\end{equation}
In this case, $ h $ has the physical interpretation of small ripples on a smooth background. Equivalently, in frequency space, if $ \bar{g} $ contains frequencies up to $ f_\text{b} $, while $ h $ is a monochromatic wave peaked around a frequency $ f $, then it must be:
\begin{equation}
  f \gg f_\text{b}
  \label{eq:f-condition}
\end{equation}
which is the case of a high-frequency perturbation over a slowly-varying background. Note that the scales $ L_\text{b} $ and $ f_\text{b} $ are a priori unrelated, hence the two condition are independent, and it suffices that one of them is satisfied: \eref{eq:l-condition} is known as the \emph{short-wave expansion}, but for GWs \eref{eq:f-condition} is usually fulfilled instead\footnotemark.

\subsection{Back-reaction on the background}

To study how GWs modify the background spacetime as they propagate, expand the $ \rr_{\mu \nu} $ in powers of $ h $:
\begin{equation*}
  \rr_{\mu \nu} = \sum_{n \in \N_0} [\rr\od{n}] = \bar{\rr}_{\mu \nu} + [\rr\od1]_{\mu \nu} + [\rr\od2]_{\mu \nu} + \smo(h^3)
\end{equation*}
where $ \bar{\rr}_{\mu \nu} $ is the Ricci tensor of the background, thus containing only low-frequency modes. On the other hand, $ [\rr\od1]_{\mu \nu} $ is linear in $ h_{\mu \nu} $, thus containing only high-frequency modes, while $ [\rr\od2]_{\mu \nu} $ is quadratic in $ h_{\mu \nu} $, thus containing both low- and high-frequency modes: indeed, in a quadratic term $ \sim h_{\mu \nu} h_{\rho \sigma} $ a mode with high $ \ve{k}_1 $ from $ h_{\mu \nu} $ can combine with a mode with high $ \ve{k}_2 \simeq - \ve{k}_1 $ from $ h_{\rho \sigma} $, yielding a low-frequency mode. This separation into low- and high-frequency modes can be imposed on the Einstein field equations, suitably written as:
\begin{equation*}
  \rr_{\mu \nu} = 8\pi \gn \left( T_{\mu \nu} - \frac{1}{2} g_{\mu \nu} T \right)
\end{equation*}
In order to do this, introduce the \bctxt{Brill--Hartle--Isaacson average} (BHI average) $ \braket{A} $ which spatially averages the quantity $ A $ over a scale of order of a few wavelengths $ \lambda $, so that:
\begin{equation*}
  \bar{g}_{\mu \nu} \equiv \braket{g_{\mu \nu}}
  \qquad \qquad
  h_{\mu \nu} = g_{\mu \nu} - \braket{g_{\mu \nu}}
\end{equation*}
This makes manifest that the BHI average eliminates oscillations on scales $ \sim \lambda $, leaving only low-frequency modes: in particular, $ \braket{h_{\mu \nu}} = 0 $, as trivially seen by the above relations. The BHI average can be shown to have the following properties (integration by parts and integration of total derivative):
\begin{equation}
  \braket{A \bar{\na}_\alpha B} = - \braket{B \bar{\na}_\alpha A} + \smo \left( \frac{\lbar^2}{L_\text{b}^2} \right)
  \qquad \qquad
  \braket{\bar{\na}_\alpha A} = 0 + \smo \left( \frac{\lbar}{L_\text{b}} \right)
  \label{eq:bhi-prop}
\end{equation}
Setting the notation $ A^\text{Low} \equiv \braket{A} $ and $ A^\text{High} \equiv A - \braket{A} $, the above observation on the mode composition of the Ricci tensor at various orders is confirmed by the expansion at NLO:
\begin{equation}
  [\rr\od1]_{\mu \nu} = \frac{1}{2} \left( \bar{\na}^\alpha \bar{\na}_\mu h_{\nu \alpha} + \bar{\na}^\alpha \bar{\na}_\nu h_{\mu \alpha} - \bar{\dal} h_{\mu \nu} - \bar{\na}_\nu \bar{\na}_\mu h \right)
\end{equation}
which yields $ \braket{[\rr\od1]_{\mu \nu}} = 0 $ by $ \braket{h_{\mu \nu}} = 0 $. On the contrary, schematically $ [\rr\od2]_{\mu \nu} \sim (\na h)^2 + h \na^2 h $, so $ \braket{[\rr\od2]_{\mu \nu}} \sim \braket{(\na h)^2} $ by integration by parts. The Einstein field equations are then split as:
\begin{equation}
  \bar{\rr}_{\mu \nu} + [\rr\od2]_{\mu \nu}^\text{Low} = 8\pi \gn \left( T_{\mu \nu} - \frac{1}{2} g_{\mu \nu} T \right)^\text{Low}
  \label{eq:gw-low}
\end{equation}
\begin{equation}
  [\rr\od1]_{\mu \nu} + [\rr\od2]_{\mu \nu}^\text{High} = 8\pi \gn \left( T_{\mu \nu} - \frac{1}{2} g_{\mu \nu} T \right)^\text{High}
  \label{eq:gw-high}
\end{equation}
The low-frequency equation gives the scale of $ h $: in vacuum $ \bar{\rr}_{\mu \nu} + [\rr\od2]_{\mu \nu} = 0 $, thus, given $ \bar{\rr} \sim L_\text{b}^{-2} $ and $ \rr\od2 \sim h^2 / \lbar^2 $, the perturbation scale is $ h \sim \lbar / L_\text{b} $; in presence of matter, instead, $ L_\text{b}^{-2} + h^2 / \lbar^2 $ equals to the matter contribution, which is dominant if $ L_\text{b}^{-2} \gg h^2 / \lbar^2 $, i.e. if $ h \ll \lbar / L_\text{b} $. In general, GWs have typical scale:
\begin{equation}
  h \lesssim \frac{\lbar}{L_\text{b}}
  \label{eq:gw-ampl-constraint}
\end{equation}
Notice that $ \bar{g}_{\mu \nu} \equiv \eta_{\mu \nu} $ means $ L_\text{b}^{-1} \ra 0 $, hence by the above relation the background spacetime cannot be flat: GWs always back-react on the background. The amplitude of a GW can violate \eref{eq:gw-ampl-constraint} only in a non-perturbative description, e.g. in presence of an exact solution.

\subsection{Energy carried by a GW}

To express the back-reaction of GWs on the background spacetime clearly, define the effective background contribution of matter\footnotemark as $ \braket{T_{\mu \nu} - \frac{1}{2} g_{\mu \nu} T} \equiv \bar{T}_{\mu \nu} - \frac{1}{2} \bar{g}_{\mu \nu} \bar{T} $, with $ \bar{T} \equiv \bar{g}_{\mu \nu} \bar{T}^{\mu \nu} $, so that \eref{eq:gw-low} can be rewritten as:
\begin{equation}
  \bar{\rr}_{\mu \nu} = - [\rr\od2]_{\mu \nu}^\text{Low} + 8 \pi \gn \left( \bar{T}_{\mu \nu} - \frac{1}{2} \bar{g}_{\mu \nu} \bar{T} \right)
  \label{eq:gw-background}
\end{equation}
This expression directly links the background curvature to the background matter contribution plus an additional gravitational contribution due to the GWs: $ \braket{\rr\od2} $ contains the large-scale back-reaction of small-scale high-frequency GWs.

\footnotetext{By definition, $ \bar{T}_{\mu \nu} $ only has low-frequency (or low-momentum) modes, and can be viewed as a smoothed form ``macroscopic" form of the fundamental ``microscopic" EMT $ T_{\mu \nu} $. The EMT generated by a macroscopic matter distribution is already quite smooth, so in this case $ \bar{T}_{\mu \nu} \simeq T_{\mu \nu} $.}

In order to study the energy carried by a GW, define the pseudo-EMT $ t_{\mu \nu} $ as:
\begin{equation}
  t_{\mu \nu} \defeq - \frac{1}{8\pi \gn} \left\langle [\rr\od2]_{\mu \nu} - \frac{1}{2} g_{\mu \nu} \rr\od2 \right\rangle
  \label{eq:pseudo-emt-def}
\end{equation}
where $ \rr\od2 \equiv g^{\mu \nu} [\rr\od2]_{\mu \nu} $. Then, defining the trace $ t \equiv g^{\mu \nu} t_{\mu \nu} $, at NNLO:
\begin{equation*}
  [t\od2]_{\mu \nu} = - \frac{1}{8\pi \gn} \left( \braket{[\rr\od2]_{\mu \nu}} - \frac{1}{2} \bar{g}_{\mu \nu} \bar{g}^{\alpha \beta} \braket{[\rr\od2]_{\alpha \beta}} \right)
  \qquad \qquad
  t\od2 = \frac{1}{8\pi \gn} \bar{g}^{\alpha \beta} \braket{[\rr\od2]_{\alpha \beta}}
\end{equation*}
This pseudo-EMT represents the self-consistent contribution of the GW to the background geometry: being defined by through the BHI average, it respects the equivalence principle, since it does not provide a localized notion of gravitational EMT. Inserting the pseudo-EMT into \eref{eq:gw-background}:
\begin{equation*}
  \bar{R}_{\mu \nu} = 8\pi \gn \left( t_{\mu \nu} - \frac{1}{2} \bar{g}_{\mu \nu} t + \bar{T}_{\mu \nu} - \frac{1}{2} \bar{g} \bar{T} \right)
\end{equation*}
or, equivalently:
\begin{equation}
  \bar{G}_{\mu \nu} = 8\pi \gn \left( t_{\mu \nu} + \bar{T}_{\mu \nu} \right)
\end{equation}
which makes it clear that this pseudo-EMT encodes the energy-momentum contribution of the GW which back-reacts on the background spacetime. The background Bianchi identities $ \bar{\na}^\mu \bar{G}_{\mu \nu} = 0 $ lead to the covariant conservation $ \bar{\na}^\mu (t_{\mu \nu} + \bar{T}_{\mu \nu}) = 0 $.

Then:
\begin{equation*}
  \int_{\pa V} \dd^2S \, n^k \braket{\pa_0 h^\text{TT}_{ij} \pa_k h^\text{TT}_{ij}} = - \int_{\pa V} \dd^2S \, n^k n_k \braket{\dot{h}^\text{TT}_{ij} \dot{h}^\text{TT}_{ij}} = - r^2 \int_{\Sn{2}_r} \dd \Omega_2 [t\od2]_{00}
\end{equation*}
where in the last step it has been assumed that $ \pa V \cong \Sn{2}_r $. The irradiated energy can then be computed as:
\begin{equation}
  \frac{\dd E_V}{\dd t} = - \frac{1}{32 \pi \gn} r^2 \int_{\Sn{2}_r} \dd \Omega_2 \braket{\dot{h}^\text{TT}_{ij} \dot{h}^\text{TT}_{ij}}
\end{equation}

\subsection{Propagation in curved spacetime}

While the low-frequency equation \eref{eq:gw-low} allows to identify the pseudo-EMT of the GW, its propagation through spacetime is determined by the high-frequency equation \eref{eq:gw-high}.

First of all, consider the limiting case of no external sources, i.e. $ T_{\mu \nu} = 0 $, so that \eref{eq:gw-high} becomes:
\begin{equation*}
  [\rr\od1]_{\mu \nu} = - [\rr\od2]_{\mu \nu}^\text{High}
\end{equation*}
Performing order-of-magnitude estimates:
\begin{equation*}
  \rr\od1 \sim \pa^2 h \sim \frac{h}{\lbar^2}
  \qquad \qquad
  [\rr\od2]^\text{High} \sim (\pa h)^2 \sim \frac{h^2}{\lbar^2}
\end{equation*}
Recalling \eref{eq:gw-ampl-constraint}, a small parameter $ \varepsilon \in \R^+ $ can be defined as $ \varepsilon \sim h \sim \lbar / L_\text{b} $, so that:
\begin{equation*}
  \rr\od1 \sim \frac{1}{\varepsilon L_\text{b}}
  \qquad \qquad
  [\rr\od2]^\text{High} \sim \frac{1}{L_\text{b}}
\end{equation*}
Since $ \varepsilon > 0 $ is small, the second term is subleading when compared to the first, hence at LO:
\begin{equation}
  [\rr\od1]_{\mu \nu} \simeq 0
\end{equation}
This equation can be written in the Lorentz gauge as:
\begin{equation*}
  \dal \bar{h}_{\mu \nu} + 2 \bar{\rr}_{\mu \alpha \nu \beta} \bar{h}^{\alpha \beta} - 2 \bar{\rr}_{\alpha (\nu} \tensor{\bar{h}}{_{\mu)}^\alpha} \simeq 0
\end{equation*}
where $ \dal \equiv \pa_\mu \pa^\mu $ since $ \bar{g}_{\mu \nu} \simeq \eta_{\mu \nu} $ at LO in $ \varepsilon $.
The two curvature terms are of order $ \sim h / L_\text{b}^2 \sim \varepsilon^3 / \lbar^2 $, while the first term is of order $ \sim h / \lbar^2 \sim \varepsilon / \lbar^2 $, hence the curvature terms are subleading and at LO:
\begin{equation}
  \dal \bar{h}_{\mu \nu} \simeq 0
\end{equation}
This is just the wave equation derived from the linearized theory, which thus remains valid in the limit of GWs being the only sources of curvature. In the absence of matter (or at least far from sources) the TT gauge can be used, and the wave equation has general solution:
\begin{equation}
  h^\text{TT}_{ij}(x) = \int_{\R^3} \frac{\dd^3k}{(2\pi)^3} \left[ \ampl_{ij}(k) e^{\img k \cdot x} + \ampl^*_{ij}(k) e^{-\img k \cdot x} \right]
\end{equation}
with $ k \equiv (\omega , \ve{k}) : \norm{\ve{k}} = \omega = 2\pi f $. Typical astrophysical sources are localized in a certain direction $ \hat{\ve{k}}_0 $, so $ \ampl_ij(k) = \ampl_{ij}(f) \delta^{(3)}(\hat{\ve{k}} - \hat{\ve{k}}_0) $. Note that for ground detectors the spatial dependence of the wave can often be ignored.

Now, consider the presence of external matter dominating the curvature. In this case, recalling the discussion on \eref{eq:gw-ampl-constraint}, $ h \ll \lbar / L_\text{b} \ll 1 $, so the expansions in $ h $ and in $ \lbar / L_\text{b} $ are different: for GWs, terms quadratic in $ h $ are extremely negligible, so retain only terms linear in $ h $ and expand them in powers of $ \lbar / L_\text{b} $. In this case, $ [\rr\od2]^\text{High} \sim h^2 $ is neglibile, while the source term is:
\begin{equation*}
  \left( T_{\mu \nu} - \frac{1}{2} g_{\mu \nu} T \right)^\text{High} \sim \frac{h}{L_\text{b}^2}
\end{equation*}
This is because $ T_{\mu \nu} $ is smooth for macroscopic matter sources, hence its high-frequency components depend on the perturbation $ h_{\mu \nu} $, thus $ T_{\mu \nu} \sim h / L_\text{b}^2 $, and similarly for $ g_{\mu \nu} T = (\bar{g}_{\mu \nu} + h_{\mu \nu}) T $. On the contrary, $ \rr\od1 \sim \pa^2 h \sim h / \lbar^2 $, so the source term is smaller by a factor $ \smo (\lbar^2 / L_\text{b}^2) $, which does not contribute at LO and NLO. The linear equation at NLO in $ \lbar / L_\text{b} $ is then:
\begin{equation*}
  [\rr\od1]_{\mu \nu} = 0
\end{equation*}
As discussed above, in the Lorentz gauge this equation leads to:
\begin{equation}
  \bar{\dal} \bar{h}_{\mu \nu} = 0
\end{equation}
where now $ \bar{\dal} \equiv \bar{\na}_\mu \bar{\na}^\mu $. This wave equation therefore determines the propagation of GWs in a curved background, in the short-wave limit $ \lbar \ll L_\text{b} $ (or equivalently $ f \gg f_\text{b} $). 

In conclusion, the separation of the Einstein field equations into low- and high-frequency components allows to describe both the effects of GWs and external matter on the background spacetime (low-frequency), and the propagation of GWs in the curved background (high-frequency).
