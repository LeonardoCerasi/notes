% local table of contents
\usepackage{etoc}

\newcommand{\localtoc}{%
  \hypersetup{linkcolor = black}%
  \par
  {\sffamily \bfseries Contents}%
  \vspace{1.5ex}%
  \hrule height 0.4pt
  \vspace{1.5ex}%
  {%
    \etocsettocstyle{}{}%
    \etocsetnexttocdepth{1}%
    \localtableofcontents%
  }%
  \par
  \vspace{1.5ex}%
  \hrule height 0.4pt
  \vspace{1.5ex}%
  \hypersetup{linkcolor = \primcol}%
  \vspace{10pt}
}

  }
